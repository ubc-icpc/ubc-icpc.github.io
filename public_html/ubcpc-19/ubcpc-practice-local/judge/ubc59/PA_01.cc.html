char change_case (char c, int i) {
    if (i != 0) 
        return std::tolower(c); 
    else
        return std::toupper(c); 
}

int main() {
    // concatenate b before a and change case
    string a, b, c;
    cin >> a; cin >> b;
    c = b + a;
    for(int i = 0; i < c.size(); i++) {
        c[i] = change_case(c[i], i);
    }
    cout << c << endl;

    return 0;
}
