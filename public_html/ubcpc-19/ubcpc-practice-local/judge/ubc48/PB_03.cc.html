using namespace std;

#define long long long
#define fi first
#define se second
typedef pair<int,int> ii;

const double nt = pow(2, 1.0/12.0), cents = pow(2, 1.0/1200.0), err = pow(2, 1.0/2000.0);

string ans[] = {"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};

int main()
{
	// ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	// freopen("input.in", "r", stdin);
	double p, q;
	cin >> p >> q;
	
	bool rev = (p > q);
	if(p > q) swap(p, q);
	// cout << p << " " << q << "\n";

	double target = q/p;
	
	for(int i = 0; i <= 300; i++)
	{
		for(int j = -25; j <= 25; j++)	
		{
			for(int k = -25; k <= 25; k++)
			{
				double ratio = pow(nt, i)*pow(cents, -j+k);
				if(ratio > target)
				{
					if(target*err >= ratio)
					{
						if(!rev)
						{
							cout << "A " <<  j << "\n";
							cout << ans[i%12] << " " << k << "\n";
						}
						else
						{
							cout << ans[i%12] << " " << k << "\n";
							cout << "A " <<  j << "\n";
						}
						return 0;
					}

				}
				else
				{
					if(ratio*err >= target)
					{
						if(!rev)
						{
							cout << "A " <<  j << "\n";
							cout << ans[i%12] << " " << k << "\n";
						}
						else 
						{
							cout << ans[i%12] << " " << k << "\n";
							cout << "A " <<  j << "\n";
						}
						return 0;
					}
				}
			}
		}
	}
}
