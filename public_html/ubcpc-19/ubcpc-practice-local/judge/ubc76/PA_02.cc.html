#include <bits/stdc++.h>
using namespace std;
string a, b;
char lower[10000];
char upper[10000];
int main() {
	cin >> a >> b;
	for (int i = 0; i < 26; i++) {
		lower['A' + i] = ('a' + i);
		upper['a' + i] = ('A' + i);
	}
	for (int i = 0; i < a.length(); i++) {
		if ('A' <= a[i] && a[i] <= 'Z')
			a[i] = lower[a[i]];
	}
	for (int i = 1; i < b.length(); i++) {
		if ('A' <= b[i] && b[i] <= 'Z')
			b[i] = lower[b[i]];
	}
	if ('a' <= b[0] && b[0] <= 'z')
		b[0] = upper[b[0]];
	cout << b << a << endl;
}
