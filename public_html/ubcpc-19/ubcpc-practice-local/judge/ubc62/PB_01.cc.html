using namespace std;

const double EPS = 1e-8;
const double limit = pow(2.0, 0.0005);

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    unordered_map<int, string> notes;
    notes[0] = "A"; notes[1] = "A#"; notes[2] = "B"; notes[3] = "C"; notes[4] = "C#"; notes[5] = "D";
    notes[6] = "D#"; notes[7] = "E"; notes[8] = "F"; notes[9] = "F#", notes[10] = "G"; notes[11] = "G#";
    double p, q;
    bool swap=0;
    cin >> p >> q;
    if(p>q) {
        swap = 1;
        double t = p;
        p = q;
        q = t;
    }
    double ratio = q/p;
    while(ratio-2>EPS)
        ratio /= 2;
    for(int a=0; a<=12; a++)
        for(int n1=-25; n1<=25; n1++)
            for(int n2=-25; n2<=25; n2++) {
                double r = pow(2.0, double(100*a+n2-n1)/1200.0);
                if(fabs(log(r)-log(ratio))<log(limit)) {
                    if(swap) {
                        cout << notes[a%12] << " " << n2 << "\n";
                        cout << notes[0] << " " << n1 << "\n";
                    }
                    else {
                        cout << notes[0] << " " << n1 << "\n";
                        cout << notes[a%12] << " " << n2 << "\n";
                    }
                    return 0;
                }
            }


    return 0;
}
