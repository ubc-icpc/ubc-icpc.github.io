#include <iostream>
#include <string>


using namespace std;

int main(int argc, const char * argv[]) {
    
//    freopen("input.in", "r", stdin);
    
    string a, b;
    
    cin >> a >> b;
    
    b.append(a);
    
    b[0] = toupper(b[0]);
    
    for(int i = 1; i < b.size(); i++) {
        b[i] = tolower(b[i]);
    }
    
    cout << b << endl;
    
    
    return 0;
}
