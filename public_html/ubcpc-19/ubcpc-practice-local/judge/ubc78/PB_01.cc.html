typedef long long ll;typedef pair<int,int> pii;typedef vector<bool> vbool;typedef vector<int> vint;typedef vector<ll> vll;typedef long double ld;typedef vector<vector<int>> vvi; 

const ll N = 100010;
const ll INF = 0x3f3f3f3f;
const ll MOD = 1e9 + 7;

string freq[12] = {"A","A#","B","C","C#","D","D#","E","F","F#","G","G#"};

int main() {
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	ld p, q; cin >> p >> q;
	ld x = max(p,q)/min(p,q);
	int sum = round(1200.0*log2l(x));
	int c = min(sum%100, 100-sum%100), d = (sum+c)/100;
	string fi = freq[0]    + " " + ((c+d*100 > sum)? to_string(c/2) : to_string(-c/2));
	string se = freq[d%12] + " " + ((c+d*100 > sum)? to_string(-(c-c/2)):to_string((c-c/2)));
	if (p < q) 
		cout << fi << nl << se << nl; 
	else
		cout << se << nl << fi << nl; 
	return 0;
}
