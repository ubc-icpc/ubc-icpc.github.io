#include<bits/stdc++.h>
using namespace std;

int main() {
	ios::sync_with_stdio(false);
	string A; cin >> A; const unsigned int size_A = A.size();
	string B; cin >> B; const unsigned int size_B = B.size();
	char str[size_A+size_B+1]; memset(str, 0, sizeof(str));
	for(unsigned i = 0; i < size_B; i++) {
		str[i] = B[i];
	}
	for(unsigned i = 0; i < size_A; i++) {
		str[i+size_B] = A[i];
	}
	str[size_B] = tolower(str[size_B]);
	cout << str << "\n";
	return 0;
}
