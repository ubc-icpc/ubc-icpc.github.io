#include <bits/stdc++.h>
const char nl = '\n';
using namespace std;
using ll = long long;
using ld = long double;
const ld INF = 1e18;
const vector<string> note = {"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  int p, q; cin >> p >> q;
  ld target = log2(p) - log2(q);
  ld best = INF; string besti;
  for (int i = 0; i < 12; i++) {
    for (int j = -25; j <= 25; j++) {
      for (int k = 0; k < 12; k++) {
        for (int l = -25; l <= 25; l++) {
          for (int o = -8; o <= 8; o++) {
            ld cur = (i/12.L + j/1200.L) - (k/12.L + l/1200.L) + o;
            if (abs(cur-target) < best) {
              best = abs(cur-target);
              besti = note[i] + " " + to_string(j) + "\n" + note[k] + " " + to_string(l);
            }
          }
        }
      }
    }
  }
  //cerr << best << nl;
  cout << besti << nl;
  return 0;
}
