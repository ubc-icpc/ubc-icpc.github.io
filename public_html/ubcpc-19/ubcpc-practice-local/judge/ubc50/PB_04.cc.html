#include<cstdio>
#include<cmath>
using namespace std;
void print(int x)
{
	switch (x)
	{
	case 0:printf("A");break; 
	case 1:printf("A#");break; 
	case 2:printf("B");break; 
	case 3:printf("C");break; 
	case 4:printf("C#");break; 
	case 5:printf("D");break; 
	case 6:printf("D#");break; 
	case 7:printf("E");break; 
	case 8:printf("F");break; 
	case 9:printf("F#");break; 
	case 10:printf("G");break; 
	case 11:printf("G#");break; 
    }
}
int p,q,ans,x,y,x1,x2,a1,a2;
int main()
{
	scanf("%d%d",&p,&q);
	while (p<q) p*=2;
	while (p>=q*2) q*=2;
	//printf("%d %d\n",p,q);
	double t=log((double)p/q)/log(2);
	//printf("%.2lf\n",t);
	double tt=t*1200;
	double l=floor(tt);
	if (tt-l>0.5) ans=l+1;
	else ans=l;
	if (ans%100<=50) x=ans/100;
	else x=ans/100+1;
	y=ans-100*x;
	x1=x;x2=0;
	if (y>=25) 
	{
		a1=25;
		a2=a1-y; 
	} 
	else if (y>-25)
	{
		a1=y;
		a2=0;
	}
	else 
	{
		a1=-25;
		a2=a1-y;
	}
	//printf("%d %d\n",x1,x2);
	if (x1==12) x1=0;
	print(x1);
	printf(" %d\n",a1);
	print(x2);
	printf(" %d",a2);
}
