#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<ll, ll> pii;

ld p, q;

vector<string> notes = {"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};

int main() {
    ios_base::sync_with_stdio(0), cin.tie(0);
    cin >> p >> q;
    swap(p, q);
    ll x = round(log((ld) p / q)/log((ld) 2)*1200);
    ll y = (x + 50)/100;
    if (x < -50) y -= 1;
    ll cts = x - y*100;
    ll m = (y + 12*10000)/12;
    ll r = (y + 12*10000)%12;

    ll d1 = cts/2;
    ll d2 = cts - d1;
    d1 *= -1;
    
    cout << 'A' << " " << d1 << endl;
    cout << notes[r] << " " << d2 << endl;
    return 0;
}
