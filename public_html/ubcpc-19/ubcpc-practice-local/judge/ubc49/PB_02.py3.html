import math, sys
NOTES = 'A, A#, B, C, C#, D, D#, E, F, F#, G, G#'.split(', ')
p, q = map(int, input().split(' '))

r = q/p

diff = math.log(r, 2)*12
b = round(diff)
diff -= b
cents = round(diff * 100)
if (cents < 0):
    c1 = 25
    c2 = cents + c1
else:
    c1 = -25
    c2 = cents + c1

a = NOTES.index('A')
b += a

print(NOTES[a] + ' ' + str(c1))
print(NOTES[b % len(NOTES)] + ' ' + str(c2))
