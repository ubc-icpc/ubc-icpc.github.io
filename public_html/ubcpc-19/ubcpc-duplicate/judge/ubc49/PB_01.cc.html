#include <bits/stdc++.h>
using namespace std; inline void fileio(const char* in, const char* out) { freopen(in, "r", stdin); freopen(out, "w", stdout); } 
#define TD typedef
#define forx(i,n) for(int i = 0; i < (n); ++i)
#define forxr(i,n) for(int i = (n)-1; i >= 0; --i)
TD long long ll; TD long double ld; TD pair<int,int> pii; TD pair<ll,ll> pll; TD vector<int> VI; TD vector<bool> VB; TD vector<ll> VL;
const char nl = '\n'; const int INF = 0x3f3f3f3f; const ll LINF = 0x3f3f3f3f3f3f3f3fll; const ld EPS = 1e-9, PI = acos(-1);
#define ff first
#define ss second

#define NDEBUG
#ifndef NDEBUG
#define trace(var)  {cerr << "Line " << __LINE__ << ": " << #var << "=" << (var) << nl;}
#else
#define trace(var)	{}
#endif

struct Fighter {
  string name;
  int hp, at, df;
  void get() {
    cin >> name >> hp >> at >> df;
  }
};

bool win(const Fighter& a, const Fighter& b) {
  int diffA = max(0, b.at - a.df);
  int diffB = max(0, a.at - b.df);
  trace(diffA);
  trace(diffB);
  trace(a.hp);
  trace(b.hp);
  int roundsA = diffA ? ((a.hp + diffA - 1) / diffA) : INF;
  int roundsB = diffB ? ((b.hp + diffB - 1) / diffB) : INF;
  trace(roundsA);
  trace(roundsB);
  return roundsB < roundsA;
}

int main() {
  atexit([](){ cerr << "Time: " << (ld)clock() / CLOCKS_PER_SEC << nl; });
  ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
  // Emily <3
  int n;
  cin >> n;
  vector<Fighter> ppl(n);
  forx(i,n) {
    ppl[i].get();
  }
  set<VI> triples;
  forx(i,n) {
    forx(j,n) {
      if (j == i) continue;
      forx(k,n) {
        if (k == j) continue;
        if (win(ppl[i], ppl[j]) && win(ppl[j], ppl[k]) && win(ppl[k], ppl[i])) {
          VI add = {i,j,k};
          sort(add.begin(), add.end());
          triples.insert(add);
        }
      }
    }
  }
  cout << triples.size() << nl;
  for (auto e : triples) {
    for(int i : e) {
      cout << ppl[i].name << ' ';
    } cout << nl;
  }
  
  return 0;
}
