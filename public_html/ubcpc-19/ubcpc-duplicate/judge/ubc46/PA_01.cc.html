#include <stdio.h>
#include <cstring>
int main()
{
    char a;
    char word[110];
    int n,i,len;
    scanf("%c", &a);
    scanf("%d", &n);
    scanf("%s", &word);
    len = strlen(word);
    if(a == 'E')
    {
        for(i=0;i<len;i++) {
            if(word[i]+n <= 'z')
                word[i] = word[i]+n;
            else word [i] = word[i] + n - 26;
        }
        for(i=0;i<len;i++)
            printf("%c", word[i]);
    }
    else if(a == 'D'){
        for(i=0;i<len;i++) {
            if(word[i]-n >= 'a')
                word[i] = word[i]-n;
            else word [i] = word[i] - n + 26;

        }
        for(i=0;i<len;i++)
            printf("%c", word[i]);
    }
    return 0;
}
