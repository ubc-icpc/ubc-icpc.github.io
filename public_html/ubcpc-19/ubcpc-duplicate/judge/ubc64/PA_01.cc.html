LL closest(LD x) {
    LD f = floor(x), c = ceil(x);
    LD a = x - f; 
    LD b = c - x; 
    if (abs(a) < abs(b)) return f;
    else return c;
}

LL getmod(LL x, LL mod) {
    if (x >= 0) return x % mod;
    else return (mod - ((-x) % mod)) % mod;
}
