LL n, hp[MAXN], at[MAXN], df[MAXN];
string names[MAXN];
set<pair<LL, PLL> > ans;

LL comp(LL i, LL j) {
    if (at[j] - df[i] <= 0) return INF;
    else return ceil((LD)hp[i]/(LD)(at[j] - df[i]));
}

LL win(LL i, LL j) {
    LL ni = comp(i, j);
    LL nj = comp(j, i);
    if (ni == nj) return 0;
    else if (ni > nj) return 1;
    else return -1;
}

bool isTrans(LL i, LL j, LL k) {
    // cout << "CALLED " << i << " " <<  j << " " << k << endl;
    LL x = win(i, j);
    if (x == 1) {   
        if (win(j, k) == 1 && win(k, i) == 1) return true;
    } else if (x == -1) {
        if (win(i, k) == 1 && win(k, j) == 1) return true;
    }
    return false;
}

    REP(i, n) {
        REPI(j, i + 1, n) {
            REPI(k, j + 1, n) {
                if (isTrans(i, j, k)) ans.INS(MP(i, MP(j, k)));
            }
        }
    }
    cout << ans.size() << endl;
    for(auto k: ans) 
        cout << names[k.F] << " " << names[k.S.F] << " " << names[k.S.S] << endl;
}
