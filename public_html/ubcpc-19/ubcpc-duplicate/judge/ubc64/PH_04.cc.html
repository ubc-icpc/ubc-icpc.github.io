    LL ans = 1, curr = 2;
    while(c != 1 && curr * curr * curr <= c) {
        LL ctr = 0;
        while(c % curr == 0) {
            ctr++;
            c /= curr;
        }
        if (ctr % 2 == 1) ans *= curr;
        curr++;
    }
    if (c != 1) {
        LL small = sqrt(c);
        if (small * small != c) ans *= c;
    }
    cout << ans << endl;
