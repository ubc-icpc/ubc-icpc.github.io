#include <iostream>


using namespace std;


int main() {
  char input;
  int shift;
  string words;

  cin >> input;
  cin >> shift;
  cin >> words;
  for(int i=0; i<words.size(); i++) {
    char temp = words[i];

    if(input=='E') {
      if(temp+shift > 'z') {
        cout << char(((temp-'a'+shift)%26)+'a');
      } else if(temp+shift<='z') {
        cout << char(temp+shift);
      }
    }
    if(input=='D') {
      if(temp-shift < 'a') {
        cout << char(temp-shift+26);
      } else if(temp-shift<='z') {
        cout << char(temp-shift);
      }
    }
  }
  cout << endl;
  return 0;
}
