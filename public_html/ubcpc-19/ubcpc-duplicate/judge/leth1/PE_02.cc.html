#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;	       
	    
int main()
{
   int x;
   int y;
   vector<int> V;
   int input;
   int output = 0;

   cin >> x >> y;

   for(int i=0;i<x;i++)
   {
      cin >> input;
      V.push_back(input);
   }
   
   sort(V.begin(), V.end());

   while(y > 0)
   {
      output += V.back();
      V.pop_back();
      y--;
   }
   
   cout << output << endl;
   return 0;
}
