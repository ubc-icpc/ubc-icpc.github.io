#include <iostream>
#include <iomanip>
#include <vector>
#include <string>
#include <array>
#include <algorithm>
#include <vector>

using namespace std;


void input() {
   int scene;
   int act;
   cin >> scene;
   cin >> act;

   vector<int> arr;
   for(int i = 0; i < scene; i++) {
      int read;
      cin >> read;
      arr.push_back(read);
   }

   


    sort(arr.rbegin(), arr.rend());

    int sum = 0;
   for(int i = 0; i < act; i++) {
      sum += arr.at(i);
    }

   cout << sum << endl;
}
	

  

   
      
			 

int main() {
   input();
   

   
   return 0;
}
