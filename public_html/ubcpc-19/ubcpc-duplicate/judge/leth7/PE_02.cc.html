#include<iostream>
#include<queue>
using namespace std;

int main(){
   int scenes, acts;
   cin>>scenes>>acts;
   priority_queue<int> p;

   for(int i=0; i<scenes; i++){
      int temp;
      cin>>temp;
      p.push(temp);
   }

   int score=0;
   for(int i=0; i<acts; i++){
      score+=p.top();
      p.pop();
   }

   cout<<score<<endl;

}
