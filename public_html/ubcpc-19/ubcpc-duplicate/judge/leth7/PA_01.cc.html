#include<iostream>
using namespace std;


int main(){
   char inst;
   int shift;
   string message;
   cin>>inst>>shift>>message;

   if(inst == 'E'){
      for(int i=0; i<message.length(); i++){
	 if(message[i]>(122-shift)){
	    message[i]+=shift-26;
	 }
	 else{
	    message[i]+=shift;
	 }
      }
   }

   if(inst == 'D'){
      for(int i=0; i<message.length(); i++){
	 if(message[i]<(97+shift)){
	    message[i]-=shift;
	    message[i]+=26;
	 }
	 else{
	    message[i]-=shift;
	 }
      }
   }

   
   cout<<message<<endl;
}
