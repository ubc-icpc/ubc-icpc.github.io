	public static String encode(String str, int shift) {
		StringBuilder ret = new StringBuilder();
		for(int i = 0; i < str.length(); i++) {
			char c = str.charAt(i);
			if(c+shift > 122) {
				int r = (c+shift) - 122;
			}
			char newC = (char)(c + shift);
			if(newC > 122) {
				newC = (char)((c+shift) - 26);
			}
			else if(newC < 97) {
				newC = (char)((c+shift) + 26);
			}
			ret.append(newC);
		}
		return ret.toString();
	}
	
	public static String decode(String str, int shift) {
		return encode(str,shift*(-1));
