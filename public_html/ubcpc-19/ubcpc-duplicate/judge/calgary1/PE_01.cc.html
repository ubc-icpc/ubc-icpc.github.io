#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const int MAX_N = 1e6+10;
const ll INF = 1000000000000000000LL;


int n, k;
int a[MAX_N];

int main() {
  cin>>n>>k;
  for(int i = 0 ; i < n ; i++)
    cin>>a[i];
  sort(a, a+n);
  reverse(a, a+n);
  int ans = 0;
  for(int i = 0 ; i < k ; i++)
    ans += a[i];
  cout<<ans<<endl;

}
