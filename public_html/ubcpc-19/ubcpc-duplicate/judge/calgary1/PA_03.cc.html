#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const int MAX_N = 1e6+10;
const ll INF = 1000000000000000000LL;


vector<char>v;

int main() {
  for(char i = 'a' ; i <= 'z' ; i++)
    v.push_back(i);
  char ch;
  int s;
  string w;
  cin>>ch>>s>>w;
  if(ch == 'E') {
    for(int i = 0 ; i < w.size() ; i++) {
      int tmp = w[i] - 'a';
      tmp += s;
      tmp %= v.size();
      w[i] = v[tmp];
    }
  }
  else if(ch == 'D') {
    for(int i = 0 ; i < w.size() ; i++) {
      int tmp = w[i] - 'a';
      tmp -= s;
      tmp += v.size();
      tmp %= v.size();
      w[i] = v[tmp];
    }
  }
  cout<<w;
}
