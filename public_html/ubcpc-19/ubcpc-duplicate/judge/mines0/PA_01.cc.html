#include <iostream>
#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;

int main() {
  cin.tie(0);
  ios::sync_with_stdio(0);
  char c;
  cin>>c;
  int s;
  cin>>s;
  string w;
  cin>>w;
  bool add;
  if(c=='E'){ 
        add=true;
   }else{ 
        add=false; 
    }
       

  for(int i=0;i<w.size();i++){ 
          int a=int(w.at(i));
          a-=97;
          if(add){
               a=(a+s);
               if(a>=26){ 
                  a-=26; 
                }
           }else{ 
               a=(a-s);
               if(a<0){ 
                    a+=26; 
                 }
           }
          a+=97; 
          w.at(i)=char(a); 
     } 
  cout<<w;

  return 0;
}
