class fighterStats{
public:
    string Name;
    int hp, atk, def;
    fighterStats()
    {
        Name="";
        hp=0, atk=0, def=0;
    }
    void set(string S, int h, int a, int d)
    {
        Name=S, hp=h, atk=a, def=d;
    }
};

bool operator > (fighterStats A, fighterStats B)
{
    int dmga=A.atk-B.def, dmgb=B.atk-A.def;
    // cerr<<A.Name<<" "<<dmga<<" vs "<<B.Name<<" "<<dmgb<<'\n';
    if (dmga<=0) return 0;
    if (dmgb<=0) return 1;
    int ta=(A.hp-1)/dmgb, tb=(B.hp-1)/dmga;
    // cerr<<ta<<" "<<tb<<"\n";
    return (ta>tb);
}

bool correctTriplet(fighterStats A, fighterStats B, fighterStats C)
{
    if ((A>B) && (B>C) && (C>A)) return 1;
    if ((B>A) && (C>B) && (A>C)) return 1;
    return 0;
}
