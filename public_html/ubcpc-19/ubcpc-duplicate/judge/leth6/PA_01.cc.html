/***********************************************************/
bool Crypted()
{
   bool encrp;
   char ch;
   cin >> ch;
   switch(ch)
   {  case 'E': encrp = true; break;
      case 'D': encrp = false; break;
   }
   return encrp;
}


/***********************************************************/
void ConvertString(string orig, int s)
{
   int w, l;
   char ch, temp;
   l = orig.length();
   for(int i =0; i < l; i++)
   {
      temp = orig.at(i);
      w = temp;
      ch = ConvertChar(w,s);
      cout << ch;
   }
   
}
/***********************************************************/
char ConvertChar(int w, int s)
{
   char ch;
   if(s>0)
   {
      if((w+s) <= 122)
	 ch = w+s;
      else
	 ch = w+s-26;
   }
   else
   {
      if((w+s) >= 97)
	 ch = w+s;
      else
	 ch = w+s+26;
   }

   
   return ch;
}
