#include <bits/stdc++.h>
using namespace std;



int main() {
    int N, K;
    cin >> N >>  K;

    priority_queue<int> scenes;
    for (int i = 0; i < N; ++i) {
        int num;
        cin >> num;
        scenes.push(num);
    }

    int total = 0;
    for (int i = 0; i < K; ++i) {
        total += scenes.top();
        scenes.pop();
    }

    cout << total << std::endl;
    return 0;
}
