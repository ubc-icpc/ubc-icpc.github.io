#include <bits/stdc++.h>
using namespace std;

string alpha = "abcdefghijklmnopqrstuvwxyz";

void shift(string* word, int amount) {
    for (int i = 0; i < word->length(); i++) {
        char c = (*word)[i];
        c = (c - 'a' + amount + 26) % 26 + 'a';
        (*word)[i] = c;
    }
}

int main() {
    char encode;
    int shiftAmount;
    string word;

    cin >> encode;
    cin >> shiftAmount;
    cin >> word;

    switch (encode) {
        case 'D':
            shiftAmount *= -1;
        case 'E':
            shift(&word, shiftAmount);
    }
    cout << word << '\n';

}
