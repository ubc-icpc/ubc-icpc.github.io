#include <bits/stdc++.h>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    string c;
    cin >> c;

    int s;
    cin >> s;

    string w;
    cin >> w;

    if (c == "E") {
        for (unsigned int i = 0; i < w.size(); ++i) {
            w[i] = ((w[i] - 'a' + s) % 26 + 'a');
        }
        cout << w << '\n';
    } else {
        for (unsigned int i = 0; i < w.size(); ++i) {
            if ((w[i] - 'a' - s) < 0) {
                w[i] = 26 + ((w[i] - 'a' - s) % 26 + 'a');
            } else {
                w[i] = ((w[i] - 'a' - s) % 26 + 'a');
            }
        }
        cout << w << '\n';
    }

    return 0;
}
