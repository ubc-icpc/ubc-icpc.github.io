#include <bits/stdc++.h>
using namespace std;

int main()
{
   char c; cin >> c;
   int a; cin >> a;
   string s; cin >> s;

   for(unsigned int i = 0; i < s.length(); ++i)
   {
      if(c == 'E')
      {
	 char d = (((s[i] + a) - 'a') % 26) + 'a';
	 //if(d > 'z') { d = (d - 'z') + ('a'-1); }
	 cout << d;
      }
      else
      {
	 char d = ((((s[i] - a) - 'a') + 26) % 26) + 'a';
	 //if(d < 'a') { d = ('z'+1) - ('a' - d); }
	 cout << d;
      }
   }

   cout << endl;
   
}
