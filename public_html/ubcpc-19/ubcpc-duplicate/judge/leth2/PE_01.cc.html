#include <bits/stdc++.h>
using namespace std;

int A[20005];

bool f(int a, int b)
{
   return a > b;
}

int main()
{
   int n, k;
   cin >> n >> k;

   for(int i = 0; i < n; ++i) cin >> A[i];

   long long int ans = 0;

   sort(A, A+n, f);

   for(int i = 0; i < k; ++i) ans += A[i];
   cout << ans << endl;
}
