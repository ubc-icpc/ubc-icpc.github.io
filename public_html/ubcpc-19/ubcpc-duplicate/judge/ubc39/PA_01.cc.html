#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);
  string code, word;
  int shift, i;
  cin >> code >> shift >> word;
  if (code[0] == 'D') {
    shift = 26-shift;
  }
  for (i = 0; i<word.length(); i++) {
    word[i] = ((word[i]-'a'+shift)%26)+'a';
  }
  cout << word << endl;
  return 0;
}
