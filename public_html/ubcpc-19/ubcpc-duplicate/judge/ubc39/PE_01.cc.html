#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);

  int numS, numA, i, ret = 0;
  cin >> numS >> numA;
  vector<int> scenes(numS,0);
  for (i = 0; i<numS; i++) {
    cin >> scenes[i];
  }
  sort(scenes.begin(),scenes.end());
  for (i = 0; i<numA; i++) {
    ret += scenes[numS-1-i];
  }
  cout << ret << endl;
  return 0;
}
