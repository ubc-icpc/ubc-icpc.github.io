#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(NULL);

	char type; cin >> type;
	int shif; cin >> shif;
	string s; cin >> s;
	if (type == 'E')
	{
		for (char &c : s)
			c = (((c - 'a') + shif) % 26) + 'a';
	}
	else
	{
		for (char &c : s)
			c = (((c - 'a') - shif + 26) % 26) + 'a';
	}
	cout << s << endl;
}
