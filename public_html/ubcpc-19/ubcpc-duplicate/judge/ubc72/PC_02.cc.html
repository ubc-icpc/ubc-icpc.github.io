#include <bits/stdc++.h>
using namespace std;

bool query(int a, int b, int c)
{
	++a; ++b; ++c;
	cout << "? " << a << ' ' << b << ' ' << c << endl;
	int ret; cin >> ret;
	return ret;
}

void output(int a)
{
	cout << "! " << a+1 << endl;
}

void swap(int &a, int &b)
{
	int c = b;
	b = a;
	a = c;
}

int main()
{
	ios::sync_with_stdio(0);

	int N; cin >> N;
	int lower, upper, next;
	lower = 0; upper = 1; next = 2;
	if (query(lower,upper,next))
	{
		++next;
	}
	else
	{
		swap(lower, upper);
		++next;
	}
	for (; next < N; ++next)
	{
		if (query(next,lower,upper))
		{
			lower = next;
		}
	}
	output(lower);
}
