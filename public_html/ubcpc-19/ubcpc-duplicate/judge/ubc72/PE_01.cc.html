#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(NULL);

	int N, K; cin >> N >> K;
	vector<int> exci(N);
	for (int i = 0; i < N; ++i)
			cin >> exci[i];
	//sum of K largest numbers
	sort(exci.rbegin(),exci.rend());
	int sum = 0;
	for (int i = 0; i < K; ++i)
			sum += exci[i];
	cout << sum << endl;
}
