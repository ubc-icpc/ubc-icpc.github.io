#include <bits/stdc++.h>

using namespace std;

int main() {
  int n, k;
  cin >> n >> k;
  vector<int> arr(n);
  for (int i = 0; i < arr.size(); i++) {
    cin >> arr[i];
  }

  sort(arr.begin(), arr.end());

  long long sum = 0;
  int index = n - 1;
  for (int i = 0; i < k; i++) {
    sum += arr[index];
    index--;
  }
  cout << sum << endl;
}
