#include <bits/stdc++.h>

using namespace std;

void encode() {
  int n;
  cin >> n;
  string s;
  cin >> s;

  for (int i = 0; i < s.length(); i++) {
    s[i] = (char)(((s[i] - 'a') + n)%26 + 'a');
  }
  cout << s << endl;
}

void decode() {
  int n;
  cin >> n;
  string s;
  cin >> s;

  for (int i = 0; i < s.length(); i++) {
    s[i] = (char) (((s[i] - 'a') - n + 26)%26 + 'a');
  }

  cout << s << endl;
}

int main() {
  char c;
  cin >> c;
  if (c == 'E') {
    encode();
  } else {
    decode();
  }
}
