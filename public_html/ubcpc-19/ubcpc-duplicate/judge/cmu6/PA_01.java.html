import java.util.*;
import java.io.*;
public class ubca {
    public static void main(String[] args) throws IOException {
        Scanner in = new Scanner(System.in);
        String code = in.next();
        int s = in.nextInt();
        String word = in.next();
        String alphabet = "abcdefghijklmnopqrstuvwxyzabcdefghijklmnopqrstuvwxyz";
        HashMap<Character, Character> cipher = new HashMap<>();
        if (code.equals("E")) {
            for (int i = 0; i < 26; i++) {
                char a = alphabet.charAt(i);
                char b = alphabet.charAt(i+s);
                cipher.put(a,b);
            }
        }
        else {
            for (int i = 0; i < 26; i++) {
                char a = alphabet.charAt(i);
                char b = alphabet.charAt(i+s);
                cipher.put(b,a);
            }
        }

        String res = "";
        for (char c : word.toCharArray()) {
            res += cipher.get(c);
        }
        System.out.println(res);
    }
}
