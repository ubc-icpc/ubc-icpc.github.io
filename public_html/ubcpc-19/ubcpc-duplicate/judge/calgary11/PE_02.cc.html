#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>
#include <numeric>

#define rep(i, min, max) for (Num i = min; i < max; ++i)
using Num = unsigned int;

using namespace std;


Num numScenes;
Num numActs;
vector<Num> scenes;

int main()
{
	cin >> numScenes;
	cin >> numActs;

	scenes.reserve(numScenes);

	rep(i, 0, numScenes)
	{
		Num tmp;
		cin >> tmp;
		scenes.push_back(tmp);
	}

	Num acc = 0;
	sort(scenes.begin(), scenes.end());
	for (int i = scenes.size() - numActs; i < scenes.size(); ++i) {
		acc += scenes[i];
	}
	cout << acc;
}
