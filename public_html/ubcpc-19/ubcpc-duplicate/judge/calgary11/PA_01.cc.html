#include <iostream>
#include <algorithm>
#include <cmath>

static const int alh = 26;
static const int abase = 97;

using namespace std;

#define rep(i, min, max) for (size_t i = min; i < max; ++i)

int main()
{
	// cout << (char)((((int)'z' - 98 + 2) % 25) + abase);
	char EorD;
	int shift;

	string str;

	cin >> EorD;
	cin >> shift;
	cin >> str;

	if (EorD == 'E')
		transform(str.begin(), str.end(), str.begin(),
			  [shift](char c) -> char {
				  return (char)(abase
						+ ((((int)c - abase) + shift)
						   % alh));
			  });
	else

		transform(str.begin(), str.end(), str.begin(),
			  [shift](char c) -> char {
				  return (char)(abase
						+ ((((int)c - abase)
						    + (alh - shift))
						   % alh));
			  });


	cout << str;
}
