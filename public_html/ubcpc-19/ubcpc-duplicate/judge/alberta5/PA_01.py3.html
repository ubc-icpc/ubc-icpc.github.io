ALPHABET = "abcdefghijklmnopqrstuvwxyz"

code = input()
s = int(input())
w = input()

if code == 'E':
    encoded_word = ''

    for letter in w:
        letter_index = ALPHABET.index(letter)
        encoded_word += ALPHABET[(letter_index + s) % 26]
    print(encoded_word)
else:
    decoded_word = ''

    for letter in w:
        letter_index = ALPHABET.index(letter)
        decoded_word += ALPHABET[(letter_index - s) % 26]
    print(decoded_word)
