#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {

    vector<int> x;
    int N,K,temp;
    int ans=0;
    cin >> N >> K;
    for (int i = 0; i < N; ++i) {
        cin >> temp;
        x.push_back(temp);
    }
    for (int k = 0; k < K; ++k) {
        int max=0;
        for (int j = 0; j < x.size(); ++j) {
            if(x.at(j)>=x.at(max))
                max=j;
        }
        ans+=x.at(max);
        x.erase( x.begin() + max );
    }


    cout << ans<< flush;

    return 0;
}
