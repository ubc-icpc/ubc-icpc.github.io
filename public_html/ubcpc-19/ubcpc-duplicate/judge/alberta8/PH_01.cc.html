#include <iostream>
#include <cmath>

using namespace std;

typedef unsigned long long ull;

int main() {
  ull n, d = 1;
  cin >> n;

  for (ull k = 2; k*k*k <= n; ++k) {
    int cnt = 0;
    while (n%k == 0) {
      n /= k;
      ++cnt;
    }
    if (cnt&1) d *= k;
  }

  ull rt = sqrt((double) n);
  while ((rt+1)*(rt+1) <= n) ++rt;
  while (rt*rt > n) --rt;

  if (rt*rt != n) d *= n;

  cout << d << endl;

  return 0;
}
