import java.util.*;

public class k 
{
	public static void main(String[] args)
	{
		Scanner sc = new Scanner(System.in);
		
		int scene = sc.nextInt();
		int act = sc.nextInt();
		
		ArrayList<Integer> ex = new ArrayList<Integer>();
		
		int i;
		for (i = 0; i < scene; i++)
		{
			int e = sc.nextInt();
			ex.add(e);
		}
		
		Collections.sort(ex);;
		
		int excite = 0;
		for (i = 0; i < act; i++)
			excite += ex.get(ex.size() - i - 1);
		System.out.println(excite);
	}
}
