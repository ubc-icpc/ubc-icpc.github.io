import java.util.*;

public class a 
{
	public static void encode(int n, String str)
	{
		String out = "";
		for (int i = 0; i < str.length(); i++)
		{
			int k = (int)str.charAt(i) + n;
			if (k > 122)
				k -= 26;
			
			char ch = (char)k;
			out += ch;
		}
		System.out.println(out);
	}
	
	public static void decode(int n, String str)
	{
		String out = "";
		for (int i = 0; i < str.length(); i++)
		{
			int k = (int)str.charAt(i) - n;
			if (k < 97)
				k += 26;
			char ch = (char)k;
			out += ch;
		}
		System.out.println(out);
	}
	
	public static void main(String[]args)
	{
		Scanner sc = new Scanner(System.in);
		
		String ed = sc.next();
		int n = sc.nextInt();
		String word = sc.next();
		
		if (ed.contentEquals("E"))
			encode(n, word);
		
		else if (ed.contentEquals("D")) 
			decode(n, word);
	}
}
