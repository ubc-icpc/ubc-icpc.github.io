#include <iostream>
#include <queue>
using namespace std;

int main() {
    int noOfScenes, acts;
    int answer = 0;
    cin>>noOfScenes;
    cin>>acts;
    if(noOfScenes == 0) {
        cout<< "0"<<endl;
        return 0;
    }
    priority_queue <int> scenes;
    for(int i = 0 ; i < noOfScenes ; i++) {
        int temp;
        cin >> temp;
        scenes.push(temp);
    }
    while(acts!=0) {
        answer = answer + scenes.top();
        acts--;
        scenes.pop();
    }
    cout<<answer;
    return 0;
}
