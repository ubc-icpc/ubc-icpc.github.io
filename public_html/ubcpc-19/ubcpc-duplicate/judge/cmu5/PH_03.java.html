import java.util.Scanner;

public class UBCH {
	
	public static boolean isSquare(long n) {
		double sqrt = Math.sqrt(n);
		long sqrtL = (long) sqrt;
		
		if (sqrtL * sqrtL == n) return true;
		if ((sqrtL - 1) * (sqrtL - 1) == n) return true;
		if ((sqrtL + 1) * (sqrtL + 1) == n) return true;
		
		return false;
	}
	
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		long n = sc.nextLong();
		
		sc.close();
		
		long power = 0;
		
		while (n % 2 == 0) {
			n /= 2;
			power += 1;
		}
		
		long out = 1;
		if (power % 2 == 1) {
			out = 2;
		}
		power = 0;
		
		long test = 3;
		while ((n/test)/test >= test) {
			while (n % test == 0) {
				power += 1;
				n /= test;
			}
				if (power % 2 == 1) {
					out *= test;
				}
				power = 0;
				test += 2;
		}
		
		if (!isSquare(n)) out *= n;
		
		System.out.println(out);

		
		
	}
}
