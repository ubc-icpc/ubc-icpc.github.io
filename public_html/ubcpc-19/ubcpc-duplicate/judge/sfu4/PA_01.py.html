import sys

lines = sys.stdin.readlines()

action = lines[0].strip()
s = int(lines[1].strip())
txt = lines[2].strip()

if action == 'E':
    ct = ''
    for char in txt:
        ct += chr((ord(char) - ord('a') + s) % 26 + ord('a'))
    print(ct)
else:
    pt = ''
    for char in txt:
        ptord = ord(char) - ord('a') - s
        if ptord < 0:
            ptord = ord('z') - abs(ptord) + 1
        else:
            ptord += ord('a')
        pt += chr(ptord)
    print(pt)
