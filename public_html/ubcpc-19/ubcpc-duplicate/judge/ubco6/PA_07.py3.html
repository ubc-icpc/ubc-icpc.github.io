def encode_char(ch, s) -> chr:
    new_char = (ord(ch) - 97) + s
    return chr(97 + ((new_char) % 26))

def decode_char(ch, s) -> chr:
    e = encode_char(ch, -s)
    return e

def encode(string, s) -> str:
    return ''.join([encode_char(x, s) for x  in string])

def decode(string, s) -> str:
    return ''.join([decode_char(x, s) for x  in string])


if __name__ == "__main__":
    mode = input()
    shift = int(input())
    if mode == 'E':
        print(encode(input(), shift))

    else:
        print(decode(input(), shift))
