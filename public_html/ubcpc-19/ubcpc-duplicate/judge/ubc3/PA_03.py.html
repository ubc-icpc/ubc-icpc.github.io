cmd = raw_input()
shift = int(raw_input())
string = raw_input()
results = []
if cmd == 'E':
    for i in string:
        results.append(chr ( ord('a')+ (ord(i)-ord('a') + shift ) %26))
    print(''.join(results))
if cmd == 'D':
    for i in string:
        results.append(chr ( ord('a')+ (ord(i)-ord('a') - shift ) %26))
    print(''.join(results))
