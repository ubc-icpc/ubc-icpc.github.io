import java.util.*;
public class E{
    public static void main (String[] args){
        Scanner input = new Scanner(System.in);
        int n = input.nextInt();
        int k = input.nextInt();

        Integer[] excite = new Integer[n];

        for (int i = 0; i < n; i++){
            excite[i] = input.nextInt();
        }

        Queue<Integer> q = new PriorityQueue<>(10000, Collections.reverseOrder());
        q.addAll(Arrays.asList(excite));
        int maxExcite = 0;

        for (int i = 0; i < k; i++){
            maxExcite += q.poll();
        }
        System.out.println(maxExcite);
    }
}
