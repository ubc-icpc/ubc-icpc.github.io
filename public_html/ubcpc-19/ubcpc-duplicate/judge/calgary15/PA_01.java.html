import java.util.*;

public class A{
    public static void main (String[] args){
        Scanner input = new Scanner(System.in);

        String whatDo = input.nextLine();
        int shift = input.nextInt();
        input.nextLine();
        String word = input.nextLine();
        String outputWord = "";

        if (whatDo.equals("E")){
            for (int i = 0; i < word.length(); i++){
                int letter = word.charAt(i);
                letter -= 'a' ;
                letter = (letter + shift) % 26;
                letter += 'a';
                outputWord +=(char) (letter);
            }
        }else{
            for (int i = 0; i < word.length(); i++){
                int letter = word.charAt(i);
                letter -= 'a' ;
                letter = (letter - shift);
                if (letter < 0)
                    letter += 26;
                letter += 'a';
                outputWord += (char)letter;
            }
        }
        System.out.println(outputWord);
    }
}
