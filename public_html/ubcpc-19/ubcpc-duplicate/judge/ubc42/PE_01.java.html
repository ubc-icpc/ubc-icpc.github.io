import java.util.*;

public class Main {

  public static void main(String[] args) {
    Scanner scanner=new Scanner(System.in);
    int sceneCount = scanner.nextInt();
    int actCount = scanner.nextInt();
    Integer[] arr = new Integer[sceneCount]; 
    int excitement = 0;

    for (int i = 0; i < sceneCount; i++) {
      arr[i] = scanner.nextInt();
    }
    Arrays.sort(arr, Collections.reverseOrder());

    for (int j = 0; j < actCount; j++) {
      excitement += arr[j];
    }

    System.out.println(excitement);
    

    scanner.close();
  }
}
