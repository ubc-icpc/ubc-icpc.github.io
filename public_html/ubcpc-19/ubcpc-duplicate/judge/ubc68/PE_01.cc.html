#include <bits/stdc++.h>
using namespace std;
#define MAX_I 2028

int N,K;
unsigned long a[MAX_I];

int main() {
	ios::sync_with_stdio(false);
	cin >> N >> K;
	for(unsigned n = 0; n < N; n++) {
		cin >> a[n];
	}
	sort(a, a+N);
	unsigned long sum = 0;
	for(unsigned k = 1; k <= K; k++) {
		sum += a[N-k];
#ifdef DEBUG
		cout << "Picked: " << a[N-k] << endl;
#endif
	}
	cout << sum << "\n";
	return 0;
}
