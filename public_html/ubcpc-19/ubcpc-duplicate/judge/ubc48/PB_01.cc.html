#define long long long
#define fi first
#define se second
typedef pair<int,int> ii;
typedef pair<ii,int> iii;

const int INF = 1e9;
string S[103];
int H[103], A[103], D[103];
vector<int> ans;

bool win(int x, int y)
{
	int decx = max(0, A[y]-D[x]);
	int decy = max(0, A[x]-D[y]);

	int tx = (decx)? (H[x]+decx-1)/decx : INF;
	int ty = (decy)? (H[y]+decy-1)/decy : INF;
	// cout << tx << " " << ty << "\n";
	return tx > ty;
}

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	// freopen("input.in", "r", stdin);
	
	int n; cin >> n;
	for(int i = 0; i < n; i++)
		cin >> S[i] >> H[i] >> A[i] >> D[i];

	int res = 0;

	// cout << win(4, 0) << "\n";
	// return 0;

	vector<iii> ans;
	for(int i = 0; i < n; i++)
	{
		for(int j = i+1; j < n; j++)
		{
			for(int k = j+1; k < n; k++)
			{
				if(win(i,j) && win(j,k) && win(k,i))
				{
					ans.push_back({{i, j}, k});
					continue;
				}
				if(win(i,k) && win(k,j) && win(j,i))
				{
					ans.push_back({{i, j}, k});
					continue;	
				}
			}
		}
	}
	cout << ans.size() << "\n";;
	for(auto i : ans) cout << S[i.fi.fi] << " " << S[i.fi.se] << " " << S[i.se] << "\n";

}
