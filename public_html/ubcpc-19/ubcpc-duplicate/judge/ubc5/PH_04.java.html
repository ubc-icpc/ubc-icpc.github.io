import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long c = sc.nextLong();
        long max = 8000000000000L;
        for (long i = 2; (i * i) <= max; i++) {
            if ((c % (i * i)) == 0) {
                c /= i;
                c /= i;
                i--;
            }
        }
        for (long i = 1; i <= 4000000; i++) {
            if (c % i == 0) {
                long test = c / i;
                long sqrt = sqrt(test);
                if (sqrt != -1) {
                    c /= sqrt;
                    c /= sqrt;
                }
            }

        }

        System.out.println(c);
    }

    static long sqrt(long l) {
        long low = 1;
        long high = (long) Math.sqrt(l) + 100;
        while (low <= high) {
            long mid = (low + high) / 2;
            long sqr = mid * mid;
            if (sqr == l) {
                return mid;
            } else if (sqr < l) {
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return -1;

    }
}
