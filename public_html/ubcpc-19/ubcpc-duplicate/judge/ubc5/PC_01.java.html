import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int end = 2;
        for (int i = 3; i <= n; i++) {
            System.out.println("? 1 " + end + " " + i);
            System.out.flush();
            int res = sc.nextInt();
            if (res == 1) {
                end = i;
            }
        }
        System.out.println("! " + end);
    }
}
