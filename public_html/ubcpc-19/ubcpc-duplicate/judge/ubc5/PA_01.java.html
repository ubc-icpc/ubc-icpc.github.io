import java.util.*;

public class Main {
    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String c = sc.next();
        int s = sc.nextInt();
        String w = sc.next();
        for (int i = 0; i < w.length(); i++) {
            char cc = w.charAt(i);
            if (c.charAt(0) == 'E') {
                System.out.print((char)((cc + 26 - 'a' + s) % 26 + 'a'));
            } else {
                System.out.print((char)((cc + 26 - 'a' - s) % 26 + 'a'));
            }
        }
        System.out.println();
    }
}
