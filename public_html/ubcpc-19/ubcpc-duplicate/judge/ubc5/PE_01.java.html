import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int K = sc.nextInt();
        int[] arr = new int[N];
        for (int i = 0; i < N; i++) {
            arr[i] = sc.nextInt();
        }
        Arrays.sort(arr);
        int total = 0;
        int count = 0;
        for (int i = N - 1; i >= 0 && count < K; i--) {
            total += arr[i];
            count++;
        }
        System.out.println(total);
    }
}
