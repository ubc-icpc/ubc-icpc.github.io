    public static boolean isForwardOrBackward(String s){
      boolean forward = true;
      if(s.equals("E")){
        forward = true;
      }
      else if(s.equals("D")){
        forward = false;
      }
      return forward;
    }

    public static char[] createCharList(){
      char[] alphabets = new char[26];
      int index =0;
      for (int i = 'a'; i <= 'z'; i++){
        alphabets[index] = (char)i;
        index++;
      }
      return alphabets;
    }

    public static int handleBoundary (int target) {
      int result = 0;
      if(target >= 26){
        result = target%26;
      }
      else if(target < 0){
        result = 26- ((-target)%26);
      }else{
        result = target;
      }
      return result;
    }

    public static char[] shiftLetter (String s, int shift, boolean forward, char[] alphabets) {
      char[] chars = s.toCharArray();
      char[] resultChars = new char[chars.length];
      for(int i =0; i < chars.length; i++){
        int val = new String(alphabets).indexOf(chars[i]);

        String index = String.valueOf(val);

        int newIndex = 0;
        if(forward){
          newIndex = val+shift;
        }else{
          newIndex = val-shift;
        }
        resultChars[i] = alphabets[handleBoundary(newIndex)];

      }
      return resultChars;
    }
}
