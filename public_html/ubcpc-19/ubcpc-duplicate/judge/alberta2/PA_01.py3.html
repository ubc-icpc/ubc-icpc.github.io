cmd = input()
shift = int(input())
word = input()
out = []
if cmd == 'E':
    for char in word:
        newChar = chr((((ord(char) - ord('a')) + shift) % 26 + ord('a')))
        out.append(newChar)
else:
    for char in word:
        newChar = chr((((ord(char) - ord('a')) - shift) % 26 + ord('a')))
        out.append(newChar)

print("".join(out))
