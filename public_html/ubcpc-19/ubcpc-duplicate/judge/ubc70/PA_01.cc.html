#include <iostream>
#include <string>

using namespace std;


void solve(char mode, int shift, string str) {
    
    const int cap = 123;
    int x, temp;
    
    
    if(mode == 'D') {
        for(int i = 0; i < str.length(); i++) {
            x = str[i];
            x -= shift;
            temp = x;
            
            if(x < 97) {
                temp = 97 - x;
                printf("%c", cap - temp);
            } else {
                printf("%c", temp % cap);
            }
            
        }
    }
    
    if(mode == 'E') {
        
        for(int i = 0; i < str.length(); i++) {
            x = str[i];
            x += shift;
            
            if(x >= cap) {
                x -= cap;
                x += 97;
            }
            printf("%c", x);
        }
    }
    printf("\n");
    
}

int main(int argc, const char * argv[]) {
    
//    freopen("input.in", "r", stdin);
    
    char mode;
    int shift;
    string str;
    
    
    cin >> mode >> shift >> str;
    
    
//    printf("%c %i %s\n", mode, shift, str.c_str());
    
    solve(mode, shift, str);
    
    
    
    return 0;
}
