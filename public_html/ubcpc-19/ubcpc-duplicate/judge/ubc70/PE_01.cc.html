#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int solve(vector<int> inVec, int actCount) {
    
    int total = 0;
    
    sort(inVec.begin(), inVec.end());
    
    for(int i = inVec.size()-1; i >= 0; i--) {
        total += inVec[i];
        actCount--;
        if(actCount == 0) {
            break;
        }
    }
    
    return total;
    
}

int main(int argc, const char * argv[]) {
    
//    freopen("input.in", "r", stdin);
    
    int s, a;
    
    cin >> s >> a;
    
    vector<int> inVec(s);
    
    for(int i = 0; i < s; i++) {
        cin >> inVec[i];
    }
    
    cout << solve(inVec, a) << endl;
    
    return 0;
}
