#include<iostream>
#include<string>

using namespace std;

int main()
{
   
   int index, s, indexAlpha, num;
   string alphabet = "abcdefghijklmnopqrstuvwxyz", message, cypher;
   char letter, letter1 ,name;
   
   cin >> name;
   cin >> s;
   cin >> message;

   for(index=0; index < message.length(); index++)
   {
      letter = message.at(index);
      if(name == 'E')
      {
	 for(indexAlpha=0; indexAlpha < alphabet.length(); indexAlpha++)
	 {
	    letter1 = alphabet.at(indexAlpha);
	    if(letter == letter1)
	    {
	       num =(indexAlpha + s)%26; 
	       letter1 = alphabet.at(num);
	       cypher = cypher + letter1;
	    }
	 }
      }
      else
      {
	    for(indexAlpha=0; indexAlpha < alphabet.length(); indexAlpha++)
	    {
	       letter1 = alphabet.at(indexAlpha);
	       if(letter == letter1)
	       {
		  if(indexAlpha-s < 0)
		  {
		     num=(26+(indexAlpha-s));
		  }
		  else
		  {
		     num =(indexAlpha-s)%26;
		  };
		  letter1 = alphabet.at(num);
		  cypher = cypher + letter1;
	       }
	    } 
      }
   }
   cout << cypher << endl;
   return 0;
   
}
