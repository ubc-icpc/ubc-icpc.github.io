def abnormal():
    strs = 'abcdefghijklmnopqrstuvwxyz'
    is_encoding = input()
    shift = int(input())
    word = input()
    data = []

    if is_encoding is "D":
        shift *= -1

    for i in word:
        if i.strip() and i in strs:
            data.append(strs[(strs.index(i) + shift) % 26])
        else:
            data.append(i)


    out = ''.join(data)
    print(out)

abnormal()
