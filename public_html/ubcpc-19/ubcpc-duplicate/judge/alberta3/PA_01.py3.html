letters = [chr(ord('a') + i) for i in range(26)]

def solve(word, n, right):
    if right == "E":
        word = [letters[(ord(letter) - ord('a') + n)%26] for letter in word]
        return "".join(word)
    else:
        word = [letters[ord(letter)-ord('a') - n] for letter in word]
        return "".join(word)

_dir = input().strip()
N = int(input().strip())

print(solve(input().strip(), N, _dir))
