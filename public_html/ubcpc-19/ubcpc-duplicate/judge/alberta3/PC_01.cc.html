#include <bits/stdc++.h>
using namespace std;

int oo = 0x3f3f3f3f;

int main() {
	int N, ans, a, b, c;
	cin >> N;
	unordered_set<int> unseen;
	for (int i = 1; i <= N; ++i) unseen.insert(i);
	a = *unseen.begin(); unseen.erase(unseen.begin());
	b = *unseen.begin(); unseen.erase(unseen.begin());
	c = *unseen.begin(); unseen.erase(unseen.begin());

	for (int i = 0; i < N; ++i) {
		cout << "? " << a << " " << b << " " << c << endl;
		cin >> ans;
		if (ans == 1) {
			if (unseen.size() == 0) {
				cout << "! " << c << endl;
				break;
			}
			a = b;
			b = c;
			c = *unseen.begin(); unseen.erase(unseen.begin()); 
		} else if (ans == 0) {
			if (unseen.size() == 0) {
				cout << "! " << b << endl;
				break;
			}
			c = *unseen.begin(); unseen.erase(unseen.begin()); 
		}
	}

	return 0;
}
