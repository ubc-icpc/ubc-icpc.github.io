import java.util.Arrays;
import java.util.Scanner;
public class excitingActs {

	public static void main(String[] args) {

		int numScenes = 0;
		int numActs = 0;
		int maxExcitement = 0;
		
		
		Scanner input = new Scanner(System.in);
		
		numScenes = input.nextInt();
		numActs = input.nextInt();
		int[] levels = new int[numScenes];
		
		for(int i = 0; i < numScenes; i++) {
		
			levels[i] = input.nextInt();
		}
		
		Arrays.sort(levels);
		
		for(int j = 0; j < numActs; j++) {
			
			maxExcitement += levels[(levels.length - 1) - j];
		}
		
		System.out.println(maxExcitement);
		
	}//End of main

}//End of class
