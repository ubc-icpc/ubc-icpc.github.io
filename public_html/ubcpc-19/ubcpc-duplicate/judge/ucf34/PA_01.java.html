import java.util.Scanner;
public class abnormWords {

	public static void main(String[] args) {

		Scanner input = new Scanner(System.in);
		
		char request = input.next().charAt(0);
		int shift = input.nextInt();
		input.nextLine();
		char[] word = input.nextLine().toCharArray();
		
		if(request == 'E') {
			
			for(int i = 0; i < word.length; i++) {
				
				if((char)(word[i] + shift) > 'z') {
					
					word[i] = (char) ('a' + ( (word[i] - 'z') + shift) - 1);
					System.out.print(word[i]);
				}
				
				else {
				
					word[i] = (char) (word[i] + shift);
					System.out.print(word[i]);
				
				}
					
			}//End for
		
		}//End request == 'E'
		
		else {
							
				for(int i = 0; i < word.length; i++) {
					
					if((char)(word[i] - shift) < 'a') {
						
						word[i] = (char) ( ('z' -  ('a' - word[i]) ) - shift + 1);
						System.out.print(word[i]);
					}
					
					else {
					
						word[i] = (char) (word[i] - shift);
						System.out.print(word[i]);
					
					}
						
				}//End for			
			
		}//End request == 'D'
		
		
		
	}//End main

}//End class
