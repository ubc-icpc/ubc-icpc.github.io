int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	ll N, K, sum = 0;
	cin >> N>>K;
	vector<ll> input(N);


	for (int i = 0; i < N; i++) {
		cin >> input[i];
	}

	// sort
	sort(input.begin(), input.end());


	// sum biggest K

	for (int i = N-1; i >= N-K; i--) {
		sum += input[i];
	}
	cout << sum << endl;

	return 0;
}
