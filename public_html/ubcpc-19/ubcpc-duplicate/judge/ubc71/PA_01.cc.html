int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);

	ll shift;
	string mode, cipher;
	cin >> mode>>shift>>cipher;

	string res = "";

	if (mode == "E") {
		shift = shift;
	}
	else {
		shift = 26 - shift; // decode by 1 means shift all by 25
		// b becomes a, which is the same as shift by 25

	}

	for (int i = 0; i < cipher.length(); i++) {
		res += ((cipher[i] - 'a') + shift )%26 + 'a';
	}

	// I'm getting the value

	cout << res << endl;

	return 0;
}
