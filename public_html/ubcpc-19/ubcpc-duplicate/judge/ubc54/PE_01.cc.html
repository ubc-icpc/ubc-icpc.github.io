#include <bits/stdc++.h>

using namespace std;

main(){
    int scene_count, acts;
    //we need k acts aka top k
    cin >> scene_count >> acts;

    priority_queue<int, vector<int>, greater<>> min_heap;
    vector<int> scenes;

    for(int i = 0; i < scene_count; i++){
        int cur;
        cin >> cur;
        if(min_heap.size() < acts)
            min_heap.push(cur);
        else if(min_heap.top() < cur){
            min_heap.pop();
            min_heap.push(cur);
        }
    }
    int ans = 0;
    while(!min_heap.empty()){
        ans += min_heap.top();
        min_heap.pop();
    }

    cout << ans << endl;
    return 0;
}
