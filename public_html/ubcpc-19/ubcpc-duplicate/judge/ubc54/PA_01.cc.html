#include <bits/stdc++.h>

using namespace std;

string encode(string s, int shift){
    string ans;
    for(int i = 0; i < s.size(); i++){
        if(s[i] + shift <= 'z')
            ans.push_back(s[i] + shift);
        else{
            int pos =  ((s[i] -'a' + shift)%26);
            char next = 'a' + pos;
            ans.push_back(next);
        }
    }
    return ans;
}

string decode(string s, int shift){
    string ans;
    for(int i = 0; i < s.size(); i++){
        if(s[i] - shift >= 'a')
            ans.push_back(s[i] - shift);
        else{
            int back_shift =  ((s[i] -'a' - shift)%26);
            char next = 'z' + back_shift + 1;
            ans.push_back(next);
        }
    }
    return ans;
}

main(){
    char type;
    int shift;
    string s;
    cin >> type >> shift >> s;

    if(type == 'E')
        cout << encode(s, shift) << endl;
    else if(type == 'D')
        cout << decode(s, shift) << endl;

}
