#include<cstdio>
using namespace std;
const int nn=1e6+10;
int n,a[110],d[110],h[110],x[nn],y[nn],z[nn];
char s[110][20];
int max(int a,int b){return a>b?a:b;}
bool beat(int i,int j)
{
	int x,y,xx,yy;
	x=max(0,a[j]-d[i]);
	y=max(0,a[i]-d[j]);
	if (x==0&&y==0) return 0;
	if (x==0) return 1;
	if (y==0) return 0;
	xx=h[i]/x; if (xx*x<h[i]) xx++;
	yy=h[j]/y; if (yy*y<h[j]) yy++;
	return xx>yy; 
}
int main()
{
	scanf("%d",&n);
	int i,j,k,ans;
	for (i=1;i<=n;i++)
	{
		scanf("%s%d%d%d",s[i],&h[i],&a[i],&d[i]);
	}
	ans=0;
	for (i=1;i<=n;i++) 
	  for (j=i+1;j<=n;j++)
	    for (k=i+1;k<=n;k++)
	    {
	    	if (j==k) continue;
	    	if (beat(i,j)&&beat(j,k)&&beat(k,i))
	    	{
	    		ans++;
	    		x[ans]=i;
	    		y[ans]=j;
	    		z[ans]=k;
			}
		}
	printf("%d\n",ans);
	for (i=1;i<=ans;i++)
	{
		printf("%s %s %s\n",s[x[i]],s[y[i]],s[z[i]]);
	}
}
