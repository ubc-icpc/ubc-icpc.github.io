#include<cstdio>
#include<cmath>
using namespace std;
typedef long long ll;

const int maxn=3e6;
bool vis[maxn+10];
int P[maxn],tot=0;
void findprime()
{
    vis[1] = 1;
    for (int i = 2; i < maxn; ++i)
    {
        if (!vis[i])
            P[++tot] = i;
        for (int j = 1; j <= tot &&  i * P[j] < maxn; ++j)
        {
            vis[i*P[j]] = 1;
            if (i % P[j] == 0)
                break;
        }
    }
}

int main()
{
	/*ll y=1;
	for (int i=1;i<=62;i++) y=y*2;
	printf("%lld\n",y);
	printf("1");
	for (int i=1;i<=18;i++) printf("0");*/
	findprime();
	ll x;
	int i;
	scanf("%lld",&x);
	ll ans=1;
	for (i=1;i<=tot;i++)
	{
		if (x%P[i]!=0) continue;
		int k=0;
		while (x%P[i]==0)
		{
			k++;
			x/=P[i];
		}
		if (k%2==1) ans*=P[i];
	}
	ll t=ll(sqrt(x));
	bool flag=0;
	for (int j=-1;j<=1;j++)
	  if ((t+j)*(t+j)==x)
	  {
	  	flag=1;
	  }
	if (flag) {printf("%lld\n",ans);return 0;}
	ans=ans*x;
	printf("%lld\n",ans);
}
