#include<cstdio> 
#include<cstring>
using namespace std;
const int maxn=1e5+10;
int l,i,s;
char s1[maxn],y[100];
int main()
{
	char c;
	scanf("%c",&c);
	scanf("%d",&s);
	scanf("%s",s1);
	if (c=='D') s=-s;
	l=strlen(s1);
	for (i=0;i<=25;i++) y[i]='a'+i;
	for (i=0;i<l;i++) 
	{
	  int x=s1[i]-'a';
	  x+=s;
	  printf("%c",y[(x%26+26)%26]);
    } 
}
