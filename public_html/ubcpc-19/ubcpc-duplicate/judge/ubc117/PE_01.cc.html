#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<ll, ll> pii;

int main() {
    ios_base::sync_with_stdio(0), cin.tie(0);
    int n, k;
    cin >> n >> k;
    vector<int> a(n);
    for(auto& x : a) cin >> x;
    sort(a.begin(), a.end());
    reverse(a.begin(), a.end());
    ll sum = 0;
    for(int i=0; i<k; ++i) {
        sum += a[i];
    }
    cout << sum << endl;
    return 0;
}
