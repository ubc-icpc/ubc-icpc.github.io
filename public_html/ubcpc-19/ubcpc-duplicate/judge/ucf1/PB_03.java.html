	public static boolean beat(dood one, dood two) {
		//turns for guy one to kill guy two
		long turnsa = 0;
		//turns for guy two to kill guy one
		long turnsb = 0;
		if(one.b<=two.c) {
			turnsa = Long.MAX_VALUE;
		}else {
			turnsa = two.a/(one.b-two.c)+1;
			if(two.a%(one.b-two.c)==0) {
				turnsa--;
			}
		}
		if(two.b<=one.c) {
			turnsb = Long.MAX_VALUE;
		}else {
			turnsb = one.a/(two.b-one.c)+1;
			if(one.a%(two.b-one.c)==0) {
				turnsb--;
			}
		}
		//System.out.println(one.name+" "+two.name+" "+(turnsa<turnsb));
		if(turnsa<turnsb) {
			return true;
		}return false;
