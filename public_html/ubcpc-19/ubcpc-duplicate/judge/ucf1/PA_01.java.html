import java.util.*;
public class a {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		char a = scan.next().charAt(0);
		int n = scan.nextInt();
		if(a=='D') {
			n *=-1;
		}
		char[] arr = scan.next().toCharArray();
		for(int i = 0;i<arr.length;i++) {
			int val = (int)(arr[i])+n;
			if(val<'a') {
				val+=26;
			}
			if(val>'z') {
				val-=26;
			}
			arr[i] = (char)(val);
		}
		for(int i = 0;i<arr.length;i++) {
			System.out.print(arr[i]);
		}
	}

}
