#include <iostream>
#include <string>
using namespace std;

int main() {
	char type;
	int shif;
	cin >> type >> shif;
	if (type == 'D') {
		shif = 26-shif;
	}

	string word;
	cin >> word;
	for (int i = 0; i < word.size(); i++) {
		if (word[i] + shif > 'z') {
			cout << static_cast<char>(word[i] + shif - 26);
		}
		else {
			cout << static_cast<char>(word[i] + shif);
		}
	}
	cout << endl;
	return 0;
}
