#include <iostream>
#include <queue>
#include <algorithm>
using namespace std; 

int main() {
	int num_scene, num_act;
	int result = 0;

	cin >> num_scene >> num_act;
	int* excitement = new int[num_scene];
	for (int i = 0; i < num_scene; i++) {
		cin >> excitement[i];
	}
	sort(excitement, excitement + num_scene);
	for (int i = 0; i < num_act; i++) {
		result += excitement[num_scene - i - 1];
	}
	cout << result << endl;
	return 0;
}
