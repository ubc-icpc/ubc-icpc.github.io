#include "bits/stdc++.h"
using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<int, int> pii;

int main() {
    char c;
    int shift;
    string s;
    cin >> c >> shift >> s;
    
    if(c=='E') {
        for(int i=0; i<s.length(); i++) {
            s[i] = 'a'+((s[i]+shift-'a')%26);
        }
    }
    else {
        for(int i=0; i<s.length(); i++) {
            s[i] = 'a'+(((s[i]-shift-'a')%26)+26)%26;
        }
    }
    cout << s << endl;
}
