#include "bits/stdc++.h"
using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<int, int> pii;

int main() {
    int N, K;
    cin >> N >> K;
    
    multiset<int> es;
    for(int i=0; i<N; i++) {
        int a;
        cin >> a;
        es.insert(a);
    }
    
    int res = 0;
    int acc = 0;
    for(auto it = es.rbegin(); it != es.rend(); it++) {
        if(acc == K) break;
        acc++;
        res += *it;
    }
    cout << res << endl;
    
}
