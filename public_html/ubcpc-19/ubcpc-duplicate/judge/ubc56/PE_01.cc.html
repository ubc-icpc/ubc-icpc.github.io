#include <iostream>
#include <vector>
#include <math.h>
#include <string>
#include <algorithm>
using namespace std;


int main()
{
	int s, a;
	cin >> s >> a;
	auto XXX = vector<int>(s);
	for (int i = 0; i < s; ++i){
		cin >> XXX[i];
	}
	// find the sum of the a biggest numbers
	sort(XXX.begin(), XXX.end());
	int sum = 0;
	for (int i = 0; i < a; ++i){
		sum += XXX[s-i-1];
	}
	cout << sum;
}
