#include <iostream>
#include <vector>
#include <math.h>
#include <string>
using namespace std;


int main()
{
	char e_or_d;
	cin >> e_or_d;
	bool e = e_or_d == 'E';
	int s;
	cin >> s;
	string w;
	cin >> w;
	if (e) {
		// ADD s
		for (int i = 0; i < w.length(); ++i){
			// get "number" of the char
			int num = w[i] - 97;
			// add s to num
			num += s;
			// % 26 to wrap around
			num %= 26;
			// add 97 to get char
			num += 97;
			w[i] = (char) num;
		}
	} else {
		// decode, SUBTRACT s
		for (int i = 0; i < w.length(); ++i){
			int num = w[i] - 97;
			// add 26 to num
			num += 26;
			// sub s from num
			num -= s;
			// % 26 to get final num
			num %= 26;
			// add 97 to get char
			num += 97;
			w[i] = (char) num;
		}
	}

	cout << w;
}
