#include <iostream>
using namespace std;

int main() {
	char c; cin >> c;
	int n; cin >> n;
	string s; cin >> s;

	if (c == 'E') {
		for (int i = 0; i < s.length(); i++){
			if (s[i] + n >'z') {
				s[i] = s[i] - 26 + n;
			} else {
				s[i] = s[i] + n;
			}
		}
		cout << s << endl;
	} else if (c == 'D') {
		for (int i = 0; i < s.length(); i++){
			if (s[i] - n < 'a') {
				s[i] = s[i] + 26 - n;
			} else {
				s[i] = s[i] - n;
			}
		}
		cout << s << endl;
	}
	return 0;
}
