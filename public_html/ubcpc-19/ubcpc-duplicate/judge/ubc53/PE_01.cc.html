#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main() {
	int n; cin>>n;
	int k; cin>>k;
	vector<int> subi;

	for (int i = 0; i < n; i++) {
		int temp; cin>> temp;
		subi.push_back(temp);
	}

	sort(subi.begin(), subi.end());
	
	int sum = 0;
	for (int i = n - k; i < n; i++) {
		sum += subi[i];
	}
	cout << sum<< endl;

	return 0;
}
