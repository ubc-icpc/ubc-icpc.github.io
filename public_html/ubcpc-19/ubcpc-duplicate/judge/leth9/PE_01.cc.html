#include <iostream>
#include <queue>

using namespace std;

int main() {
   int N, K;
   int sum = 0;
   priority_queue<int> q;
   cin >> N >> K;
   
   for(int i = 0; i < N; i++) {
      int temp;
      cin >> temp;
      q.push(temp);
   }

   for(int j = 0; j < K; j++) {
      sum += q.top();
      q.pop();
   }

   cout << sum << endl;

   return 0;
}
