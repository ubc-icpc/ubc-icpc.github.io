#include <iostream>
#include <string>

using namespace std;

int main() {
   char choice;
   int shift, n;
   string word;

   cin >> choice >> shift >> word;
   if(choice == 'E') 
      for(unsigned int i = 0; i < word.length(); i++)
	 word[i] = char(((word[i] - 'a' + shift) % 26) + 'a');

   else if(choice == 'D')
      for(unsigned int i = 0; i < word.length(); i++) {
	 n = word[i] - 'a' - shift;
	 if(n < 0)
	    n += 26;
	 word[i] = char((n % 26) + 'a');
      }
 

   cout << word << endl;
   
   return 0;
}
