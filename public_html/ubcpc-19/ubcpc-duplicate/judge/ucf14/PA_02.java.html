import java.util.*;
import java.io.*;

public class A {
    public static void main(String args[]) {
        Scanner in = new Scanner(new BufferedReader(new InputStreamReader(System.in)));

        String type = in.next();
        int shift = in.nextInt();
        String word = in.next();

        if (type.equals("E")) {
            encode(shift, word);
        } else {
            decode(shift, word);
        }
        System.out.println();

        in.close();
    }

    public static void encode(int shift, String word) {
        int len = word.length();
        for (int i = 0; i < len; i++) {
            int c = word.charAt(i) - 'a';

            System.out.print((char) ((c + shift) % 26 + 'a'));
        }
    }

    public static void decode(int shift, String word) {
        int len = word.length();
        for (int i = 0; i < len; i++) {
            int c = word.charAt(i) - 'a';

            System.out.print((char) (((c - shift + 26) % 26) + 'a'));
        }
    }
}
