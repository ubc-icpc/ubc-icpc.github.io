import java.util.*;
import java.io.*;

public class E {
    public static void main(String args[]) {
        Scanner in = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        int sceneNum = in.nextInt();
        int actNum = in.nextInt();

        int list[] = new int[sceneNum];
        for (int i = 0; i < sceneNum; i++) {
            list[i] = in.nextInt();
        }

        Arrays.sort(list);

        int result = 0;
        for (int i = 0; i < actNum; i++) {
            int index = sceneNum - i - 1;
            if (index >= 0 & index < sceneNum)
                result += list[sceneNum - i - 1];
        }

        System.out.println(result);

        in.close();
    }
}
