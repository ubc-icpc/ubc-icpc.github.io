#include <iostream>
using namespace std;
int main(void) {
	char c;
	cin >> c;
	int s;
	cin >> s;
	string w;
	cin >> w;
	if (c == 'D') 
		for (int i = 0; i < w.size(); ++i) 
			cout << (char) ('a' + ((w[i] - 'a') - s + 26) % 26);
	else
		for (int i = 0; i < w.size(); ++i) 
			cout << (char) ('a' + ((w[i] - 'a') + s + 26) % 26);
	cout << '\n';
	return 0;
}
