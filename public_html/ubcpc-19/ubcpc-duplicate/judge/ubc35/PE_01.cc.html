#include <iostream>
#include <algorithm>

using namespace std;

int as[3000];

int main() {
		ios::sync_with_stdio(0);
		cin.tie(0);
		int n, k;
		cin >> n >> k;
		for (int i = 0; i < n; ++i) {
			cin >> as[i];
			as[i] = -as[i];
		}
		sort(as, as + n);

		int res = 0;
		for (int i = 0; i < k; ++i)
			res -= as[i];
		cout << res << '\n';
		return 0;
}
