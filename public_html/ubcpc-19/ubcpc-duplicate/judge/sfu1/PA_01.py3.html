def solution():
    action = str(input())
    s = int(input())
    data = str(input())
    result = ""
    
    if (action == 'D'):
        s = s * (-1)
    
    for i in range(len(data)):
        t = ord(data[i])
        t += s
        if (t > 122):
            t = t - 26
        elif (t < 97):
            t = t + 26
        result = result + chr(t)
    
    print(result)
    
if __name__ == "__main__":
    solution()
