#include <iostream>
#include <cmath>

using namespace std;
    long n,d=1;
int main()
{
    cin>>n;
    long p;
    while(n>1) {
        p = cbrt(n);
        bool ok=false;
        long i;
        for(i = 2; i<= p; i++) {
            if(n % i == 0) {ok=true;break;}
        }
        if(ok) { 
            n/=i;
            if(n%i==0) n/=i; else d*=i;
        } else {
            long r = sqrt(n);
            if(r *r == n) n =1;
            else break;
        }
    }
    cout<<d * n;
    return 0;
}
