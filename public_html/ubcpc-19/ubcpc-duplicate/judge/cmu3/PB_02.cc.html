#include <iostream>

using namespace std;
    int n,s;
    string na[100], ty;
    int hp[100],at[100],df[100];
bool win(int i,int j) {
    int ri = max(0,at[j]-df[i]);
    int rj = max(0,at[i]-df[j]);
    if(rj==0) return false;
    if(ri==0) return true;
    return (hp[i]-1)/ri > (hp[j]-1)/rj;
}
int main()
{
    cin>>n;
    for(int i=0; i <n;i++) cin>>na[i]>>hp[i]>>at[i]>>df[i];
    for(int i=0;i<n;i++) for(int j=i+1;j<n;j++) for(int k=j+1;k<n;k++) {
        if((win(i,j)&&win(j,k)&&win(k,i)) || (win(j,i)&&win(k,j)&&win(i,k))) s++,ty+=(na[i]+' '+na[j]+' '+na[k]+"\n");
    }
    cout<<s<<endl<<ty;

    return 0;
}
