ed = input()
shift = int(input())
text = input()

bytetext = bytearray()
bytetext.extend(map(ord, text))
if ed == "E":
  for i in range(len(bytetext)):
    bytetext[i] += shift
    if bytetext[i] >= 123:
      bytetext[i] -= 26
elif ed == "D":
  for i in range(len(bytetext)):
    bytetext[i] -= shift
    if bytetext[i] <= 96:
      bytetext[i] += 26

print(str(bytetext, "utf-8"))
