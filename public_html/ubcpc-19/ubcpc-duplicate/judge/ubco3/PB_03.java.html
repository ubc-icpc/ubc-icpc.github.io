import java.util.ArrayList;
import java.util.Scanner;

public class B {

	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		int n = in.nextInt();
		Fighter[] fighters = new Fighter[n];
		ArrayList<Fighter> correct = new ArrayList<Fighter>();
		for(int i = 0; i < n; i++) {
			fighters[i] = new Fighter(in.next(),in.nextInt(),in.nextInt(),in.nextInt());
		}
		
		for(int i = 0; i < n; i++)
			for(int j = i+1; j < n; j++)
				for(int k = j+1; k < n; k++) {
					if(check(fighters[i],fighters[j],fighters[k])){
						correct.add(fighters[i]);
						correct.add(fighters[j]);
						correct.add(fighters[k]);
					}
				}
		System.out.println(correct.size()/3);
		for(int i = 0; i < correct.size(); i+=3) {
			System.out.println(correct.get(i).name + " " + correct.get(i+1).name + " " + correct.get(i+2).name);
		}
	}
	
	public static boolean check(Fighter A, Fighter B, Fighter C) {
		int i = A.fight(B)+B.fight(C)+C.fight(A);
		if(i ==3 || i ==  -3) {
			return true;
		}
		return false;
	}

}
class Fighter{
	
	String name;
	public double hp;
	public double at;
	public double df;
	
	public Fighter(String name, double hp, double at, double df) {
		super();
		this.name = name;
		this.hp = hp;
		this.at = at;
		this.df = df;
	}
	
	int fight(Fighter enemy) {
		if(Math.ceil(hp/Math.max(0, enemy.at-df)) > Math.ceil(enemy.hp/Math.max(0, at-enemy.df)))
			return 1;
		else if(Math.ceil(hp/Math.max(0, enemy.at-df)) < Math.ceil(enemy.hp/Math.max(0, at-enemy.df)))
			return -1;
		return 0;
	}
	

}
