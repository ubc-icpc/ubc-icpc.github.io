import java.util.Collections;
import java.util.PriorityQueue;
import java.util.Scanner;

public class E {

	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		int n = in.nextInt();
		int k = in.nextInt();
		PriorityQueue<Integer> scenes = new PriorityQueue(Collections.reverseOrder());
		for(int i = 0; i < n; i++) {
			scenes.offer(in.nextInt());
		}
		int sum = 0;
		for(int i = 0; i < k; i++) {
			sum += scenes.poll();
		}
		System.out.println(sum);
	}

}
