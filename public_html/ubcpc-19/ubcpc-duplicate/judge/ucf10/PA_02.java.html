import java.io.BufferedReader;
import java.util.Scanner;

public class a {

	public static void main(String[] args) {


		Scanner kb = new Scanner(System.in);
		String type = kb.next();
		int shift = kb.nextInt();
		String s = kb.next();
		if(type.equals("E"))
		{
			
			int i = 0;
			while(i < s.length())
			{
				System.out.print((char)(((((int)s.charAt(i) - 97) + shift) % 26)+97));
				i++;
			}
		}
		else
		{
			int i = 0;
			while(i < s.length())
			{
				System.out.print((char)(((((int)s.charAt(i) - 97) + (26 - shift)) % 26)+97));
				i++;
			}
		}
	}

}
