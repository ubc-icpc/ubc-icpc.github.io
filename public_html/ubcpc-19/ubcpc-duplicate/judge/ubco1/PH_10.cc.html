#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

bool is_square(ll x){

	ll root = (ll)sqrt(x);

	return (root*root == x);

}

int main(){

	ll N;
	ll ans = 1;
	cin >> N;

	int cnt2 = 0;
	while(N % 2 == 0){
		cnt2++;
		N = N/2;
	}
	if(cnt2 % 2 == 1) ans *= 2; 

	ll limit = (ll)cbrt(N) + 1;

	for(ll i = 3; i <= limit; i+= 2){
		if(N % i != 0) continue;
		int cnt = 0;
		while(N % i == 0){
			cnt++;
			N = N/i;
		}
		if(cnt % 2 == 1) ans *= i;
	}

	limit = (ll)cbrt(N) + 1;
	
	ll minP = N;
	for(ll i = 1; i <= limit; i++){
		if(N % i == 0){
			ll Q2 = N/i;
			if(is_square(Q2)){
				minP = min(minP, i);
			}
			ll j = i*i;
			if(N % j == 0){
				minP = min(minP, N/j);
			}
		}
	}
	ans *= minP;

	cout << ans << endl;

	return 0;

}
