import java.util.Arrays;
import java.util.Scanner;

public class e {
	public static void main(String[] Args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int k = sc.nextInt();
		int[] vals = new int[n];
		for (int i = 0; i < n; i++)
			vals[i] = sc.nextInt();
		Arrays.sort(vals);
		int ret = 0;
		for (int i = 0; i < k; i++)
			ret += vals[n - i - 1];
		System.out.println(ret);
	}
}
