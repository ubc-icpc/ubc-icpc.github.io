import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.*;

public class Scratch {
    public static void main(String[] args) {
        Scanner reader = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
    
        String first = reader.nextLine();
        String second = reader.nextLine();
        String third = reader.nextLine();
        
        Map<Integer, Character> mapping = new HashMap<>();
        Map<Character, Integer> mapping2 = new HashMap<>();
        
        char start = 'a';
        for (int i = 0; i < 26; i++) {
            char letter = (char) (start + i);
            mapping.put(i, letter);
            mapping2.put(letter, i);
        }
        
        int shift = Integer.parseInt(second);
        
        if (first.equals("E")) {
            String[] split = third.split("");
            String encoded = "";
            for (String c : split) {
                char individual = c.toCharArray()[0];
                encoded += mapping.get((mapping2.get(individual) + shift) % 26);
            }
            System.out.println(encoded);
        } else {
            String[] split = third.split("");
            String encoded = "";
            for (String c : split) {
                char individual = c.toCharArray()[0];
                encoded += mapping.get((26 + mapping2.get(individual) - shift) % 26);
            }
            System.out.println(encoded);
        }
    }
}
