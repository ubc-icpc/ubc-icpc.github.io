import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.Scanner;

public class E {
    public static void main(String[] args) {
        Scanner reader = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        
        String line = reader.nextLine();
        int scenes = Integer.parseInt(line.split(" ")[0]);
        int acts = Integer.parseInt(line.split(" ")[1]);
        
        String numbers = reader.nextLine();
        System.out.println(Arrays.stream(numbers.split(" ")).map(Integer::parseInt).sorted((a, b) -> b - a).limit(acts).reduce(0, Integer::sum));
    }
}
