#include <bits/stdc++.h>

using namespace std;

int INF = 1000000000;

int N;
string name[105];
int H[105], A[105], D[105];
	vector<string> output;

bool wins(int i, int j)
{
	int turns_i = INF; // rounds for i to defeat j
	if (A[i] > D[j])
		turns_i = (H[j] + A[i] - D[j] - 1) / (A[i] - D[j]);
	int turns_j = INF;
	if (A[j] > D[i])
		turns_j = (H[i] + A[j] - D[i] - 1) / (A[j] - D[i]);
	return (turns_i < turns_j);
}

int main()
{
	ios::sync_with_stdio(false);
	cin >> N;
	for (int i = 0; i < N; i++)
		cin >> name[i] >> H[i] >> A[i] >> D[i];
	for (int i = 0; i < N; i++)
		for (int j = i + 1; j < N; j++)
			for (int k = i + 1; k < N; k++)
				if (wins(i, j) && wins(j, k) && wins(k, i))
					output.push_back(name[i] + ' ' + name[j] + ' ' + name[k]);
	cout << output.size() << endl;
	for (int i = 0; i < output.size(); i++) cout << output[i] << '\n';
}
