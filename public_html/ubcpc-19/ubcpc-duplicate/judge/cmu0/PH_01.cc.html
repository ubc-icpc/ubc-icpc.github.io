#include <bits/stdc++.h>

using namespace std;

typedef unsigned long long ll;

int main()
{
	ll C;
	ll D = 1;
	cin >> C;
	for (ll x = 2; x <= 2097151; x++)
	{
		if (C % x == 0)
		{
			int p = 0;
			while (C % x == 0)
			{
				p++;
				C /= x;
			}
			if (p % 2 == 1)
				D *= x;
		}
	}
	ll sqrtC = (sqrt(C) + 0.5);
	//cout << sqrtC << endl;
	if (sqrtC * sqrtC != C)
		D *= C;
	cout << D << endl;
}
