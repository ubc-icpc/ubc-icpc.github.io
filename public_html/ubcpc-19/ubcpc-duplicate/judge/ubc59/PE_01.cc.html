int main() {
    int N, K, acc; acc = 0;
    cin >> N >> K;
    int a[N];

    for(int i = 0; i < N; i++) {
        cin >> a[i];
    }
    sort(a, a+N);
    for(int j = N - 1; j > N - 1 - K; j--) {
        acc += a[j];
    }
    cout << acc << endl;

    return 0;
}
