import java.util.*;

public class a {
	
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		
		boolean encode = scan.nextLine().equals("E");
		int shiftAmount = scan.nextInt();
		scan.nextLine();
		String input = scan.nextLine();
		
		if (!encode) {
			shiftAmount *= -1;
		}
		String output = "";
		for (int i = 0; i < input.length(); i++) {
			char character = (char) ((input.charAt(i) + shiftAmount - 'a') % 26 + 'a');
			while (character < 'a') {
				character += 26;
			}
			output += character;
		}
		
		System.out.println(output);
		scan.close();
	}
	
}
