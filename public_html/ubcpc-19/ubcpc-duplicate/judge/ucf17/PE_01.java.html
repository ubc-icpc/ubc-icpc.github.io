import java.util.*;

public class e {
	
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		
		int numScenes = scan.nextInt();
		int numActs = scan.nextInt();
		
		int[] sceneExcitementLevels = new int[numScenes];
		for (int i = 0; i < numScenes; i++) {
			sceneExcitementLevels[i] = scan.nextInt();
		}
		
		// Find the top numActs values in the array
		int[] actExcitementLevels = new int[numActs];
		for (int i = 0; i < numActs; i++) {
			int maxSceneExcitementLevelIndex = 0;
			for (int j = 0; j < numScenes; j++) {
				if (sceneExcitementLevels[j] > sceneExcitementLevels[maxSceneExcitementLevelIndex]) {
					maxSceneExcitementLevelIndex = j;
				}
			}
			
			actExcitementLevels[i] = sceneExcitementLevels[maxSceneExcitementLevelIndex];
			sceneExcitementLevels[maxSceneExcitementLevelIndex] = Integer.MIN_VALUE;
		}
		
		int playExcitementLevel = 0;
		for (int i = 0; i < numActs; i++) {
			playExcitementLevel += actExcitementLevels[i];
		}
		
		System.out.println(playExcitementLevel);
		
		scan.close();
	}
	
}
