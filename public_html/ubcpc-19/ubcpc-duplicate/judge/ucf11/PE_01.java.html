import java.util.Arrays;
import java.util.Scanner;

public class acts {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int k = sc.nextInt();
		int[] scenes = new int[n];
		for(int i = 0; i < n; i++) {
			scenes[i] = sc.nextInt();
		}
		int sum = 0;
		Arrays.sort(scenes);
		int index = scenes.length - 1;
		while(k > 0) {
			sum+=scenes[index];
			index--;
			k--;
		}
		System.out.print(sum);
		sc.close();
	}
	
}
