def encode(x,n):
	new_string = ''
	for char in x:
		i = ord(char)
		i +=-97
		i+=n
		i = i%26
		i+=97
		new_string += chr(i)
	return new_string

def decode(x,n):
	return encode(x,-n)

method = raw_input()
n = int(raw_input())
x = raw_input()

if method=="E":
	print encode(x,n)
if method=="D":
	print decode(x,n)
