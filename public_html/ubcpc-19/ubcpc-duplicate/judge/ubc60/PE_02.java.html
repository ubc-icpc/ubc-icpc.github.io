import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.Scanner;

public class E {
    public static Scanner scanner = new Scanner(new BufferedReader(new InputStreamReader(System.in)));

    public static void main(String[] args) {
        int numScenes = scanner.nextInt();
        int numActs = scanner.nextInt();
        scanner.nextLine();

        if (numScenes == 0) {
            System.out.println("0");
            System.exit(0);
        }
        int[] scenes = new int[numScenes];

        for (int i = 0; i < numScenes; i++) {
            scenes[i] = (scanner.nextInt());
        }

        Arrays.sort(scenes);

        int sum = 0;
        for (int i = 0, j = numScenes - 1; i < numActs; i++, j--) {
            sum += scenes[j];
        }
        System.out.println(sum);
    }
}
