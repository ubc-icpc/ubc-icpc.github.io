import java.util.*;

public class a {

	public static void main(String[] args) {
		Scanner s=new Scanner(System.in);
		int numpos=s.nextInt();
		int numchoose=s.nextInt();
		int[] acts=new int[numpos];
		for(int i=0;i<numpos;i++) {
			acts[i]=s.nextInt();
		}
		Arrays.sort(acts);
		int ans=0;
		int numActs=Math.min(numpos, numchoose);
		for(int i=0;i<numActs;i++) {
			ans+=acts[numpos-i-1];
		}
		System.out.println(ans);
		s.close();

	}

}
