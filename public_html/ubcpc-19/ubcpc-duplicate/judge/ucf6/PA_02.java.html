import java.util.*;

public class d {

	public static void main(String[] args) {
		Scanner s=new Scanner(System.in);
		String encdec=s.next();
		int amount=s.nextInt();
		char[] input=s.next().toCharArray();
		if(encdec.equals("E")) {
			for(int i=0;i<input.length;i++) {
				input[i]=(char)('a'+(input[i]-'a'+amount)%26);
			}
		}
		else {
			for(int i=0;i<input.length;i++) {
				input[i]=(char)('a'+(input[i]-'a'-amount+26)%26);
			}
		}
		for(char c:input)System.out.print(c);
		
		
		s.close();
	}

}
