a = input('')
b = int(input(''))
c = list(input(''))
output = ''

if a == 'E':
    for i in range(0, len(c)):
        if ord(c[i]) + b > 122:
            output += chr(ord(c[i]) + b - 26)

        else:
            output += (chr(ord(c[i]) + b))

else:
    for i in range(0, len(c)):
        if ord(c[i]) - b < 97:
            output += chr(ord(c[i]) - b + 26)

        else:
            output += (chr(ord(c[i]) - b))


print(output)
