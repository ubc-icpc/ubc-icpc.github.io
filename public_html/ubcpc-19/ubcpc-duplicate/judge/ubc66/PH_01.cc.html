#include <bits/stdc++.h>

using namespace std;

int n;
long long p;
long long ans=1;

int main(){
	scanf("%lld",&p);
	for(int i=2;i<=p && i<=5000000;i++){
	//	printf("%d\n",i);
		if(p%i==0){
			int cnt=0;
			while(p%i==0){
				p/=i;
				cnt++;
			}
			if(cnt%2==1)
				ans*=(long long)i;
		}
	}
	long long q=sqrt(p);
	if(q*q!=p)
		ans*=p;
	
	printf("%lld\n",ans);

	return 0;
}
