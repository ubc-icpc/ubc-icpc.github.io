#include <bits/stdc++.h>

using namespace std;

int n;
char DE[5];
char str[110];

int main(){
	scanf("%s",DE);
	scanf("%d",&n);
	scanf("%s",str);

	int len=strlen(str);
	for(int i=0;i<len;i++){
		for(int j=1;j<=n;j++){
			if(DE[0]=='E'){
				str[i]=str[i]+1;
				if(str[i]>'z')
					str[i]='a';
			}else{
				str[i]=str[i]-1;
				if(str[i]<'a')
					str[i]='z';
			}
		}
	}
	printf("%s\n",str);

	return 0;
}
