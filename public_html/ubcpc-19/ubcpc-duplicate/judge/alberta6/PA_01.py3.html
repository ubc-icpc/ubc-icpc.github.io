what = input()
shift = int(input())

text = input()
newText = ""

if what == "E":
	for char in text:
		if(ord(char) + shift > ord('z')):
			newText += chr(ord(char) + (shift - 26))
		else:
			newText += chr(ord(char) + shift)

else:
	for char in text:
		if(ord(char) - shift < ord('a')):
			newText += chr(ord(char) - shift + 26)
		else:
			newText += chr(ord(char) - shift)

print(newText)
