#include <iostream>
#include <vector>

using namespace std;

int main() {
	int N;
	cin >> N;
	int current = 2;
	for (int i = 3; i < N + 1; i++) {
		cout << "? " << 1 << " " << current << " " << i << "\n";
		int judge_output;
		cin >> judge_output;
		if (judge_output == -1) {
			return 0;
		}
		if (judge_output == 1) {
			current = i;
		}
	}
	cout << "! " << current << "\n";
	return 0;
}
