int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);

    string alpha = "abcdefghijklmnopqrstuvwxyz";
    string task, str;
    int s;

    cin >> task >> s >> str;
    if(task[0]=='E') {
        for(int i=0; i<str.length(); i++)
            str[i] = alpha[(str[i]-'a'+s)%alpha.length()];
    }
    else {
        for(int i=0; i<str.length(); i++)
            str[i] = alpha[(str[i]-'a'+alpha.length()-s)%alpha.length()];
    }
    cout << str << "\n";


    return 0;
}
