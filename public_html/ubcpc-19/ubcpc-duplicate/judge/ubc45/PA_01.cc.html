#include <bits/stdc++.h>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    char c;
    int n;
    string s;
    cin >> c >> n >> s;
    int shift = c == 'E'? n : 26 - n;
    for (int i = 0; i < s.size(); i++) {
        int idx = 1 * s[i];
        s[i] = ((s[i] - 'a' + shift) % 26) + 'a';
    }
    
    cout << s << '\n';
}
