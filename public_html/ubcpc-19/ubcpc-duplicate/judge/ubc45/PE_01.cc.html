#include <bits/stdc++.h>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N, A;
    cin >> N >> A;
    
    vector<int> v(N);
    for (int i = 0; i < N; i++) cin >> v[i];

    sort(v.begin(), v.end());

    int sum = 0;
    for (int i = N-A; i < N; i++) sum += v[i];
    cout << sum << '\n';
}
