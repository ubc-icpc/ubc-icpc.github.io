#include <bits/stdc++.h>
const char nl = '\n';
using namespace std;
using ll = long long;
using ld = long double;
const int N = 2010;

int n, k;
int a[N];

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  cin >> n >> k;
  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }
  sort(a, a + n, greater<int>());
  ll ans = 0;
  for (int i = 0; i < k; i++) {
    ans += a[i];
  }
  cout << ans << nl;
  return 0;
}
