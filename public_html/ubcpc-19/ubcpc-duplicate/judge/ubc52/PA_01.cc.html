#include <bits/stdc++.h>
const char nl = '\n';
using namespace std;
using ll = long long;
using ld = long double;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  char c; cin >> c;
  int k; cin >> k;
  string s; cin >> s;
  if (c == 'E') {
    for (char& c : s) {
      c = ((c - 'a') + k) % 26 + 'a';
    }
  } else {
    for (char& c : s) {
      c = ((c - 'a') - k + 26) % 26 + 'a';
    }
  }
  cout << s << nl;
  return 0;
}
