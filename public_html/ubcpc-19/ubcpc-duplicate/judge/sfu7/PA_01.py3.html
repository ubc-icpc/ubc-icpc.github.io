def encode(word):
    return [ord(c)-ord('a') for c in word]
def decode(encode):
    return ''.join([chr(ord('a') + n) for n in encode])


def c(word, shift):
    e = encode(word)
    e = [(n + shift) % 26 for n in e]
    print(decode(e))
en = input()
shift = int(input())
word = input()
if (en == 'D'):
    shift *= -1

c(word, shift)
