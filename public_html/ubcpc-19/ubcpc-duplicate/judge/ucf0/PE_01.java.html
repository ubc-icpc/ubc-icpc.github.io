import java.util.Arrays;
import java.util.Scanner;

public class play {

	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		
		int numscenes = in.nextInt();
		int numacts = in.nextInt();
		
		int[] scenes = new int[numscenes];
		int[] maxes = new int[numacts];
		for (int i = 0; i < numscenes; i++) {
			scenes[i] = in.nextInt();
			if (scenes[i] < maxes[0]) continue;
			maxes[0] = scenes[i];
			for (int j = 1; j < numacts; j++) {	
				if (scenes[i] > maxes[j]) {
					maxes[j - 1] = maxes[j];
					maxes[j] = scenes[i];
				} else break;
			}
		}
		//System.out.println(Arrays.toString(scenes));
		//System.out.println(Arrays.toString(maxes));
		int ans = 0;
		for (int i = 0; i < maxes.length; i++) {
			ans+= maxes[i];
		}
		System.out.println(ans);
		in.close();
	}

}
