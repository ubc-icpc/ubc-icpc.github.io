import java.util.Scanner;

public class Solution {

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);

        boolean encode = scanner.next().contains("E");
        int shift = Integer.parseInt(scanner.next());
        char[] string = scanner.next().toCharArray();

        shift = encode ? shift : 26 - shift;

        for(int i=0; i<string.length; i++) {
            string[i] = (char)(((string[i] - 'a') + shift)%26 + 'a');
        }

        System.out.println(String.copyValueOf(string));

    }

}
