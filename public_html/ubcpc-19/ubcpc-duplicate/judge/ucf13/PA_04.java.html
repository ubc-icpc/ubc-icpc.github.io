import java.util.Scanner;

public class one {
	
	public static void main(String[] args) {
		
		Scanner kb = new Scanner(System.in);
		String opt = kb.next();
		int s = kb.nextInt();
		String str = kb.next();
		String end = "";
		if(opt.equals("E")) {
		for(int x=0;x<str.length();x++) {
			String out = Character.toString((char) (str.charAt(x)+s));
			if(str.charAt(x)+s>122) {
				//System.out.println((str.charAt(x)+s)-122);
				out =  Character.toString((char) (96+((str.charAt(x)+s)-122)));
			}
			end = end + out;
		}
		System.out.println(end);
		}
		else {
			for(int y=0;y<str.length();y++) {
				String out = Character.toString((char) (str.charAt(y)-s));
				if(str.charAt(y)-s<97) {
					out =  Character.toString((char) (123-(97-(str.charAt(y)-s))));
					//System.out.println(out.charAt(0)+0);
				}
				
				end = end+out;
			}
			System.out.println(end);
		}
	}

}
