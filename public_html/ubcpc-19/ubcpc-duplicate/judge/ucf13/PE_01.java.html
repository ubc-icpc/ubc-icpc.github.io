import java.util.Scanner;

public class second {
	
	public static void main(String[] args) {
		Scanner kb = new Scanner(System.in);
		int scenes = kb.nextInt();
		int acts = kb.nextInt();
		int[]values = new int[scenes];
		int[] wt = new int[scenes];
		for(int x =0;x<scenes;x++) {
			values[x] = kb.nextInt();
			wt[x] = 1;
		}
		System.out.println(bottomUpDP(values,wt,acts));
	}
	  public static int bottomUpDP(int val[], int wt[], int W){
	        int K[][] = new int[val.length+1][W+1];
	        for(int i=0; i <= val.length; i++){
	            for(int j=0; j <= W; j++){
	                if(i == 0 || j == 0){
	                    K[i][j] = 0;
	                    continue;
	                }
	                if(j - wt[i-1] >= 0){
	                    K[i][j] = Math.max(K[i-1][j], K[i-1][j-wt[i-1]] + val[i-1]);
	                }else{
	                    K[i][j] = K[i-1][j];
	                }
	            }
	        }
	        return K[val.length][W];
	    }

}
