import java.util.*;
public class h {
	static int max = 1 << 21 + 1;
	static boolean[] prime = new boolean[max];
	static ArrayList<Integer> primes = new ArrayList<>();
	public static void main(String[] args) {
		Arrays.fill(prime, true);
		prime[0] = false;
		prime[1] = false;
		for(int i = 0; i < max; ++i) {
			if(prime[i]) {
				for(int j = i + i; j < max; j += i) prime[j] = false;
			}
		}
		
		for(int i = 0; i < max; ++i) {
			if(prime[i]) primes.add(i);
		}
		Scanner sc = new Scanner(System.in);
		long c = sc.nextLong();
		for(int a : primes) {
			long i = (long)a * a;
			while(c % i == 0) c /= i;
		}
		
		long out = c;
		for(int i : primes) {
			if(c % i == 0) c /= i;
		}
		
		long sqrt = (long)Math.round(Math.sqrt(c));
		if(sqrt * sqrt == c) out /= c;
//		double sqrtD = Math.sqrt(c), sqrtI = (int)Math.round(sqrtD);
//		if(Math.abs(sqrtD - sqrtI) <= 1e-8) out /= c;
		
		System.out.println(out);
	}
}
