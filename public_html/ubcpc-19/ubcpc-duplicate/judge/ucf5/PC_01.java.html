import java.util.*;
public class c {
	static int n;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		int piv = 2;
		for(int i = 3; i <= n; ++i) {
			System.out.printf("? %d %d %d\n", 1, piv, i);
			System.out.flush();
			if(sc.nextInt() == 1) piv = i;
		}
		System.out.println("! " + piv);
		System.out.flush();
	}
}
