import java.util.*;
public class b {
	static int n;
	static String[] names;
	static int[] hp, at, df;
	static boolean[][] win;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		names = new String[n];
		hp = new int[n];
		at = new int[n];
		df = new int[n];
		win = new boolean[n][n];
		for(int i = 0; i < n; ++i) {
			names[i] = sc.next();
			hp[i] = sc.nextInt();
			at[i] = sc.nextInt();
			df[i] = sc.nextInt();
		}
		
		for(int i = 0; i < n; ++i) {
			for(int j = 0; j < n; ++j) {
				if(i == j) continue;
				if(df[i] >= at[j] && df[j] >= at[i]) continue; //draw
				
				int numForI = Integer.MAX_VALUE;
				if(at[i] > df[j]) numForI = (hp[j] + at[i] - df[j] - 1) / (at[i] - df[j]);
				int numForJ = Integer.MAX_VALUE;
				if(at[j] > df[i]) numForJ = (hp[i] + at[j] - df[i] - 1) / (at[j] - df[i]);
				
				if(numForI < numForJ) win[i][j] = true;
				if(numForJ < numForI) win[j][i] = true;
			}
		}
		
		ArrayList<String> out = new ArrayList<>();
		for(int i = 0; i < n; ++i) {
			for(int j = i + 1; j < n; ++j) {
				for(int k = j + 1; k < n; ++k) {
					if(win[i][j] && win[j][k] && win[k][i] || win[j][i] && win[k][j] && win[i][k]) {
						out.add(names[i] + " " + names[j] + " " + names[k]);
					}
				}
			}
		}
		
		System.out.println(out.size());
		for(String s : out) System.out.println(s);
	}
}
