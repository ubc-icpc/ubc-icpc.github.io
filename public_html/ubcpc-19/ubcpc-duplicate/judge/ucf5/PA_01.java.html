import java.util.*;
public class a {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		char in = sc.next().charAt(0);
		int off = 1;
		if(in == 'D') off = -1;
		off *= sc.nextInt();
		char[] arr = sc.next().toCharArray();
		for(char a : arr) {
			int b = (int)(a - 'a') + off;
			if(b < 0) b += 26;
			else if(b >= 26) b -= 26;
			
			char c = (char)(b + 'a');
			System.out.print(c);
		}
		System.out.println();
	}
}
