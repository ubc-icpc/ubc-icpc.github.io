import java.util.*;
import java.io.*;

public class A
{
	public static void main(String[] args) throws IOException
	{
		BufferedReader f = new BufferedReader(new InputStreamReader(System.in));
		
//		StringTokenizer token = new StringTokenizer(f.readLine());
//		int val1 = Integer.parseInt();
		
		String code = f.readLine();
		int shift = Integer.parseInt(f.readLine());
		String o = f.readLine();
		
		if(code.charAt(0) == 'D')
		{
			shift *= -1;
		}
		
		//System.out.println("old string = "+o);
		
		String newString = "";
		for(int i = 0;i<o.length();i++)
		{
			char oChar = o.charAt(i);
			int charVal = oChar - 'a';
			charVal = (charVal + shift + 26)%26;
			charVal += 'a';
			
			char newChar = (char) (charVal);
			newString += newChar;
			//System.out.println("new char = "+newChar+", charVal = "+charVal);
		}
		//System.out.println(newString.charAt(0));
		
		System.out.println(newString);
		
		f.close();
		System.exit(0);
	}
}
