import java.util.*;
import java.io.*;

public class E
{
	public static void main(String[] args) throws IOException
	{
		BufferedReader f = new BufferedReader(new InputStreamReader(System.in));
		
		StringTokenizer token = new StringTokenizer(f.readLine());
		int scenes = Integer.parseInt(token.nextToken());
		int acts = Integer.parseInt(token.nextToken());
		
		
		token = new StringTokenizer(f.readLine());
		
		ArrayList<Integer> list = new ArrayList<Integer>();
		for(int i = 0;i<scenes;i++)
		{
			list.add(Integer.parseInt(token.nextToken()));
		}
		
		int total = 0;
		for(int i = 0;i<acts;i++)
		{
			int index = returnGreatestIndex(list);
			total += list.get(index);
			
			list.remove(index);
		}
		
		System.out.println(total);
		
		f.close();
		System.exit(0);
	}
	
	public static int returnGreatestIndex(ArrayList<Integer> list)
	{
		int toReturn = -1;
		
		Iterator it = list.iterator();
		
		int currentIndex = 0;
		while(it.hasNext())
		{
			Integer nextVal = (Integer) it.next();
			
			if(currentIndex ==  0)
			{
				toReturn = 0;
			}
			else
			{
				if(nextVal > list.get(toReturn))
				{
					toReturn = currentIndex;
				}
			}
			
			currentIndex++;
		}
		
		return toReturn;
	}
}
