#include <iostream>

using namespace std;

int main() {
    char l;
    cin >> l;
    int s;
    cin >> s;
    string word;
    cin >> word;


    if (l == 'D') {
        for (int i = 0; i < word.size(); i++) {
            for (int j = s; j > 0; j--) {
                word[i] = word[i] - 1;
                if (word[i] == 96) word[i] = 122;
            }
        }
    }
    else {
        for (int i = 0; i < word.size(); i++) {
            for (int j = s; j > 0; j--) {
                word[i] = word[i] + 1;
                if (word[i] == 123) word[i] = 97;
            }
        }

    }

    cout << word;

    return 0;
}
