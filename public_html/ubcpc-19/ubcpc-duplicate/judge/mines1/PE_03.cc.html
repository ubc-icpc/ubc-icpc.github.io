#include <iostream>
#include <sstream>
#include <vector>
using namespace std;

int main() {
    int n, k, temp;
    vector<int> exciting;
    vector<int> mostExcite;

    cin >> n >> k;
    for(int i = 0; i < n; i++) {
        cin >> temp;
        exciting.push_back(temp);
    }

    int sum = 0;
    int greatest = 0;

    for (int i = 0; i < k; i++) {
        for (int j = 0; j < exciting.size(); j++) {
            if(exciting[j] > exciting[greatest]) {
                greatest = j;
            }
        }
        mostExcite.push_back(exciting[greatest]);
        exciting[greatest] = 0;
    }

    for (int k = 0; k < mostExcite.size(); k++) {
        sum += mostExcite[k];
    }

    cout << sum;

    return 0;
}
