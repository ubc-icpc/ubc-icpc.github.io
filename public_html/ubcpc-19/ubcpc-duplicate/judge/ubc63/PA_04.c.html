#include <stdio.h>
#include <string.h>


int main(){
   char word[200];
   int shift;
   char mode;  
   scanf("%s",&mode);
   scanf("%d",&shift);
   scanf("%s",&word);
   int i,j;
   if (mode=='E'){
      for (i=0;i<strlen(word);i++){
         
         if (word[i]+shift>'z'){
             word[i] = word[i] + shift - 26;
         }
         else{
         word[i] += shift;
         }
      }
    }
   else if(mode=='D'){
      for (i=0;i<strlen(word);i++){
         word[i] -= shift;
         if (word[i]<'a'){
            word[i] += 26;
         }
      }
   }
   
   printf(word);
   printf("\n");
}
