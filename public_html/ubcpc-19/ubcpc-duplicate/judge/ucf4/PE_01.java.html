import java.util.Arrays;
import java.util.Scanner;

public class E {
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		
		int N = in.nextInt();
		int K = in.nextInt();
		int[] excitement = new int[N];
		
		for (int i = 0; i < N; i ++) {
			excitement[i] = in.nextInt();
		}
		
		Arrays.sort(excitement);
		
		int total = 0;
		
		for (int i = 0; i < K; i ++) {
			total += excitement[N - 1 - i];
		}
		
		System.out.println(total);
	}
}
