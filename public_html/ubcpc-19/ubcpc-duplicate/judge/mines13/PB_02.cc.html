#include<string>
#include<iostream>
using namespace std;
int nto(int h,int a,int d){
  if(d>=a)return -1;
  return (h+a-d-1)/(a-d); 
}
int H[100];
int A[100];
int D[100];
string name[100];
int wins(int x,int y){
  int a=nto(H[x],A[y],D[x]);
  int b=nto(H[y],A[x],D[y]);
  if(a==-1 && b==-1)return 0;
  if(a==b)return 0;
  if(a==-1)return 1;
  if(b==-1)return -1;
  return a<b?-1:1;
} 
int main(){
  cin.tie(0);
  int n;
  cin>>n;
  for(int i=0;i<n;++i)cin>>name[i]>>H[i]>>A[i]>>D[i];
  int count = 0;
  for(int i=0;i<n;++i){
    for(int j=i+1;j<n;++j){
      for(int k=j+1;k<n;++k){
        if(wins(i,j)!=0){
          if(wins(i,j) == wins(j,k) && wins(k,i)==wins(i,j) ) ++count;
        }
      }
    }
  }
  cout<<count<<endl;
  for(int i=0;i<n;++i){
    for(int j=i+1;j<n;++j){
      for(int k=j+1;k<n;++k){
        if(wins(i,j)!=0){
	  if(wins(i,j) == wins(j,k) && wins(k,i)==wins(i,j) ) cout<<name[i]<<" "<<name[j]<<" "<<name[k]<<endl;
        }
      }
    }
  }
  cout<<flush;
  return 0;
}
