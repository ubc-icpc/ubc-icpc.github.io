import java.util.*;
public class B {
	
static String[] names;
static int[][] stats;

public static boolean win(int a , int b) {

	int turnsA = (int)Math.ceil((double)stats[a][0] / (Integer.max(stats[b][1] - stats[a][2] , 0)));
	
	int turnsB = (int)Math.ceil((double)stats[b][0] / (Integer.max(stats[a][1] - stats[b][2] , 0)));
	
	return turnsA > turnsB;
}

public static boolean t(int a , int b , int c) {
	return win(a,b) && win(b,c) && win(c,a);
}

public static void main(String[] args) {
Scanner scan = new Scanner(System.in);

int N = scan.nextInt();

names = new String[N];
stats = new int[N][3];

for (int i = 0; i < N; i++)
	{
	names[i] = scan.next();
	
	for (int j = 0; j < 3; j++)
		stats[i][j] = scan.nextInt();
	}

int count = 0;
ArrayList<String> trips = new ArrayList<String>();

for (int i = 0; i < N; i++)
	for (int j = i+1; j < N; j++)
		for (int k = j+1; k < N; k++)
			if(t(i,j,k) || t(i,k,j) || t(j,i,k) || t(j,k,i) || t(k,i,j) || t(k,j,i))
				{
				count++;
				trips.add(names[i] + " " + names[j] + " " + names[k]);
				}

System.out.println(count);

for (String trip : trips)
	System.out.println(trip);

scan.close();
	}
}
