#include <stdio.h>
#include <stdlib.h>

int main()
{
    char which_code;
    int shift;
    char word[101];

    scanf("%c %d %s", &which_code, &shift, &word);

    if (which_code =='E'){
        int i=0;
        while(word[i]!= '\0'){
            int letnum = (int)word[i];
            int shifted_character = letnum + shift;

            if (shifted_character> 122){
                int new_int = shifted_character - 122;
                shifted_character = new_int + 96;
            }
            word[i] = shifted_character;
            i++;
        }
    } else {
        int i =0;
        while(word[i]!= '\0'){
            int letnum = (int)word[i];
            int shifted_character = letnum - shift;

            if (shifted_character < 97){
                int new_int = shifted_character + 122;
                shifted_character = new_int - 97 + 1;
            }
            word[i] = (char)shifted_character;
            i++;
        }
    }

    printf("%s", word);
    return 0;
}
