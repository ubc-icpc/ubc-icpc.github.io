import java.util.*;
public class B 
{
	public static void main(String args[])
	{
		Scanner in = new Scanner (System.in);
		int n = in.nextInt();
		int k = in.nextInt();
		int[] arr = new int [n];
		int sum=0;
		for (int i=0;i<n;i++)
		{
			arr[i]=in.nextInt();
			arr[i]*=-1;
		}
		Arrays.sort(arr);
		for (int i= 0;i<k;i++)
		{
			sum+=(arr[i]*-1);
		}
		System.out.println(sum);
	}
}
