using namespace std;

int main(){
    char ED;
    cin >> ED;
    int shift;
    cin >> shift;
    char inp[100];
    char oup[100];
    cin >> inp;
    if(ED == 'D') shift *= -1;
    
    int i=0;
    for(; inp[i] != 0; ++i){
        oup[i] = (inp[i] - 'a' + shift + 26) % 26 + 'a';
    }
    oup[i] = 0;
    cout << oup << endl;
    return 0;
}
