#include<iostream>

using namespace std;

int main () {
  string d; cin >> d;
  int s; cin >> s;
  string str; cin >> str;
  int diff;
  if (d=="E") {
    diff = s;   
  } else {
    diff = 26-s;
  }
  
//   cout << diff << endl;
  string ans = "";
  for (int i=0; i<str.size(); i++) {
    int cha = (int) (str[i] - 'a');
    int ncha = (cha + diff) % 26;
    
    // cout << cha << " " << ncha << endl;
    ans += 'a' + ncha;
  }
  cout << ans << endl;
}
