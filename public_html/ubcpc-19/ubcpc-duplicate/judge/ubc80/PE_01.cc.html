#include<iostream>
#include<algorithm>

using namespace std;


int arr[3000];
int main () {
    int n,k;
    cin >> n >> k;
    
    
    for (int i=0; i<n; i++) {
        cin >> arr[i];
    }
    
    sort(arr, arr+n);
    
    
    int ans = 0;
    for (int i=0; i<k; i++) {
        
        // cout  <<  ans << " " << arr[n-1-i] << endl;
        ans += arr[n-1-i];
    }
    
    cout << ans << endl;
}
