import java.util.Arrays;
import java.util.Scanner;

public class excitingActs {

	public static void main(String[] args) {

		Scanner s = new Scanner(System.in);
		
		int numScenes = s.nextInt();
		int numActs = s.nextInt();
		
		int[] scenesEx = new int[numScenes];//excitement of scenes

		for(int i = 0; i < numScenes; i++) {//read in excitement
			scenesEx[i] = s.nextInt();
		}
		Arrays.sort(scenesEx);//sort excitement
		
		int sum = 0;
		
		for(int j = 0; j < numActs; j++ ) {//add up most exciting scenes 
											//(only as many as there are acts)
			sum += scenesEx[(numScenes-1)-j];
		}
		
		System.out.println(sum);
	}

}
