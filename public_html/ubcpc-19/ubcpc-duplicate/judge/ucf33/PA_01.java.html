import java.util.Scanner;

public class abnormalWords {

	public static void main(String[] args) {
		
		Scanner s = new Scanner(System.in);
		
		String E_or_D = s.next();
		int shift = s.nextInt();
		String str = s.next();
		char[] word = str.toCharArray();
		String result = "";
		
		if(E_or_D.equals("E")) {
			
			for(int i = 0; i < str.length(); i++) {
				int charIndex = word[i] - 'a';
				int newCharIndex = (charIndex + shift)%26;
				char newChar = (char)('a' + newCharIndex);
				result += newChar;
				
			}
			System.out.println(result);
		}
		
		else if(E_or_D.equals("D")) {
			
			for(int i = 0; i < str.length(); i++) {
				int charIndex = word[i] - 'a';
				int newCharIndex;
				if(((charIndex - shift)%26) < 1){
					newCharIndex = (26-(shift-charIndex))%26;
				}
				else {
					newCharIndex = (charIndex - shift)%26;
				}
				char newChar = (char)('a' + newCharIndex);
				result += newChar;
				
			}
			System.out.println(result);
		}
		
		else
			System.out.println("I can't believe you've done this.");
		

	}

}
