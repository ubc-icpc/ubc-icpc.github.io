import java.util.Scanner;

public class a {
	public static void main(String[] args) {
		Scanner stdin = new Scanner(System.in);
		
		char ch = stdin.next().charAt(0);
		stdin.nextLine();
		int s = stdin.nextInt();
		
		if(ch == 'D')
			s = (0-s);
		
		stdin.nextLine();
		
		String str = stdin.nextLine();
		
		String res = "";
		
		for(int i = 0; i < str.length(); i++) {
			int num = ((int)str.charAt(i) + s);
			
			if(num <= (int)'z' && num >= 'a')
				res = res + (char)num;
			
			else if(num >= 'z'){
				num -= 26;
				res = res + (char)num;
			}
			
			else {
				num += 26;
				res = res + (char)num;
			}
			
		}
		System.out.println(res);
		stdin.close();
	}
}
