mode = input()
shift = int(input())

if mode == 'D':
	shift *= -1

word = list(input())

int_list= []

new_list = []

for i, char in enumerate(word):
	int_list.append(ord(char))

for i, num in enumerate(int_list):
	int_list[i] = ((num - 97) + 26 + shift) % 26

for i, num in enumerate(int_list):
	new_list.append(chr(num + 97))

print(''.join(new_list))
