import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Scanner;

public class A {
    public static void main(String[] args) {
        Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        String str = sc.next();
        int n = sc.nextInt();
        String str1 = sc.next();

        String def = "abcdefghijklmnopqrstuvwxyz";
        String res = "";

        if (str.equals("E")) {
            for (int i = 0; i < str1.length(); i++ ) {

                int num = (Character.getNumericValue(str1.charAt(i)) - 10 + n) % 26;
                if (num < 0) {
                    num = 26 + num;
                }
                res = res + def.charAt((Character.getNumericValue(str1.charAt(i)) - 10 + n) % 26 );
            }
        } else {
            for (int i = 0; i < str1.length(); i++ ) {
                //System.out.println((Character.getNumericValue(str1.charAt(i)) - 10 + n) % 26);
                int num = (Character.getNumericValue(str1.charAt(i)) - 10 - n) % 26;
                if (num < 0) {
                    num = 26 + num;
                }
                res = res + def.charAt(num);
            }
        }
        System.out.println(res);
    }
}
