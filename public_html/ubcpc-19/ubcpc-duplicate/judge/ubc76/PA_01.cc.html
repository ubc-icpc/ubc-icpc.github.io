#include <bits/stdc++.h>
using namespace std;

string s;
char type;
int am;
string alpha = "abcdefghijklmnopqrstuvwxyz";
int conv[10000];
int main() {
	for (int i = 0; i < 26; i++) {
		conv['a' + i] = i;
	}
	cin >> type >> am >> s;
	if (type == 'E') {
		for (int i = 0; i < s.length(); i++) {
			s[i] = alpha[(conv[s[i]] + am) % 26];
		}
	} else {
		for (int i = 0; i < s.length(); i++) {
			s[i] = alpha[(conv[s[i]] - am + 26) % 26];
		}
	}
	cout << s << '\n';
}
