#include <bits/stdc++.h>
using namespace std;
long long res = 1;
int main() {
	long long n;
	cin >> n;
	for (long long i = 2; i <= 3000000; i++) {
		while (n % (i * i) == 0) {
			n /= (i * i);
		} 
		if (n % i == 0) {
			n /= i;
			res *= i;
		}
	}
	long long v = sqrt(n);
	if (v * v == n)
		cout << res << '\n';
	else
		cout << n * res << '\n';
}
