import java.io.*;
import java.util.*;

public class Solution{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int k = sc.nextInt();
        PriorityQueue<Integer> q = new PriorityQueue<Integer>(k);
        int[] arr = new int[n];
        for(int i=0; i<n; i++)
            arr[i] = sc.nextInt();
        for(int i: arr){
            q.offer(i);
 
            if(q.size()>k){
                q.poll();
            }
        }
        long sum=0;
        for(int i=0; i<k; i++)
            sum += q.poll();
        System.out.println(sum);
    }
}
