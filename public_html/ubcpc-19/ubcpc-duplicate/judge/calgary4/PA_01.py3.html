person = input()
shift = int(input())
word = input()
answer = ""

if person == "E":
    for i in range(len(word)):
        casted = ord(word[i])
        newletter = casted + shift
        if newletter > 122:
            newletter = 96 + (newletter - 122)
        answer += chr(newletter)
else:
    for i in range(len(word)):
        casted = ord(word[i])
        newletter = casted - shift
        if newletter < 97:
            newletter = 122 - (96 - newletter)
        answer += chr(newletter)

print(answer)
