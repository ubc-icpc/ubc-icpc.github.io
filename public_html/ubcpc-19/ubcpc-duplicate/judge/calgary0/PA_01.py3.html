EorD = input()
shift = int(input())
text = input()
alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
alphabet = alphabet.lower()
TextNum = []
text2 = ""

length = len(text)

for i in range(length):
    TextNum.append(alphabet.index(text[i]))

if EorD == "E":
    for i in range(length):
        TextNum2 = (TextNum[i] + shift) % 26
        text2 += alphabet[TextNum2]
    print(text2)

if EorD == "D":
    for i in range(length):
        TextNum2 = (TextNum[i] - shift) % 26
        text2 += alphabet[TextNum2]
    print(text2)
