import java.util.*;
public class acts
{
   
    public static void main(String args[]){
        Scanner m = new Scanner(System.in);
        
        //System.out.println("hi");
        
        int N = m.nextInt();
        int K = m.nextInt();
        
        int max = 0;
        int sum = 0;
        int[] scenes = new int[N];
        for (int i = 0; i < N; i++){
            int temp = m.nextInt();
            if (temp > max){
                max = temp;
            }
            sum += temp;
            scenes[i] = temp;
        }
        
        if (K == 1){
            System.out.println(max);
            return;
        } else if (K == N){
            System.out.println(sum);
            return;
        }
        
        Arrays.sort(scenes);
        sum = 0;
        
        for (int i = 1; i <= K; i++){
            sum += scenes[N-i];
        }
        System.out.println(sum);
    }
    
}
