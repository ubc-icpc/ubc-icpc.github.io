import java.util.*;
public class abnormal
{
    public static void main(String args[]){
        Scanner m = new Scanner(System.in);
        
        char[] alpha = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'};
        //System.out.println("hi");
        String type = m.nextLine();
        int shift = Integer.parseInt(m.nextLine());
        String phrase = m.nextLine();
        
        char[] letters = phrase.toCharArray();
        if (type.equals("D")){
            shift = 26 - shift;
        }
        for (char l : letters){
            System.out.print(shift(l, shift, alpha));
        }
    }
    
    public static char shift(char c, int shift, char[] alpha){
        int i = c - 'a';

        i += shift;
        i %= 26;
        
        return alpha[i];
    }
}
