C = input()
S = int(input())
message = input()

ans = ""

if C == 'E':
    for c in message:
        i = abs(ord(c) - ord('a'))
        i = (i + S) % 26
        ans += chr(i + ord('a'))

if C == 'D':
    for c in message:
        i = abs(ord(c) - ord('a'))
        i = (i - S) % 26
        ans += chr(i + ord('a'))


print(ans)
