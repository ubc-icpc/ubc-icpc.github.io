using namespace std;

int main() {
	string s; cin >> s;
	int n; cin >> n;
	string msg; cin >> msg;
	if (s[0] == 'E') {
		for (int i = 0; i < msg.size(); i++) {
			char l = (msg[i] - 'a' + n)%26 + 'a';
			cout << l;
		}
	} else {
		for (int i = 0; i < msg.size(); i++) {
			char l = (msg[i] - 'a' - n + 26)%26 + 'a';
			cout << l;
		}
	}
	cout << endl;
	return 0;
}
