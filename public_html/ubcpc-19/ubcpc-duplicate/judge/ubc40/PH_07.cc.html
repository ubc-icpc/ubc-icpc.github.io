using namespace std;

int main() {
	long long int c; cin >> c;
	while (c%4 == 0) c /= 4;
	while (c%9 == 0) c /= 9;
	for (long long int i = 1; i <= 2000 && i < c; i++) {
		if (c % i == 0) {
			long double d = c/i;
			long long int m = floor(sqrt(d));
			while (c % (m * m) == 0 && m != 1) c /= (m*m);
		}
	}
	for (long long int i = 5; i <= 67909405; i+=2) {
		while (c % (i*i) == 0) c /= (i*i);
	}
	cout << c << endl;
	return 0;
}
