import java.util.Arrays;
import java.util.Scanner;

/**
 * 
 */

/**
 * @author charl
 */
public class E {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		int n = in.nextInt(), k = in.nextInt();
		int[] x = new int[n];
		for (int i = 0; i < n; i++) {
			x[i] = in.nextInt();
		}
		Arrays.sort(x);
		int sum = 0;
		for (int i = 0; i < k; i++) {
		
			sum+=x[n-i-1];
		}	
		System.out.println(sum);

	}

}
