import sys
import math
import functools
from collections import deque
sin = sys.stdin

ed = sin.readline()[:-1]
# chr
shift = int(sin.readline())
word = sin.readline()[:-1]
ret = ""
if ed == 'E':
    for i in word:
        ret += chr(((ord(i) - 97 + shift) % 26) + 97)
else:
    for i in word:
        ret += chr(((ord(i) - 97 - shift) % 26) + 97)
print(ret)
