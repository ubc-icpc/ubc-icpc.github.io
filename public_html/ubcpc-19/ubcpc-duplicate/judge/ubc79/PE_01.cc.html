#include <bits/stdc++.h> 
using namespace std; 

int main() { 

  int n, k; cin >> n >> k;
  map<int,int> m;
  int res = 0;
  for (int i = 0; i < n; i++) {
    int a; cin >> a;
    m[a]++;
  }
  for (auto it = m.rbegin(); it != m.rend(); it++) {
    if (k <= 0) break;
    res += it->first * min(k, it->second);
    k -= min(k, it->second);
  } 
  cout << res << endl;

	return 0; 
}
