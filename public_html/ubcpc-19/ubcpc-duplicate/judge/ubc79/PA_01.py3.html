a = input()
b = int(input())
c = input()
d = ""

if a == 'E':
  for e in c:
    if ((ord(e)+b) > 122):
      extra = ord(e)+b-122
      d += chr(97+extra-1)
    else:
      d += chr((ord(e)+b))
else:
  for e in c:
    if ((ord(e)-b) < 97):
      extra = 97-(ord(e)-b)
      d += chr(122-extra+1)
    else:
      d += chr((ord(e)-b))

print(d)
