import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
	// write your code here
        Scanner sc = new Scanner(System.in);
        String instr = sc.nextLine();
        int s = sc.nextInt();
        sc.nextLine();
        String wrd = sc.nextLine();
        int len = wrd.length();
        StringBuilder ans0 = new StringBuilder();
        if(instr.equals("E")){
            for(int i =0;i<len;i++){
                int ascii = wrd.charAt(i)+s;
                if(wrd.charAt(i)+s>122)
                    ascii = ascii-122+96;
                ans0.append((char) ascii);
            }
        }else{
            for(int i =0;i<len;i++){
                int ascii = wrd.charAt(i)-s;
                if(wrd.charAt(i)-s<97)
                    ascii = 123-(97-ascii);
                ans0.append((char) ascii);
            }
        }
        System.out.println(ans0.toString());
    }
}
