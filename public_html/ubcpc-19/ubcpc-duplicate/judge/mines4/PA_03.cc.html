#include <iostream>
#include <string.h>

#define ALPHA_LEN 26

using namespace std;

string shift(string toShift, int count) {

    string toReturn;
    for(int i = 0; i < toShift.length(); i++){
        int ccount = count;
        int charInd = toShift.at(i) - 0x61;
        // cout << charInd << endl;
        if(charInd + ccount > 25)
            ccount -= 26;
        toReturn += (unsigned char) (0x61 + charInd + ccount);
    }

    return toReturn;
}

string unshift(string toShift, int count) {

    string toReturn;
    for(int i = 0; i < toShift.length(); i++){
        int ccount = count; 
        int charInd = toShift.at(i) - 0x61;
        // cout << charInd << endl;
        if(charInd - ccount < 0)
            ccount -= 26;
        toReturn += (unsigned char) (0x61 + charInd - ccount);
    }

    return toReturn;
}

int main() {

    char choice;
    int count;
    string text, output;

    cin >> choice >> count >> text;

    if(choice == 'E') {
        output = shift(text, count);
    } else if(choice == 'D') {
        output = unshift(text, count);
    }

    cout << output << endl;

}
