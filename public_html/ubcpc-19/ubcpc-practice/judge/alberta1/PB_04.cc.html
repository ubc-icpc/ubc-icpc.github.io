#include <bits/stdc++.h>
using namespace std;

string names[] = { "A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#" };
double x = pow(2, (1.0/12.0));

int main() {
    int p, q;
    cin >> p >> q;

    double dp = (double) p;
    double dq = (double) q;

    double A = 12 * (log2(dq) - log2(dp));
    double RA = round(A);
    double ratio = dp / dq;
    double C_D = 1200 * log2(ratio * pow(x, RA));
    // cout << RA << "|" << ratio << "\n";

    // cout << C_D << "cd\n";
    C_D = round(C_D);
    int C, D;
    if (C_D < 0) {
        D = 25;
        C = D + C_D;
    } else {
        D = -25;
        C = D + C_D;
    }


    string s = names[((((int) RA) % 12) + 12) % 12];
    cout << "A" << " " << C << '\n' << s << " " << D << '\n' ;
}
