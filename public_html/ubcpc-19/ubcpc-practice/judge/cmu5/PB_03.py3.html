import math

noteName = ['A', 'A#', 'B','C','C#','D','D#','E','F','F#','G','G#']

a, b = list(map(int,input().split()))

##rev = False
##if a > b:
##    a,b = b,a
##    rev = True

r = math.log2(b / a)

cents = round(1200 * r)

notes = cents // 100
mod = cents % 100

if mod > 50:
    mod -= 100
    notes += 1

modA = (-mod)//2
modB = mod//2

print(noteName[0],modA)
print(noteName[notes % 12],modB)
