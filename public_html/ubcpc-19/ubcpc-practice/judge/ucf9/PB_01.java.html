import java.util.Scanner;
public class harmony {
	
public static double log2(double num) {
	return Math.log(num) / Math.log(2);
}

static String[] notes = {"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};

public static void main(String[] args) {
Scanner scan = new Scanner(System.in);

int a = scan.nextInt();
int b = scan.nextInt();

double x = log2(a) - log2(b);

int centdiff = (int)Math.round((1.0/12)*(Math.round(12*x) - (12*x)) * 1200);

//System.out.println("Cent diff: " + centdiff);

System.out.println(notes[((int)Math.round(12*x) + 120000) % 12] + " " + (-centdiff / 2));


System.out.println("A " + (centdiff - (centdiff / 2)));

scan.close();
	}
}
