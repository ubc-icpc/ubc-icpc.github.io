#include <iostream>
#include <string>

using namespace std;

int main() {

    // declare 2 string variables to hold inputs
    string a;
    string b;

    //set variables to inputs
    cin >> a >> b;

    //set the first letter of a to lowercase
    a.at(0) = tolower(a.at(0));

    //create new string and concatenate the two inputs
    string ba = b + a;

    //output concatenated string
    cout << ba;

    return 0;
}
