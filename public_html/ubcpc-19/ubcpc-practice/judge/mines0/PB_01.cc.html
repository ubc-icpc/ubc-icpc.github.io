#include<bits/stdc++.h>
#include<iostream>
using namespace std;

int main() {
  vector<string> notes = { "A", "A#", "B","C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};
  int p, q;
  cin >> p >> q;
  double ratio = (double)q / p;
  while (ratio > 2) ratio /= 2;
  while (ratio < 1) ratio *= 2;
  double cents = 1200.0 * log(ratio) / log(2);
  int note = (cents+50) / 100;
  int res = abs(cents - note*100) + 0.5;
  bool inward = (note*100 > cents);
  int s1 = res/2 * (inward?1:-1);
  int s2 = (res-abs(s1)) * (inward?-1:1);
  cout << notes[0] << " " << s1 << endl;
  cout << notes[note] << " " << s2 << endl;
  return 0;
}
