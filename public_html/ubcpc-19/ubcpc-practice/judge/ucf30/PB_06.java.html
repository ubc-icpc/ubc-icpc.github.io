import java.util.Scanner;

public class ProbB
{ 
	static final String[] keys = {"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};
	public static void main(String[] args)
	{
	
		Scanner sc = new Scanner(System.in);

		int a  = sc.nextInt();
		int b = sc.nextInt();

		double aBase = Math.log(a) / Math.log(2);
		double bBase = Math.log(b) / Math.log(2);

		double diff = bBase - aBase;

		int shift = (int)(Math.round(diff * 1200.0));

		int baseDiff = shift / 100;
		int cents = shift % 100;

		
		if (cents > 50)
		{
			baseDiff++;
			cents -= 100;
		}
		else if (cents < -50)
		{
			baseDiff--;
			cents += 100;
		}

		int aN = 0;
		int bN = (baseDiff) % 12;

		while (bN < 0)
			bN += 12;

		int centsA = cents;
		int centsB = 0;

		// System.out.printf("cents now %d\n", cents);

		if (cents > 25)
		{
			centsA = cents - 25;
			centsB = -25;
		}
		else if (cents < -25)
		{
			centsA = cents + 25;
			centsB = 25;
		}

		System.out.println(keys[aN] + " " + centsB);
		System.out.println(keys[bN] + " " + centsA);
		// System.out.printf("%s %d\n", keys[aN], centsB);
		// System.out.printf("%s %d\n", keys[bN], centsA);
	}
}
