import java.util.*;
public class b {
	static double ratio, l;
	static String[] notes = {"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int p = sc.nextInt(), q = sc.nextInt();
		boolean inverse = false;
		if(q > p) {
			int temp = p;
			p = q;
			q = temp;
			inverse = true;
		}
		ratio = 1.0 * p / q;
		while(ratio >= 2) ratio /= 2;
		
		for(int i = 0; i < 12; ++i) {
			for(int oi = -25; oi <= 25; ++oi) {
				for(int j = 0; j < 12; ++j) {
					for(int oj = -25; oj <= 25; ++oj) {
						int numi = 100 * i + oi;
						int numj = 100 * j + oj;
						int num = (numj - numi);
						
						double curr = Math.pow(2, num / 1200.0);
						
						if(check(curr)) {
//							System.out.println(curr);
							if(!inverse) System.out.println(notes[j] + " " + oj + "\n" + notes[i] + " " + oi);
							else System.out.println(notes[i] + " " + oi + "\n" + notes[j] + " " + oj);
//							System.out.println("bueno");
							return;
						}
					}
				}
			}
		}
	}
	static boolean check(double in) {
		double a = in, b = ratio;
		if(b > a) {
			double temp = a;
			a = b;
			b = temp;
		}
		
		return a / b <= Math.pow(2, 1.0/2000);
	}
}
