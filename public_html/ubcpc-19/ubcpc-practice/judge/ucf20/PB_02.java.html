import java.util.Scanner;

public class b {

	public static void main(String[] args) {
		// 4/5 = 2^(x/12)
		Scanner scan = new Scanner(System.in);
		int p = scan.nextInt();
		int q = scan.nextInt();
		boolean wop = false;
		if (p > q) {
			q ^= p;
			p ^= q;
			q ^= p;
			wop = true;
		}
		double foo = Math.log(q/(double)p) / Math.log(2) * 12;
		double cents = -(Math.round(foo) - foo) * 100;
		String[] notes = new String[] {
				"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#" };
		// 2^(7/12)
		// 4/5 = .8
		// 7/12 = .58
		int c = (int)Math.round(cents);
		int c1 = 0;
		int c2 = c;
		if (c2 > 25) {
			c1 = c2 - 25;
			c2 = 25;
			c1 = -c1;
		}
		else if (c2 < -25) {
			c2 = -c2;
			c1 = c2 - 25;
			c2 = 25;
			c2 = -c2;
		}
		if (wop) {
			System.out.printf("%s %d\n", notes[
			      ((int)Math.round(foo) + 12) % 12
			      ], c2);
			System.out.printf("A %d\n", c1);
		}
		else {
			System.out.printf("A %d\n", c1);
			System.out.printf("%s %d\n", notes[
			      ((int)Math.round(foo) + 12) % 12
			      ], c2);
		}
	}

}
