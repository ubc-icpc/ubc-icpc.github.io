# Problem B. Harmony
from math import log

p, q = map(int,raw_input().split())
a, b = log(p, 2)*1200, log(q, 2)*1200
d = round(b-a)
note_d = int(d//100)  # multiple of 2^1/12
cent_d = int(d%100)  # total cents needed
if cent_d>50:
    # move together
    note_d+=1
    cent_d-=100
    c2 = max(cent_d, -25)
    c1 = max(-25-cent_d, 0)
else:
    # move apart
    c2 = min(cent_d, 25)
    c1 = min(25-cent_d, 0)
note_d %=12

# tcents =  (c2-c1) + 100*note_d
# diff = tcents - (b-a)
# print "cents:", tcents, "original:", (b-a), "diff:", diff
notes = ['A', 'A#', 'B', 'C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#']
n1, n2 = notes[0], notes[note_d]
print n1, c1
print n2, c2
