from math import log2

names = ["A","A#","B","C",
         "C#","D","D#","E",
         "F","F#","G","G#"]

p, q = input().split(" ")
cents = round(1200 * log2(int(q) / int(p)))

notes = round(cents / 100)
cents -= 100 * notes

start_note = cents % 12 # this value is arbitrary
end_note = (start_note + notes) % 12
end_cents = max(-25, min(25, cents))
start_cents = end_cents - cents

print("{} {}".format(names[start_note], start_cents))
print("{} {}".format(names[end_note], end_cents))
