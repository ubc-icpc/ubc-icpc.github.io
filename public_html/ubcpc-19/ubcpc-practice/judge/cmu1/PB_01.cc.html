#include "bits/stdc++.h"
using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<int, int> pii;

string notes[12] = {"A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"};

int main() {
    int p,q;
    cin >> p >> q;
    int sum = (int)round(1200*(log2(p)-log2(q)));
    int rem = ((sum%100)+100)%100;
    
    int a,b;
    int c,d;
    if(sum >= 0) {
        if(rem >= 50) {
            int cminusd = rem-100;
            c = max(cminusd, -25);
            d = c-cminusd;
            
            int aminusb = sum/100+1;
            a = aminusb;
            b = 0;
        }
        else {
            int cminusd = rem;
            c = min(25, cminusd);
            d = c-cminusd;
            
            int aminusb = sum/100;
            a = aminusb;
            b = 0;
        }
    }
    else {
        if(rem >= 50) {
            int cminusd = rem-100;
            c = max(cminusd, -25);
            d = c-cminusd;
            
            int aminusb = sum/100;
            a = 0;
            b = -aminusb;
        }
        else {
            int cminusd = rem;
            c = min(25, cminusd);
            d = c-cminusd;
            
            int aminusb = sum/100-1;
            a = 0;
            b = -aminusb;
        }
    }
    cout << notes[((a%12)+12)%12] << " " << c << endl;
    cout << notes[((b%12)+12)%12] << " " << d << endl;
}
