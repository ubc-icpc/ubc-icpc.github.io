#include <iostream>
#include <algorithm>
using namespace std;

int d[100000];

int main()
{
    int n, g, s, r;
    cin >> n >> g;
    int p = 0;
    for (int i = 0; i < n; i++)
    {
        cin >> s >> r;
        d[i] = s-r;
    }
    sort(d, d+n);
    for (int i = n-1; i >= 0; i--)
    {
        if (d[i] > 0)
            p += 3;
        else if (d[i] == 0)
        {
            if (g)
            {
                g--;
                p += 3;
            }
            else
                p++;
        }
        else if (-d[i] <= g)
        {
            g += d[i];
            p++;
            if (g)
            {
                g--;
                p += 2;
            }
        }
    }
    cout << p << endl;
    return 0;
}
