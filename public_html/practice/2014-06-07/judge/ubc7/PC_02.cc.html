#include <iostream>
typedef long long ll;
using namespace std;
int main() {
    ll a, b;
    cin>>a>>b;
        ll a_= 0, b_=0, c=0, d=1, e=1;
        
        while(b_ < b){
            a_ = a & d;
            b_ = b & d;
            
            if(b-a < 2*e && a_ <= b_) {
                c += max(b_+1, e) - max(a_, e);
            }else {
                c += min(2*e - a_, e); // to end
                c += max((ll)0, b_+1-e); // to st
                ll tr = 2*e - a_ + b_;
                //cerr << "tr " << tr << endl;
                c += (b-a-tr)/2;
            }

            d <<= 1;
            d |= 1;
            e <<= 1;
            //cerr << a_ << ' ' << b_ << ' ' << c << endl;
        }
        cout << c << endl;
    
    return 0;
}
