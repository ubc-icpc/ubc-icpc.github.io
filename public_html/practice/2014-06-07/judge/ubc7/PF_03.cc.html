#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
#define MAXN 100000
int d[MAXN];

int main() {
    int n,g,s,r,p,t,l;
    cin>>n>>g;
        memset(d, 0, sizeof(d));
        p=0, t=0;
        for(int i=0;i <n;++i) {
            cin>>s>>r;
            if(s > r) p++;
            else if(s==r) t++;
            else d[i-p-t]=r-s;
        }
        l = n-p-t;
        sort(d, d+l);
        p *= 3;
        p += t;
        if(g > 0) {
            p += min(g, t) * 2;
        }
        g -= t;
        for(int i=0;i<l && g >= d[i];++i) {
            p += 1, g -= d[i];
            if(g > 0) p += 2, g--;
        }
        cout << p << endl;
    
    return 0;
}
