#include <bits/stdtr1c++.h>

using namespace std;

int del[100005];
int main() {
	int n, g; cin >> n >> g;
	for (int i = 0; i < n; i++) {
		int a, b; cin >> a >> b;
		del[i] = b-a;
	}
	sort(del, del+n);
	int s = 0;
	for (int i = 0; i < n; i++) {
		if (del[i] >= 0) {
			if (g > del[i]) {
				g -= del[i]+1;
				del[i] = -1;
			} else {
				del[i] -= g;
				g -= min(g, del[i]+g);
			}
		}
		
		if (del[i] < 0) s += 3;
		if (del[i] == 0) s += 1;
	}
	cout << s << endl;
	return 0;
}
