#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

ll g(ll n) {
	if (n <= 1) return n;
	return 2LL*g(n-1) + (1LL<<(n-1));
}

ll cnt(ll n) {
	if (n <= 1) return n;
	
	int i = 0;
	while ((1LL<<i) <= n) i++;
	i--;
	return n-(1LL<<i)+1 + cnt(n-(1LL<<i)) + g(i); 
}

int main() {
	ll a, b; cin >> a >> b;
	cout << cnt(b) - cnt(a-1) << endl;
	return 0;
}
