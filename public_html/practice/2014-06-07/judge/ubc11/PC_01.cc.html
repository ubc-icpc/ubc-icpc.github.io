#include <iostream>
#include <climits>
using namespace std;

typedef long long ll;

ll ones_power(ll num){
	ll res = 0;
	ll c = 0;
	
	for (int i = 62; i >= 0; i--){
		c = 1LL << i;
		if (c & num){
			res = (c & num) + 2*ones_power(num/2);
			break;
		}
	}
	return res;
}

ll arbitrary_ones(ll num){
	ll res = 0;
	ll c = 0;
	for (int i = 62; i >= 0; i--){
		c = 1LL << i;
		if (c & num){
			res += (ones_power(c-1) + (num-c+1) + arbitrary_ones(num-c));
			break;
		}
	}
	return res;
}
