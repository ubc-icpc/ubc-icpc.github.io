#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <algorithm>

using namespace std;

bool cd(string s){
	for(int i=0;i<s.length()-1;i++){
		if(s[i]!='C') continue;
		if(s[i+1]=='D') return true;
	}
	return false;
}

int main() {
	int N, count = 0;
	string s;
	cin>>N;
	for(int i=0;i<N;i++){
		cin>>s;
		if(cd(s)) count++;
	}
	cout<<N-count;
    return 0;
}
