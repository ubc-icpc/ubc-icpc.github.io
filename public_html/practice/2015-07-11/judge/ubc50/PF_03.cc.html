#include<bits/stdtr1c++.h>

using namespace std;

typedef pair<int, int> ii;

struct cmp {
  bool operator()(const int &x, const int &y) {
    return x > y;
  }
};

int k, n, y, p;
vector<ii> moose;
set<int, cmp> participants;
bool flag = false;

int main() {
  cin >> k >> n >> y >> p;
  moose.push_back(make_pair(y,p));
  for (int i=0; i<(n+k-2); i++) {
    int year, strength; cin >> year >> strength;
    moose.push_back(make_pair(year, strength));
  }
  sort(moose.begin(), moose.end());
  for (int i=0; i<k-1; i++)
    participants.insert(moose[i].second);
  for (int i=k-1; i<(n+k-1); i++) {
    participants.insert(moose[i].second);
    if (*participants.begin() == p) {
      cout << moose[i].first << '\n';
      flag = true;
      break;
    }
    participants.erase(participants.begin());
  }
  if (!flag)
    cout << "unknown" << '\n';
}
