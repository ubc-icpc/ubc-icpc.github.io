#include<bits/stdtr1c++.h>

using namespace std;

int n;

int main() {
  cin >> n;
  int ans = 0;
  bool flag = false;
  for (int i=0; i<n; i++) {
    string s; cin >> s;
    for (int j=0; j<s.size()-1; j++) {
      if (s[j]=='C' && s[j+1]=='D') {
	flag = true;
	break;
      }
    }
    if (flag) {
      flag = !flag;
      ans++;
    }
  }
  cout << n-ans << endl;
}
