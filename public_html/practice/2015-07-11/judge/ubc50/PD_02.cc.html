#include<bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

int f, s, g, u, d;
int arr[1000001];

int main() {
  memset(arr, -1, sizeof arr);
  cin >> f >> s >> g >> u >> d;
  arr[s] = 0;
  queue<int> q;
  q.push(s);
  while(!q.empty()) {
    int x = q.front();
    q.pop();
    if (x==g) break;
    if (x+u <= f && arr[x+u] == -1) {
      arr[x+u] = arr[x]+1;
      q.push(x+u);
    }
        if (x-d >= 1 && arr[x-d] == -1) {
      arr[x-d] = arr[x]+1;
      q.push(x-d);
    }
  }
  if (arr[g] >= 0)
    cout << arr[g] << '\n';
  else
    cout << "use the stairs" << '\n';
}
