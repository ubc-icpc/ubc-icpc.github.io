#include <bits/stdtr1c++.h>
using namespace std;

struct moose {
  int y, p, k;
  bool operator< (const moose& b) const { return p < b.p; }
} M[200005];
bool comp (const moose& a, const moose& b) { 
  if (a.y != b.y) return a.y < b.y;
  return a.p < b.p;
}
int K, N, RES;
priority_queue<moose> q;

int main () {
  cin >> K >> N;
  cin >> M[0].y >> M[0].p; M[0].k = true;
  for (int i = 1; i <= K + N - 2; ++i) { cin >> M[i].y >> M[i].p; M[i].k = false; }
  sort(M, M + K + N - 1, comp);
  for (int i = 0; i < K - 1; ++i) q.push(M[i]);
  for (int i = K - 1; i <= K + N - 2; ++i) {
    q.push(M[i]);
    if (q.top().k == true) { RES = M[i].y; break; }
    q.pop();
  }
  if (RES) cout << RES << endl;
  else cout << "unknown" << endl;
  return 0;
}
