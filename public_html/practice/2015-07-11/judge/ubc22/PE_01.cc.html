#include<bits/stdtr1c++.h>
using namespace std;

string pattern, S;
int N;

int main () {
  ios::sync_with_stdio(0);
  cin.tie(0);
  cin >> pattern;
  int i = 0;
  while ((i = pattern.find('.', i)) != -1) {
    pattern.insert(i, "\\");
    i += 2;
  }
  i = 0;
  while ((i = pattern.find('*', i)) != -1) {
    pattern.insert(i, ".");
    i += 2;
  }
  regex e(pattern);
  cin >> N;
  while (N--) {
    cin >> S;
    if (regex_match(S, e)) cout << S << endl;
  }
  return 0;
}
