#include <bits/stdtr1c++.h>
using namespace std;

int F, S, G, U, D, M[1000005];

int main () {
  cin >> F >> S >> G >> U >> D;
  queue<pair<int, int> > q; q.push(make_pair(S, 0));
  while (!q.empty()) {
    pair<int, int> t = q.front(); q.pop();
    if (M[t.first]) continue;
    M[t.first] = t.second + 1;
    if (t.first + U <= F)
      q.push(make_pair(t.first + U, t.second + 1));
    if (t.first - D >= 0)
      q.push(make_pair(t.first - D, t.second + 1));
  }
  if (M[G]) cout << M[G] - 1 << endl;
  else cout << "use the stairs" << endl;
  return 0;
}
