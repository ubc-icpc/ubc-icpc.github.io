testCases = int(raw_input())

wins = 0
for T in range(testCases):
    currCase= raw_input()

    winThisBattle = True
    prevChar = None
    for char in currCase:
        if prevChar == 'C' and char == 'D':
            winThisBattle = False
            break
        prevChar = char
        
    if winThisBattle:
        wins += 1

print(wins)
