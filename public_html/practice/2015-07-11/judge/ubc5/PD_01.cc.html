#include <iostream>
#include <queue>
using namespace std;

int v[1000001];

int main() {
  int f, s, g, u, d;
  cin >> f >> s >> g >> u >> d;
  queue<int> q;
  q.push(s);
  v[s] = 1;
  while (q.size()) {
    int c = q.front();
    q.pop();
    if (c == g)
      break;
    if (c+u <= f && !v[c+u]) {
      v[c+u] = v[c]+1;
      q.push(c+u);
    }
    if (c-d && !v[c-d]) {
      v[c-d] = v[c]+1;
      q.push(c-d);
    }
  }
  if (v[g])
    cout << v[g]-1 << "\n";
  else
    cout << "use the stairs\n";
  return 0;
}
