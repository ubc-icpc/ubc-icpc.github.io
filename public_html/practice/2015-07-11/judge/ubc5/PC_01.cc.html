#include <iostream>
using namespace std;

int main() {
  int n; cin >> n;
  int w = 0;
  for (int i = 0; i < n; i++) {
    string s; cin >> s;
    bool g = true;
    for (int j = 0; j < s.size()-1; j++) {
      if (s[j] == 'C' && s[j+1] == 'D') {
        g = false;
        break;
      }
    }
    if (g)
      w++;
  }
  cout << w << "\n";
  return 0;
}
