using namespace std;

typedef long long ll;

int ky, kp;
int main() {
    int n, k; cin >> k >> n;
    vector<pair<int, int>> m;
    cin >> ky >> kp;
    
    m.push_back({ky, kp});
    for (int i = 0; i < n+k-2; i++) {
        int x, y; cin >> x >> y;
        m.push_back({x,y});
    }
    sort(m.begin(), m.end());
    
    set<int, greater<int>> s;
    for (int i = 0; i < k-1; i++) {
        s.insert(m[i].second);
    }
    
    for (int i = k-1; i < n+k-1; i++) {
        s.insert(m[i].second);
        if (*s.begin() == kp) {
            cout << m[i].first << endl;
            return 0;
        }
        s.erase(s.begin());
    }
    cout << "unknown" << endl;
	return 0;
}
