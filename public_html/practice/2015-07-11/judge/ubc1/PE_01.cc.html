#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    string pat; cin >> pat;
    string rpat = "";
    for (char c : pat) {
        if (c == '.') rpat += "\\.";
        else if (c == '*') rpat += ".*";
        else rpat.push_back(c);
    }
    
    regex e(rpat);
    int n; cin >> n;
    for (int i = 0; i < n; i++) {
        string s; cin >> s;
        if (regex_match(s, e)) cout << s << endl;
    }
	return 0;
}
