#include <bits/stdtr1c++.h>

using namespace std;

int dist[1000005];
int main() {
    int f, s, g, u, d;
    cin >> f >> s >> g >> u >> d;
    
    memset(dist, -1, sizeof dist);
    queue<int> q({s});
    dist[s] = 0;
    while (!q.empty()) {
        int c = q.front(); q.pop();
        if (c+u <= f && dist[c+u] == -1) {
            dist[c+u] = dist[c] + 1;
            q.push(c+u);
        }
        if (c-d >= 1 && dist[c-d] == -1) {
            dist[c-d] = dist[c] + 1;
            q.push(c-d);
        }
    }
    
    if (dist[g] != -1) cout << dist[g] << endl;
    else cout << "use the stairs" << endl;
	return 0;
}
