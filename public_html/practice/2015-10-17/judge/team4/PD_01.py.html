import math
def nCr(n,r):
    f = math.factorial
    return f(n)//f(r)//f(n-r)

T = int(raw_input())
for i in xrange(T):
    C,B = [int(x) for x in raw_input().split()]
    member = [int(x) for x in raw_input().split()]
    # first caculate the denominator
    total = sum(member)
    deno = nCr(total,B)

    # then nominator
    dp = [0] * (B + 1)
    dp[0] = 1
    for i in member:
        F = [0] * (B + 1)
        for j in range(B, 0, -1):
            for k in range(1, i + 1):
                if j >= k:
                    F[j] += dp[j - k] * nCr(i, k)
        dp = F[:]

    print format(float(dp[B])/deno,'.6f')
