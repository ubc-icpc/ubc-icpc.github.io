#include <bits/stdc++.h>
using namespace std;
#define endl '\n'

int check(string inp, int pos) {
	string p = "path";
	int curr = 0;
	int start = -1,end = -1;
	for (int i=pos; i<inp.length(); i++) {
		if (p[curr] == inp[i]) {
			if (start == -1) start = i;
			if (curr == 3) {
				end = i; break;
			}
			curr++;
		}
	}
	if (end != -1) return end-start;
	else return 505;
}

int main() {
	int n; cin >> n;
	while(n--) {
		string p = "path";
		string inp; cin >> inp;
		int curr = 0;
		int ans = 505;
		for (int i=0; i<inp.length(); i++) {
			if (inp[i] == 'p') {
				ans = min(ans,check(inp,i));
			}
		}
		if (ans != 505) {
			cout << ans+1 << endl;
		} else {
			cout << -1 << endl;
		}
		
	}
	
	return 0;
}
