#include <bits/stdc++.h>
using namespace std;

const double PI = 3.1415926535;

int main() {
	int n; cin >> n;
	while(n--) {
		double r,d; cin >> r >> d;
		double h = abs(r-d);
		double vcut = PI*h*h/3.0*(3.0*r-h);
		double v = 4.0/3.0*PI*r*r*r;
		cout << fixed << setprecision(2) << v-vcut << endl;
	}
	return 0;
}
