#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
	int t; cin >> t;
	while(t--) {
		int n; int k;
		cin >> n >> k;
		ll inp[n];
		for (int i=0; i<n; i++) {
			cin >> inp[i];
			inp[i] *= 2;
		}
		ll maxr = 1e9;
		ll minr = 0;
		while(maxr-minr>1) {
			ll next = inp[0] + minr+maxr;
			ll nump = 1;
			for (int i=0; i<n; i++) {
				if (inp[i] <= next) continue;
				else {
					next = inp[i] + minr+maxr;
					nump++;
				}
			}
			if (nump > k) {
				minr = (minr+maxr) / 2+1;
			} else {
				maxr = (minr+maxr) / 2;
			}
		}
		ll next = inp[0] + minr+minr;
		ll nump = 1;
		for (int i=0; i<n; i++) {
			if (inp[i] <= next) continue;
			else {
				next = inp[i] + minr+minr;
				nump++;
			}
		}
		if (nump > k ) cout << maxr/2 << '.' << (maxr%2 ? '5' : '0') << endl;
		else cout << minr/2 << '.' << (minr%2 ? '5' : '0') << endl;
		
	}
	return 0;
}
