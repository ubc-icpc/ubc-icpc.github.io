#include <iostream>
#include <map>
#include <cstring>

using namespace std;

int nxt[505];
map<char, char> nchar;
int main() {
	nchar['p'] = 'a';
	nchar['a'] = 't';
	nchar['t'] = 'h';
	nchar['h'] = 'z';
	
	int t; cin >> t;
	while (t--) {
		string s; cin >> s;
		int n = s.size();
		memset(nxt, -1, sizeof nxt);
		for (int i = 0; i < n; i++) {
			for (int j = i+1; j < n; j++) {
				if (s[j] == nchar[s[i]]) {
					nxt[i] = j;
					break;
				}
			}
		}
		
		int ans = -1;
		for (int i = 0; i < n; i++) {
			if (s[i] == 'p') {
				int r = i;
				for (int k = 0; k < 3; k++) {
					r = nxt[r];
					if (r == -1) break;
				}
				if (r != -1) {
					if (ans == -1) ans = r-i+1;
					else ans = min(ans, r-i+1);
				}
			}
		}
		cout << ans << endl;
	}
	return 0;
}
