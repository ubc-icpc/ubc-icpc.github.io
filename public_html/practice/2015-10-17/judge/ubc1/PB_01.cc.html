#include <iostream>
#include <cmath>
#include <iomanip>

using namespace std;

const long double pi = acos(-1.0L);
int main() {
	int t; cin >> t;
	while (t--) {
		long double r, d;
		cin >> r >> d;
		cout << fixed << setprecision(2) << 2*pi/3 *r*r*r  + pi * d * r * r - pi * d*d*d/3 << endl;
	}
	return 0;
}
