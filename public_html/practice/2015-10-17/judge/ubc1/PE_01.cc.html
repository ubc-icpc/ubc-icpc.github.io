#include <iostream>

using namespace std;

typedef long long ll;
const int N = 1e5 + 10;
int n, k;
ll x[N];

bool f(ll len) {
	long long ans = 0, i = 0, last = x[0];
	while (i < n) {
		while (i < n && x[i] <= last+len) i++;
		ans++;
		last = x[i];
	}
	return ans <= k;
}

int main() {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	
	while (t--) {
		cin >> n >> k;
		for (int i = 0; i < n; i++) {
			cin >> x[i];
			x[i] *= 2;
		}
		
		long long l = 0, r = 10000000000;
		while (r-l > 1) {
			int m = (l+r)/2;
			if (f(2*m)) {
				r = m;
			} else {
				l = m+1;
			}
		}
		
		if (f(2*l)) cout << l/2 << '.' << (l%2 ? '5' : '0') << endl;
		else cout << r/2 << '.' << (r%2 ? '5' : '0') << endl;
	}
	return 0;
}
