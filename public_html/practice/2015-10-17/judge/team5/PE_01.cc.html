#include<iostream>
#include<iomanip>

using namespace std;
int n,k;
int p[100000+5];

bool ok(long long m) {
    int i=0, at=p[0],ku=0;
    while(i<n) {
        while(i<n && p[i]<at+m) i++;
        at=p[i];
        ku++;
    }
    return ku<=k;
}
int main(void) {
    int T;cin>>T;
    while(T--) {
        cin>>n>>k;
        for(int i=0; i<n; i++) {
            cin>>p[i];
            p[i]*=2;
        }
        
        
        long long l=0,r=10e9,m=r/2;
        while(r-l>1) {
            if(ok(m))
                r=m;
            else
                l=m;
            m=(l+r)/2;
        }
        cout << fixed << setprecision(1) << (double)m/4 << endl;
    }
}
