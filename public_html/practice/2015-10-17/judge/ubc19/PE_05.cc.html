#include <iostream>

using namespace std;
typedef long long ll;

ll pn[100000]; int n;


double covers(ll rad, int k) {
  int pos = 0;
  while (pos < n) {
    ll start = pn[pos];
    while (pn[pos+1] <= start + rad && pos < n) {
      pos++;
    }
    pos++;
    k--;
  }

  return k>=0;
}

      cin >> pn[j];
    }
    
    ll low = 0; ll high = pn[n-1] - pn[0];
    while (low < high) {
      if (covers(low, k)) {
        break;
      }
     // cout << low << " " << high << endl;
      ll mid = (low + high)/2;
      if (covers(mid, k)) {
