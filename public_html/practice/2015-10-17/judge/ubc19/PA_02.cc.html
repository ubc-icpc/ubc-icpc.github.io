#include <iostream>
#include <vector>
#include <math.h>

using namespace std;
typedef long long ll;

int main() {
  int t; cin >> t;
  
  for (int i=0; i<t; i++) {
    ll x; cin >> x;
    ll balls = 0; ll rows = 1;
    
    while (balls < x) {
      balls += rows*(rows+1)*0.5;
      rows++;
    }
    rows--;
    if (balls == x) {
      cout << rows  << endl;
      continue;
    }
    
    ll bls = 0;
    int rem = 0;
    
    while (bls < x) {
      bls += rows*(rows+1)*0.5;
      rows--;
      rem++;
    }
    
    cout << rem << endl;
    
    
  }
}
