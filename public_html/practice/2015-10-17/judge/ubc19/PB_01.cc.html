#include <iostream>
#include <vector>
#include <math.h>

using namespace std;

int main() {
  int t; cin >> t;
  
  for (int i=0; i<t; i++) {
    double r, d; cin >> r >> d;
    d = r - d;
    double sphere = 4.0*M_PI*r*r*r/3.0;
    double cap = M_PI*d*d*(3.0*r-d)/3.0;
    
    printf("%.2f\n", sphere - cap);
  }
}
