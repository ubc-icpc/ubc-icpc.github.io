#include <iostream>
#include <vector>

using namespace std;

int main() {
  int t; cin >> t;
  
  for (int i=0; i<t; i++) {
    string s; cin >> s;
    vector<int> p_ind;
    vector<int> a_ind;
    vector<int> t_ind;
    vector<int> h_ind;
    
    for (int j=0; j<s.size(); j++) {
      if (s[j] == 'p') p_ind.push_back(j);
      if (s[j] == 'a') a_ind.push_back(j);
      if (s[j] == 't') t_ind.push_back(j);
      if (s[j] == 'h') h_ind.push_back(j);
    }
    
    if (p_ind.size()*a_ind.size()*t_ind.size()*h_ind.size() == 0) {
      cout << -1 << endl;
      continue;
    }
    
    int mini = 501;
    
    for (int j=0; j<p_ind.size(); j++) {
      int pos = 0;
      while (a_ind[pos] < p_ind[j] && pos<a_ind.size()) {
        pos++;
      }
      if (pos == a_ind.size()) {
        break;
      }
      
      int a_pos = a_ind[pos];
      pos = 0;
      while (t_ind[pos] < a_pos && pos<t_ind.size()) {
        pos++;
      }
      if (pos == t_ind.size()) {
        break;
      }
      
      int t_pos = t_ind[pos];
      pos = 0;
      while (h_ind[pos] < t_pos && pos<h_ind.size()) {
        pos++;
      }
      if (pos == h_ind.size()) {
        break;
      }
      
      int len =  h_ind[pos] - p_ind[j] + 1;
      if (len < mini) mini = len;
    }
    
    if (mini != 501) cout << mini << endl;
    else cout << -1 << endl;
    
  }
}
