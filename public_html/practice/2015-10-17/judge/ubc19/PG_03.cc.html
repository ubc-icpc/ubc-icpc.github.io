#include <iostream>

using namespace std;

int main() {
  int t; cin >> t;
  auto sizes = new int[2000][2000]();
  
  for (int i=0; i<t; i++) {
    int n; cin >> n;
    int maxsize = 0;
    
    for (int j=0; j<n; j++) {
      string s; cin >> s;
      for (int k=0; k<n; k++) {
        if (s[k] == 'F') {
          sizes[j][k] = 0;
        } else {
          sizes[j][k] = 1;
        }
        
        if (j!=0 && k!=0 && sizes[j][k]!=0) {
          sizes[j][k] = min(sizes[j-1][k], min(sizes[j][k-1], sizes[j-1][k-1])) + 1;
        }
        if (sizes[j][k] > maxsize) maxsize = sizes[j][k];
      }
    }
    cout << maxsize << endl;
  }
}
