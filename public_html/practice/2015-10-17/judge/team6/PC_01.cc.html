#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int temp_min = 600;
		string s;
		cin >> s;
		for (int i = 0; i < s.size(); i++)
		{
			int index = 999;
			if(s[i] == 'p')
			{
				for (int j = i + 1; j < s.size(); j++)
				{
					if (s[j] == 'a')
					{
						index = j;
						break;
					}
					index = 999;
				}
				if (index == 999)
					break;
				if (index == s.size() - 1)
					break;
				for (int j = index + 1; j < s.size(); j++)
				{
					if (s[j] == 't')
					{
						index = j;
						break;
					}
					index = 999;
				}
				if (index == 999)
					break;
				if (index == s.size() - 1)
					break;
				for (int j = index + 1; j < s.size(); j++)
				{
					if (s[j] == 'h')
					{
						index = j;
						break;
					}
					index = 999;
				}
				if (index == 999)
					break;
				if (temp_min > index - i + 1)
				{
					temp_min = index - i + 1;
				}
			}
		}
		if (temp_min == 600)
		{
			cout << -1 << endl;
		}
		else
		{
			cout << temp_min << endl;
		}
	}
	return 0;
}
