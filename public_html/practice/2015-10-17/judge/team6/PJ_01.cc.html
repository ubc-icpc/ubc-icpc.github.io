#include <iostream>

using namespace std;

int main()
{
	int x, y;
	char c;
	int t;
	cin >> t;
	while (t--)
	{
		cin >> x >> y >> c;
		if (c == '>')
		{
			if (x > y)
			{
				cout << "Ready" << endl;
			}
			else
			{
				cout << "Reboot" << endl;
			}
		}
		else
		{
			if (x < y)
			{
				cout << "Ready" << endl;
			}
			else
			{
				cout << "Reboot" << endl;
			}
		}
	}
	return 0;
}
