#include <iostream>
#include <vector>
#include <utility>
#include <cstring>
#include <algorithm>

using namespace std;

typedef pair<int, int> pii;
char grid[2005][2005];
int sgrid[2005][2005];
vector<pii> pairs;
int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int size;
		cin >> size;
		int temp_max = 0;
		memset(sgrid, 0, sizeof(sgrid));
		for (int i = 0; i < size; i++)
		{
			for (int j = 0; j < size; j++)
			{
				char c; 
				cin >> c;
				grid[i][j] = c;
			}
		}
		for (int i = 0; i < size; i++)
		{
			if (grid[0][i] == 'F')
			{
				sgrid[0][i] = 0;
			}
			else
			{
				sgrid[0][i] = 1;
			}
			if (grid[i][0] == 'F')
			{
				sgrid[i][0] = 0;
			}
			else
			{
				sgrid[i][0] = 1;
			}
		}
		for (int i = 1; i < size; i++)
		{
			for (int j = 1; j < size; j++)
			{
				if (grid[i][j] == 'F')
				{
					sgrid[i][j] = 0;
				}
				else
				{
					sgrid[i][j] = min(sgrid[i - 1][j], min(sgrid[i - 1][j - 1], sgrid[i][j - 1])) + 1;
					if (sgrid[i][j] > temp_max)
						temp_max = sgrid[i][j];
				}
			}
		}
		cout << temp_max << endl;
	}
}
