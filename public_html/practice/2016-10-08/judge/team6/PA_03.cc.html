#include <iostream>
#include <cmath>

using namespace std;

int main() {
  int T, n, m;
  bool prime;
  cin >> T;
  while(T--) {
    cin >> m >> n; 

    prime = true;
    for(int p = -200; p <= 200; p++) {
      for(int q = -200; q <= 200; q++) {
        if(p == 0 && q == 0) {
          continue;
        }

        if((m*p + n*q) % (p*p + q*q) == 0 && (n*p - m*q) % (p*p + q*q) == 0) {
          if((abs(m) == abs(p) && abs(n) == abs(q)) || (abs(n) == abs(p) && abs(m) == abs(q)) || (abs(p) == 1 && q == 0) || (abs(q) == 1 && p == 0)) {
            continue;
          }

          prime = false; 
        }
      }
    }

    if(prime) {
      cout << "P\n";
    } else {
      cout << "C\n";
    }
  }
}
