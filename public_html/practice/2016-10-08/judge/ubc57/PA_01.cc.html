#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef complex<int> pt;

bool isInt(long double x){
    int f=x;
    if (abs(f-x)<1e-6) return true;
    f++;
    if (abs(f-x)<1e-6) return true;
    return false;
}
int main(){
    int t; cin>>t;
    while (t--){
        int a,b; cin>>a>>b; 
        pt c(a,b);
        int size=a*a+b*b;
        int f=0;
        for (int x=0;x*x<=size;x++){
            for (int y=0;y*y+x*x<=size;y++) if (((x*x+y*y)>0)&&((size%(x*x+y*y))==0)){
                //cout<<x<<' '<<y<<endl;
                pt res=c/pt(x,y);
                
                if ((isInt(real(res)))&&(isInt(imag(res)))) f+=2;
            }
        }
        if (f==8) cout<<"P"<<endl;
        else cout<<"C"<<endl;
    }
}
