#include <iostream>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int cases, m, n;
    cin >> cases;
    for (int test = 0; test < cases; ++test) {
        cin >> m >> n;
        if (m * m + n * n <= 1) {
            cout << "C\n";
            goto end;
        }
        for (int i = -200; i < 200; ++i)
            for (int j = -200; j < 200; ++j) {
                if ((i == 0 && j == 0) ||
                    (i == 1 && j == 0) || (i == 0 && j == 1) ||
                    (i == -1 && j == 0) || (i == 0 && j == -1) ||
                    (i == m && j == n) || (i == -n && j == m) ||
                    (i == -m && j == -n) || (i == n && j == -m))
                    continue;
                int div = i * i + j * j;
                if ((i * m + j * n) % div == 0 &&
                    (i * n - j * m) % div == 0) {
                    cout << "C\n";
                    goto end;
                }
            }
        cout << "P\n";
        end:;
    }
    return 0;
}
