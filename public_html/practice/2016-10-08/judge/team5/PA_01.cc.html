#include <bits/stdtr1c++.h>

using namespace std;

int main() {
  int datasets; cin >> datasets;
  while (datasets--) {
    int m, n; cin >> m >> n;
    if (m*m + n*n <= 1) {
      cout << "C" << endl;
      continue;
    }
    int cnt = 0;
    for (int p = 0; p <= 150; p++) {
      for (int q = 0; q <= 150; q++) {
        if (p==0 && q ==0) continue;
        if (p*p + q*q >= 20000) continue;
        if (cnt > 8) break;
        if ((p*m+q*n) % (p*p+q*q) == 0
            && (p*n-q*m) % (p*p+q*q) == 0) {
          if (p != 0 && q != 0) {
            cnt += 4;
          } else {
            cnt += 2;
          }
        }
      }
      if (cnt > 8) break;
    }
    if (cnt ==8)
      cout << "P" << endl;
    else
      cout << "C" << endl;
  }
}
