#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int ncase; cin >> ncase;
    while (ncase--) {
        int p, q; cin >> p >> q;
        int cnt = 0;
        for (int m = -200; m <= 200; m++) {
            for (int n = -200; n <= 200; n++) {
                int a = m*p+n*q;
                int b = m*q-n*p;
                int z = m*m+n*n;
                if (z == 0) continue;
                if (a%z == 0 && b%z == 0) {
                    cnt++;
                }
            }
        }
        cout << (cnt == 8 ? "P" : "C") << endl;
    }
}
