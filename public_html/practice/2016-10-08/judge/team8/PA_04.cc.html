#include <bits/stdc++.h>

using namespace std;

vector<bool> prime(20000, true);

void pre() {
	for (int i = 2; i <= 20000; i++) {
		if (prime[i]) {
			for (int j = i*i; j <= 20000; j += i) {
				prime[j] = false;
			}
		}
	}
	prime[0] = false;
	prime[1] = false;
}


int main() {
	int p; cin >> p;
	pre();
/*	for (int i = 0; i < 20000; i++) {
		if(prime[i]) cout << i << endl;
	}*/
	for (int i = 0; i < p; i++) {
		int m, n; cin >> m >> n;
		if (m == 0) {
			if (prime[abs(n)] && abs(n) % 4 == 3) {
				cout << "P" << endl;
			} else cout << "C" << endl;
		} else if (n == 0) {
			if (prime[abs(m)] && abs(m) % 4 == 3) {
				cout << "P" << endl;				
			} else cout << "C" << endl;
		} else {
			if (prime[m*m + n*n]) {
				cout << "P" << endl;				
			} else cout << "C" << endl;
		}
	}
	return 0;
}
