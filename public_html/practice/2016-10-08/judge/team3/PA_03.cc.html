#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t; cin >> t;
	while (t--) 
	{
		int a, b; cin >> a >> b;
		int lenab = a * a + b * b;
		int max = (int) sqrt(lenab) + 1;
		int d = 0;

		if (lenab <= 1)
		{
			cout << "C" << endl;
			continue;
		}

		for (int m = -max; m < max; m++)
		{
			int nmax = (int) sqrt(lenab - m * m) + 1;
			for (int n = -nmax; n < nmax; n++)
			{
				int lenmn = m * m + n * n;
				if (lenmn <= 1) continue;

				if (lenab % lenmn != 0) continue;

				// remainder = (a + ib) / (m + in)

				if ((a * m + b * n) % lenmn == 0 && (b * m - a * n) % lenmn == 0)
				{
					d++;
					if (d > 4) goto done;
				}
			}
		}
done:
		cout << (d > 4 ? 'C' : 'P') << endl;
		// cout << flush << endl;
	}
}
