#include <cmath>
#include <cstdio>
#include <vector>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
//typedef long double ld;
#define double long double

#define epsilon 0.000001

//------------------
int d;
double f[1000];
//------------------

double calc_val(int x, int forbid1, int forbid2){

  double ret = 0.0;
  for(int i = 0;i <= d+2;i++){

      if(i == forbid1 || i == forbid2) continue;

      double p = f[i];
      for(int j = 0;j <= d+2;j++){

        if(j == forbid1 || j == forbid2 || j == i) continue;

        p *= (double)(x - j);
        p /= (double)(i - j);
      }

      ret += p;
  }

  return ret;
}


bool check(int k){

  for(int i = 0;i <= d+2;i++){

    if(i == k) continue;

    double val = calc_val(k, i, k);

    if(abs(val - f[k]) < 0.5) return false;
  }

  return true;
}

int main(){

  while( scanf("%d", &d) && d ){

    for(int i = 0;i <= d+2;i++) cin >> f[i];

    //cout << answer() << endl;
    for(int i = 0;i <= d+2;i++)
      if(check(i)){
        cout << i << endl;
        break;
      }
  }


  return 0;
}
