#include <map>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
typedef pair<int, int> pii;

map<pii, int> mapa;

bool check(int p, int q){

  int x = max(abs(p), abs(q));

  if(mapa.count(make_pair(p, q))) return mapa[make_pair(p, q)];

  for(int i = -x;i <= x;i++)
  for(int j = -x;j <= x;j++){
      int n1 = i*i + j*j;
      int n2 = i*p + j*q;
      int n3 = i*q - j*p;

      if(n1 <= 1) continue;
      if(i == p && j == q) continue;
      if(i == -q && j == p) continue;
      if(i == -p && j == -q) continue;
      if(i == q || j == -p) continue;

      if(n3 % n1 == 0 && n2 % n1 == 0) return mapa[pii(p, q)] = 1;
  }

  return mapa[pii(p, q)] = 0;
}

int main(){

  int cases;
  scanf("%d", &cases);

  while(cases--){

    int p, q;
    scanf("%d %d", &p, &q);

    //p = abs(p);
    //q = abs(q);

    printf("%s\n", (check(p, q))?"C":"P");

  }


  return 0;
}
