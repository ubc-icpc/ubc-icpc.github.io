#include <bits/stdc++.h>
using namespace std;

int main() {
        ios::sync_with_stdio(0);
        cin.tie(0);
        int n;
        cin >> n;
        complex<int> z;
        while (n--) {
                int p, q;
                cin >> p >> q;
                int count = 0;
                for (int m = -142; m <= 142; m++) {
                        for (int n = -142; n <= 142; n++) {
                                int a = m * p + n * q;
                                int b = m * q - n * p;
                                int m2n2 = m * m + n * n;
                                if (m2n2 != 0 && a % m2n2 == 0 && b % m2n2 == 0) {
                                        count++;
                                }
                        }
                }
                if (count == 8) {
                        cout << "P\n";
                } else {
                        cout << "C\n";
                }
        }
       
        return 0;
}
