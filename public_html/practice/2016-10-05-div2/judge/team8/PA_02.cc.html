#include<bits/stdc++.h>

using namespace std;

#define endl '\n'

double a, b, c, d, m, t;

double f(double x) {
	return a * m * x*x*x
		+ b * m * x*x
		+ c * m * x
		+ d * m - t;
}

int main() {
	while (cin >> a) {
		cin >> b >> c >> d >> m >> t;
		double l = 0.001;
		double r = t / (c * m);
		while (r - l > 2e-6) {
			if (f((l+r) / 2) > 0) {
				r = (l+r) / 2;
			} else {
				l = (l+r) / 2;
			}
		}
		// int numDigits = ceil(log10((l+r)/2));
		// int ans = (int) (l+r) / 2 * pow(10, numDigits+2);
		// double k = (double) ans / pow(10, numDigits+2);
		// cout << setprecision(2) << fixed << k << endl;
		int ans = (int) ((l + r) / 2 * 100 + 1e-10);
		double ansd = ans / 100.0;
		cout << fixed << setprecision(2) << ansd << endl;
	}
}
