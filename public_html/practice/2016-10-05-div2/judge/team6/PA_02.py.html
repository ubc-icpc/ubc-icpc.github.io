import sys

def f(a,b,c,d,m,t,v):
  return 1.0/m - (a*v**3 + b*v**2 + c*v + d)/t

def ff(a,b,c,d,m,t,v):
  return -1.0/t * (3*a*v**2 + 2*b*v + c)

for line in sys.stdin:
  a,b,c,d,m,t = map(float, line.split())

  v = 0
  while abs(f(a,b,c,d,m,t,v)) > 10e-10:
    v -= f(a,b,c,d,m,t,v)/ff(a,b,c,d,m,t,v)

  print("%.2f"%(v-0.005))
