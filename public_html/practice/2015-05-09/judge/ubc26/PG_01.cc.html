#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

double L[32] = {};
int ms[32] = {};
int n;

double f(int m) {
  if (m >= n + 1) {
    throw 1;
  } else if (ms[m] > 0) {
    ms[m]--;
    return 0;
  } else {
    double a = f(m+1);
    double b = f(m+1);
    return L[m+1] + a + b;
  }
}

int main() {
  for (int i = 0; i < 32; i++)
    L[i] = pow(2.0, (1.0 - 2.0*(double)i) / 4.0);

  ms[0] = ms[1] = 0;

  cin >> n;
  for (int i = 0; i < n-1; i++) cin >> ms[2 + i];
  try {
    printf("%.14f\n", f(1));
  } catch (int e) {
    cout << "impossible" << endl;
  }
}
