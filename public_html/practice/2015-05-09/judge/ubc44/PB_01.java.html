import java.util.Scanner;

public class blackfriday {

	public static void main(String[] args) {
		Scanner s = new Scanner(System.in);
		int[] count = new int[6]; //use i-1 indices
		int size = s.nextInt();
		int[] a = new int[size];
		for (int i=0; i<size ; i++) {
			int a_i = s.nextInt();
			a[i] = a_i;
			count[(a_i-1)]++;
		}
		int high = -1;
		for (int k=6; k>0; k--) {
			int index = k-1;
			if (count[index] == 1) {
				high = k;
				break;
			}
		}
		if (high == -1)
			System.out.println("none");
		else {
			for (int i=0; i<size ; i++) {
				if (a[i] == high) {
					System.out.println(i+1);
					break;
				}
			}
		}
		s.close();
	}

}
