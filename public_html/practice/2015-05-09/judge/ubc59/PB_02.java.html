import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
public class BlackFriday2 {
	public static void main(String[] args) throws IOException{
		BufferedReader kinput=new BufferedReader(new InputStreamReader(System.in));
		String in=kinput.readLine();
		int ppl=Integer.parseInt(in);
		in=kinput.readLine();
		String[] split=new String[ppl];
		split=in.split(" ");
		int ans=0;
		int[]dice=new int[6];
		for(int i=0;i<ppl;i++){
			dice[Integer.parseInt(split[i])-1]++;
		}
		for(int i=0;i<6;i++){
			if(dice[i]==1)ans=i+1;
		}
		if(ans==0){
			System.out.print("none");
		}else{
			for(int i=0;i<ppl;i++)if(Integer.parseInt(split[i])==ans)System.out.print(i+1);
		}
	}
}
