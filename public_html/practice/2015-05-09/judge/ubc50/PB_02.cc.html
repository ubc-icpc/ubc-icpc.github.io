#include<bits/stdtr1c++.h>

using namespace std;

int n;
map<int, int> m;
vector<int> v;

int f(int k) {
  for (int i=0; i<n; i++) {
    if (v[i] == k)
      return i+1;
  }
}

int main() {
  cin >> n;
  for (int i=0; i<n; i++) {
    int x; cin >> x;
    m[x]++;
    v.push_back(x);
  }
  bool flag = false;
  int ans = 0;
  auto it = m.end();
  --it;
  while (it!=m.begin()) {
    if (it->second == 1) {
      ans = f(it->first);
      flag = true;
      break;
    }
    --it;
  }
  if (!flag) {
    if (it->second == 1) {
      ans = f(it->first);
      flag = true;
    }
  }
  if (flag)
    cout << ans << endl;
  else
    cout << "none" << endl;
  return 0;
}
