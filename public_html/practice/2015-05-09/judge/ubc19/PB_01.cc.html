#include <iostream>
#include <stdio.h>
#include <string.h>

#include <algorithm>
#include <cmath>

#include <sstream>
#include <complex>
#include <vector>
#include <set>

using namespace std;

int main()
{ 
  int t;
  cin >> t;
  int p[t];
  int res [7] = {0,0,0,0,0,0,0};
  for (int i = 0; i < t; i++) {
    cin >> p[i];
    res[p[i]]++;
  } 
  int ans = 0;
  for (int i=6; i>0; i--) {
    if (res[i] == 1) {
      ans = i;
      break;
    }
  }
  if (ans == 0) {
    cout << "none";
    return 0;
  }
  for (int i = 0; i < t; i++) {
    if (p[i] == ans) {
      cout << i+1;
      break;
    }
  } 
}
