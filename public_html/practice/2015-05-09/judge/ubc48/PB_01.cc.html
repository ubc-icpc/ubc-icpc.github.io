#include<bits/stdtr1c++.h>
using namespace std;
int main() {
	int n; cin >> n;
	vector < pair < int, int > > cust(n);
	vector < int > roll(7);
	for (int i = 0; i < n; ++i) {
		int temp; cin >> temp;
		cust[i] = make_pair(i, temp);
		++roll[temp];
	}
	vector < int > list;
	for (int i = 0; i < roll.size(); ++i) {
		if (roll[i] == 1) {
			list.push_back(i);
		}
	}

	if (list.size() == 0) {
		cout << "none" << endl;
	}
	else {
		sort(list.begin(), list.end());
		for (int i = 0; i < cust.size(); ++i) {
			if (cust[i].second == list[list.size() - 1]) {
				cout << cust[i].first + 1 << endl;
				return 0;
			}
		}
	}
	return 0;
}
