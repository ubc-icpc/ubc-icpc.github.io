#include <iostream>
#include <iomanip>
#include <algorithm>
#include <cmath>
using namespace std;

typedef long long ll;
typedef long double ld;

ll a[31];

int main() {
  cout << fixed << setprecision(11);
  int n; cin >> n;
  ll tot = 0;
  for (int i = 2; i < n+1; i++) {
    cin >> a[i];
    tot += (1LL<<(30-i))*a[i];
  }
  if (tot < (1LL<<29))
    cout << "impossible\n";
  else {
    ld tape = 0.0L;
    ll need = 1;
    for (int i = 1; i < 31; i++) {
      need -= min(need,a[i]);
      if (!need)
        break;
      tape += need*pow(2,-0.25L*(2*i+1));
      need *= 2;
    }
    cout << tape << "\n";
  }
  return 0;
}
