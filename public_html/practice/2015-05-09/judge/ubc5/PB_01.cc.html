#include <iostream>
#include <cstring>
#include <set>
using namespace std;

int a[100];
set<int> f[7];

int main() {
  int n;
  while (cin >> n) {
    for (int i = 1; i < 7; i++)
      f[i].clear();
    for (int i = 0; i < n; i++) {
      cin >> a[i];
      f[a[i]].insert(i+1);
    }
    for (int i = 6; i; i--) {
      if (f[i].size() == 1) {
        cout << *f[i].begin() << "\n";
        goto VELOCIRAPTOR;
      }
    }
    cout << "none\n";
    VELOCIRAPTOR:
    ;
  }
  return 0;
}
