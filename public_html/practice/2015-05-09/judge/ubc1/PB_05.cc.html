#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int n; cin >> n;
    map<int, vector<int> > m;
    for (int i = 0; i < n; i++) {
        int x; cin >> x;
        m[x].push_back(i+1);
    }
    
    int ans = -1;
    for (auto it = m.rbegin(); it != m.rend(); it++) {
        if (it->second.size() == 1) {
            ans = it->second[0];
            break;
        }
    }
    
    if (ans == -1) cout << "none" << endl;
    else cout << ans << endl;
	return 0;
}
