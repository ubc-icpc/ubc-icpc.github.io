#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;

long long a[35], b[35];
int main() {
    int n; cin >> n;
    
    int need = 2;
    bool poss = false;
    for (int i = 0; i < n-1; i++) {
        int x; cin >> x;
        if (x >= need) {
            b[i] = need;
            poss = true;
            break;
        } else {
            need -= x;
            need *= 2;
            b[i] = x;
        }
    }
    
    if (!poss) cout << "impossible" << endl;
    else {
        ld ans = 0, width = pow(2, -0.75), length = pow(2, -1.25);
        for (int i = n-2; i >= 0; i--) {
            ans += (b[i]/2) * width * pow(2, -i/2.0);
            if (i > 0) b[i-1] += b[i]/2;
        }
        cout << setprecision(11) << fixed << ans << endl;
    }
	return 0;
}
