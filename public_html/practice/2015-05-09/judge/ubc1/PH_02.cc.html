#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

map<ll, ll> memo;
ll f(ll n) {
    if (memo.count(n)) return memo[n];
    ll k = 1LL << int(log2(n+1)+0.5);
    while (k > n) k /= 2;
    return memo[n] = f(k-1) + 2*f(n-k);
}

int main() {
    ll n; cin >> n;
    memo[0] = 1;
    cout << f(n-1) << endl;
	return 0;
}
