import java.util.*;
import java.io.*;
public class B {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int[] arr = new int[7];
		int[] arr2 = new int[7];
		for (int i = 0; i < n; i++) {
			int lol = sc.nextInt();
			arr[lol]++;
			arr2[lol] = i+1;
		}
		int max = 6; while (max > 0 && arr[max] != 1) max--;
		if (max == 0) System.out.println("none");
		else System.out.println(arr2[max]);
	}
}
