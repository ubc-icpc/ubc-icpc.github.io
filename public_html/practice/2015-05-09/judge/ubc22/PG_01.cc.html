#include<bits/stdtr1c++.h>
using namespace std;

int N, M[35], R = 1;

int main () {
  cin >> N;
  for (int i = 1; i < N; ++i) cin >> M[i];
  double res = 0;
  for (int i = 0; i < N; ++i) {
    if ((R -= M[i]) <= 0) break;
    res += R * pow(2, -0.75 - i * 0.5);
    R *= 2;
  }
  if (R <= 0) cout << fixed << setprecision(10) << res << endl;
  else cout << "impossible" << endl;
}
