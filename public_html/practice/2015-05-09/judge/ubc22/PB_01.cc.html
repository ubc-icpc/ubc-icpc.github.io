#include<bits/stdtr1c++.h>
using namespace std;

int N, V[105], C[10];

int main () {
  cin >> N;
  for (int i = 1; i <= N; ++i) { cin >> V[i]; ++C[V[i]]; }
  int i = 6, j = 1;
  for (; i >= 1; --i) if (C[i] == 1) break;
  if (i) {
    for (; j <= N; ++j) if (V[j] == i) break;
    cout << j << endl; }
  else cout << "none" << endl;
  return 0;
}
