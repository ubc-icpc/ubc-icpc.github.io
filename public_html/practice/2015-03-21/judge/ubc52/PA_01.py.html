T=int(raw_input())
for t in range(T):
	N=int(raw_input())
	l=[]
	for n in xrange(N):
		l.append(raw_input())
	l.sort()
	works=True
	for i in xrange(1,len(l)):
		if len(l[i])>len(l[i-1]):
			longer=l[i]
			shorter=l[i-1]
		else:
			longer=l[i-1]
			shorter=l[i]
		if longer[:len(shorter)]==shorter:
			works=False
			break
	if works:
		print "YES"
	else:
		print "NO"
