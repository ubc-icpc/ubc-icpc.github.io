#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

bool comp (int i, int j) {
  return i>j;
}

int main (void) {
  
  int T; cin >> T;
  while(T--) {
    int n; cin >> n;
    vector< int > arr;
    for( int i=0; i<n; i++ ) {
      int t; cin >> t;
      arr.push_back(t);
    }
    
    sort(arr.begin(), arr.end(), comp);
    
    int res = 0;
    for ( int i=0; i<n; i+=3) {
      if( n-i >=3 ) {
	res += arr[i+2];
      }
    }
    cout << res << endl;
  }

  

  return 0;
}
