#include<bits/stdtr1c++.h> 

using namespace std;

int main (void) {
  int T; cin >> T;

  while (T--) {

    vector<int> arr;
    int n; cin >> n;
    for (int i=0; i<n; i++) {
      int t; cin >> t;
      arr.push_back(t);
    }

    int min=1000000, max=-1;
    for (int i=0; i<n; i++) {
      if (arr[i]<min) {
	min = arr[i];
      }
      if( arr[i]>max ) {
	max = arr[i];
      }
    }

    cout << (max-min)*2 << endl;
   
  }

  return 0;
}
