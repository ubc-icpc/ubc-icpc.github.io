#include<iostream>
#include<vector>
#include<string>
#include<algorithm>

using namespace std;

void f(vector<string> arr, int n) {
  for ( int i=0; i<n-1; i++ ) {
    string min = arr[i], max = arr[i+1];
    max = max.substr(0, min.size());
    if(max.compare(min) == 0) {
      cout << "NO"<< endl;
      return;
    }
  }
  cout << "YES" << endl;
}
int main ( void ) {
  int T; cin >> T;
  while (T--) {
    int n; cin >> n;
    vector<string> arr;
    for( int i=0; i<n; i++) {
      string t; cin >> t;
      arr.push_back(t);
    }
    sort(arr.begin(), arr.end());
    f(arr, n);
  }
  return 0;
}
