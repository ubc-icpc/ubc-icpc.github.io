from sys import stdin

t = int(stdin.readline().strip())

for _ in range(t):
    n = int(stdin.readline().strip())
    numbers = ["" for i in range(n)]
    for i in range(n):
        numbers[i] = stdin.readline().strip()
    numbers.sort()
    result = True
    for i in range(n-1):
        currNum = numbers[i]
        nextNum = numbers[i+1]
        if len(currNum) >= len(nextNum):
            result = True
        elif nextNum[:len(currNum)] == currNum:
            result = False
        if not result:
            break
    print "YES" if result else "NO"
