#include <iostream>

using namespace std;

int main() {
    int t; cin >> t;
    while (t--) {
        int n; cin >> n;
        int h = 0;
        int l = 100;
        while (n--) {
            int p; cin >> p;
            h = p > h ? p : h;
            l = p < l ? p : l;
        }
        cout << 2 * (h - l) << endl;
    }
}
