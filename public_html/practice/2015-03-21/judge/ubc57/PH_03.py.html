cases=int(raw_input())
for i in range(0,cases):
    number=int(raw_input())
    a=raw_input().split()
    L=[]
    for i in range(0,number):
        L.append(int(a[i]))
    L.sort()
    count=number-3
    answer=0
    while count>=0:
      answer=answer+L[count]
      count=count-3
    print(answer)
