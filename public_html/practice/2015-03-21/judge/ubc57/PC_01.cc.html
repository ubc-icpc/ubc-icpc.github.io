#include <iostream>
using namespace std;


int main()
{
	int cases;
	cin >> cases;
	int stores;
	for (int i = 1; i <= cases; i++){
		cin >> stores;
		int small;
		int big;
		int n;
		cin >> n;
		small = n;
		big = n;
		int a;
		for (int j = 2; j <= stores; j++){
			cin >> a;
			if (a > big){
				big = a;
			}
			if (a < small){
				small = a;
			}
		}
		int answer;
		answer = 2 * (big - small);
		cout << answer << endl;
	}
	return 0;
}
