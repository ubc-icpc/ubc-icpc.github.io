#include<bits/stdtr1c++.h>
using namespace std;

int main() {
  int T; cin >> T;
  while (T--) {
    int N; cin >> N;
    int MAX = -1, MIN = 100;
    while (N--) {
      int temp; cin >> temp;
      MAX = max(MAX, temp);
      MIN = min(MIN, temp);
    }
    cout << (MAX - MIN) * 2 << endl;
  }
  return 0;
}
