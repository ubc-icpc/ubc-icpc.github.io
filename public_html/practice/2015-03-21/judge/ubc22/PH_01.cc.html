#include<bits/stdtr1c++.h>
using namespace std;

int main () {
  int T; cin >> T;
  while (T--) {
    int N; cin >> N;
    vector<int> v(N);
    for (int i = 0; i < N; ++i) cin >> v[i];
    sort(v.begin(), v.end());
    int total = 0;
    for (int i = N - 3; i >= 0; i -= 3) total += v[i];
    cout << total << endl;
  }
  return 0;
}
