#include<bits/stdtr1c++.h>
using namespace std;

int main () {
  int T; cin >> T;
  while (T--) {
    int N; cin >> N;
    vector<string> v(N);
    for (int i = 0; i < N; ++i) cin >> v[i];
    sort(v.begin(), v.end());
    bool yes = true;
    for (int i = 0; yes && i < N - 1; ++i) {
      int END = min(v[i].size(), v[i + 1].size());
      bool b = true;
      for (int j = 0; j < END; ++j)
	if (v[i][j] != v[i + 1][j]) b = false;
      if (b) yes = false;
    }
    if (yes) cout << "YES" << endl;
    else cout << "NO" << endl;
  }
  return 0;
}
