#include <iostream>
#include <algorithm>
using namespace std;

int main() {
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    int mn = 100, mx = -1;
    for (int i = 0; i < n; i++) {
      int s; cin >> s;
      mn = min(mn, s);
      mx = max(mx, s);
    }
    cout << 2*(mx-mn) << "\n";
  }
  return 0;
}
