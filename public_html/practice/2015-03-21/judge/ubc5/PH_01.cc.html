#include <iostream>
#include <algorithm>
using namespace std;

int p[20000];

int main() {
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    for (int i = 0; i < n; i++)
      cin >> p[i];
    sort(p,p+n);
    int d = 0;
    for (int i = n-3; i >= 0; i -= 3)
      d += p[i];
    cout << d << "\n";
  }
  return 0;
}
