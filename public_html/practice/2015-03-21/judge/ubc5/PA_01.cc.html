#include <iostream>
#include <algorithm>
using namespace std;

string p[10000];

int main() {
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    for (int i = 0; i < n; i++)
      cin >> p[i];
    sort(p,p+n);
    bool ok = true;
    for (int i = 0; i < n-1; i++) {
      if (p[i].size() > p[i+1].size())
        continue;
      bool pm = true;
      for (int j = 0; j < p[i].size(); j++) {
        if (p[i][j] != p[i+1][j]) {
          pm = false;
          break;
        }
      }
      if (pm) {
        ok = false;
        break;
      }
    }
    cout << (ok ? "YES" : "NO") << "\n";
  }
  return 0;
}
