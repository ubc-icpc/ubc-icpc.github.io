import java.util.Scanner;

/**
 * Created by Peter on 2015-03-21.
 */
public class Shopping {
    public static void main(String[] args) {
        int[] shops;
        Scanner s = new Scanner(System.in);
        int noOfTests = Integer.parseInt(s.nextLine());

        for(int k = 0; k < noOfTests; k++) {
            int noOfShops = Integer.parseInt(s.nextLine());
            shops = new int[noOfShops];


            String[] line = s.nextLine().split("\\s+");

            for(int i = 0; i < noOfShops; i++){
                shops[i] = Integer.parseInt(line[i]);
            }
            int min = Integer.MAX_VALUE;
            int max = Integer.MIN_VALUE;
            for(int n : shops){
                if(n<min){
                    min = n;
                }
                if(n>max){
                    max = n;
                }
            }

            System.out.println((max-min)*2);
        }
    }
}
