import java.util.Arrays;
import java.util.Scanner;

/**
 * Created by Peter on 2015-03-21.
 */
public class Discount {

    public static void main(String[] args) {
        int[] prices;
        Scanner s = new Scanner(System.in);
        int noOfTests = Integer.parseInt(s.nextLine());
        int totalDiscount;

        for (int k = 0; k < noOfTests; k++) {
            totalDiscount =0;
            int noOfPrices = Integer.parseInt(s.nextLine());
            //System.out.println("noOfPrices: " + noOfPrices);
            prices = new int[noOfPrices];

            String str =s.nextLine();
            //System.out.println("str: " + str);

            String[] line = str.split("\\s+");

            for(int i = 0; i < noOfPrices; i++){
                prices[i] = Integer.parseInt(line[i]);
            }

            Arrays.sort(prices);
            int counter = 1;
            for(int j= prices.length -1 ; j >=0; j--){
                 if(counter % 3 == 0){
                     totalDiscount += prices[j];
                 }
                counter +=1;
            }
            System.out.println(totalDiscount);
        }
    }
}
