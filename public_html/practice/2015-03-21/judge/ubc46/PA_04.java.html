import java.util.Arrays;
import java.util.Scanner;

/**
 * Created by Peter on 2015-03-25.
 */
public class PhoneNumber {
    private static String [] phoneList;// ={"911", "333456", "911234"};


    public static void main(String[] args) {

        Scanner s = new Scanner(System.in);
        int noOfTests = Integer.parseInt(s.nextLine());

        for(int k = 0; k < noOfTests; k++) {
            int noOfNumbers = Integer.parseInt(s.nextLine());
            phoneList = new String[noOfNumbers];
            for(int j = 0; j < noOfNumbers; j++){
                phoneList[j] = s.nextLine();
            }
            Arrays.sort(phoneList);


            if (anyPrefix(phoneList)) {
                System.out.println("NO");
            } else {
                System.out.println("YES");
            }
        }
    }
    private static boolean anyPrefix (String[] str){
        for(int i = 0; i < phoneList.length-1; i++){
            String first = phoneList[i];
            String second = phoneList[i+1];
            if(second.length() >= first.length()) {
                String sub =second.substring(0, first.length());
                if (first.equals(sub)) {
                    return true;
                }
            }
        }
        return false;
    }
}
