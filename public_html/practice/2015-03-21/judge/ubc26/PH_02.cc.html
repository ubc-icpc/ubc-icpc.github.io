#include <iostream>
#include <algorithm>
using namespace std;

int main() {
  int t; cin >> t;
  while(t--) {
    int n; cin >> n;
    int* aa = new int[n];
    int nn=n;while(nn)cin >> aa[--nn];
    sort(&aa[0], &aa[n]);

    //for (int i = 0; i < n; i++)  cout << aa[i] << "  ";
    //cout << endl;

    int sss=0, k=0;
    for (int i = n-1; i >= 0; i--, k++) {
      if ((k%3)==2) sss += aa[i];
    }
    cout << sss << endl;

    delete[] aa;
  }
}
