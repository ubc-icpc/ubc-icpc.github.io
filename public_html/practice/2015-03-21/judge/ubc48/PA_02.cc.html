#include<iostream>
#include<string>
#include<algorithm>
#include<utility>
#include<vector>
using namespace std;

int main() {
	int t; cin >> t;
	while (t--) {
		int n; cin >> n;
		vector < string > dic(n);
		for (int i = 0; i < n; ++i) {
			string temp; cin >> temp;
			dic[i] = temp;
		}
		bool flag = true;
		sort(dic.begin(), dic.end());
		for (int i = 0; i < dic.size() - 1 && flag; ++i) {
			if (dic[i + 1].at(0) == dic[i].at(0)) {
				if (dic[i + 1].substr(0, dic[i].length()) == dic[i]) {
					flag = false;
					break;
				}
			}
		}
		if (flag) {
			cout << "YES" << endl;
		}
		else {
			cout << "NO" << endl;
		}
	}
}
