#include<iostream>
#include<string>
#include<algorithm>
#include<utility>
#include<vector>
using namespace std;

int main() {
	int t; cin >> t;
	while (t--) {
		int n; cin >> n;
		vector < int > list(n);
		for (int i = 0; i < n; ++i) {
			cin >> list[i];
		}
		sort(list.begin(), list.end());
		int total = 0;
		for (int i = n - 3; i >= 0; i -= 3) {
			total += list[i];
		}
		cout << total << endl;
	}
}
