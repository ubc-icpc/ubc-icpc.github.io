for i in range(t):
    n = int(raw_input())
    consistent = "YES"
    lst = []
    for j in range(n):
        lst.append(raw_input())
    lst.sort()
    for i in range(n-1):
        if len(lst[i]) < len(lst[i+1]):
            if lst[i] == lst[i+1][0:len(lst[i])]:
                consistent = "NO"
