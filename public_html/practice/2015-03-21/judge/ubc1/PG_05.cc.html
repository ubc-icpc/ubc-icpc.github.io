#include <bits/stdtr1c++.h>

using namespace std;

typedef pair<int, int> pii;

pii arr[20005];
int main() {
    int t; cin >> t;
    while (t--) {
        int n; cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> arr[i].first >> arr[i].second;
            arr[i].second *= -1;
        }
        sort(arr, arr+n);
        
        multiset<int> s;
        for (int i = 0; i < n; i++) {
            int x = arr[i].second;
            auto it = s.upper_bound(x);
            if (s.empty() || it == s.end()) s.insert(x);
            else {
                s.erase(it);
                s.insert(x);
            }
        }
        cout << s.size() << endl;
    }
	return 0;
}
