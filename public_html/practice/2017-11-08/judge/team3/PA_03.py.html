good_vals = [1,2,3,3,4,10]
bad_vals = [1,2,2,2,3,5,11]
T = int(raw_input())
for tc in range(1, T+1):
  good = [int(x) for x in raw_input().split()]
  bad = [int(x) for x in raw_input().split()]
  for i in range(len(good)):
    good[i] = good[i]*good_vals[i]
  for i in range(len(bad)):
    bad[i] = bad[i]*bad_vals[i]
  res = ''
  if sum(good) == sum(bad):
    res = 'No victor on this battle field'
  elif sum(good) > sum(bad):
    res = 'Good triumphs over Evil'
  else:
    res = 'Evil eradicates all trace of Good'
  print('Battle {}: {}'.format(tc, res))
