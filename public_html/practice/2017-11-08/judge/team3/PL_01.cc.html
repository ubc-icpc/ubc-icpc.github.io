#include<bits/stdc++.h>

using namespace std;

char vs[] = {'a', 'i', 'y','e','o','u'};
char cs[] = {'b', 'k', 'x', 'z', 'n', 'h', 'd', 'c', 'w', 'g', 'p', 'v', 'j', 'q', 't', 's', 'r', 'l', 'm', 'f'};

bool isVowel(char c) {
  for(int i=0; i<6; i++)
    if(c == vs[i]) return true;
  return false;
}

map<char,int> vid,cid;
int main() {
  string str;
  for(int i=0; i<6; i++) {
    vid[vs[i]] = i;
  }
  for(int i=0; i<20; i++) cid[cs[i]]=i;
  while(getline(cin, str)) {
    string ans = "";
    for(char c:str) {
      if(!(c>='A' && c<='Z') && !(c>='a' && c<='z')) {
        ans +=c; continue;
      }
      if(c>='A' && c<='Z') {
        char x = (c-'A')+'a';
        int idx;
        if(isVowel(x)) {
          idx = (vid[x]+3)%6;
          ans += (vs[idx] - 'a' + 'A');
        } else {
          idx = (cid[x]+10)%20;
          ans += (cs[idx] - 'a' + 'A');
        }
      } else {
        int idx;
        if(isVowel(c)) {
          idx = (vid[c]+3)%6;
          ans += (vs[idx]);
        } else {
          idx = (cid[c]+10)%20;
          ans += (cs[idx]);
        }
      }
    }
    cout << ans << endl;
  }

}
