#include<bits/stdc++.h>
#define int long long

using namespace std;
int n;
int dp[66][66];
//m choose r
int chooose(int m, int r) {
  return dp[m][r];
}

int f(int x, int y) {
  int ans = 0;

  for(int i=3-y; i<=x; i+=3) {
    ans += chooose(x,i);
  }
  return ans;
}

#undef int
int main() {
#define int long long
  //cout << f(6) << endl;
  //int x = 1;
  //x = x << 60;
  //cout << x << endl;
  for(int i=0; i<65; i++) {
    for(int j=0; j<65; j++) {
      if(i==j) dp[i][j] = 1;
      else if(i==0) dp[i][j] = 0;
      else if(j==0) dp[i][j] = 1;
      else if(i<j) dp[i][j] = 0;
      else dp[i][j] = dp[i-1][j] + dp[i-1][j-1];
    }
  }
  while(cin >> n) {
  int cnt = 0;
  int ans = 0;
  for(int i=60; i>=0; i--) {
    int t = 1;
    if(!(n & (t<<i))) continue;
    ans += f(i,cnt%3);

    cnt++;
  }
  ans += cnt/3;
  cout << "Day " << n << ": Level = " << ans << endl;
  }
}
