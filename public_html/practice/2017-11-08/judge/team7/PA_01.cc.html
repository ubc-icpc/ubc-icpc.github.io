/*
 * GoodVsEvil.cpp
 */

#include <bits/stdc++.h>
using namespace std;

int main() {
	int n;
	int good[6];
	int evil[7];

	cin >> n;

	for (int i = 0; i < n; i++) {
		scanf("%d %d %d %d %d %d", &good[0], &good[1], &good[2], &good[3],
				&good[4], &good[5]);
		scanf("%d %d %d %d %d %d %d", &evil[0], &evil[1], &evil[2], &evil[3],
				&evil[4], &evil[5], &evil[6]);

		int goodWorth = good[0] + 2*good[1] + 3*good[2] + 3*good[3] + 4*good[4] + 10*good[5];
		int evilWorth = evil[0] + 2*evil[1] + 2*evil[2] + 2*evil[3] + 3*evil[4] + 5*evil[5] + 11*evil[6];

		cout << "Battle " << i+1 << ": ";
		if (goodWorth < evilWorth) {
			cout << "Evil eradicates all trace of Good";
		} else if (goodWorth > evilWorth) {
			cout << "Good triumphs over Evil";
		} else {
			cout <<"No victor on this battle field";
		}
		cout << "\n";
	}
}
