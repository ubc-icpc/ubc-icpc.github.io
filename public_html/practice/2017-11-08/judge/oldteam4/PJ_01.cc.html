#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

long long int dp[1000001];

int main() {
    long long H, B, T, c1, c2, c3;
    while(cin >> H >> B >> T >> c1 >> c2 >> c3) {
        memset(dp, 0, sizeof dp);
        long long start = min(c1, min(c2,c3));
        for (long long h = start; h <= H; ++h) {
            long long v1 = 0, v2 = 0, v3 = 0;
            if (c1 <= h) {
                long long n = ((B-T)*(H-h) + T*H) / (c1*H);
                long long v = n*n*c1*c1*c1;
                v1 = dp[h-c1] + v;
            }
            if (c2 <= h) {
                long long n = ((B-T)*(H-h) + T*H) / (c2*H);
                long long v = n*n*c2*c2*c2;
                v2 = dp[h-c2] + v;
            }
            if (c3 <= h) {
                long long n = ((B-T)*(H-h) + T*H) / (c3*H);
                long long v = n*n*c3*c3*c3;
                v3 = dp[h-c3] + v;
            }
            dp[h] = max(dp[h-1], max(v1, max(v2, v3)));
        }
        cout << dp[H] << endl;
    }
    return 0;
}
