#include <iostream>

using namespace std;

int main(){
    int n, g, gsum, b, bsum, gs[6] = {1, 2, 3, 3, 4, 10}, bs[7] = {1,2,2,2,3,5,11};
    cin>>n;
    for(int j = 1; j<= n ; j++){
        gsum = 0; bsum = 0;
        for(int i = 0;i<6;i++){
            cin>>g;
            gsum += gs[i]*g; 
        }
        for(int i = 0;i<7;i++){
            cin>>b;
            bsum += bs[i]*b; 
        }
        if(gsum == bsum){
            cout<<"Battle "<<j<<": No victor on this battle field"<<endl;
            continue;
        }
        if(gsum > bsum){
            cout<<"Battle "<<j<<": Good triumphs over Evil"<<endl;
            continue;
        }
        
        cout<<"Battle "<<j<<": Evil eradicates all trace of Good"<<endl;
        
    
    }
    return 0;
}
