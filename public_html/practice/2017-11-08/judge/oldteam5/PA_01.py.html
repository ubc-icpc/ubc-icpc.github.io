import sys

stdin = sys.stdin

T = int(stdin.readline().strip())
for i in range(T):
    good = list(map(int, stdin.readline().strip().split()))
    evil = list(map(int, stdin.readline().strip().split()))
    good = good[0] + 2 * good[1] + 3 * good[2] + 3 * good[3] + 4 * good[4] + 10 * good[5]
    evil = evil[0] + 2 * evil[1] + 2 * evil[2] + 2 * evil[3] + 3 * evil[4] + 5 * evil[5] + 11 * evil[6]
    string = "Battle " + str(i+1) + ": "
    if good > evil:
        string += "Good triumphs over Evil"
    elif good == evil:
        string += "No victor on this battle field"
    else:
        string += "Evil eradicates all trace of Good"
    print(string)
