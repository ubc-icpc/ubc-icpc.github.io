import fileinput

final = set(map(str, range(10)))

def digits(n):
    return set(str(n))

for line in fileinput.input():
    n = int(line.strip())
    x = n
    digit = digits(x)
    count = 1
    while digit != final:
        x += n
        digit = digit.union(digits(x))
        count += 1
    print(count)
