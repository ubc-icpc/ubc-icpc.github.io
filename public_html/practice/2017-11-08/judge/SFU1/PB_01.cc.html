#include <cstdio>
#include <set>

using namespace std;

int main() {
	long long i;
	char buf[100];

	while(scanf("%lld",&i) == 1) {
		set<char> s;
		int m;

		for(m=1;true;m++) {
			long long d = i *m;
		
			sprintf(buf,"%lld",d);

			for(char* p = buf;*p != 0;p++) {
				s.insert(*p);
			}

			if(s.size() == 10) { break; }
		}

		printf("%d\n",m);
	}

	return 0;
}
