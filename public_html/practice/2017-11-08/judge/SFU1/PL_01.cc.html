#include <cstdio>
#include <map>
#include <cctype>
#include <cstring>

using namespace std;

int main() {
	char line[200];
	map<char,int> vowels;
	map<char,int> cons;

	const char* vowS = "aiyeou";
	const char* conS = "bkxznhdcwgpvjqtsrlmf";
	
	for(int i = 0;i < strlen(vowS);i++) {
		vowels[vowS[i]] = i;
	}

	for(int i = 0;i < strlen(conS);i++) {
		cons[conS[i]] = i;
	}

	while(gets(line)) {
		int len = strlen(line);

		for(int i = 0;i < len;i++) {
			bool upper = false;

			if(isupper(line[i])) {
				line[i] = tolower(line[i]);
				upper = true;
			}

			if(vowels.count(line[i]) == 1) {
				line[i] = vowS[(vowels[line[i]]+3)%6];
			}

			if(cons.count(line[i]) == 1) {
				line[i] = conS[(cons[line[i]]+10)%20];
			}

			if(upper) {
				line[i] = toupper(line[i]);
			}
		}

		puts(line);
	}

	return 0;
}
