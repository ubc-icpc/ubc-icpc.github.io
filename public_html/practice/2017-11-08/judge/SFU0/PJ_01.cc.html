#include <iostream>
#include <cstdlib>
#include <cstring>

using namespace std ;

long long ans[1000005];

int main() 
{
   long long h, l, t ;
   int a[3] ;
   while (cin >> h >> l >> t >> a[0] >> a[1] >> a[2]) {
      memset(ans, 0, sizeof(ans)) ;
      ans[0] = 0 ;
      long long r = 0 ;
      for (int i=1; i<=h; i++)
         for (int b=0; b<3; b++)
            if (i >= a[b] && ans[i-a[b]] >= 0) {
               long long w =
                (t + (l - t) * (h-i) / h) / a[b] ;
               long long l = ans[i-a[b]] + w * w * a[b]*a[b]*a[b] ;
               if (l > ans[i]) {
                  ans[i] = l;
                  if (l > r)
                     r = l;
               }
            }
      cout << r << endl ;
   }
}
