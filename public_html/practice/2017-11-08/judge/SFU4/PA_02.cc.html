#include <iostream>

using namespace std;

typedef long long ll;
int main(){
    int T;
    cin >> T;
    int k=0;
    int go[6] = {1,2,3,3,4,10};
    int ev[7] = {1,2,2,2,3,5,11};
    while(T--){
        k++;
        ll good=0, evil = 0, tmp;
        for(int i=0;i<6;i++){
            cin >> tmp;
            good += tmp * go[i];
        }
        for(int i=0;i<7;i++){
            cin >> tmp;
            evil += tmp*ev[i];
        }
        if(good>evil){
            cout<<"Battle "<<k<<": Good triumphs over Evil"<<endl;
        }
        else if(evil>good){
            cout<<"Battle "<<k<<": Evil eradicates all trace of Good"<<endl;
        }
        else if(evil == good){
            cout<<"Battle "<<k<<": No victor on this battle field"<<endl;
        }
    }
    return 0;
}
