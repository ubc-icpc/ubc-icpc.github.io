#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
#include<algorithm>
#include<vector>
#include<cmath>
#include<map>
using namespace std;

typedef long long LL;
LL ans;
LL bottom, height, top;
LL c[10];
LL f[2000000];


LL vol(int i, LL nn){
	long double d = top + (bottom - top) * (double) nn / height;
	LL num = (LL) floor(d / (long double) c[i]);
	return c[i] * c[i] * c[i] * num * num ;
}

int main(){
	while (cin >> height){
		cin >> bottom >> top >> c[0] >> c[1] >> c[2];
		if (bottom < top) swap(bottom, top);
		memset(f, 0, sizeof(f));
		ans = 0;
		for (int k = height; k >= 0; --k){
			for (int i = 0; i < 3; ++i){
				if (k + c[i] <= height){
					LL tmp = f[k+c[i]] + vol(i, k);
					f[k] = max(tmp, f[k]);
					ans = max(ans, f[k]);
				}
			}
		}
		cout << ans << endl;
		
	}

	return 0;
	
}
