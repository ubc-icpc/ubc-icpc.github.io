#include <iostream>
#include <cstring>
using namespace std;
typedef long long ll;
int main(){
	ll n;

	while (cin >> n){
		bool visited[10];
		memset(visited,false,sizeof(visited));
		ll k = 1;
		while (1){
			bool all_ok = true;
			for (int i = 0; i < 10; i++){
				if (!visited[i]) all_ok = false;
			}
			if (all_ok) break;
			
			ll num = k * n;
			while (num != 0){
				visited[num%10] = true;
				num /= 10;
			}
			k++;
		}
		cout << k-1 << endl;
	}

	return 0;
}
