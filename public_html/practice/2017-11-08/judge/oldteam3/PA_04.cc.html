#include <iostream>
using namespace std;
#define N 10
int g[6] = {1, 2,3,3,4,10};
int e[7] = {1, 2, 2, 2, 3, 5, 11};

int main() {
	int T, Tn;
	cin>>T;
	Tn = 1;
	
	while(Tn <= T) {
		int gg= 0, ee = 0, x;
		for(int i = 0 ;i < 6;++i) {
			cin>>x;
			gg += g[i] * x;
		}
		for(int i=0; i< 7; ++i) {
			cin>>x;
			ee += e[i] * x;
		}
		cout << "Battle " << Tn++ << ": ";
		if(gg < ee) {
			cout << "Evil eradicates all trace of Good" << endl;
		}else if (gg == ee ) {
			cout << "No victor on this battle field" << endl;

		} else {
			cout << "Good triumphs over Evil" << endl;
		}		
	}
	return 0;
}
