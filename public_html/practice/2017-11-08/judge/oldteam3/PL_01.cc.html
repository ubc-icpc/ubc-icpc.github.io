#include <iostream>
#include <map>
using namespace std;
char vow[6] = {'a', 'i', 'y', 'e', 'o', 'u'};
char cons[20] = {'b', 'k', 'x', 'z', 'n', 'h', 'd', 'c', 'w', 'g', 'p', 'v', 'j', 'q', 't', 's', 'r', 'l', 'm', 'f'};
int main(){
	map<char, char> translate;
	for (int i = 0; i < 6; i++){
		translate[vow[i]] = vow[(i+6-3)%6];
		translate[vow[i]-'a'+'A'] = vow[(i+6-3)%6] - 'a' +'A';
		//cout << (char)(vow[i] -'a'+'A')<< " " <<  (char)(vow[(i+6-3)%6] -'a'+'A')<< endl;
	}
	for (int i = 0; i < 20; i++){
		translate[cons[i]] = cons[(i+20-10)%20];
		translate[cons[i] -'a'+'A'] =  cons[(i+20-10)%20] - 'a' + 'A';
		//cout << cons[i] << " " <<  cons[(i+20-10)%20] << endl;
	}
	/*for (map<char,char>::iterator it = translate.begin(); it != translate.end(); it++){
		cout << (*it).first << " " << (*it).second << endl;
	}*/

	string s; 
	while(getline(cin, s)){
		for (int i = 0; i < s.length(); i++) {
			if ((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z')) cout << translate[s[i]];
			else cout << s[i];
		} cout << endl;
	}
	return 0 ;
}
