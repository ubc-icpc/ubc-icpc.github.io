#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

ll h, d1, d2, c1, c2, c3;

ll memo[1000001];

ll dp(ll curH) {
    if(memo[curH] != -1) {
        return memo[curH];
    }
    
    ll best = 0;
    ll n1 = ((d1 * h + (d2 - d1) * (curH + c1)) / h) / c1;
    ll n2 = ((d1 * h + (d2 - d1) * (curH + c2)) / h) / c2;
    ll n3 = ((d1 * h + (d2 - d1) * (curH + c3)) / h) / c3;
    //cout << curH << " " << n1 << " " << n2 << " " << n3 << endl;
    if(curH + c1 <= h) {
        best = max(best, dp(curH + c1) + n1*n1*c1*c1*c1);
    }
    if(curH + c2 <= h) {
        best = max(best, dp(curH + c2) + n2*n2*c2*c2*c2);
    }
    if(curH + c3 <= h) {
        best = max(best, dp(curH + c3) + n3*n3*c3*c3*c3);
    }

    return memo[curH] = best;
}

int main(){
  ios_base::sync_with_stdio(0); cin.tie(0);
  
  while(cin >> h) {
    cin >> d1 >> d2 >> c1 >> c2 >> c3;
    memset(memo, -1, sizeof memo);
    cout << dp(0) << endl;
  }
  
}
