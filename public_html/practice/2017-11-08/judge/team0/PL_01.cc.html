#include <bits/stdc++.h>

using namespace std;

int main(){
  ios_base::sync_with_stdio(0); cin.tie(0);
  
  string vowels = "aiyeou";
  string consonants = "bkxznhdcwgpvjqtsrlmf";
  
  set<char> vSet, cSet;
  for(char cur : vowels) {
    vSet.insert(cur);
  }
  for(char cur : consonants) {
    cSet.insert(cur);
  }
  
  string s; 
  
  while(getline(cin, s)) {
    string out = s;
      for(int i = 0; i < s.size(); i++) {
        bool isUpper = (s[i] >= 'A' && s[i] <= 'Z');
        if(isUpper) {
            s[i] = s[i] - 'A' + 'a';
        }
        if(vSet.count(s[i])) {
            for(int j = 0; j < vowels.size(); j++) {
                if(vowels[j] == s[i]) {
                    out[i] = vowels[(j + 3) % 6];
                    break;
                }
            }
        } else if (cSet.count(s[i])) {
            for(int j = 0; j < consonants.size(); j++) {
                if(consonants[j] == s[i]) {
                    out[i] = consonants[(j + 10) % 20];
                    break;
                }
            }
        }
        if(isUpper) {
            out[i] = out[i] - 'a' + 'A';
        }
      }
      cout << out << endl;
  }
}
