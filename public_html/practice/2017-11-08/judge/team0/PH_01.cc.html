#include <bits/stdc++.h>
using namespace std;
typedef long long lli;

//-----------------------
#define MAXN 100100

int n;
lli bit[MAXN];
map<string, int> mapa;
//-----------------------

lli sum(int x) {
  lli s = 0;
  while (x > 0) {
    s += bit[x];
    x -= (x & -x);
  }
  return s;
}

void update(int x) {
  while (x < MAXN) {
    bit[x]++;
    x += (x & -x);
  }
}

int main(){
  ios_base::sync_with_stdio(0); cin.tie(0);

  while (cin >> n && n) {
    mapa.clear();
    memset(bit, 0, sizeof bit);

    for (int i = 1; i <= n; i++) {
      string s;
      cin >> s;
      mapa[s] = i;
    }

    lli answer = 0LL;
    for (int i = 1; i <= n; i++) {
      string s;
      cin >> s;
      int cur = mapa[s];
      answer += sum(n) - sum(cur);
      update(cur);
    }

    cout << answer << endl;
  }
}
