#include <bits/stdc++.h>

using namespace std;

int main(){
  ios_base::sync_with_stdio(0); cin.tie(0);
  
  vector<int> v1 = {1, 2, 3, 3, 4, 10};
  vector<int> v2 = {1, 2, 2, 2, 3, 5, 11};
  
  int N; cin >> N;
  
  for(int i = 0; i < N; i++) {
    int a1 = 0, a2 = 0;
    for(int j = 0; j < v1.size(); j++) {
        int cur; cin >> cur; a1 += v1[j]*cur;
    } 
    for(int j = 0; j < v2.size(); j++) {
        int cur; cin >> cur; a2 += v2[j]*cur;
    }
    cout << "Battle " << i + 1 << ": ";
    if(a1 > a2) {
        cout << "Good triumphs over Evil" << endl;
    } else if(a1 < a2) {
        cout << "Evil eradicates all trace of Good" << endl;
    } else {
        cout << "No victor on this battle field" << endl;
    }
  }
}
