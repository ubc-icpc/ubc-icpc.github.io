#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main(){
  ios_base::sync_with_stdio(0); cin.tie(0);
  
  ll n;
  while(cin >> n) {
    ll k = 1;
    set<char> seen;
    while(true) {
        string s = to_string(k*n);
        for(char cur : s) {
            seen.insert(cur);
        }
        
        if(seen.size() == 10) {
            cout << k << endl;
            break;
        }
        k++;
    }
  }
}
