#include <iostream>

using namespace std;

int n,i,j,k,tmp,sumgood,sumevil;
int good[6]={1,2,3,3,4,10},evil[7]={1,2,2,2,3,5,11};

int main()
{
	cin>>n;
	for (k=0;k<n;k++)
	{
		sumgood = 0;
		for (i=0;i<6;i++)
		{
			cin>>tmp;
			sumgood+=tmp*good[i];
		}
		sumevil = 0;
		for (i=0;i<7;i++)
		{
			cin>>tmp;
			sumevil+=tmp*evil[i];
		}
		//cout<<sumgood<<' '<<sumevil<<endl;
		cout<<"Battle "<<k+1<<": ";
		if (sumgood<sumevil)
			cout<<"Evil eradicates all trace of Good"<<endl;
		else if (sumgood>sumevil)
			cout<<"Good triumphs over Evil"<<endl;
		else
			cout<<"No victor on this battle field"<<endl;
	}
	return 0;
}
