#include<iostream>
#include<stdio.h>
char exchage(char c)
{
	char dic[26]={'e','p','l','r','a','g','f','s','o','x','v','c','w','t','i','b','z','d','h','n','y','k','m','j','u','q'};
	if(c>='a' && c<='z') return dic[c-'a'];
	if(c>='A' && c<='Z') return dic[c-'A']-'a'+'A';
	return c;
}
int main(void)
{
	char c;
	int mark;
	mark = scanf("%c",&c);
	while(mark!=EOF)
	{
	printf("%c",exchage(c));
	mark = scanf("%c", &c);
	}
}
