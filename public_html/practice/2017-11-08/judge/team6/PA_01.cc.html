#include <bits/stdc++.h>
using namespace std;

int n, sg, sb, e;
int g[] = {1, 2, 3, 3, 4, 10};
int b[] = {1, 2, 2, 2, 3, 5, 11};
int main() {
    cin >> n;
    for (int k = 1; k <= n; k++) {
        sg = 0, sb = 0;
        cout << "Battle " << k << ": ";
        for (int i = 0; i < 6; i++) {
            cin >> e;
            sg += g[i] * e;
        }
        for (int i = 0; i < 7; i++) {
            cin >> e;
            sb += b[i] * e;
        }
        if (sg == sb) {
            cout << "No victor on this battle field\n";
        } else if (sg > sb) {
            cout << "Good triumphs over Evil\n";
        } else {
            cout << "Evil eradicates all trace of Good\n";
        }
    }
    return 0;
}
