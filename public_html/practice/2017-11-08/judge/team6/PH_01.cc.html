#include <iostream>
#include <map>

using namespace std;
#define N (1<<17)
#define ll long long

ll spuu[2*N];
void add(int i){
	i+=N;
	while(i){
		spuu[i]++;
		i/=2;
	}
}

ll q(int a, int b){
	a+=N; b+=N;
	ll ret = 0;
	while(a < b){
		if(a % 2 == 1){
			ret += spuu[a];
			a++;
		}
		if(b % 2 == 0){
			ret += spuu[b];
			b--;
		}
		a/=2;
		b/=2;
	}
	if(a==b)
		ret += spuu[a];
	return ret;
}

void clear(){
	for(int i=0; i<=2*N; i++)
		spuu[i] = 0;
}

int main(){
	bool _ = 1;
	for(;_;){
		int n;
		cin>>n;
		if(n==0)
			break;
		clear();
		
		map<string, int> idx;
		for(int i=0; i<n; i++){
			string s;
			cin>>s;
			idx[s] = i;
		}

		ll ret = 0;

		for(int i=0; i<n; i++){
			string s;
			cin>>s;
			ret += q(idx[s]+1, N);
			add(idx[s]); 
		}
		cout << ret << endl;
	}
	return 0;
}
