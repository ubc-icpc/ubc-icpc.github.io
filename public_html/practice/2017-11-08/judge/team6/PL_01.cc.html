char lett[] = {'e','p','l','r','a','g','f','s','o','x','v','c','w','t','i','b','z','d','h','n','y','k','m','j','u','q'};
char Lett[] = 
{'E','P','L','R','A','G','F','S','O','X','V','C','W','T','I','B','Z','D','H','N','Y','K','M','J','U','Q'};
int main(){
	string s;
	
	while(getline(cin,s,'\n')){
		for(int i=0; i<256; i++){
			if(s[i] <= 'Z' && s[i] >= 'A'){
				s[i] = Lett[s[i] - 'A'];
			}
			if(s[i] <= 'z' && s[i] >= 'a'){
				s[i] = lett[s[i] - 'a'];
			}
		}
		cout<<s<<endl;
	}
	return 0;
}
