#include <iostream>
#include <vector>
#include <string>
#include <cctype>

std::string vowels = "aiyeou";
std::string conson = "bkxznhdcwgpvjqtsrlmf";

char condupper(bool cond, char c) {
	if(!cond) return toupper(c);
	return c;
}

char trans(char c) {
	bool lower = std::islower(c);
	c = std::tolower(c);
	if(vowels.find(c) != std::string::npos) {
		return condupper(lower, vowels[(vowels.find(c)+3)%vowels.size()]);
	}
	if(conson.find(c) != std::string::npos) {
		return condupper(lower, conson[(conson.find(c)+10)%conson.size()]);
	}
	return c;
}

int main() {

	std::string line;
	while(std::getline(std::cin, line)) {
		for(int i = 0; i < line.size(); i ++) line[i] = trans(line[i]);

		std::cout << line << std::endl;
	}

	return 0;
}
