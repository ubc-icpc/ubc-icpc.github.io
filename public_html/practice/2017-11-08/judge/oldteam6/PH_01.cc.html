#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
typedef long long ll;

ll solve(int * arr, int n) {
	if (n == 1) return 0;
	ll res = 0;
	int mid = n/2;
	res += solve(arr, mid) + solve (arr+mid, n-mid);
	sort(arr, arr+mid);
	sort(arr+mid, arr+n);
	int * a1 = arr; int * a1end = arr+mid;
	int * a2 = arr+mid; int * a2end = arr+n;
	while (a1 < a1end && a2 < a2end) {
		while (a1 < a1end && *a1 < *a2) {
			a1++;
		}
		res += a1end - a1;
		a2++;
	}
	return res;
}

int main() {
	int n; cin >> n;
	while (n > 0) {
		map<string, int> m;
		for (int i = 0; i < n; i++) {
			string s; cin >> s; m[s] = i;
		}
		int arr[n];
		for (int i = 0; i < n; i++) {
			string s; cin >> s; arr[i] = m[s];
		}
		cout << solve(arr, n) << endl;
		cin >> n;
	}
}
