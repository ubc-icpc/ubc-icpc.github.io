#include <bits/stdc++.h>

using namespace std;

int c[] = {1,2,3,3,4,10}, d[] = {1,2,2,2,3,5,11};
int main() {
	int t, ca = 0; cin >> t;
	while (t--) {
		int good = 0, evil = 0;
		
		for (int i = 0; i < 6; i++) {
			int x; cin >> x;
			good += c[i]*x;
		}
		
		for (int i = 0; i < 7; i++) {
			int x; cin >> x;
			evil += d[i]*x;
		}

		cout << "Battle " << ++ca << ": ";
		if (good == evil) {
			cout << "No victor on this battle field" << endl;
		} else if (good > evil) {
			cout << "Good triumphs over Evil" << endl;
		} else {	
			cout << "Evil eradicates all trace of Good" << endl;
		}
	}
}
