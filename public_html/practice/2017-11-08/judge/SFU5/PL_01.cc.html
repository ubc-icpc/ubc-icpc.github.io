#include <stdio.h>
#include <string.h>
#include <ctype.h>

char s[10000];

char v[6] = {'a', 'i', 'y', 'e', 'o', 'u'};
char c[20] = {'b', 'k', 'x', 'z', 'n', 'h', 'd', 'c', 'w', 'g', 'p', 'v', 'j', 'q', 't', 's', 'r', 'l', 'm', 'f'};

char subs[140];

int main() {
	for (int i = 0; i < 140; i++) subs[i] = i;
	for (int i = 0; i < 6; i++) {
		subs[v[i]] = v[(i + 3) % 6];
		subs[v[i]-32] = v[(i + 3) % 6]-32;
	}
	for (int i = 0; i < 20; i++) {
		subs[c[i]] = c[(i + 10) % 20];
		subs[c[i]-32] = c[(i + 10) % 20]-32;
	}
	while (gets(s)) {
		int n = strlen(s);
		for (int i = 0; i < n; i++) {
			s[i] = subs[s[i]];
		}
		puts(s);
	}
}
