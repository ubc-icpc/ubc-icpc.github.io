#include <iostream>
#include <string>
using namespace std;

char v[6] = {'a', 'i', 'y', 'e', 'o', 'u'};
char C[20] = {'b', 'k', 'x', 'z', 'n', 'h', 'd', 'c', 'w', 'g', 'p', 'v', 'j', 'q', 't', 's', 'r', 'l', 'm', 'f'};

int main()
{
    string s;
    while (getline(cin, s))
    {
        string s2 = "";
        for (int i = 0; i < s.size(); i++)
        {
            char c = s[i];
            if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z'))
            {
                bool isl = (c >= 'a' && c <= 'z');
                if (!isl)
                    c |= 32;
                int ind = 0;
                while (ind < 6 && v[ind] != c)
                    ind++;
                if (ind < 6)
                {
                    c = v[(ind+3)%6];
                    if (!isl)
                        c ^= 32;
                }
                else
                {
                    ind = 0;
                    while (C[ind] != c)
                        ind++;
                    c = C[(ind+10)%20];
                    if (!isl)
                        c ^= 32;
                }
            }
            s2.push_back(c);
        }
        cout << s2 << "\n";
    }
    return 0;
}
