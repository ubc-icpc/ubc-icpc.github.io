#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long ll;
typedef long double ld;

ll dp[1000001];

int main()
{
    ll h, sb, st, b1, b2, b3;
    while (cin >> h >> sb >> st >> b1 >> b2 >> b3)
    {
        memset(dp, 0, sizeof dp);
        ll tm = 0;
        for (ll i = 0; i < h; i++)
        {
            if (i != 0 && !dp[i])
                continue;
            ll v1 = 0, v2 = 0, v3 = 0;
            if (i+b1 <= h)
            {
                ld s = sb+(ld)(i+b1)*(st-sb)/h;
                ll l = s/b1;
                v1 = b1*b1*b1*l*l;
                //cout << i << " " << b1 << " " << i+b1 << " " << s << " " << l << " " << v1 << " " << dp[i] << " " << dp[i]+v1 << "," << dp[i+b1] << endl;
                dp[i+b1] = max(dp[i+b1], dp[i]+v1);
                tm = max(tm, dp[i+b1]);
            }
            if (i+b2 <= h)
            {
                ld s = sb+(ld)(i+b2)*(st-sb)/h;
                ll l = s/b2;
                v2 = b2*b2*b2*l*l;
                //cout << i << " " << b2 << " " << i+b2 << " " << s << " " << l << " " << v2 << " " << dp[i] << " " << dp[i]+v2 << "," << dp[i+b2] <<endl;
                dp[i+b2] = max(dp[i+b2], dp[i]+v2);
                tm = max(tm, dp[i+b2]);
            }
            if (i+b3 <= h)
            {
                ld s = sb+(ld)(i+b3)*(st-sb)/h;
                ll l = s/b3;
                v3 = b3*b3*b3*l*l;
                //cout << i << " " << b3 << " " << i+b3 << " " << s << " " << l << " " << v3 << " " << dp[i] << " " << dp[i]+v3 << "," << dp[i+b3] <<endl;
                dp[i+b3] = max(dp[i+b3], dp[i]+v3);
                tm = max(tm, dp[i+b3]);
            }
        }
        cout << tm << "\n";
    }
    return 0;
}
