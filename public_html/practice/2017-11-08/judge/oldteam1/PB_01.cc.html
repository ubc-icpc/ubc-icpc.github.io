#include <iostream>
#include <cstring>
using namespace std;

typedef long long ll;

bool d[10];

int main()
{
    ll n;
    while (cin >> n)
    {
        memset(d, 0, sizeof d);
        int k = 0;
        ll o = n;
        while (true)
        {
            bool br = true;
            for (int i = 0; i < 10; i++)
            {
                if (!d[i])
                {
                    br = false;
                    break;
                }
            }
            if (br)
            {
                cout << k << "\n";
                break;
            }
            ll nn = n;
            while (nn)
            {
                d[nn%10] = true;
                nn /= 10;
            }
            n += o;
            k++;
        }
    }
    return 0;
}
