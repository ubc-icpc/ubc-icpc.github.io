#include <iostream>
using namespace std;

int a[13];
int v[13] = {1, 2, 3, 3, 4, 10, 1, 2, 2, 2, 3, 5, 11};

int main()
{
    int t; cin >> t;
    for (int T = 1; T <= t; T++)
    {
        for (int i = 0; i < 13; i++)
        {
            cin >> a[i];
            a[i] *= v[i];
        }
        int g = 0, e = 0;
        for (int i = 0; i < 13; i++)
        {
            if (i < 6)
                g += a[i];
            else
                e += a[i];
        }
        cout << "Battle " << T << ": ";
        if (g > e)
            cout << "Good triumphs over Evil\n";
        else if (e > g)
            cout << "Evil eradicates all trace of Good\n";
        else
            cout << "No victor on this battle field\n";
    }
    return 0;
}
