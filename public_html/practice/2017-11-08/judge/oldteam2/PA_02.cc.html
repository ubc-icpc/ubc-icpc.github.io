#include <iostream>

using namespace std;

int main() {
    int t; cin >> t;
    for (int i = 1; i <= t; i++) {
        unsigned int good, evil;
        unsigned int a, b, c, d, e, f, g;
        cin >> a >> b >> c >> d >> e >> f;
        good = a + 2 * b + 3 * (c + d) + 4 * e + 10 * f;
        cin >> a >> b >> c >> d >> e >> f >> g;
        evil = a + 2 * (b + c + d) + 3 * e + 5 * f + 11 * g;
        cout << "Battle " << i << ": ";
        cout << ((good < evil) ? "Evil eradicates all trace of Good" : (good == evil) ? "No victor on this battle field" : "Good triumphs over Evil") << endl;
    }
    return 0;
}
