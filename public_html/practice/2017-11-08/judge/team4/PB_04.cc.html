#include <bits/stdc++.h>
using namespace std;

bool nogood(long long n, long long k) {
	bool cool[10];// = {false,false,false,false,false,false,false,false,false,false};
	memset(cool, false, 10);
	for (long long i = 1; i <= k; ++i) {
		//cout << i << ',';
		long long val = n * i;

		while (val > 0) {
			cool[val % 10] = true;
			val /= 10;
		}
	}
	bool coolio = true;
	for (long long i = 0; i < 10; ++i)
		coolio = coolio && cool[i];
	return !coolio;
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(NULL);

	long long n;
	while (cin >> n)
	{
		//long long n;
		//cin >> n;

		long long k = 1;
		while (nogood(n,k))
			++k;
		cout << k << endl;
	}
}
