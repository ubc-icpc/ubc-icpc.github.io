#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(NULL);

	int n;
	cin >> n;
	for (int i = 0; i < n; ++i)
	{
		int a[6];
		cin >> a[0] >> a[1] >> a[2] >> a[3] >> a[4] >> a[5];
		a[1]*=2;a[2]*=3;a[3]*=3;a[4]*=4;a[5]*=10;
		int sum1 = 0;
		for (int i = 0; i < 6; ++i)
			sum1+=a[i];

		int b[7];
		cin >> b[0] >> b[1] >> b[2] >> b[3] >> b[4] >> b[5] >> b[6];
		b[1]*=2;b[2]*=2;b[3]*=2;b[4]*=3;b[5]*=5;b[6]*=11;
		int sum2 = 0;
		for (int i = 0; i < 7; ++i)
			sum2+=b[i];
		cout << "Battle " << i+1 << ": ";
		if (sum1 == sum2)
			cout << "No victor on this battle field";
		else if (sum1 > sum2)
			cout << "Good triumphs over Evil";
		else
			cout << "Evil eradicates all trace of Good";
		cout << endl;
	}
}
