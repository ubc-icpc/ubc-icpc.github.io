#include <iostream>
using namespace std;
int g_w[6] = {1,2,3,3,4,10};
int b_w[7] = {1,2,2,2,3,5,11};
int main() {
	int test;
	cin>>test;

	for(int z = 1; z <= test; z++) {
		int good = 0;
		int bad = 0;
		int temp;
		for(int i = 0; i < 6; i++) {
			cin>>temp;
			good += temp * g_w[i];
		}
		for(int i = 0; i < 7; i++) {
			cin>>temp;
			bad += temp * b_w[i];
		}
		cout<<"Battle "<<z<<": ";
		if(good > bad) {
			
			cout<<"Good triumphs over Evil"<<endl;
		}
		else {
			if(good < bad) {
				cout<<"Evil eradicates all trace of Good"<<endl;
			}
			else {
				cout<<"No victor on this battle field"<<endl;
			}
		}
	}
}
