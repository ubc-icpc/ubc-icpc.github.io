#include <bits/stdc++.h>
#define endl '\n'
using namespace std;
typedef long long ll;
typedef long double ld;
typedef complex<ld> pt;
const int MOD = 1e9 + 7;

const vector<int> good = {1, 2, 3, 3, 4, 10}, bad = {1, 2, 2, 2, 3, 5, 11};

int main() {
//	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	int T; cin >> T;
	for (int t = 1; t <= T; t++) {
		vector<int> a(6), b(7);
		for (int i = 0; i < 6; i++)
			cin >> a[i];
		for (int i = 0; i < 7; i++)
			cin >> b[i];
		int k = 0, l = 0;
		for (int i = 0; i < 6; i++)
			k += good[i] * a[i];
		for (int i = 0; i < 7; i++)
			l += bad[i] * b[i];
		printf("Battle %d: ", t);
		if (k > l) puts("Good triumphs over Evil");
		else if (k < l) puts("Evil eradicates all trace of Good");
		else puts("No victor on this battle field");
	}
	return 0;
}
