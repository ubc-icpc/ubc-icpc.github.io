#include <bits/stdc++.h>
#define endl '\n'
using namespace std;
typedef long long ll;
typedef long double ld;
typedef complex<ld> pt;
const int MOD = 1e9 + 7;

const vector<int> vow = {'a', 'i', 'y', 'e', 'o', 'u'}, cons = {'b', 'k', 'x', 'z', 'n', 'h', 'd', 'c', 'w', 'g', 'p', 'v', 'j', 'q', 't', 's', 'r', 'l', 'm', 'f'};

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	string s;
	map<char, int> svow, scons;
	for (int i = 0; i < vow.size(); i++)
		svow[vow[i]] = i;
	for (int i = 0; i < cons.size(); i++)
		scons[cons[i]] = i;
	while (getline(cin, s)) {
		for (char &c : s) {
			if (svow.count(c)) {
				c = vow[(svow[c] + 3)%vow.size()];
			} else if (scons.count(c)) {
				c = cons[(scons[c] + 10)%cons.size()];
			} else if (svow.count(c - 'A' + 'a')) {
				c = vow[(svow[c - 'A' + 'a'] + 3)%vow.size()] - 'a' + 'A';
			} else if (scons.count(c - 'A' + 'a')) {
				c = cons[(scons[c - 'A' + 'a'] + 10)%cons.size()] - 'a' + 'A';
			}
		}
		cout << s << endl;
	}
	return 0;
}
