#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>
#include <queue>
#include <climits>
#include <stack>
#include <string>

using namespace std;

int main() {
	string vowels = "aiyeou";
	string consonants = "bkxznhdcwgpvjqtsrlmf";
	string s;
	while (getline(cin, s)) {
		for (int i = 0; i < s.size(); i++) {
			bool isUpper = false;
			if (s[i] >= 'A' && s[i] <= 'Z') {
				isUpper = true;
				s[i] = s[i] - 'A' + 'a';
			}
			int vi = vowels.find(s[i]);
			int ci = consonants.find(s[i]);
			//cout << vi << ' ' << ci << endl;
			char c;
			if (vi != -1) {
				c = vowels[(vi + 3) % 6];
			} else if (ci != -1) {
				c = consonants[(ci + 10) % 20];
			} else {
				c = s[i];
			}
			if (isUpper) {
				c += 'A' - 'a';
			}
			cout << c;
		}
		cout << endl;
	}
	return 0;
}
