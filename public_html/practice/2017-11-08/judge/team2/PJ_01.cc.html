#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>
#include <queue>
#include <climits>
#include <stack>
#include <string>
#include <set>
#include <complex>
#include <math.h>

using namespace std;

int main() {
	long long int h, b, t, b1, b2, b3;
	while(cin >> h >> b >> t >> b1 >> b2 >> b3) {
		long long int dp[h + 1];
		double x = (t*h+0.0)/(b-t);
		dp[0] = 0;
		for (int i = 1; i <= h; i++) {
			dp[i] = dp[i - 1];
			if (i - b1 >= 0)
				dp[i] = max(dp[i], dp[i-b1] + (long long) (floor(((x+i-b1)*b/(x+h))/b1)*floor(((x+i-b1)*b/(x+h))/b1)*b1*b1*b1));
			if (i - b2 >= 0)
				dp[i] = max(dp[i], dp[i-b2] + (long long) (floor(((x+i-b2)*b/(x+h))/b2)*floor(((x+i-b2)*b/(x+h))/b2)*b2*b2*b2));
			if (i - b3 >= 0)
				dp[i] = max(dp[i], dp[i-b3] + (long long) (floor(((x+i-b3)*b/(x+h))/b3)*floor(((x+i-b3)*b/(x+h))/b3)*b3*b3*b3));
		}
		cout << dp[h] << endl;
	}
	return 0;
}
