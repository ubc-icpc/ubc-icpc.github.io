#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>
#include <queue>
#include <climits>
#include <stack>
#include <string>
#include <set>

using namespace std;

int main() {
	string s;
	while (getline(cin, s)) {
		set<int> digits;
		for (int i = 0; i <= 9; ++i)
			digits.insert(i);
		long long n = stoi(s);
		long long k = 0;
		while (!digits.empty()) {
			++k;
			long long kn = k * n;
			while (kn > 0) {
				digits.erase(kn % 10);
				kn /= 10;
			}
		}
		cout << k << '\n';
	}
	return 0;
}
