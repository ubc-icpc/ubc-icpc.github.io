#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>
#include <queue>
#include <climits>
#include <stack>
#include <string>

using namespace std;

int main() {
	int cases;
	cin >> cases;
	for (int test = 1; test <= cases; ++test) {
		int a, b, c, d, e, f, g;
		cin >> a >> b >> c >> d >> e >> f;
		int good = a + 2 * b + 3 * c + 3 * d + 4 * e + 10 * f;
		cin >> a >> b >> c >> d >> e >> f >> g;
		int bad = a + 2 * b + 2 * c + 2 * d + 3 * e + 5 * f + 11 * g;
		cout << "Battle " << test << ": ";
		if (good > bad)
			cout << "Good triumphs over Evil\n";
		else if (bad > good)
			cout << "Evil eradicates all trace of Good\n";
		else
			cout << "No victor on this battle field\n";
	}
}
