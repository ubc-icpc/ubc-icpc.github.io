#include <bits/stdc++.h>
using namespace std;

string a = "aAiIyYeEoOuU";
string b = "bBkKxXzZnNhHdDcCwWgGpPvVjJqQtTsSrRlLmMfF";
int main() {
    string s;
    while (getline(cin, s)) {
        string r;
        for (int i = 0; i < s.size(); i++) {
            int pa = a.find(s[i]);
            int pb = b.find(s[i]);
            if (pa != -1) {
                r += a[(pa + 6) % a.size()];
            } else if (pb != -1) {
                r += b[(pb + 20) % b.size()];
            } else {
                r += s[i];
            }
        }
        cout << r << '\n';
    }
    return 0;
}
