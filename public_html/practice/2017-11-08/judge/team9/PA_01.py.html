u = [1, 2, 3, 3, 4, 10]
v = [1, 2, 2, 2, 3, 5, 11]
n = int(raw_input())

for i in range(n):
    print "Battle " + str(i + 1) + ": ",
    g, b = 0, 0
    g = sum([x * y for (x, y) in zip(u, map(int, raw_input().split()))])
    b = sum([x * y for (x, y) in zip(v, map(int, raw_input().split()))])
    if g == b:
        print "No victor on this battle field"
    elif g > b:
        print "Good triumphs over Evil"
    else:
        print "Evil eradicates all trace of Good"
