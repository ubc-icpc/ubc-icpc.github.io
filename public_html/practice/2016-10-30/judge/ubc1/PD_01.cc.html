#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int q, n, D; 
string s, t; 
ll memo[105][205];
ll dp(int i, int j) {
	if (j < 0) return 0;
	if (i == n) return (j == 0);
	if (memo[i][j] != -1) return memo[i][j];
	
	if (s[i] == t[i]) memo[i][j] = (q-1) * dp(i+1, j-2) + dp(i+1, j);
	else memo[i][j] = 2 * dp(i+1, j-1) + (q-2) * dp(i+1, j-2);
	
	return memo[i][j];
}

int main() {
	ios::sync_with_stdio(0);
	cin >> q >> n >> D;
	cin >> s >> t;
	memset(memo, -1, sizeof memo);
	cout << dp(0, D) << endl;
	return 0;
}
