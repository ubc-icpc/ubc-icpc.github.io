#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

int main() {
	ios::sync_with_stdio(0);
	ll n, m; cin >> n >> m;
	int t = 0;
	while (min(n, m) > 0) {
		if (n < m) swap(n, m);
		if (n%m == 0 || n/m > 1) break;
		t ^= 1;
		n %= m;
	}
	
	cout << (t == 0 ? "win" : "lose") << endl;
	return 0;
}
