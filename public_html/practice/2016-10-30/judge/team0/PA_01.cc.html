#include <bits/stdtr1c++.h>
using namespace std;
typedef long long lli;

int win(lli a, lli b){

  if(a < b) return win(b, a);

  if((a % b) == 0LL) return 1;

  if( (a/b) == 1 ) return 1 - win(b, a-b);

  return 1;
}

int main(){

  lli a, b;
  cin >> a >> b;

  printf("%s\n", (win(a, b))?"win":"lose");

  return 0;
}
