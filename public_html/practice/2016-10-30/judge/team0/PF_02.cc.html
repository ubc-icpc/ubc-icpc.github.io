#include <bits/stdtr1c++.h>
using namespace std;

//---------------------
#define MAXN 1010

int n, k;

int idx;

int certain;
int conta[MAXN];
int carta[MAXN];

map<string, int> mapa;
//---------------------

int main(){

  cin >> n >> k;

  int found = 0;
  for(int i = 1;i <= k;i++){

    int p1, p2;
    string n1, n2;
    cin >> p1 >> p2 >> n1 >> n2;

    int s1, s2;
    if(!mapa.count(n1)) mapa[n1] = ++idx;
    if(!mapa.count(n2)) mapa[n2] = ++idx;

    s1 = mapa[n1];
    s2 = mapa[n2];

    if(s1 == s2) found++;

    carta[p1] = s1;
    carta[p2] = s2;
  }

  for(int i = 1;i <= n;i++) conta[carta[i]]++;

  for(int i = 1;i <= n;i++) if(conta[i] == 2) certain++;

  certain -= found;

  if(2*idx == n-2){
    if(2*(found + certain) + 2 == n) cout << 1 + certain << endl;
    else cout << certain << endl;
  }

  else if(2*idx < n) cout << certain << endl;
  else cout << (n/2) - found << endl;

  return 0;
}
