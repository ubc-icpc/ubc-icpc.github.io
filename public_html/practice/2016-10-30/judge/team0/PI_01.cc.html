#include<bits/stdtr1c++.h>
using namespace std;

int M, S, I, L[1000005];
vector<int> s;
vector<map<int, int> > F;
vector<priority_queue<int, vector<int>, greater<int> > > Q;

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> M >> S;
  F.resize(M-1);
  Q.resize(M-1);
  s.resize(M-1, 0);

  for (int i = 0; i < S; ++i)
    cin >> L[i];

  for (int i = 0; i < M-1; ++i) {
    for (int j = 0; j < S; ++j) {
      int a; cin >> a;
      F[i][a] = j;
    }
  }

  for (; I < S; ++I) {
     for (int i = 0; i < M-1; ++i) {
       Q[i].push(F[i][L[I]]);
     }
     int good = true;
     for (int i = 0; i < M-1; ++i) {
       while (!Q[i].empty() && Q[i].top() == s[i]) {
         Q[i].pop();
         ++s[i];
       }
       if (!Q[i].empty()) good = false;
     }
     if (good) break;
  }

  ++I;

  cout << I << endl;
  sort(L, L+I);
  cout << L[0];
  for (int i = 1; i < I; ++i) {
    cout << " " << L[i];
  }
  cout << endl;

  return 0;
}
