#include<bits/stdtr1c++.h>
using namespace std;

int x[3], y[3], perm[3];

int main() {
  cin >> x[0] >> y[0];
  cin >> x[1] >> y[1];
  cin >> x[2] >> y[2];
  perm[0] = 0;
  perm[1] = 1;
  perm[2] = 2;

  do {
    int dx1 = x[perm[1]] - x[perm[0]];
    int dy1 = y[perm[1]] - y[perm[0]];
    int dx2 = x[perm[2]] - x[perm[0]];
    int dy2 = y[perm[2]] - y[perm[0]];
    if (dx1 * dx2 + dy1 * dy2 == 0) {
      cout << dx1 + dx2 + x[perm[0]] << " " << dy1 + dy2 + y[perm[0]] << endl;
      return 0;
    }
  } while (next_permutation(perm, perm + 3));

  return 0;
}
