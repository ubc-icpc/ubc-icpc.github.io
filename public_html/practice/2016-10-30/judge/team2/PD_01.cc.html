#include <bits/stdc++.h>

using namespace std;

int prefs[1000002];

int main() {
	int q, n, D; cin >> q >> n >> D;
	string s1, s2; cin >> s1 >> s2;

	long long dp[101][201]; memset(dp, 0, sizeof dp);
	dp[0][0] = 1;

	for(int i = 1; i <= n; i++) {
		for(int j = 0; j <= D; j++) {
			if(s1[i-1] == s2[i-1]) {
				dp[i][j] = dp[i-1][j] + (j >= 2 ? (q-1) * dp[i-1][j-2] : 0);
			} else {
				dp[i][j] = (j >= 1 ? 2*dp[i-1][j-1] : 0) + (j >= 2 ? (q-2) * dp[i-1][j-2] : 0);
			}
		}
	}

	cout << dp[n][D] << endl;
}
