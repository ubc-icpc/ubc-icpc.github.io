#include <bits/stdc++.h>

using namespace std;

int main() {
	vector<complex<int>> coords;
	set<int> nums;
	int x, y;
	cin >> x >> y; coords.push_back({x, y});
	cin >> x >> y; coords.push_back({x, y});
	cin >> x >> y; coords.push_back({x, y});
	complex<int> test = coords[0] - coords[1];
	int minLength = test.real() * test.real() + test.imag() * test.imag();
	test = coords[0] - coords[2];
	if(test.real() * test.real() + test.imag() * test.imag() == minLength) {
		//0-1 and 0-2 are legs
		complex<int> final = coords[2] - coords[0] + coords[1];
		cout << final.real() << " " << final.imag() << endl;
	} else if (test.real() * test.real() + test.imag() * test.imag() < minLength) {
		//0-1 is hypotenuse
		complex<int> final = coords[1] - coords[2] + coords[0];
		cout << final.real() << " " << final.imag() << endl;
	} else {
		//0-2 is hypotenuse
		complex<int> final = coords[2] - coords[1] + coords[0];
		cout << final.real() << " " << final.imag() << endl;
	}
}
