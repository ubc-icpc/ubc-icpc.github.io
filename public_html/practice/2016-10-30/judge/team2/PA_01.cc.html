#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

bool gcd(ll a, ll b) {
	ll p = a / b, q = a % b;
	if (q == 0) return true;
	if (p > 1) return true;
	return !gcd(b, q);
}

int main() {
	ll a, b; scanf("%lld%lld", &a, &b);
	if (a < b) {
		ll t = b;
		b = a;
		a = t;
	}
	printf(gcd(a, b) ? "win\n" : "lose\n");

	return 0;
}
