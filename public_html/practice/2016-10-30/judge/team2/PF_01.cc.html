#include <bits/stdc++.h>

using namespace std;

int main() {
	int N; cin >> N;
	int K; cin >> K;
	map<string, int> cards;
	set<string> matchableCards;
	set<string> matchedCards;
	set<string> unmatched;
	for(int i = 0; i < K; i++) {
		int C1, C2; cin >> C1 >> C2;
		string p1, p2; cin >> p1 >> p2;
		if(p1 == p2) {
			matchedCards.insert(p1);
			matchableCards.erase(p1);
			unmatched.erase(p1);
		} else {
			if(cards.count(p1) && cards[p1] != C1 && !matchedCards.count(p1)) {
				matchableCards.insert(p1);
				unmatched.erase(p1);
			} else if(!matchableCards.count(p1)){
				cards[p1] = C1;
				unmatched.insert(p1);
			}
			if(cards.count(p2) && cards[p2] != C2 && !matchedCards.count(p2)) {
				matchableCards.insert(p2);
				unmatched.erase(p2);
			} else if(!matchableCards.count(p2)){
				cards[p2] = C2;
				unmatched.insert(p2);
			}
		}
	}
	
	if(matchableCards.size() + unmatched.size() + matchedCards.size() == N / 2) {
		cout << matchableCards.size() + unmatched.size() << endl;
	} else if (matchableCards.size() + matchedCards.size() == N/2 - 1) {
		cout << matchableCards.size() + 1 << endl;
	} else {
		cout << matchableCards.size() << endl;
	}
}
