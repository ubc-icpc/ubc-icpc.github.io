#include<bits/stdc++.h>
using namespace std;

int n, m;

vector<vector<int>> x;
int main(){
    cin >> n >>m;
    int a;
    for(int i=0;i<n;i++){
        vector<int> tmp;
        for(int j=0;j<m;j++){
            cin >>a;
            tmp.push_back(a);
        }
        x.push_back(tmp);
    }
    set<int> b;
    for(int j=0;j<m;j++){
        for(int i=0;i<n;i++){
            b.insert(x[i][j]);
        }
        if (j+1==(int) b.size()){
            cout << b.size() << endl;
            for(auto k: b){
                cout << k << " ";
            }
            return 0;
        }
    }
}
