#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <cmath>
#include <math.h>

using namespace std;

int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		int usa[3], rus[3];
		int usum = 0, rsum = 0;
		for(int j=0;j<3;j++){
			cin>>usa[j];
			cout<<usa[j]<<' ';
			usum+=usa[j];
		}
		for(int j=0;j<3;j++){
			cin>>rus[j];
			cout<<rus[j];
			if(j!=2) cout<<' ';
			rsum+=rus[j];
		}
		cout<<endl;
		bool count=false, color=false;
		if(usum>rsum) count=true;
		if(usa[0]>rus[0]) color=true;
		if((usa[0]==rus[0]) && (usa[1]>rus[1])) color=true;
		if((usa[0]==rus[0]) && (usa[1]==rus[1]) && (usa[2]>rus[2])) color=true;
		
		if(count&&color) cout<<"both"<<endl;
		else if(count) cout<<"count"<<endl;
		else if(color) cout<<"color"<<endl;
		else cout<<"none"<<endl;
	}
	return 0;
}
