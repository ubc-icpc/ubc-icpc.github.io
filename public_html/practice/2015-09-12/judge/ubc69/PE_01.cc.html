#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <cmath>
#include <math.h>

using namespace std;

int main() {
	cout<<fixed<<setprecision(2);
	int n;
	cin>>n;
	while(n--){
		double s,r;
		cin>>s>>r;
		if(r>s/(sqrt(2))){
			cout<<s*s<<endl;
			continue;
		}
		if(r<s/2){
			cout<<r*r*M_PI<<endl;
			continue;
		}
		double l = sqrt(r*r-s*s/4);
		double t = 2*acos(s/(2*r));
		cout<<r*r*M_PI-4*(r*r*t/2 - l*s/2)<<endl;
	}
	return 0;
}
