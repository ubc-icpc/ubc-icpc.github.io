#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <cmath>
#include <math.h>

using namespace std;

int main() {
	int n;
	cin>>n;
	while(n--){
		int d,x,s;
		cin>>d>>x>>s;
		int cups[d], lemons[d], sugar[d];
		for(int i=0;i<d;i++){
			cin>>cups[i]>>lemons[i]>>sugar[i];
			if(i>0&&lemons[i]>lemons[i-1]) lemons[i]=lemons[i-1];
			if(i>0&&sugar[i]>sugar[i-1]) sugar[i]=sugar[i-1];
		}
		int cost=0, amtsugar=0;
		for(int i=0;i<d;i++){
			cost+=lemons[i]*x*cups[i];
			while(amtsugar<cups[i]*s){
				amtsugar+=80;
				cost+=sugar[i];
			}
			amtsugar-=cups[i]*s;
		}
		cout<<cost<<endl;
	}
	return 0;
}
