#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <cmath>
#include <math.h>

using namespace std;

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		int s,b,m;
		cin>>s>>b>>m;
		cout<<"Practice #"<<i<<": "<<s<<' '<<b<<endl;
		for(int j=0;j<m;j++){
			int k; cin>>k;
			while(b<=k) b*=2;
			b-=k;
			cout<<k<<' '<<b<<endl;
		}
	}
	return 0;
}
