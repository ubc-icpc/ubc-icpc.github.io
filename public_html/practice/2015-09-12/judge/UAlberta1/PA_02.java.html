import java.util.Scanner;

public class twins {

	public static void main(String[] args){
		System.out.println(" ");
		Scanner in = new Scanner(System.in);
		String G = in.nextLine();
		for(int i = 0; i < Integer.parseInt(G); i++){
			String input = in.nextLine();
			boolean zack = input.contains("17");
			boolean mack = input.contains("18");
			System.out.println(input);
			if(zack && mack)System.out.println("both");
			if(!zack && mack)System.out.println("mack");
			if(zack && !mack)System.out.println("zack");
			if(!zack && !mack)System.out.println("none");
			System.out.println();
		}
	}
}
