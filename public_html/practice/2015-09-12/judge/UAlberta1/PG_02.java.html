import java.util.Scanner;

public class hanoi {

	static Scanner in = new Scanner(System.in);
	static String input;
	public static void main(String[] args){
		System.out.println(" ");
		String G = in.nextLine();
		for(int g = 0; g < Integer.parseInt(G); g++){
			input = in.nextLine();
			int n = Integer.parseInt(input.split(" ")[0]);
			int d = Integer.parseInt(input.split(" ")[1]);
			if(n > d*d - 2*(d-1))System.out.println("Grid #" + (g+1) + ": impossible");
			else System.out.println("Grid #" + (g+1) + ": " + (2*n*(d-1)));
			System.out.println();
		}
	}
}
