import java.util.Scanner;
import java.text.DecimalFormat;

public class rain {

	static Scanner in = new Scanner(System.in);
	static String input;
	public static void main(String[] args){
		System.out.println(" ");
		String G = in.nextLine();
		for(int g = 0; g < Integer.parseInt(G); g++){
			input = in.nextLine();
			double pi = 3.14159265358979;
			double s = Integer.parseInt(input.split(" ")[0]);
			s/=2;
			double r = Integer.parseInt(input.split(" ")[1]);
			double v = (r <= s) ? pi*r*r : ((r < s*Math.sqrt(2)) ? pi*r*r - 4*(r*r*Math.acos(s/r) - s*Math.sqrt(r*r-s*s)) : 4*s*s);
			System.out.println(new DecimalFormat("#.00").format(v));
		}
	}
}
