import java.util.Scanner;

public class medal {

	public static void main(String[] args){
		Scanner in = new Scanner(System.in);
		String G = in.nextLine();
		for(int i = 0; i < Integer.parseInt(G); i++){
			String input = in.nextLine();
			System.out.println(input);
			int[] m = new int[6];
			for(int j = 0; j < 6; j++)
				m[j] = Integer.parseInt(input.split(" ")[j]);
			boolean color = (m[0] != m[3]) ? (m[0] > m[3]) : ((m[1] != m[4]) ? (m[1] > m[4]) : (m[2] > m[5]));
			boolean count = (m[0] + m[1] + m[2] > m[3] + m[4] + m[5]);
			if(color && count)System.out.println("both");
			else if(color)System.out.println("color");
			else if(count)System.out.println("count");
			else System.out.println("none");
			System.out.println();
		}
	}
}
