#include <bits/stdtr1c++.h>
using namespace std;

int main(){
	int teams; cin >> teams;
	int cur = 1;
	while(teams--){
		int s, b, m;
		cin >> s >> b >> m;

		cout << "Practice #" <<cur<<": "<<s<<" "<<b<< endl;
		cur++;
		while(m--){
			int k; cin >> k;
			while(k >= b){
				b *= 2;
			}
			b = b - k;
			cout <<k<<" "<<b<<endl;
			if (m == 0){
				cout << endl;
			}
		}

	}
}
