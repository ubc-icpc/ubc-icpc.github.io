#include <bits/stdtr1c++.h>
using namespace std;

int main(){
	int s; cin >> s;

		bool mack, zack;
		for (int i = 1; i <= 10*s; i++){
			int k;
			cin >> k;
			if (k == 17){zack = true;}
			if (k == 18){mack = true;}
			if (i%10 == 0){
				cout << k << endl;
				if (zack && mack){
					cout << "both" << endl;
					zack = false;
					mack = false;
				}
				else if (zack){
					zack = false;
					cout << "zack" << endl;
				}
				else if (mack){
					mack = false;
					cout << "mack" << endl;
				}
				else {cout << "none" << endl;}		
				
			}
			else{
				cout << k << " ";
			}
		}

}
