#include <bits/stdtr1c++.h>

using namespace std;

int main(){
	int s; cin >> s;
	while(s--){
		int ut,ug,us,ub,rt,rg,rs,rb;
		bool count = false;
		bool color = false;
		cin >>ug>>us>>ub>>rg>>rs>>rb;
		cout<<ug<<" "<<us<<" "<<ub<<" "<<rg<<" "<<rs<<" "<<rb<<" "<< endl;

		ut = ug + us + ub;
		rt = rg + rs + rb;

		if (ut > rt){count = true;}

		if (ug > rg){color = true;}
		else if (ug == rg){
			if (us > rs){color = true;}
			else if (us == rs){
				if (ub > rb){color = true;}
			}
		}
		if (count && color){
			cout << "both" << endl << endl;
		}
		else if (count){cout << "count"<<endl<<endl;}
		else if (color){cout << "color"<<endl<<endl;}
		else {cout << "none" <<endl<<endl;}

		ut = 0; rt = 0; count = 0; color = 0;

	}
}
