import java.util.Scanner;


public class C12 {

	static Scanner in;
	public static void main(String[] args) {
		
		in = new Scanner(System.in);
		

			int caseSize = in.nextInt();
			int[][] testCase = new int[caseSize][];
			for (int k = 0; k < caseSize; k++) {
				int numOfStudents = in.nextInt();
				int numOfBrownies = in.nextInt();
				int numOfGroups = in.nextInt();	
				testCase[k] = new int[numOfGroups];
				System.out.println("Practice #" + (k + 1) +":" + " " + numOfStudents + " " + numOfBrownies);
				for (int j = 0; j < numOfGroups; j++) {
					testCase[k][j] = in.nextInt();
				}
	
				solve(numOfGroups, numOfBrownies, testCase[k]);
			}
		
		return;
	}
	
	public static void solve(int numOfGroups, int numOfBrownies, int[] arr) {
		for (int i = 0; i < numOfGroups; i++) {
			while (arr[i] >= numOfBrownies) {
				numOfBrownies *= 2;
			}
			numOfBrownies -= arr[i];
			System.out.println(arr[i] + " " + numOfBrownies);
			}
			
		
	}

}
