import java.util.Scanner;


public class B12 {

	static Scanner in;
	public static void main(String[] args) {
		
		in = new Scanner(System.in);
		int testNum = in.nextInt();
		

		for (int i = 0; i < testNum; i++) {
			int[] testCase = new int[6];
			for (int j = 0; j < 6; j++) {
				testCase[j] = in.nextInt();
				
			}

			solve(testCase);
		}
		return;
	}
	
	public static void solve(int[] arr) {
		boolean count = false;
		boolean color = false;
		for (int i = 0; i < 6; i++) {
			System.out.print(arr[i] + " ");
		}
		System.out.println();
		if (arr[0] > arr[3]) {
			color = true;
		} else if (arr[0] == arr[3]) {
			if (arr[1] > arr[4])
			{
				color = true;
			} else if(arr[1] == arr[4]){
				if(arr[2] > arr[5]) {
					color = true;
				} 
			}
		}
		
		if (arr[0] + arr[1] + arr[2] > arr[3] + arr[4] + arr[5]) {
			count = true;
		} 
		if (count && color)
			System.out.println("both");
		else if (count) 
			System.out.println("count");
		else if (color)
			System.out.println("color");
		else 
			System.out.println("none");
		System.out.println();
	}

}
