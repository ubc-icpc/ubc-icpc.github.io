#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n; cin >> n;
	while(n--) {
		int ug, us, ub, rg, rs, rb;
		bool count = false, color = false;
		cin >> ug >> us >> ub >> rg >> rs >> rb;
		cout << ug << " " << us << " " << ub << " " << rg << " " << rs << " " << rb << endl;
		if ((ug + us + ub) > (rg + rs + rb)) count = true;
		if (ug > rg) {
			color = true;
		} else if (ug == rg && us > rs) {
			color = true;
		} else if (ug == rg && us == rs && ub > rb) {
			color = true;
		}
		if (count && color) {
			cout << "both";
		} else if (count) {
			cout << "count";
		} else if (color) {
			cout << "color";
		} else {
			cout << "none";
		}
		cout << endl << endl;

	}
	return 0;
}
