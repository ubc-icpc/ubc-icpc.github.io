#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n; cin >> n;
	for (int N=1; N<=n; N++) {
		int s, b, m; cin >> s >> b >> m;
		cout << "Practice #" << N << ": " << s << " " << b << endl;
		for (int i=0; i<m; ++i) {
			int g; cin >> g;
			if (g < b) {
				b -= g;
			} else {
				while (b <= g) {
					b *= 2;
				}
				b -= g;
			}
			cout << g << " " << b << endl;
		}
		cout << endl;
	}
	return 0;
}
