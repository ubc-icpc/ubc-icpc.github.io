#include <bits/stdtr1c++.h>
using namespace std;

const double PI = 3.14159265358979;

int main() {
	int n; cin >> n;
	while(n--) {
		double r,s; cin >> s >> r;
		double ans = 0;
		if (s >= r*2) {
			ans = PI*r*r;
		} else if (s*sqrt(2.0) <= r*2) {
			ans = s*s;
		} else {
			double h = r - s/2.0;
			double theta = 2*acos(1 - h/r);
			double excess = r*r/2*(theta - sin(theta));
			ans = PI*r*r - excess*4;
		}
		cout << setprecision(2) << fixed << ans << endl;
	}
	return 0;
}
