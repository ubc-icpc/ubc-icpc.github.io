int main() {
	int n; cin >> n;
	while(n--) {
		int d, x, s; cin >> d >> x >> s;
		int minl = 1000, mins = 1000;
		int cost = 0, currs = 0;
		for (int i=0; i<d; ++i) {
			int c, pl, ps; cin >> c >> pl >> ps;
			if (pl < minl) {
				minl = pl;
			}
			if (ps < mins) {
				mins = ps;
			}
			cost += x*c*minl;
			while(currs < c*s) {
				currs += 16*5;
				cost += mins;
			}
			currs -= c*s;
		}
		cout << cost << endl;
	}
	return 0;
}
