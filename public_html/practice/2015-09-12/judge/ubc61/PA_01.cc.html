#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n; cin >> n;
	while(n--) {
		bool m = false, z = false;
		for (int i=0; i<10; ++i) {
			int inp; cin >> inp;
			if (inp == 17) z = true;
			if (inp == 18) m = true;
			cout << inp << " ";
		}
		cout << endl;
		if (z && m) {
			cout << "both";
		} else if (z) {
			cout << "zack";
		} else if (m) {
			cout << "mack";
		} else {
			cout << "none";
		}
		cout << endl << endl;
	}	
	return 0;
}
