#include <iostream>
#include <vector>

using namespace std;

int main() {
  int t;
  int g[1000][1000];
  
  cin >> t;
  while (t--) {
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i)
      for (int j = 0; j < n; ++j)
	cin >> g[i][j];

    vector<int> seq(1, 0);

    for (int i = 1; i < n; ++i) {
      if (g[i][seq[0]]) seq.insert(seq.begin(), i);
      else if (g[seq[i-1]][i]) seq.insert(seq.end(), i);
      else {
	for (int j = 0; ; ++j)
	  if (g[seq[j]][i] && g[i][seq[j+1]]) {
	    seq.insert(seq.begin()+j+1, i);
	    break;
	  }
      }
    }

    for (int i = 0; i < n; ++i)
      cout << seq[i]+1 << (i+1==n?'\n':' ');
  }

  return 0;
  
}
