#include <iostream>

using namespace std;

int a[6];

bool clr() {
  for (int i = 0; i < 3; ++i)
    if (a[i] > a[i+3]) return true;
    else if (a[i] < a[i+3]) return false;
  return false;
}

int main() {
  int t;
  cin >> t;
  while (t--) {
    for (int i = 0; i < 6; ++i) {
      cin >> a[i];
      cout << a[i] << (i==5?'\n':' ');
    }

    bool count = (a[0] + a[1] + a[2] > a[3] + a[4] + a[5]);
    bool colour = clr();

    if (count) {
      if (colour) cout << "both";
      else cout << "count";
    }
    else {
      if (colour) cout << "color";
      else cout << "none";
    }
    cout << "\n\n";
  }
}
