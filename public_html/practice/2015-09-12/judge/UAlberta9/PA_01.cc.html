#include <iostream>

using namespace std;

int main() {
  int t;
  cin >> t;
  while (t--) {
    bool z = false, m = false;
    for (int i = 0; i < 10; ++i) {
      int a;
      cin >> a;
      cout << a << (i==9?'\n':' ');
      z |= (a == 17);
      m |= (a == 18);
    }
    if (z) {
      if (m) cout << "both";
      else cout << "zack";
    }
    else {
      if (m) cout << "mack";
      else cout << "none";
    }
    cout << "\n\n";
  }
  return 0;
}
