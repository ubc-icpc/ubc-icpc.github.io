#include <iostream>

using namespace std;

int main() {
  int t;
  cin >> t;
  for (int cno = 1; cno <= t; ++cno) {
    int s, b;
    cin >> s >> b;

    cout << "Practice #" << cno << ": " << s << ' ' << b << endl;

    int m;
    cin >> m;

    while (m--) {
      int k;
      cin >> k;

      while (b <= k) b *= 2;
      b -= k;
      cout << k << ' ' << b << endl;
    }
    cout << endl;
  }
  return 0;
}
