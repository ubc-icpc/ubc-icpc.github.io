#include <iostream>
#include <cmath>
#include <iomanip>

using namespace std;

const double PI = 3.14159265358979;

int main() {
  cout.setf(ios::fixed);
  cout.precision(2);

  int t;
  cin >> t;
  while (t--) {
    double s, r;
    cin >> s >> r;

    double as = s*s, ar = PI*r*r;

    if (r <= s/2) cout << ar << endl;
    else if (r*r >= s*s/2) cout << as << endl;
    else {
      double nx = s/(2*r);
      double ny = sqrt(1-nx*nx);
      double theta = atan2(ny,nx);
      double slice = r*r*theta/2;
      double triang = s/2*ny*r/2;
      
      cout << ar - 8*(slice-triang) << endl;
    }
  }
}
