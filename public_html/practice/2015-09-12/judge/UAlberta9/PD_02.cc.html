#include <iostream>
#include <algorithm>

#define OZ (5*16)

using namespace std;

typedef long long LL;

int main() {
  int t, d, x, s;
  
  cin >> t;
  
  while (t--) {
    cin >> d >> x >> s;
    LL bl = 1000, bs = 1000, cost = 0, rem = 0;
    
    for (int i = 0; i < d; ++i) {
      LL num, pl, ps;
      cin >> num >> pl >> ps;

      bl = min(bl, pl);
      bs = min(bs, ps);

      LL bags = (s*num - rem + OZ-1)/OZ;

      rem = (rem + OZ*bags - s*num)%OZ; //the mod probably is not necessary
      
      cost += x*num*bl + bags*bs;
    }

    cout << cost << endl;
  }

}
