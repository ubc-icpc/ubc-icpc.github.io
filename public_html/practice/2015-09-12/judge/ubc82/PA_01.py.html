inputNumber = int(raw_input())

counter = 0
teamArrayArray = []
teamStringArray = []
while counter < inputNumber:
    team = raw_input()
    teamStringArray.append(team)
    teamArray = []

    inputCounter = 0
    inputBuffer = ""
    
    for i in team:
        if inputCounter < 2:
            inputBuffer += i
        else:
            inputCounter = -1
            teamArray.append(int(inputBuffer))
            inputBuffer = ""
        inputCounter += 1

    teamArrayArray.append(teamArray)
    
    counter += 1
