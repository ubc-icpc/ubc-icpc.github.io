n = int(raw_input())
for x in range(0,n):
    line = raw_input()
    listIn = [int(i) for i in line.split()]
    if 17 in listIn and 18 in listIn:
        s = 'both'
    elif 17 in listIn:
        s = 'zack'
    elif 18 in listIn:
        s = 'mack'
    else:
        s = 'none'
    print str(line) + "\n" + s
