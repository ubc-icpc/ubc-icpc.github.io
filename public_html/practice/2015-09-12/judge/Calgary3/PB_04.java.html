import java.util.Scanner;

public class B {

	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);

		int n = input.nextInt();
		int usa[] = new int[3];
		int russia[] = new int[3];

		for (int count = 0; count < n; count++) {
			int usaCount = 0;
			int russiaCount = 0;
			for (int i = 0; i < 6; i++) {
				if (i < 3) {
					usa[i] = input.nextInt();
					usaCount += usa[i];
					System.out.print(usa[i] + " ");
				} else {
					russia[i - 3] = input.nextInt();
					russiaCount += russia[i - 3];
					if (i < 5) {
						System.out.print(russia[i - 3] + " ");
					} else {
						System.out.print(russia[i - 3]);
					}
				}
			}
			
			System.out.println();

			int amount = 0;
			String type = "";
			
			if (usaCount > russiaCount) {
				amount++;
				type = "count";
			}

			if (usa[0] > russia[0]){
				amount++;
				type="color";
			}else if (usa[0] == russia[0]){
				if (usa[1] > russia [1]){
					amount++;
					type="color";
				}else if (usa[1] == russia [1]){
					if (usa[2] > russia[2]){
						amount++;
						type = "color";
					}
				}
				
			}

			if (amount == 2) {
				System.out.println("both"+"\n");
			} else if (amount == 0) {
				System.out.println("none"+"\n");
			} else {
				System.out.println(type+"\n");
			}

		}

	}

}
