import java.util.*;

public class A {

	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);

		int n = input.nextInt();
		int players[] = new int[10];
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < 10; j++) {
				players[j] = input.nextInt();
			}
			for (int j = 0; j < 10; j++) {
				if (j < 9) {
					System.out.print(players[j] + " ");
				}else{
					System.out.print(players[j]);
				}
			}
			System.out.println();
			String name = "";
			int amount = 0;
			for (int j = 0; j < 10; j++) {
				if (players[j] == 18) {
					name = "mack";
					amount++;
				} else if (players[j] == 17) {
					name = "zack";
					amount++;
				}
			}

			if (amount == 2) {
				System.out.println("both");
			} else if (amount == 0) {
				System.out.println("none");
			} else {
				System.out.println(name);
			}

		}
	}

}
