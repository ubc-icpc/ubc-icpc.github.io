#include <iostream>
using namespace std;

string a[4] = {"none", "count", "color", "both"};

int m[6];

int main() {
  int t; cin >> t;
  while (t--) {
    int x = 0;
    for (int i = 0; i < 6; i++) {
      cin >> m[i];
      if (i)
        cout << " ";
      cout << m[i];
    }
    if (m[0]+m[1]+m[2] > m[3]+m[4]+m[5])
      x |= 1;
    if ((m[0] > m[3]) || (m[0] == m[3] && m[1] > m[4]) ||
        (m[0] == m[3] && m[1] == m[4] && m[2] > m[5]))
      x |= 2;
    cout << "\n" << a[x] << "\n\n";
  }
  return 0;
}
