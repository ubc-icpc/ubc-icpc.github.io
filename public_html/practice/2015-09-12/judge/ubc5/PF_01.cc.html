#include <iostream>
using namespace std;

typedef long long ll;

ll never_trust_cmath_pow(ll b, ll e) {
  ll r = 1;
  while (e) {
    e--;
    r *= b;
  }
  return r;
}

inline bool is_vowel(char c) {
  return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' || c == 'y';
}

int main() {
  int n; cin >> n;
  for (int N = 1; N <= n; N++) {
    cout << "String #" << N << ": ";
    string s; cin >> s;
    ll a1 = 1;
    for (int i = 0; i < s.size(); i++) {
      char c = s[i];
      if (i % 2 == is_vowel(c) || c == '?') {
        if (c == '?')
          a1 *= (i % 2 ? 6 : 20);
      }
      else {
        a1 = 0;
        break;
      }
    }
    ll a2 = 1;
    for (int i = 0; i < s.size(); i++) {
      char c = s[i];
      if (i % 2 != is_vowel(c) || c == '?') {
        if (c == '?')
          a2 *= (i % 2 ? 20 : 6);
      }
      else {
        a2 = 0;
        break;
      }
    }
    cout << a1+a2 << "\n\n";
  }
  return 0;
}
