#include <iostream>
using namespace std;

string a[4] = {"none", "mack", "zack", "both"};

int main() {
  int t; cin >> t;
  while (t--) {
    int x = 0;
    for (int i = 0; i < 10; i++) {
      int d; cin >> d;
      if (i)
        cout << " ";
      cout << d;
      if (d == 18)
        x |= 1;
      else if (d == 17)
        x |= 2;
    }
    cout << "\n" << a[x] << "\n\n";
  }
  return 0;
}
