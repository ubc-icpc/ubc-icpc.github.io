#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

typedef long double ld;

const ld pi = 3.1415926535897932384626433832795L;

int main() {
  cout << fixed << setprecision(2);
  int n; cin >> n;
  while (n--) {
    ld l, r; cin >> l >> r;
    if (r >= l/sqrt(2.0L)) {
      cout << l*l << "\n";
      continue;
    }
    else if (r < l/2) {
      cout << pi*r*r << "\n";
      continue;
    }
    cout << pi*r*r-(r*r*acos(l/2/r)-sqrt(r*r-l*l/4)*l/2)*4 << "\n";
  }
  return 0;
}
