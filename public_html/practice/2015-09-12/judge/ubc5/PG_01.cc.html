#include <iostream>
using namespace std;

int main() {
  int g; cin >> g;
  for (int G = 1; G <= g; G++) {
    int d, n;
    cin >> d >> n;
    cout << "Grid #" << G << ": ";
    if (d > (n-1)*(n-1)+1) {
      cout << "impossible\n\n";
      continue;
    }
    cout << d*2*(n-1) << "\n\n";
  }
  return 0;
}
