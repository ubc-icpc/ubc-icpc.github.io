#include <bits/stdc++.h>

using namespace std;

typedef long double ld;

ld PI = 3.14159265358979L;

int main() {
  int N; cin >> N;
  while (N--) {
    ld s, r;
    cin >> s >> r;
    if (2*s*s <= 4*r*r) cout << setprecision(2) << fixed << s*s << endl;
    else if (s >= 2*r) cout << setprecision(2) << fixed << PI*r*r << endl;
    else {
      ld theta = 2*acos(s/2/r);
      ld base = 2*sqrtl(r*r - s*s/4);
      ld tri = base * s/4;
      ld diff = r*r*theta/2 - tri;
      ld area = PI*r*r - 4*diff;
      cout << setprecision(2) << fixed << area << endl;
    }
  }
  return 0;
}
