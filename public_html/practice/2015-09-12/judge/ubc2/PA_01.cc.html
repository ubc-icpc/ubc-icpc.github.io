#include <bits/stdc++.h>

using namespace std;

int main() {
  int N; cin >> N;
  while(N--) {
    bool zack = false, mack = false;
    for (int i = 0; i < 10; ++i) {
      int num; cin >> num; cout << num;
      if (i < 9) cout << " ";
      if (num == 17) zack = true;
      else if (num == 18) mack = true;
    }
    cout << endl;
    if (zack && mack) cout << "both";
    else if (zack) cout << "zack";
    else if (mack) cout << "mack";
    else cout << "none";
    cout << endl << endl;
  }
  return 0;
}
