#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main() {
  int N; cin >> N;
  while (N--) {
    ll d, l, s; cin >> d >> l >> s;
    ll clsf, minpl, minps; cin >> clsf >> minpl >> minps;
    ll cssf = clsf;
    ll cost = 0, xs = 0;
    for (int i = 1; i < d; ++i) {
      ll c, pl, ps; cin >> c >> pl >> ps;
      if (pl < minpl) {
        cost += clsf * l * minpl;
        clsf = 0;
        minpl = pl;
      }
      if (ps < minps) {
        ll ounces_needed = cssf * s - xs;
        ll sugar_needed = (ounces_needed + 5*16 - 1) / (5*16);
        xs = sugar_needed*5*16 - ounces_needed;
        cost += sugar_needed * minps;
        cssf = 0;
        minps = ps;
      }
      clsf += c; cssf += c;
    }
    cost += clsf * l * minpl;
    ll ounces_needed = cssf * s - xs;
    ll sugar_needed = (ounces_needed + 5*16 - 1) / (5*16);
    cost += sugar_needed * minps;
    cout << cost << endl;
  }
  return 0;
}
