#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

bool isVowel(char c) {
  return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' || c == 'y';
}

ll ways(string s, bool vowel) {
  ll total = 1;
  for (int i = 0; i < s.size(); ++i) {
    if (s[i] == '?') {
      if (vowel) total *= 6;
      else total *= 20;
    }
    else {
      if (vowel != isVowel(s[i])) return 0;
    }
    vowel = !vowel;
  }
  return total;
}

int main() {
  int T; cin >> T;
  for (int t = 1; t <= T; ++t) {
    string s; cin >> s;
    ll total = 0;
    if (s[0] == '?') total = ways(s, false) + ways(s, true);
    else total = ways(s, isVowel(s[0]));
    cout << "String #" << t << ": " << total << endl << endl;
  }
  return 0;
}
