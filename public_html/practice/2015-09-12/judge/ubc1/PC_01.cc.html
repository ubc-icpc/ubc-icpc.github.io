#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int main() {
    int t; cin >> t;
    for (int ca = 1; ca <= t; ca++) {
        int n, m; cin >> n >> m;
        cout << "Practice #" << ca << ": " << n << " " << m << endl;
        int k; cin >> k;
        for (int i = 0; i < k; i++) {
            int x; cin >> x;
            while (m <= x) m *= 2;
            m -= x;
            cout << x << " " << m << endl;
        }
        cout << endl;
    }
	return 0;
}
