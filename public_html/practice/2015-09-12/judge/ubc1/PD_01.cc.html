#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int t; cin >> t;
    while (t--) {
        int d, x, s; cin >> d >> x >> s;
        int bl = 100000, bs = 1000000;
        int cl = 0, cs = 0, cost = 0;
        for (int i = 0; i < d; i++) {
            int c, pl, ps; cin >> c >> pl >> ps;
            bl = min(bl, pl);
            bs = min(bs, ps);
            
            int nl = x * c, ns = s * c;
            if (cl < nl) {;
                cost += (nl-cl)*bl;
                cl += (nl-cl);
                
            }
            
            if (cs < ns) {
                int nb = (ns-cs+79)/80;
                cost += nb * bs;
                cs += nb * 80;
            }
            cl -= c * x;
            cs -= c * s;
        }
        cout << cost << endl;
    }
	return 0;
}
