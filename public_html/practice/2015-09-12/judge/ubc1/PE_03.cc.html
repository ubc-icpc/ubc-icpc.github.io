#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;

const ld pi = acos(-1.0L), eps = 1e-8;
int main() {
    int t; cin >> t;
    while (t--) {
        cout << fixed << setprecision(2);
        ld s, r; cin >> s >> r;
        if (r >= s/sqrt(2) - eps) {
            cout << s*s << endl;
            continue;
        } else if (r <= s/2 + eps) {
            cout << pi * r * r << endl;
            continue;
        } else {        
            ld t = 2*acos(s/(2*r));
            ld A = 2*r*r*sin(t) + (pi-2*t)*r*r;
            cout << A << endl;
        }
    }
	return 0;
}
