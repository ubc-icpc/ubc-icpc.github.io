#include <bits/stdtr1c++.h>

using namespace std;

int adj[505][505];
int main() {
    ios::sync_with_stdio(0);
    int t; cin >> t;
    while (t--) {
        int n; cin >> n;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                cin >> adj[i][j];
            }
        }
        
        list<int> path({0});
        for (int k = 1; k < n; k++) {
            auto it = path.begin();
            while (true) {
                if (adj[k][*it]) {
                    path.insert(it, k);
                    break;
                } else {
                    ++it;
                    if (it == path.end()) {
                        path.insert(it, k);
                        break;
                    }
                }
            }
        }
        for (int v : path) cout << v + 1 << " ";
        cout << endl;
    }
	return 0;
}
