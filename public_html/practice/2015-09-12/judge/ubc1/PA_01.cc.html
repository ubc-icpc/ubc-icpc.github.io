#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int t; cin >> t;
    while (t--) {
        bool has17 = false, has18 = false;
        for (int i = 0; i < 10; i++) {
            int x; cin >> x;
            has17 = has17 || (x == 17);
            has18 = has18 || (x == 18);
            cout << x << " ";
        }
        cout << endl;
        
        if (has17 && has18) {
            cout << "both" << endl;
        } else if (has17) {
            cout << "zack" << endl;
        } else if (has18) {
            cout << "mack" << endl;
        } else {
            cout << "none" << endl;
        }
        cout << endl;
    }
	return 0;
}
