#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int main() {
    int t; cin >> t;
    for (int ca = 1; ca <= t; ca++) {
        int n, d; cin >> d >> n;
        cout << "Grid #" << ca << ": ";
        if ((n-1)*(n-1)+1 >= d) cout << d * (2*n-2) << endl;
        else cout << "impossible" << endl;
        cout << endl;
    }
	return 0;
}
