#include <bits/stdtr1c++.h>

using namespace std;

typedef tuple<int, int, int> tup;
int main() {
    int t; cin >> t;
    while (t--) {
        int a, b, c, d, e, f;
        cin >> a >> b >> c >> d >> e >> f;
        cout << a << " " << b << " " << c << " " << d << " " << e << " " << f << endl;
        
        bool bcnt = false, bcol = false;
        if (a+b+c > d+e+f) bcnt = true;
        if (tup(a,b,c) > tup(d,e,f)) bcol = true;
        
        if (bcnt && bcol) {
            cout << "both" << endl;
        } else if (bcnt) {
            cout << "count" << endl;
        } else if (bcol) {
            cout << "color" << endl;
        } else {
            cout << "none" << endl;
        }
        cout << endl;
    }
	return 0;
}
