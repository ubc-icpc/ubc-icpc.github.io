#include<iostream>
#include<vector>

using namespace std;

bool winColor(int g1, int s1, int b1, int g2, int s2, int b2) {
    if (g1 > g2) {
        return true;
    } else if (g1 == g2) {
        if (s1 > s2) {
            return true;
        } else if (s1 == s2) {
            if (b1 > b2) {
                return true;
            } else {
                return false;
            }
        } else {
            return false;
        }
    } else {
        return false;
    }
}

void solveLine() {
    int USAGold, USASilver, USABronze,
        RGold, RSilver, RBronze;
    cin >> USAGold >> USASilver >> USABronze >> RGold >> RSilver >> RBronze;

    cout << USAGold << " " <<
        USASilver << " " <<
        USABronze << " " <<
        RGold << " " <<
        RSilver << " " <<
        RBronze << "\n";

    bool USAColor = winColor(USAGold, USASilver, USABronze, RGold, RSilver, RBronze);
    bool USACount = USAGold + USASilver + USABronze > RGold + RSilver + RBronze;

    if (USAColor && USACount) {
        cout << "both\n\n";
    } else if (USAColor && !USACount) {
        cout << "color\n\n";
    } else if (!USAColor && USACount) {
        cout << "count\n\n";
    } else {
        cout << "none\n\n";
    }

}

int main(int argc, char* argv[]) {
    // Setup
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int numSets;
    cin >> numSets;

    while (numSets--) {
        solveLine();
        cout << endl;
    }

    return 0;
}
