#include<iostream>
#include<vector>

using namespace std;
const int ZACKS_LOC = 17;
const int MACKS_LOC = 18;

void solveLine() {
    vector<bool> present(200, false);
    int player;
    for (int i = 0; i < 9; i++) {
        cin >> player;
        present[player] = true;
        cout << player << " ";
    }

    cin >> player;
    present[player] = true;
    cout << player << endl;

    if (present[ZACKS_LOC] && present[MACKS_LOC]) {
        cout << "both" << endl;
    } else if (present[ZACKS_LOC] && !present[MACKS_LOC]) {
        cout << "zack" << endl;
    } else if (!present[ZACKS_LOC] && present[MACKS_LOC]) {
        cout << "mack" << endl;
    } else {
        cout << "none" << endl;
    }

}

int main(int argc, char* argv[]) {
    // Setup
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int numSets;
    cin >> numSets;

    while (numSets--) {
        solveLine();
        cout << endl;
    }

    return 0;
}
