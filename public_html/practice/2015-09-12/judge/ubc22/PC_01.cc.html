#include<bits/stdtr1c++.h>
using namespace std;

int T, N, M, O, P;

int main () {
    cin >> T;
    for (int t = 1; t <= T; ++t) {
        cin >> N >> M >> O;
        printf("Practice #%d: %d %d\n", t, N, M);
        while (O--) {
            cin >> P;
            while (M <= P) M *= 2;
            M -= P;
            printf("%d %d\n", P, M);
        }
        printf("\n");
    }
    return 0;
}
