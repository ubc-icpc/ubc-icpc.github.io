#include<bits/stdtr1c++.h>
using namespace std;

int N, A[6], K;

int main () {
    cin >> N;
    while (N--) {
        for (int i = 0; i < 6; ++i) {
            cin >> A[i];
            cout << A[i] << " ";
        }
        cout << "\n";
        K = 0;
        if (A[0] + A[1] + A[2] > A[3] + A[4] + A[5])
            K |= 1;
        if (A[0] < A[3] || (A[0] == A[3] && A[1] < A[4]) || (A[0] == A[3] && A[1] == A[4] && A[2] <= A[5])){}
        else K |= 2;
        switch (K) {
            case 0: cout << "none"; break;
            case 1: cout << "count"; break;
            case 2: cout << "color"; break;
            case 3: cout << "both"; break;
        }
        cout << "\n\n";
    }
    return 0;
}
