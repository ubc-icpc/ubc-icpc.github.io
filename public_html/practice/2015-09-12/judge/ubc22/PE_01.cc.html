#include<bits/stdtr1c++.h>
using namespace std;

int N;
const double PI = acos(-1.0);
double L, R, A;

int main () {
	cin >> N;
	while (N--) {
		cin >> L >> R;
		if (R * 2 <= L)
			A = PI * R * R;
		else if (R * sqrt(2) >= L)
			A = L * L;
		else {
			double t = 2 * acos((L / 2) / R);
			A = R * R * (2 * sin(t) + PI - 2 * t);
		}
		cout << fixed << setprecision(2) << A << endl;
	}
	return 0;
}
