#include<bits/stdtr1c++.h>
using namespace std;

long long T, D, X, S, C, PX, PS, MX, MS, res, R;

int main () {
    cin >> T;
    while (T--) {
        MX = MS = numeric_limits<long long>::max();
        res = R = 0;
        cin >> D >> X >> S;
        while (D--) {
            cin >> C >> PX >> PS;
            MX = min(MX, PX);
            MS = min(MS, PS);
            res += MX * C * X;
            R -= S * C;
            int n = (max(0ll, -R) + 79) / 80;
            R += 80 * n;
            res += MS * n;
        }
        cout << res << endl;
    }
    return 0;
}
