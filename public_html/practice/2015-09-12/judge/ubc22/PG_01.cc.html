#include<bits/stdtr1c++.h>
using namespace std;

int N, A, B;

int main() {
    cin >> N;
    for (int n = 1; n <= N; ++n) {
        cin >> A >> B;
        cout << "Grid #" << n << ": ";
        if (A - 1 <= (B - 1) * (B - 1))
            cout << A * (B - 1) * 2;
        else
            cout << "impossible";
        cout << "\n\n";
    }
    return 0;
}
