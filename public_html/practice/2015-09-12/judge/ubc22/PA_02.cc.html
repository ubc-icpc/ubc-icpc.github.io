#include<bits/stdtr1c++.h>
using namespace std;

int N, M, K;

int main () {
    cin >> N;
    while (N--) {
        K = 0;
        for (int i = 0; i < 10; ++i) {
            cin >> M;
            if (M == 18) K |= 2;
            if (M == 17) K |= 1;
            cout << M << " ";
        }
        cout << "\n";
        switch (K) {
            case 0: cout << "none"; break;
            case 1: cout << "zack"; break;
            case 2: cout << "mack"; break;
            case 3: cout << "both"; break;
        }
        cout << "\n\n";
    }
    return 0;
}
