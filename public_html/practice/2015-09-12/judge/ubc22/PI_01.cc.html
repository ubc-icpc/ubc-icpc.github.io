#include<bits/stdtr1c++.h>
using namespace std;

int T, N, M[105][105];
list<int> R;

int main () {
    cin >> T;
    while (T--) {
        cin >> N;
        for (int i = 0; i < N; ++i)
            for (int j = 0; j < N; ++j)
                cin >> M[i][j];
        R = list<int>();
        R.push_back(0);
        for (int i = 1; i < N; ++i) {
            if (M[i][*R.begin()]) R.push_front(i);
            else if (M[*(--R.end())][i]) R.push_back(i);
            else {
                list<int>::iterator it;
                for (it = R.begin(); M[*it][i]; ++it);
                R.insert(it, i);
            }
        }
        for (list<int>::iterator it = R.begin(); it != R.end(); ++it)
            cout << *it + 1 << " ";
        cout << "\n";
    }
    return 0;
}
