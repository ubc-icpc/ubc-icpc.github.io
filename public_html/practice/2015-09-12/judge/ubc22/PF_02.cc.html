#include<bits/stdtr1c++.h>
using namespace std;

int T, B[105];
long long res;
string S;

int isv(int i) {
    switch (S[i]) {
        case '?':
            return 0;
        case 'a':
        case 'e':
        case 'i':
        case 'o':
        case 'u':
        case 'y':
            return 1;
        default:
            return 2;
    }
}

int main () {
    cin >> T;
    for (int t = 1; t <= T; ++t) {
        res = 0;
        cin >> S;
        int start = 0;
        for (int i = 0; i < S.size(); ++i) {
            B[i] = isv(i);
            if (B[i]) start = (B[i] - 1 + i) % 2 + 1;
        }
        if (start == 0) {
            long long temp1 = 1;
            long long temp2 = 1;
            for (int i = 0; i < S.size(); ++i) {
                temp1 *= 6 + 14 * (i % 2);
                temp2 *= 20 - 14 * (i % 2);
            }
            res = temp1 + temp2;
        } else {
            res = 1;
            for (int i = 0; i < S.size(); ++i) {
                if (B[i] == 0) res *= 6 + 14 * ((start-1+i) % 2);
                else if (B[i] != ((start-1+i) % 2) + 1) res = 0;
            }
        }
        cout << "String #" << t << ": " << res << "\n\n";
    }
    return 0;
}
