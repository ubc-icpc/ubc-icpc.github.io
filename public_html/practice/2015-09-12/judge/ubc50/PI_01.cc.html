#include<iostream>
#include<list>
using namespace std;

int mat[505][505];

int main() {
  int test; cin >> test;
  while (test--) {
    int v; cin >> v;
    for (int i=1; i<=v; i++) {
      for (int j=1; j<=v; j++) {
	cin >> mat[i][j];
      }
    }
    list<int> res;
    // say we start from 1
    res.insert(res.begin(),1);
    for (int i=2; i<=v; i++) {
      auto it=res.begin();
      while (true) {
	// if we can go from i to current node
	// add i before current node
	// if we reach the end of the answer, add this node
	// to the end, guaranteed that mat[*it-1][i]=1
	if (it==res.end() || mat[i][*it]) {
	  res.insert(it,i);
	  break;
	}
	// otherwise move onto next node in our answer so far
	// loop again to see can we go from i to this node
	else {
	  ++it;
	}
      }
    }
    for (auto x : res)
      cout << x << " ";
    cout << endl;
  }
}
