#include<iostream>
using namespace std;

int main() {
  int test; cin >> test;
  for (int i=1; i<=test; i++) {
    cout << "Grid #" << i << ": ";
    int d,n; cin >> d >> n;
    if (d >1+(n-1)*(n-1))
      cout << "impossible" << endl;
    else
      cout << d*(n-1)*2 << endl;
    cout << endl;
  }
}
