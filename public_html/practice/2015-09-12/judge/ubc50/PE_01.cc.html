#include<iostream>
#include<iomanip>
#include<cmath>
using namespace std;

const double PI = 3.14159265358979;

int main() {
  int test; cin >> test;
  while (test--) {
    double s,r; cin >> s >> r;
    double diag=2*s*s;
    if (r<=(s/2)) {
      cout << fixed << setprecision(2) << PI*r*r << endl;
    }
    else if (r*r>=(s*s*0.5)) {
      cout << fixed << setprecision(2) << s*s << endl;
    }
    else {
      double deg=acos(s/(2*r));
      deg*=2;
      double x=(2*PI-4*deg)/4;
      double area_x=r*r*0.5*x;
      double area_t=0.5*r*r*sin(deg);
      double ans = 4*(area_x+area_t);
      cout << fixed << setprecision(2) << ans << endl;
    }
  }
}
