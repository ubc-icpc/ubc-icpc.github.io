#include<iostream>
using namespace std;

int t;

int main() {
  cin >> t;
  while (t--) {
    int arr[6];
    int us[3], russia[3];
    int ucnt=0, rcnt=0;
    bool col=false, cnt=false;
    
    for (int i=0; i<6; i++) cin >> arr[i];
    for (int i=0; i<3; i++) { us[i]=arr[i]; ucnt+=us[i]; }
    for (int i=0; i<3; i++) { russia[i]=arr[i+3]; rcnt+=russia[i];}
    if (ucnt>rcnt) cnt=true;
    if (us[0]>russia[0]) col=true;
    else if (us[0]==russia[0] && us[1]>russia[1]) col=true;
    else if (us[0]==russia[0] && us[1]==russia[1] && us[2]>russia[2]) col=true;
    for (int i=0; i<6; i++) cout << arr[i] << " ";
    cout << endl;
    if (col&cnt) cout << "both" << endl;
    else if (col) cout << "color" << endl;
    else if (cnt) cout << "count" << endl;
    else cout << "none" << endl;
  }
}
