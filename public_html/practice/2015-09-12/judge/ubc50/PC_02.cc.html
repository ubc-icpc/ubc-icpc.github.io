#include<iostream>
#include<vector>
using namespace std;

int main() {
  int t; cin >> t;
  for (int test=1; test<=t; test++) {
    int s=0,b=0,m=0;
    cin >> s >> b >> m;
    vector<int> g;
    for (int i=0; i<m; i++) {
      int x; cin >> x; g.push_back(x);
    }
    cout << "Practice #" << test << ": " << s << " " << b << endl;
    for (int i=0; i<m; i++) {
      while (b<=g[i]) b*=2;
      b-=g[i];
      cout << g[i] << " " << b << endl;
    }
    cout << endl;
  }
}
