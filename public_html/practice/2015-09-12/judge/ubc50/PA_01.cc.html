#include<iostream>
using namespace std;

int main() {
  int t; cin >> t;
  while (t--) {
    int arr[10];
    bool zack=false, mack=false;
    for (int i=0; i<10; i++) {
      cin >> arr[i];
      if (arr[i]==18) mack=true;
      if (arr[i]==17) zack=true;
    }
    for (int i=0; i<10; i++)
      cout << arr[i] << " ";
    cout << endl;
    if (mack&zack) cout << "both" << endl;
    else if (mack) cout << "mack" << endl;
    else if (zack) cout << "zack" << endl;
    else  cout << "none" << endl;
  }
}
