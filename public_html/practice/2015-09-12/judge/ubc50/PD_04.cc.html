#include<iostream>
#include<vector>
#include<limits>
#include<algorithm>
#include<cmath>
using namespace std;
typedef long long ll;

const int INF=numeric_limits<int>::max();

int main() {
  int test; cin >> test;
  while (test--) {
    ll ans=0;
    int ml=INF, ms=INF;
    int cur=0;
    int cup[1005], lemon[1005], sugar[1005];
    int d,x,s; cin >> d >> x >> s;
    for (int i=0; i<d; i++) { cin >> cup[i] >> lemon[i] >> sugar[i];}
    for (int i=0; i<d; i++) {
      ml=min(ml,lemon[i]);
      ms=min(ms,sugar[i]);
      int numl=cup[i]*x;
      int nums=cup[i]*s;
      ans+=(numl*ml);
      int req=nums-cur;
      int bags=ceil(static_cast<float>(req)/80.0);
      int tmp=0;
      tmp=max(0,bags);
      ans+=(ms*tmp);
      //cur += max(0,(tmp*80-nums));
      cur+=(tmp*80-nums);
    }
    cout << ans << endl;
  }
}
