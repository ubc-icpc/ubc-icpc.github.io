#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	while (n--)
	{
		bool mack = false;
		bool zack = false;
		for (int i = 0; i < 10; i++)
		{
			int num;
			cin >> num;
			cout << num;
			if (i != 9)
			{
				cout << " ";
			}
			else
			{
				cout << endl;
			}
			if (num == 18)
			{
				mack = true;
			}
			else if (num == 17)
			{
				zack = true;
			}
		}
		if (mack && zack)
		{
			cout << "both" << endl << endl;
		}
		else if (mack)
		{
			cout << "mack" << endl << endl;
		}
		else if (zack)
		{
			cout << "zack" << endl << endl;
		}
		else
		{
			cout << "none" << endl << endl;
		}
	}
	return 0;
}
