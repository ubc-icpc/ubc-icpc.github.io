#include <iostream>
#include <cstring>
#include <climits>

using namespace std;

int main()
{
	int n;
	cin >> n;
	while (n--)
	{
		int mins = INT_MAX;
		int minl = INT_MAX;
		int total = 0;
		int sugar_count = 0;
		//d = days x =lemons s = ounc of sgar 5lb= 80 ounc
		int d, x, s;
		cin >> d >> x >> s;
		for (int i = 0; i < d; i++)
		{
			//cups, price lem, price sug
			int c, pl, ps;
			cin >> c >> pl >> ps;
			if (pl < minl)
			{
				minl = pl;
			}
			if (ps < mins)
			{
				mins = ps;
			}
			//sugar
			total += x*c*minl;
			while (c*s > sugar_count)
			{
				total += mins;
				sugar_count += 80;
			}
			sugar_count -= c*s;
		}
		cout << total << endl;
	}
	return 0;
}
