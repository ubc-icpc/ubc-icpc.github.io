#include <iostream>
#include <list>

using namespace std;

bool arr[505][505];

int main(void)
{
	int n;
	cin >> n;
	while (n--)
	{
		int nodes;
		cin >> nodes;
		for (int j = 0; j < nodes; j++)
		{
			for (int i = 0; i < nodes; i++)
			{
				cin >> arr[j][i];
			}
		}
		list<int> path;
		path.insert(path.begin(), 0);
		for (int i = 1; i < nodes; i++)
		{
			auto it = path.begin();
			for (int j = 0; j < nodes; j++)
			{
				if (arr[i][*it])
				{
					path.insert(it, i);
					break;
				}
				it++;
				if (it == path.end())
				{
					path.insert(it, i);
					break;
				}
			}
		}
		for (auto node : path)
		{
			cout << node + 1 << " ";
		}
		cout << endl;
	}
	return 0;
}
