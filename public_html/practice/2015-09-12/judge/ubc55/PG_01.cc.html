#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int count = 0;
	while (n--)
	{
		count++;
		int d, grid;
		cin >> d >> grid;
		if (d-1 <= (grid - 1)*(grid - 1))
		{
			cout << "Grid #" << count << ": " << 2 * (grid-1) * d << endl << endl;
		}
		else
		{
			cout << "Grid #" << count << ": " << "impossible" << endl << endl;
		}
	}
}
