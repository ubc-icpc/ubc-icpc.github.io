#include <iostream>
#include <string>
#include <cstring>

using namespace std;

typedef long long ll;

int is_vowel(char a)
{
	if (a == 'a' || a == 'e' || a == 'i' || a == 'o' || a == 'u' || a == 'y')
	{
		return 1;
	}
	else if (a == '?')
	{
		return -1;
	}
	else
		return 0;
}
int main()
{
	int n;
	int count = 0;
	cin >> n;
	while (n--)
	{
		int start = -1;
		count++;
		string word;
		cin >> word;
		cout << "String #" << count << ": ";
		for (int i = 0; i < word.length(); i++)
		{
			int b = is_vowel(word[i]);
			if (b == 1)
			{
				if (i % 2 == 0)
				{
					start = 1;
					break;
				}
				else
				{
					start = 0;
					break;
				}
			}
			else if (b == 0)
			{
				if (i % 2 == 0)
				{
					start = 0;
					break;
				}
				else
				{
					start = 1;
					break;
				}
			}
		}
		if (start == -1)
		{
			long long total1 = 1, total2 = 1;
			for (int i = 0; i < word.length(); i++)
			{
				if (i % 2 == 0)
				{
					total1 *= 6;
					total2 *= 20;
				}
				else
				{
					total1 *= 20;
					total2 *= 6;
				}
			}
			cout << total1 + total2 << endl << endl;
		}
		//start with non vowel
		else if (start == 0)
		{
			ll total = 1;
			for (int i = 0; i < word.length(); i++)
			{
				int b = is_vowel(word[i]);
				int remainder = i % 2;
				if (b == 1 && remainder == 0)
				{
					total = 0;
					break;
				}
				if (b == 0 && remainder == 1)
				{
					total = 0;
					break;
				}
				else
				{
					if (b == -1 && remainder == 1)
					{
						total *= 6;
					}
					else if (b == -1 && remainder == 0)
					{
						total *= 20;
					}
				}
			}
			cout << total << endl << endl;
		}
		else
		{
			ll total = 1;
			for (int i = 0; i < word.length(); i++)
			{
				int b = is_vowel(word[i]);
				int remainder = i % 2;
				if (b == 1 && remainder == 1)
				{
					total = 0;
					break;
				}
				if (b == 0 && remainder == 0)
				{
					total = 0;
					break;
				}
				else
				{
					if (b == -1 && i%2 == 0)
					{
						total *= 6;
					}
					else if (b == -1 && i%2 == 1)
					{
						total *= 20;
					}
				}
			}
			cout << total << endl << endl;
		}
	}
	return 0;
}
