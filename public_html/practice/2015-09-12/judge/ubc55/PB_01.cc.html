#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	while (n--)
	{
		bool count = false;
		bool color = false;
		int arr[6];
		for (int i = 0; i < 6; i++)
		{
			int	m;
			cin >> m;
			cout << m;
			if (i == 5)
			{
				cout << endl;
			}
			else
			{
				cout << " ";
			}
			arr[i] = m;
		}
		int utotal = arr[0] + arr[1] + arr[2];
		int rtotal = arr[3] + arr[4] + arr[5];
		if (utotal > rtotal)
			count = true;
		for (int i = 0; i < 3; i++)
		{
			if (arr[i] > arr[i + 3])
			{
				color = true;
				break;
			}
			else if (arr[i] < arr[i + 3])
			{
				break;
			}
		}
		if (color && count)
		{
			cout << "both";
		}
		else if (color)
		{
			cout << "color";
		}
		else if (count)
		{
			cout << "count";
		}
		else
		{
			cout << "none";
		}
		cout << endl << endl;
	}
	return 0;
}
