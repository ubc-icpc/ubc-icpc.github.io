#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int count = 0;
	while (n--)
	{
		count++;
		int s, b;
		cin >> s >> b;
		cout << "Practice #" << count << ": " << s << " " << b << endl;
		int g; cin >> g;
		while (g--)
		{
			int num;
			cin >> num;
			while (num >= b)
			{
				b *= 2;
			}
			b -= num;
			cout << num << " " << b << endl;
		}
		cout << endl;
	}
}
