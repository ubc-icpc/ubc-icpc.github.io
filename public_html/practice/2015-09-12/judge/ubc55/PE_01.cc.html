#include <iostream>
#include <cmath>

using namespace std;

#define pi 3.14159265358979

int main()
{
	int n; 
	cin >> n;
	cout.precision(2);
	while (n--)
	{
		double s, r;
		cin >> s >> r;
		if (s/2 > r)
		{
			cout << fixed << pi*r*r << endl;
		}
		else if (r >= s* sqrt(2) / 2)
		{
			cout << fixed << s*s << endl;
		}
		else
		{
			double h = r - s/2;
			cout << fixed << r*r*pi - 4 *( r*r*acos((r - h) / r) - (r - h)*(sqrt(2 * r*h - h*h))) << endl;
		}
	}
	return 0;
}
