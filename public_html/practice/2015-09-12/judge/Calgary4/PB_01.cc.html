#include <iostream>

using namespace std;

int main(){
  int n; cin >> n;
    while(n--)
      {
	int medal[6];
	int countU, countR; countU = 0; countR =0;
	for(int i = 0; i < 6; i++){
	  cin >> medal[i];
	  cout << medal[i] << " ";
	  if(i <3)
	    countU += medal[i];
	  else
	    countR += medal[i];
	}
	cout << endl;

	bool count = false;
	if(countU > countR)
	  count = true;

	bool color = false;
	if(medal[0] > medal[3])
	  color = true;
	else if(medal[0] == medal[3] && medal[1] > medal[4])
	  color = true;
	else if(medal[0] == medal[3] && medal[1] == medal[4] && medal[2] > medal[5])
	  color = true;

	if(color && count){
	  cout << "both\n\n";
	  continue;
	}else if(color){
	  cout << "color\n\n";
	  continue;
	}else if (count){
	  cout << "count\n\n";
	  continue;
	}else
	  cout << "none\n\n";

      }

}
