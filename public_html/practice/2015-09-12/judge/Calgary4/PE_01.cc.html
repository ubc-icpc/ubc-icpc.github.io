#include<iostream>
#include<cmath>

using namespace std;

#define PI 3.14159265358979

int main(){
  int n;
  cin >> n;
  while(n--){
    double s, r; cin >> s >> r;
    if (2*r <= s){ // circle < square
      printf("%.2f\n", PI *r *r);
    }else if(sqrt(s/2*s/2 + s/2*s/2) < r){ // square is inside circle
      printf("%.2f\n", s*s);
    }else{
      double areaTri = sqrt((r*r) - ((s/2)*(s/2)))  * (s/2);
     
      double angleR = acos((s/2) /r) *2;

      double areaCir = (PI * r *r) * (angleR /(2*PI));
      double extra = 4 *(areaCir -areaTri);
      printf("%.2f\n",( r *r * PI) - extra);
    }
  }
}
