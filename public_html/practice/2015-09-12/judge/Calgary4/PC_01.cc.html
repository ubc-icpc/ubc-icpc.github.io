#include<iostream>

using namespace std;

int main(){
  int n, index = 1; cin >> n;
  while(n--){
    int s, b, m; cin >> s >> b >> m;
    int member[m];
    for(int i = 0; i < m; i++){
      cin >> member[i];
      
    }

    cout << "Practice #" << index << ": " << s << " "<< b << endl;
    for(int i = 0; i < m; i++){
      cout << member[i] << " ";
      while(b <= member[i]){
	b *= 2;
      }
      b -= member[i];
      cout << b << "\n";
    }
    cout << endl;
    index++;
  }
}
