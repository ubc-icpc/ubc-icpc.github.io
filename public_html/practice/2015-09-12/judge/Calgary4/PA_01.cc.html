#include <iostream>

using namespace std;

int main(){
  int n; cin >> n;
  while(n--){
    int player[10] = {0};
    bool mack, zack; mack = false; zack = false;
    for(int i = 0; i<10; i++){
      cin >> player[i];
      if (player[i] == 18)
	mack = true;
      if (player[i] == 17)
	zack = true;
    }

    for(int i : player){
      cout << i << " ";
    }
    cout << endl;

    if(mack && zack){
      cout << "both\n"<<endl;
      continue;
    }
    if(mack){
      cout << "mack\n"<<endl;
      continue;
    }
    if(zack){
      cout << "zack\n"<<endl;
      continue;
    }
    cout << "none\n"<<endl;
       
  }

}
