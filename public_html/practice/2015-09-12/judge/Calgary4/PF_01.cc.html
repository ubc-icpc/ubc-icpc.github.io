#include <iostream>
#include<string>

using namespace std;

signed long long int isBalanced(const string& S){
  string vowels = "aeiouy";
  signed long long int balanced = 1;
  for(int i = 0; i < S.length(); i++){
    if(i % 2 == 0 && S[i] == '?'){
      balanced *= 6;
    }else if (i % 2 == 1 && S[i] == '?' )
      balanced *= 20;
    else if((i%2 == 0 && vowels.find(S[i]) == string::npos) ||
	    (i%2 == 1 && vowels.find(S[i]) != string::npos)){
      balanced *= 0;
      break;
    }
  }
  signed long long int balance = 1;
  for(int i = 0; i < S.length(); i++){
    if(i % 2 == 0 && S[i] == '?'){
      balance *= 20;
    }else if (i % 2 == 1 && S[i] == '?' )
      balance *= 6;
    else if((i%2 == 0 && vowels.find(S[i]) != string::npos) ||
	    (i%2 == 1 && vowels.find(S[i]) == string::npos)){
      balance *= 0;
      break;
    }
  }
  return balanced + balance;
  
}

int main(){
  int n; cin >> n;
  for(int i =1 ; i <= n; i++){
    string S; cin >> S;
    cout <<"String #" << i << ": " << isBalanced(S)<< endl <<endl;
  }
  return 0;
}
