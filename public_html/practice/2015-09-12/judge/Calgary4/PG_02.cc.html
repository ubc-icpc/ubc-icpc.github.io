#include<stdio.h>
#include<iostream>

using namespace std;

int main(){
  int tc; cin >> tc;
  for(int i = 1; i <= tc; i++ ){
    int d, n; cin >> d >> n;
    if(d > n*n - 2*(n-1)){
      printf("Grid #%d: impossible\n\n",i);
    }else{
      printf("Grid #%d: %d\n\n", i, d*((n-1)+(n-1)));
    }
  }
  return 0;
}
