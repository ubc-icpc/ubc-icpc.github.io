#include <iostream>
#include <cmath>

using namespace std;

int main(){
  int n; cin >> n;
  while(n--){
    int d, x, s; cin >> d >> x >> s;
    int total[d][3];
    for(int i = 0; i < d; i ++){
      for(int j = 0; j < 3; j++){
	cin >> total[i][j];
      }
    }

    int minl, mins; minl = 10000; mins = 10000;
    int costl; costl = 0;
    int costs; costs = 0;
    int needs, current; needs = 0; current = 0;
    for(int i = 0; i < d; i++){
      total[i][1] = minl = minl < total[i][1] ? minl : total[i][1];
      total[i][2] = mins = mins < total[i][2] ? mins : total[i][2];

      costl += x * total[i][0] * total[i][1];

      needs = ceil((double)((total[i][0] * s)- current) / 80);

      costs += needs * total[i][2];
      current += needs* 80  - total[i][0]*s;
    }

    cout << costl + costs<< endl;
  }
  return 0;
}
