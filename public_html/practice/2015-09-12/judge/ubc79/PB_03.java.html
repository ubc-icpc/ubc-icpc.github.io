import java.util.ArrayList;
import java.util.Scanner;
public class Medal {
	public static void main(String[] args){
		Scanner scan=new Scanner(System.in);
		int round=scan.nextInt();
		while(round>0){
			ArrayList<Integer> set = new ArrayList<Integer>();
            for (int i = 0; i < 6; i ++) {
                set.add(scan.nextInt());
            }
            for (int i : set) {
                System.out.print(i+" ");
            }
			System.out.print("\n");
			if(color(set)&&count(set)) System.out.println("both");
			else if(color(set)) System.out.println("color");
			else if(count(set)) System.out.println("count");
			else System.out.println("none");
			
			round--;
		}
		scan.close();
	}
	private static boolean count(ArrayList<Integer> list){
		int usa=0;
		for(int i=0;i<3;i++){
			usa=usa+list.get(i);
		}
		int russia=0;
		for(int j=3;j<6;j++){
			russia=russia+list.get(j);
		}
		return usa>russia;
	}
	private static boolean color(ArrayList<Integer> list){
		if(list.get(0)>list.get(3)) return true;
		else if(list.get(0)<list.get(3)) return false;
		
		else if(list.get(1)>list.get(4)) return true;
		else if(list.get(1)<list.get(4)) return false;
		
		else if(list.get(2)>list.get(5)) return true;
		else if(list.get(2)<list.get(5)) return false;
		
		else if(list.get(2)==list.get(5)) return false;
		else return false;
		
	}
}
