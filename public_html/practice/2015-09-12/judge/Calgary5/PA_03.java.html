import java.util.*;
import java.io.*;

public class A {
    public static void main(String[] args) {
        Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        int T = sc.nextInt();

        for (int t = 0; t < T; t++) {
            int[] data = new int[10];
            int[] dataUnsorted = new int[10] ;
            for (int i = 0; i < 10; i++) {
                data[i] = sc.nextInt();
                dataUnsorted[i] = data[i];
            }

            Arrays.sort(data);

            boolean m = Arrays.binarySearch(data, 18) >= 0;
            boolean z = Arrays.binarySearch(data, 17) >= 0;
            String out;

            if (m && z) {
                out = "both";
            } else if (m) {
                out = "mack";
            } else if (z) {
                out = "zack";
            } else {
                out = "none";
            }

            for (int i = 0; i < 9; i++) {
                System.out.print(dataUnsorted[i] + " ");
            }
            System.out.println(dataUnsorted[9]);
            System.out.println(out);
            System.out.println();
        }
    }
}
