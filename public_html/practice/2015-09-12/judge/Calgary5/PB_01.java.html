// by count - # of medals
// by ranking of medals

import java.util.*;
import java.io.*;

public class B {
    public static void main(String[] args) {
        Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        int T = sc.nextInt();

        for (int tc = 0; tc < T; tc++) {
            int[] usa = new int[3];
            int[] rus = new int[3];

            for (int i = 0; i < 3; i++) {
                usa[i] = sc.nextInt();
            }
            for (int i = 0; i < 3; i++) {
                rus[i] = sc.nextInt();
            }

            boolean count = usa[0] + usa[1] + usa[2] > rus[0] + rus[1] + rus[2];
            boolean color = false;
            if (usa[0] > rus[0]) {
                color = true;
            } else if (usa[0] == rus[0]) {
                if (usa[1] > rus[1]) {
                    color = true;
                } else if (usa[1] == rus[1]) {
                    if (usa[2] > rus[2]) {
                        color = true;
                    } 
                }
            }

            String out;
            if (count && color) {
                out = "both";
            } else if (count) {
                out = "count";
            } else if (color) {
                out = "color";
            } else {
                out = "none";
            }

            for (int i = 0; i < 3; i++) {
                System.out.print(usa[i] + " ");
            }
            for (int i = 0; i < 2; i++) {
                System.out.print(rus[i] + " ");
            }
            System.out.println(rus[2]);
            System.out.println(out + "\n");
        }
    }
}
