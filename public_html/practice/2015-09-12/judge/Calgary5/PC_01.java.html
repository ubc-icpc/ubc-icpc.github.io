// INPUT:
// #nNumberOfPractices
// #studentsAtPractice (1-30) #browniesBroughtThatDay (60-600)
// #mNumberOfGroupsOfStudentsThatTookBrownies (1, #numberOfStudentsInPractice)
// m-lines #NumberOfStudentsInEachGroup
//
// If # of brownies is LESS THAN or EQUAL to #numberOfStudentsInGroup, cut brownies in half (double number)
// Each student in group grabs ONE
// CAN CUT MORE THAN ONCE.

import java.util.*;
import java.io.*;

public class C {
    public static void main(String[] args) {
        Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        for (int tc = 1, tcs = sc.nextInt(); tc <= tcs; tc++) {
            int s = sc.nextInt(), 
                b = sc.nextInt(),
                m = sc.nextInt();

            System.out.printf("Practice #%d: %d %d\n", tc, s, b);

            for (int g = 0; g < m; g++) {
                int n = sc.nextInt();
                
                while (b <= n) {
                    b *= 2;
                }

                b -= n;
                System.out.printf("%d %d\n", n, b);
            }
            System.out.println();
        }
    }
}
