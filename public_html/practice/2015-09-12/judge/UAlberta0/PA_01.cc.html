#include <iostream>
#include <string>

using namespace std;

const string names[2] = {"mack", "zack"};

int main() {
        int T, in, flag;

        cin >> T;
        while (T--) {
                flag = 0;
                for (int i = 0; i < 10; ++i) {
                        cin >> in;
                        cout << in;
                        if (i != 9) {
                                cout << ' ';
                        }

                        if (in == 18) {
                                flag |= 1;
                        } else if (in == 17) {
                                flag |= 2;
                        }
                }

                cout << endl;
                if (flag == 3) {
                        cout << "both";
                } else if (flag == 0) {
                        cout << "none";
                } else {
                        cout << names[flag - 1];
                }
                cout << endl << endl;
        }
        
        return 0;
}
