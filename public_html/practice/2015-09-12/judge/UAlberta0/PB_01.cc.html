#include <iostream>
#include <string>

using namespace std;

const string medals[2] = {"count", "color"};

int main() {
        int T, c1, c2, in, flag;
        int counts[2][3];
        
        cin >> T;
        while(T--) {
                flag = c1 = c2 = 0;
                for(int i = 0; i < 6; ++i) {
                        cin >> in;
                        c1 += i > 2 ? 0 : in;
                        c2 += i > 2 ? in : 0;
                        counts[i/3][i % 3] = in;
                        cout << in;
                        if (i != 5) {
                                cout << ' ';
                        }
                }

                if (c1 > c2) {
                        flag |= 1;
                }
                for(int i = 0; i < 3; ++i) {
                        if (counts[0][i] > counts[1][i]) {
                                flag |= 2;
                                break;
                        } else if (counts[0][i] < counts[1][i]) {
                                break;
                        }
                }

                cout << endl;
                if (flag == 3) {
                        cout << "both";
                } else if (flag == 0) {
                        cout << "none";
                } else {
                        cout << medals[flag - 1];
                }
                cout << endl << endl;
        }
        
        return 0;
}
