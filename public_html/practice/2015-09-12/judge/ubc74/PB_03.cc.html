#include<iostream>

using namespace std;

int main() {
	int n; cin >> n;
	for (int i = 1; i <= n; ++i) {
		int us[4] = {}, ra[4] = {}, message = 0;
		for (int j = 0; j < 3; ++j) {
			cin >> us[j];
			cout << us[j] << " ";
			us[3] += us[j];
		}
		for (int j = 0; j < 3; ++j) {
			cin >> ra[j];
			cout << ra[j] << " ";
			ra[3] += ra[j];
		}
		cout << endl;
		if (us[3] > ra[3])
			message++;
		us[3] = 0; ra[3] = 0;
		for (int j = 0; j < 3; ++j) {
			us[3] += us[j];
			ra[3] += ra[j];
			if (us[3] > ra[3]) {
				message += 2;
				break;
			}
			else if (us[3] < ra[3])
				break;
		}
		if (message == 0)
			cout << "none";
		else if (message == 1)
			cout << "count";
		else if (message == 2)
			cout << "color";
		else if (message == 3)
			cout << "both";
		cout << endl << endl;
	}
}
