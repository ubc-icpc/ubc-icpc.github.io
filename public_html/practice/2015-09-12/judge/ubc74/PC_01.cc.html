#include<iostream>

using namespace std;

int check(int g, int brownie) {
	if (brownie <= g)
		brownie = check(g, brownie * 2);
	return brownie;
}

int main() {
	int n, student, brownie, m, g; cin >> n;
	for (int i = 1; i <= n; ++i) {
		cin >> student >> brownie >> m;
		cout << "Practice #" << i << ": " << student << " " << brownie << endl;
		for (int j = 0; j < m; ++j) {
			cin >> g;
			brownie = check(g, brownie) - g;
			cout << g << " " << brownie << endl;
		}
		cout << endl;
	}
}
