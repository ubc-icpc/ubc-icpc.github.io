#include<iostream>

using namespace std;

int main() {
	int n, player[10] = {}; cin >> n;
	for (int i = 1; i <= n; ++i) {
		int message = 0;
		for (int j = 0; j < 10; ++j) {
			cin >> player[j];
			if (player[j] == 17)
				message++;
			else if (player[j] == 18)
				message += 2;
		}			
		for (int j = 0; j < 10; ++j)
			cout << player[j] << " ";
		cout << endl;
		if (message == 0)
			cout << "none";
		else if (message == 1)
			cout << "zack";
		else if (message == 2)
			cout << "mack";
		else if (message == 3)
			cout << "both";
		cout << endl << endl;
	}
}
