#include<iostream>
#include<math.h>
#include<iomanip>

using namespace std;

const double pi = 3.14159265358979;

int main() {
	int T; cin >> T;
	while (T--) {
		cout << fixed << setprecision(2);
		double s, r; cin >> s >> r;
		if (r >= s / sqrt(2))
			cout << pow(s, 2) << endl;
		else if (r <= s / 2)
			cout << pi*pow(r, 2) << endl;
		else {
			double theta = acos(s/(2*r));
			double phi = pi / 2 - 2*theta;
			cout << 4 * (s / 2 * sqrt(r*r - s*s / 4) + r*r*phi/2) << endl;
		}
	}
	return 0;
}
