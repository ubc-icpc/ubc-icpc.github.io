#include<iostream>
#include<math.h>

using namespace std;

int main() {
	int n; cin >> n;
	for (int i = 1; i <= n; ++i) {
		int d, x, s, expenses = 0, l_s = 0, minl = 50, mins = 500; cin >> d >> x >> s;
		for (int j = 0; j < d; ++j) {
			int c, pl, ps; cin >> c >> pl >> ps;
			if (pl < minl)
				minl = pl;
			if (ps < mins)
				mins = ps;
			expenses += c*x*minl;
			double sugar = c*s - l_s;
			int buy_s = ceil(sugar / (5.0*16));
			expenses += buy_s*mins;
			l_s = buy_s * (5*16) + l_s - c*s;
		}
		cout << expenses << endl;
	}
	return 0;
}
