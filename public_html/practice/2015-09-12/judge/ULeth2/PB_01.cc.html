#include<iostream>
using namespace std;

bool col(int U[], int R[])
{
   if(U[0]>R[0])
      return true;
   else if(U[0]==R[0])
   {
      if(U[1]>R[1])
	 return true;
      else if(U[1]==R[1])
	 return (U[2]>R[2]);
      else
	 return false;
   }
   else
      return false;
}

int main()
{
   int n;
   cin >> n;
   int U[3], R[3];
   for(int k=0; k<n; k++)
   {
      cin >> U[0] >> U[1] >> U[2];
      cin >> R[0] >> R[1] >> R[2];
      for(int i=0; i<3; i++)
      {
	 if(i==0)
	    cout << U[i];
	 else
	    cout << " " << U[i];
      }
      for(int i=0; i<3; i++)
	 cout << " " << R[i];
      cout << endl;
      bool count=false, colour=false;
      int Ucount=U[0]+U[1]+U[2], Rcount=R[0]+R[1]+R[2];
      if(Ucount>Rcount)
	 count=true;
      colour=col(U, R);
      if(count and colour)
	 cout << "both" << endl;
      else if(count and !colour)
	 cout << "count" << endl;
      else if(colour and !count)
	 cout << "color" << endl;
      else
	 cout << "none" << endl;
      cout << endl;
   }

   return 0;
}
