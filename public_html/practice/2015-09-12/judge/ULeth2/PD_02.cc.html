#include<iostream>
#include<algorithm>
using namespace std;

int d, l, s;

int main()
{
   int n;
   cin >> n;
   while(n--)
   {
      cin >> d >> l >> s;
      int C, Pl=1000000, Ps=1000000, cost=0, currS=0;
      for(int i=0; i<d; i++)
      {
	 int tmpL, tmpS;
	 cin >> C >> tmpL >> tmpS;
	 Pl=min(tmpL, Pl);
	 Ps=min(tmpS, Ps);
	 int needL=C*l;
	 cost+=needL*Pl;
	 int needS=C*s - currS;
	 if(needS<=0)
	    currS-=C*s;
	 else
	 {
	    //int bags=(needS/(16*5))+1;
	    int bags=0;
	    while(needS>0)
	    {
	       bags++;
	       needS-=5*16;
	    }
	    //cout << bags << endl;
	    cost+=bags*Ps;
	    currS+=bags*16*5;
	    currS-=C*s;
	 }
      }
      cout << cost << endl;
   }

   return 0;
}
