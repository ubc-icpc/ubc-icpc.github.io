#include<string>
#include<fstream>
#include<vector>
#include<set>
#include<math.h>
#include<algorithm>
#include<iomanip>

using namespace std;


int main(){

	double t,r,x;
	double pi = 3.14159265358979;
	cin>>t;
	while(t--){
		cin>>x>>r;
		if(2 * r <= x){
			cout<<fixed<<setprecision(2)<<pi*r*r<<endl;
		}
		else if( x*x*2 <= 4*r*r){
			cout<<fixed<<setprecision(2)<<x*x<<endl;
		}
		else{
			double al = ((acos((x/2)/r)*180.0/pi)/180.0)*(pi*r*r);
			double ne = sqrt((r*r)-((x*x)/4));
			double w = al - (ne*x)/2;
			cout<<fixed<<setprecision(2)<<pi*r*r - 4*w<<endl;
		}
	}
	
	return 0;
}
