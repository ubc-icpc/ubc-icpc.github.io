#include<iostream>
#include<map>
#include<string>
#include<fstream>
#include<vector>
#include<set>
#include<algorithm>

using namespace std;


int main(){

	long long d,t,l,s,x,y,z;
	long long ms,ml;
	
	cin>>t;
	while(t--){
		long long r=0;
		long long cl = 0;
		long long cs = 0;
		cin>>d>>l>>s;
		for(long long i=0;i<d;i++){
			cin>>x>>y>>z;
			if(i==0){
				ms = z;
				ml = y;
			}
			ms = min(ms , z);
			ml = min(ml , y);
			
			if(cl < l*x){
				r+=(l*x - cl)*ml;
				cl += (l*x - cl);
			}
			cl -= l*x;
			if(cs < s*x){
				if((s*x-cs)%80==0){
					r+=((long long)((s*x-cs)/80))*ms;
					cs += ((long long)((s*x-cs)/80))*80;
				}
				else{
					r+=((long long)((s*x-cs)/80)+1)*ms;
					cs += ((long long)((s*x-cs)/80)+1)*80;
				}
			}
			cs -= s*x;
		}
		cout<<r<<endl;
	}

	

    return 0;
}
