import java.text.DecimalFormat;
import java.util.Scanner;


public class Rain {
	public final static double PI = 3.14159265358979;

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner s = new Scanner(System.in);
		int numLines = Integer.parseInt(s.nextLine());
		
		for (int i = 0; i < numLines; i++) {
			int lengthSquareSide = s.nextInt();
			int radius = s.nextInt();
			double area = 0;
			
			if ((lengthSquareSide / (2.0 * radius)) >= 1) {
				// Circle fits in square
				area = PI * radius * radius;
			} else if (lengthSquareSide * lengthSquareSide <= 2 * radius * radius) {
				// Square fits in circle
				area = lengthSquareSide * lengthSquareSide;
			} else {
				double theta = Math.acos(lengthSquareSide / (2.0 * radius));
				double lune = ((PI * radius * radius) / (PI / theta)) - (lengthSquareSide / 2.0 * Math.sqrt((radius * radius) - (lengthSquareSide * lengthSquareSide / 4.0)));
				//System.out.println(lune + " " + ((PI * radius * radius) / (PI / theta)) + " " + (lengthSquareSide * Math.sqrt(radius * radius - (lengthSquareSide * lengthSquareSide / 4.0))));
				area = (PI * radius * radius) - (4 * lune);
			}
			
			DecimalFormat df = new DecimalFormat("#.00");
			System.out.println(df.format(area));
		}
		s.close();
	}

}
