import java.util.Scanner;


public class Brownie {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner s = new Scanner(System.in);
		int numTests = Integer.parseInt(s.nextLine());
		
		for (int i = 0; i < numTests; i++) {
			int numStudents = s.nextInt();
			int numBrownies = s.nextInt();
			int originalNumBrownies = numBrownies;
			s.nextLine();
			int numGroups = Integer.parseInt(s.nextLine());
			
			System.out.println("Practice #" + (i+1) + ": " + numStudents + " " + originalNumBrownies);
			for (int j = 0; j < numGroups; j++) {
				int numInGroup = Integer.parseInt(s.nextLine());
				while (numBrownies <= numInGroup) {
					numBrownies = numBrownies << 1;
				}
				numBrownies -= numInGroup;
				
				System.out.println(numInGroup + " " + numBrownies);
			}
			System.out.println();
		}
		s.close();
	}

}
