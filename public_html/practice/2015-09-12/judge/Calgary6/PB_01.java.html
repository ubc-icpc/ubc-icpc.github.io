import java.util.Scanner;


public class Medal {

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		// TODO Auto-generated method stub
		Scanner s = new Scanner(System.in);
		int numLines = Integer.parseInt(s.nextLine());
		
		for (int i = 0; i < numLines; i++) {
			String line = s.nextLine();
			Scanner l = new Scanner(line);
			
			int usaMedals[] = new int[3];
			int russiaMedals[] = new int[3];
			for (int j = 0; j < 3; j++) {
				usaMedals[j] = l.nextInt();
			}
			for (int j = 0; j < 3; j++) {
				russiaMedals[j] = l.nextInt();
			}
			l.close();
			
			boolean countWin = false, colorWin = false;
			
			int usaTotal = usaMedals[0] + usaMedals[1] + usaMedals[2];
			int russiaTotal = russiaMedals[0] + russiaMedals[1] + russiaMedals[2];
			if (usaTotal > russiaTotal) {
				countWin = true;
			}
			
			if (usaMedals[0] > russiaMedals[0]) {
				colorWin = true;
			} else if (usaMedals[0] == russiaMedals[0]) {
				if (usaMedals[1] > russiaMedals[1]) {
					colorWin = true;
				} else if (usaMedals[1] == russiaMedals[1] && usaMedals[2] > russiaMedals[2]) {
					colorWin = true;
				}
			}
			
			System.out.println(line);
			if (countWin && colorWin) {
				System.out.println("both");
			} else if (countWin) {
				System.out.println("count");
			} else if (colorWin) {
				System.out.println("color");
			} else {
				System.out.println("none");
			}
			System.out.println();
		}
		s.close();
	}

}
