#include <iostream>
using namespace std;


int n_practice = 0;
int n_students = 0;
int n_brownies = 0;
int n_approach = 0;
int n_pergroup = 0;

int main() {

    cin >> n_practice;

    for (int i = 0; i < n_practice; i++) {
        cin >> n_students;
        cin >> n_brownies;
        cin >> n_approach;

        cout << "Practice #" << i+1;
        cout << ": " << n_students << " " << n_brownies;
        cout << "\n";

        for (int j = 0; j < n_approach; j++){

            cin >> n_pergroup;

            while (n_pergroup >= n_brownies) {
                n_brownies = n_brownies * 2;
            }
            
            n_brownies = n_brownies - n_pergroup;
            cout << n_pergroup << " " << n_brownies << "\n";

        }
        cout << "\n";
    }
}
