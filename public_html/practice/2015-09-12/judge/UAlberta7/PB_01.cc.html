#include <iostream>
using namespace std;

#define N_SET 6

bool colour = 0;
bool count = 0;

int a = 0;
int b = 0;

int x[N_SET] = {0,0,0,0,0,0};
int n = 0;

int main(){
    cin >> n;

    for (int i = 0; i < n; i++){
        colour = 0;
        count = 0;
        for (int j = 0; j < N_SET; j++){
            cin >> x[j];
            cout << x[j] << " ";
        }
        a = x[0]+x[1]+x[2];
        b = x[3]+x[4]+x[5];

        if (a>b) count = true;

        if (x[0] > x[3]) colour = true;
        else if (x[0] == x[3]) {
            if (x[1] > x[4]) colour = true;
            else if (x[1] == x[4]) {
                if (x[2] > x[5]) colour = true;
            }
        }
        
        cout << "\n";
        if (colour & count) cout << "both";
        else if (colour) cout << "color";
        else if (count) cout << "count";
        else cout << "none";

        cout << "\n";
    }
}
