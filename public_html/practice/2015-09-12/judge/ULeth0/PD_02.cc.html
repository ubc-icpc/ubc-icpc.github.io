#include<iostream>
#include<climits>
using namespace std;

int main(){
   int n; cin >> n;
   while(n--){
      int d,x,s; cin >> d >> x >> s;
      int curl=0, curs=0;
      int curpl= INT_MAX, curps = INT_MAX;
      long long cost = 0;
      for(int i = 0; i < d; i++){
	 int c, pl, ps; cin >> c >> pl >> ps;
	 if(pl < curpl){
	    curpl = pl;
	 }
	 if(ps < curps){
	    curps = ps;
	 }
	 if(curl < c*x){
	    cost += (c*x - curl)*curpl;
	    curl = c*x;
	 }
	 if(curs < c*s){
	    cost += ((c*s - curs + 79)/80)*curps;
	    curs += ((c*s - curs + 79)/80)*80;
	 }
	 curl -= c*x;
	 curs -= c*s;
      }
      cout << cost << endl;
   }
   return 0;
}
