#include<iostream>
#include<sstream>
using namespace std;

int main(){
   int n;
   cin >> n;
   string s;
   getline(cin,s);
   for(int i = 0; i < n; i++){
      bool M=false,Z=false;
      getline(cin, s);
      cout << s << endl;
      istringstream ss(s);
      for(int j = 0; j < 10; j++){
	 int x;
	 ss >> x;
	 if(x==18){
	    M = true;
	 }
	 else if(x == 17){
	    Z = true;
	 }
      }
      if(M and Z){
	 cout << "both";
      }
      else if(M and !Z){
	 cout << "mack";
      }
      else if(!M and Z){
	 cout << "zack";
      }
      else{
	 cout << "none";
      }
      cout << endl << endl;
   }
   return 0;
}
