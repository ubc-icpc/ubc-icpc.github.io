#include<iostream>
#include<sstream>
using namespace std;

int main(){
   int n;
   cin >> n;
   string s;
   getline(cin,s);
   while(n--){
      getline(cin,s);
      cout << s << endl;
      bool count=false,color=false;
      istringstream ss(s);
      for(int i = 0; i < 6; i++){
	 int g1,g2,s1,s2,b1,b2;
	 ss >> g1 >> s1 >> b1 >> g2 >> s2 >> b2;
	 if(g1+s1+b1 > g2+s2+b2)
	    count = true;
	 if(g1 > g2 or (g1 == g2 and s1 > s2) or (g1 == g2 and s1 == s2 and b1 > b2)){
	    color = true;
	 }	 
      }
      if(count and color){
	 cout << "both";
      }
      else if(!count and color){
	 cout << "color";
      }
      else if(count and !color){
	 cout << "count";
      }
      else{
	 cout << "none";
      }
      cout << endl << endl;
   }
   return 0;
}
