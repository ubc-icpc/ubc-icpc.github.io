#include<iostream>

using namespace std;

int main(){
   int n; cin >> n;
   for( int c = 1; c <= n; c++){
      int s,b;
      cin >> s >> b; 
      cout << "Practice #" << c << ": " << s << " " << b << endl;
      int m; cin >> m;
      for(int i = 0; i < m; i++){
	 int num;
	 cin >> num;
	 while(b <= num){
	    b*=2;
	 }
	 b -= num;
	 cout << num << " " << b << endl;
      }
      cout << endl;
   }
   return 0;
}
