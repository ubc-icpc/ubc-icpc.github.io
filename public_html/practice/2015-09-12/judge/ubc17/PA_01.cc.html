#include <bits/stdc++.h>

using namespace std;

int main() {
    int n;
    cin >> n;
    while (n--) {
        bool m = false, z = false;
        for (int i = 0; i < 10; ++i) {
            int x;
            cin >> x;
            if (x == 17) {
                z = true;
            }
            if (x == 18) {
                m = true;
            }
            if (i) cout << " ";
            cout << x;
        }
        cout << "\n";
        if (m) {
            if (z) {
                cout << "both";
            } else {
                cout << "mack";
            }
        } else {
            if (z) {
                cout << "zack";
            } else {
                cout << "none";
            }
        }
        cout << "\n\n";
    }
}
