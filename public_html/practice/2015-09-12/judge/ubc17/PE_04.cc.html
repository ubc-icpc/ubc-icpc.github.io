#include <bits/stdc++.h>

using namespace std;

#define PI 3.14159265358979
#define ld long double

int main() {
    int n;
    cin >> n;
    while (n--) {
        int s, r;
        cin >> s >> r;
        if (r * 1.0 >= s / sqrt(2.0)) {
            // wholly covered
            printf("%.02f\n", 1.0 * s * s);
            continue;
        } else if (r * 2 <= s) {
            // wholly inside
            printf("%.02f\n", PI * r * r);
            continue;
        }

        ld theta = acos(s / 2.0l / r);

        ld x  = sqrt(r * r - s * s / 4.0l);

        ld covered = 0.0l;

        covered += 2.0l * x * s;

        ld rem_angle = 2.0l * PI - 8.0l * theta;

        covered += rem_angle * r * r / 2.0l;

        printf("%.02f\n", (double)covered);
    }
}
