#include <bits/stdc++.h>

using namespace std;

int main() {
    int n;
    cin >> n;
    for (int q = 1; q <= n; ++q) {
        printf("Grid #%d: ", q);

        int d, n;
        cin >> d >> n;
        if (d > ((n - 1) * (n - 1) + 1)) {
            printf("impossible\n");
        } else {
            printf("%d\n", (n - 1) * 2 * d);
        }
    }
}
