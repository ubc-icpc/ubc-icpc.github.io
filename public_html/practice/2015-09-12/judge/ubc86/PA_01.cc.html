#include <iostream>

using namespace std;

int main()
{
	int a;
	cin >> a;
	bool zack = false;
	bool mack = false;
	
	for (int k = 0; k < a; k ++)
	{
		for (int i = 0; i < 10; i++)
		{
			int b;
			cin >> b;
			cout << b;
			cout << " ";
			if (b == 17)
			{
				zack = true;
			}
			if (b == 18)
			{
				mack = true;
			}
		}
		cout << endl;
		if (zack == true)
		{
			if (mack == true)
			{
				cout << "both";
			}
			else
			{
			cout << "zack";
			}
		}
		if (mack == true && zack == false)
		{
			cout << "mack";
		}
		if (zack == false && mack == false)
		{
			cout << "none";
		}
		cout << endl;
		cout << endl;
		zack = false;
		mack = false;
	}
}
