import java.util.ArrayList;
import java.util.HashMap;
import java.util.Random;
import java.util.Scanner;


public class FindTheTwins {
	
	public static void checkForTwins() {
		
		HashMap<Integer, ArrayList<Integer>> numbers = new HashMap<Integer, ArrayList<Integer>>();
		
		
		Scanner sc = new Scanner(System.in);
		int numberOfDataSets = sc.nextInt();
		//System.out.println(numberOfDataSets);

		for (int i = 0; i < numberOfDataSets; i++) {
			ArrayList<Integer> row = new ArrayList<Integer>();
			String rowOfNumbers = "";
			for (int j = 0; j < 10; j++) {
				int number = sc.nextInt();
				row.add(number);	//add the numbers to the ArrayList
				rowOfNumbers += number + " ";
			}
			//System.out.println(rowOfNumbers);
			numbers.put(i, row);	//put the ArrayList into the HashMap
		}
		
		for(int i = 0; i < numbers.size(); i++) {
			String row = "";
			for (int j = 0; j < 10; j++) {
				row += numbers.get(i).get(j) + " ";
			}
			System.out.println(row);
			if (numbers.get(i).contains(18) && numbers.get(i).contains(17)) {
				System.out.println("both\n");
				continue;
			} else if (!(numbers.get(i).contains(18)) && !(numbers.get(i).contains(17))) {
				System.out.println("none\n");
				continue;
			} else if (numbers.get(i).contains(18)) {
				System.out.println("mack\n");
				continue;
			} else {
				System.out.println("zack\n");
				continue;
			}
		}
		
	}
	
	public static void main(String[] args) {
		
		//call checkForTwins
		checkForTwins();
		
		

	}

}
