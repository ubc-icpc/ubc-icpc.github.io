#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
typedef long long LL;
typedef pair<int,int> PT;

int main() {
	int tc;
	cin >> tc;
	while (tc--) {
		int ans=0, cs=0, lpx = INT_MAX, lps = INT_MAX, D, X, S;
		cin >> D >> X >> S;
		FOR(d,0,D) {
			int c, px, ps;
			cin >> c >> px >> ps;
			lpx = min(lpx, px);
			lps = min(lps, ps);
			int ux = c*X, us = c*S;
			ans += ux*lpx;
			int bs = max(0, (us-cs+80-1)/80);
			ans += bs*lps;
			cs += bs*80 - us;
		}
		cout << ans << endl;
	}
}
