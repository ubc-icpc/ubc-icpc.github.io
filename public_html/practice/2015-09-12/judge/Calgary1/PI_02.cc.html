#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
typedef long long LL;
typedef pair<int,int> PT;

int main() {
	int tc, V;
	bool mat[500][500];
	scanf("%d", &tc);
	while (tc--) {
		scanf("%d", &V);
		FOR(i,0,V) FOR(j,0,V) {
			int x; scanf("%d", &x);
			mat[i][j] = (x == 1);
		}
		list<int> stuff(1, 0);
		FOR(i,1,V) {
			for (auto it = begin(stuff);; ++it) {
				if (it != end(stuff) && mat[*it][i]) continue;
				if (it != begin(stuff) && mat[i][*prev(it)]) continue;
				stuff.insert(it, i);
				break;
			}
		}
		for (auto it = begin(stuff); it != end(stuff); ++it) {
			if (it != begin(stuff)) printf(" ");
			printf("%d", *it+1);
		}
		printf("\n");
	}
}
