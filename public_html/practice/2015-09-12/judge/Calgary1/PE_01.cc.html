#include <bits/stdc++.h>
using namespace std;

const double EP = 1e-9;
const double PI = 3.14159265358979;

int main() {
	int tc;
	cin >> tc;
	while (tc--) {
		double ans, S, R;
		cin >> S >> R;
		if (R + EP > sqrt(2*0.25*S*S)) {
			ans = S*S;
		} else if (R - EP < 0.5*S) {
			ans = PI*R*R;
		} else {
			double theta1 = acos(0.5*S/R);
			double theta2 = 0.5*M_PI-2.0*theta1;
			double nca = 0.25*S*S-0.5*S*R*sin(theta1)-0.5*R*R*theta2;
			ans = S*S - 4*nca;
		}
		cout << fixed << setprecision(2) << ans << endl;
	}
}
