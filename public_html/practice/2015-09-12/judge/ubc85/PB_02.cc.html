#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;

int main () {
	int n;
	cin >> n;
	for (int i=0; i<n; ++i) {
		int a, b, c, d, e, f;
		bool x = false;
		bool y = false;
		cin >> a >> b >> c >> d >> e >> f;
		if ((a+b+c)>(d+e+f)) {
			x = true;
		}
		if ((a>d) || ((a==d)&&(b>e)) || ((a==d)&&(b==e)&&(c>f))) {
			y = true;
		} 
		cout << a << " " << b << " " << c << " " << d << " " << e << " " << f << endl;
		if ((x == true)&&(y == true)) {
			cout << "both" << endl;
		} else if (x == true) {
			cout << "count" << endl;
		} else if (y == true) {
			cout << "color" << endl;
		} else {
			cout << "none" << endl;
		}
	}
	return 0;
}
