#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;

int main () {
	int n;
	cin >> n;
	for (int i=0; i<n; ++i) {
		int s, b;
		cin >> s >> b;
		int c;
		cin >> c;
		cout << "Practice #" << i+1 << ": " << s << " " << b << endl;
		for (int i=0; i<c; ++i) {
			int x; cin >> x;
			while (b<=x) {
				b = 2*b;
			}
			b = b - x;
			cout << x << " " << b << endl;
		}
	}
	return 0;
}
