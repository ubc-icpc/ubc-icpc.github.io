#include<bits/stdtr1c++.h>
using namespace std;

const long double PI = acos(-1.0);

int main () {
  int n; cin >> n;
  for (int i = 0; i < n; ++i) {
    double l,r;
    cin >> l >> r;
    if (r*2 <= l) {
      double a = PI*r*r;
      cout << fixed << setprecision(2) << a << endl;
    } else if (l <= r*sqrt(2)) {
      double a = l*l;
      cout << fixed << setprecision(2) << a << endl;
    } else {
      double a = r*r*(2*sin(2*acos(l/(2*r)))+2*(PI/2-2*acos(l/(2*r))));
      cout << fixed << setprecision(2) << a << endl;
    }
  }
  return 0;
}
