import java.util.Scanner;

public class UbcPB0905 {
	public static void main(String[] args) {

		Scanner input = new Scanner(System.in);

		int size = input.nextInt();

		boolean[] count = new boolean[size];
		boolean[] colour = new boolean[size];

		int[][] data = new int[size][6];

		// READ & SET
		for (int x = 0; x < size; x++) {

			int utotal = 0;
			int rtotal = 0;
			for (int y = 0; y < 6; y++) {
				data[x][y] = input.nextInt();
				
				if (y < 3)
					utotal += data[x][y];
				else {
					rtotal += data[x][y];
				}
			}
			
			
			if (utotal > rtotal)
				count[x] = true;
		}
		
		// CHECK
		for (int i = 0; i < size; i++) {

			for (int g = 0; g < 6; g++) {

				System.out.print(data[i][g] + " ");
				
				if (g == 3 && data[i][0] > data[i][3]) {
					colour[i] = true;
				} else if (g == 3 && data[i][0] == data[i][3]) {

					if(data[i][1] > data[i][4]){
						colour[i] = true;
					}else if(data[i][1] == data[i][4]){
						if(data[i][2] > data[i][5]){
							colour[i] = true;

						}
					}
				}
			}

			System.out.println();
			
			if(count[i] == true && colour[i] == true)
				System.out.println("both");
			else if(count[i] == true && colour[i] == false)
				System.out.println("count");
			else if(count[i] == false && colour[i] == true)
				System.out.println("color");
			else 
				System.out.println("none");
			
			System.out.println();
		}

	}
}
