import java.util.Scanner;

public class P5 {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int prac = sc.nextInt();
		int x;
		
		for (int i=0; i<prac; i++){
			int std = sc.nextInt();
			int brwn = sc.nextInt();
			System.out.println("Practice #" +(i+1)+": "+std+" "+brwn);
			
			int n = sc.nextInt();
			
			for(int j=0; j<n; j++){
				x = sc.nextInt();
				while( x >= brwn){
					brwn = brwn*2;
				}
				brwn = brwn-x;
				System.out.println(x +" "+brwn);
			}
		}
		return;
				
		
	}
}
