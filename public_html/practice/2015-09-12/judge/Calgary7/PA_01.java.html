import java.util.Scanner;
import java.util.ArrayList;

public class UbcPA0905 {

	public static void main(String [] args){
		Scanner input = new Scanner(System.in);
		
		int size = input.nextInt(); 
		
		int [][] listarr = new int[size][10];
		boolean [] ansZ = new boolean[size];
		boolean [] ansM = new boolean[size];
		// READ
		for(int y = 0; y < size; y++){
			for(int z = 0; z < 10; z++){
				
				listarr[y][z] = input.nextInt();
				if(listarr[y][z] == 18)
					ansM[y] = true;
				else if(listarr[y][z] == 17)
					ansZ[y] = true;
			}
		}
		
		// CHECK
		for(int i = 0; i < size; i++){
				boolean zack = ansZ[i];
				boolean mack = ansM[i];
			for(int g = 0; g < 10; g++){
				System.out.print(listarr[i][g] + " ");				
			}
			
			System.out.println();
			if(zack == true && mack == true)
				System.out.println("both");
			else if(zack == true && mack == false)
				System.out.println("zack");
			else if(zack == false && mack == true)
				System.out.println("mack");
			else
				System.out.println("none");
			
			System.out.println();
		}
		
	}
}
