import java.util.Scanner;

public class mahremonadesbringalldemboystodayard {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int n = sc.nextInt();

		int d; // DAYZZ
		int x; // # of remons
		int s; // # of sugar yes plz

		for (int i = 0; i < n; i++) {

			d = sc.nextInt();
			x = sc.nextInt();
			s = sc.nextInt();

			int[][] remons = new int[d][3];

			for (int j = 0; j < d; j++) {
				for (int k = 0; k < 3; k++) {
					// heheh k <3

					remons[j][k] = sc.nextInt();

				}
			}

			int remonTotal = 0;
			int remonMin = remons[0][1];

			for (int j = 0; j < d; j++) {
				if (remonMin > remons[j][1]) {
					remonMin = remons[j][1];
					remonTotal += remonMin * remons[j][0] * x;
				} else {
					remonTotal += remonMin * remons[j][0] * x;
				}
			}

			int sugahTotal = 0;
			int sugahMin = remons[0][2];
			int sugahBags = 0;
			int sugahCount = 0;

			for (int j = 0; j < d; j++) {
				sugahBags=0;
				while (((sugahBags * 80) + sugahCount) < remons[j][0] * s) {
					sugahBags++;
				}

				if (sugahMin > remons[j][2]) {
					sugahMin = remons[j][2];
				}
				//System.out.println("weed bags: " + sugahBags);
				//System.out.println("sugarMin: " + sugahMin);
				sugahTotal += sugahBags*sugahMin;
				sugahCount = (sugahCount + (sugahBags * 80)) - (remons[j][0] * s);

				

			}
			
			System.out.println(remonTotal+sugahTotal);
			
		}

	}
}
