#include <iostream>
using namespace std;
#include <vector>

bool done(vector<int> stats, vector<int> search) {
	for (int i=0;i<search.size();i++) {
		if (stats[search[i]]==1) return false;
	}
	return true;
}

int main() {
	int T;
	cin>>T;
	for (int t=0;t<T;t++) {
		int n;
		cin>>n;
		vector<int> first;
		vector<int> second;
		vector<int> stats(n);
		for (int i=0;i<n;i++) {
			int j;
			cin>>j;
			first.push_back(j);
		}
		for (int i=0;i<n;i++) {
			int j;
			cin>>j;
			second.push_back(j);
		}
		
		int place=1;
		vector<int> search;
		for (int j=0;j<n;j++) {
			stats[first[j]-1]=(stats[first[j]-1]+1)%2;
			stats[second[j]-1]=(stats[second[j]-1]+1)%2;
			search.push_back(first[j]-1);
			search.push_back(second[j]-1);
			if (done(stats, search)) {
				cout<<place<<"-"<<j+1<<" ";
				place=j+2;
				search.clear();
			}
		}
		cout<<""<<endl;
	}
	return 0;
}
