#include <iostream>
#include <cstdio>
#include <string>
#include <list>
using namespace std;

int main() {
  int t;
  char c;
  scanf("%d\n", &t);
  while (t--) {
    list<char> pass{};
    list<char>::iterator pos = pass.begin();
    while ((c = getchar()) != '\n' && c != '\r' && c != EOF) {
      switch (c) {
        case '-':
          if (pos != pass.begin()) {
            auto back = pos;
            --back;
            pos = pass.erase(back);
          }
          break;
        case '<':
          if (pos != pass.begin()) --pos;
          break;
        case '>':
          if (pos != pass.end()) ++pos;
          break;
        default:
          pos = pass.insert(pos, c);
          pos++;
          break;
      }
    }
    for (auto it=pass.begin(); it!=pass.end(); ++it)
      cout << *it;
    cout << endl;
  }
}
