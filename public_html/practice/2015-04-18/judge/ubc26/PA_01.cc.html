#include <iostream>
#include <set>
using namespace std;

int a[100002];
int b[100002];

set<int> ones;
set<int> twos;

void add(int k) {
  auto x1 = ones.find(k);
  if (x1 == ones.end()) {
    ones.insert(k);
  } else {
    ones.erase(k);
    twos.insert(k);
  }
}

int main() {
  int t; cin >> t;
  while (t--) {
    ones.clear();
    twos.clear();
    int n; cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];
    for (int i = 0; i < n; i++) cin >> b[i];

    int last = 0;
    for (int i = 0; i < n; i++) {
      add(a[i]);
      add(b[i]);
      if (ones.empty()) {
        if (last != 0) cout << " ";
        cout << (last+1) << "-" << (i+1);
        last = i+1;
      }
    }
    cout << endl;
  }
}
