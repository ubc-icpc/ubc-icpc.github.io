#include <iostream>
#include <list>
#include <string>

using namespace std;

const int MAXL = 1000*1000 + 7;

int T;

string seq;

int main()
{
	cin>>T;
	while(T--)
	{
		cin>>(seq);

		list<char> pass; pass.push_back('F');

		auto it= pass.begin();

		for(int j=0;j<seq.size();j++)
		{
			//cout<<"seq "<<seq[j]<<endl;
			if(seq[j] == '<' )
			{
				if((it)!=pass.begin()) it--;
			}
			else if(seq[j] == '>' )
			{
				it++;
				if(it==pass.end()) it--;
			}
			else if(seq[j] == '-' )
			{
				if(it!=pass.begin())
				{
					it = pass.erase(it);
					it--;
				}
			}
			else
			{
				it++;
				pass.insert( it, seq[j]);
			//	cout<<seq[j]<<endl;
				it--;
			}
		}

		it = pass.begin(); it++;
		for(; it!=pass.end(); it++)
			cout<<(*it);
		cout<<endl;
	}
	return 0;
}
