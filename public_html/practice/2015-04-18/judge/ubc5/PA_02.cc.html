#include <iostream>
#include <set>
#include <iomanip>
#include <cstring>
using namespace std;

int f[100001], g1[100000], g2[100000];

int main() {
  ios::sync_with_stdio(0);
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    for (int i = 1; i <= n; i++)
      f[i] = 0;
    for (int i = 0; i < n; i++)
      cin >> g1[i];
    for (int i = 0; i < n; i++)
      cin >> g2[i];
    int f1 = 0, f2 = 0, s = 0;
    bool first = true;
    for (int i = 0; i < n; i++) {
      f[g1[i]]++;
      if (f[g1[i]] == 1)
        f1++;
      else {
        f1--;
        f2++;
      }
      f[g2[i]]++;
      if (f[g2[i]] == 1)
        f1++;
      else {
        f1--;
        f2++;
      }
      if (!f1) {
        if (first)
          first = false;
        else
          cout << " ";
        cout << s+1 << "-" << i+1;
        s = i+1;
      }
    }
    cout << "\n";
  }
  return 0;
}
