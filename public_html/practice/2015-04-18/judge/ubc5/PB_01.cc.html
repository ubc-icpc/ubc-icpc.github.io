#include <iostream>
#include <cstring>
#include <iomanip>
#include <algorithm>
using namespace std;

typedef long long ll;

ll d[50000], s[50000], ps[50000];
ll n, m, a, b;

ll f(int x) {
  if (x == n)
    return 0;
  if (d[x] != -1)
    return d[x];
  int xx = x;
  ll p = 0;
  while (x < n && p+s[x] <= m) {
    p += s[x];
    x++;
  }
  //cout << "at " << xx << " to " << x << " p " << p << endl;
  ll m1 = 42424242424242LL;
  if (x < n)
    m1 = (p+s[x]-m)*a+f(x+1);
  if (x < n)
    m1 = min(m1,(p+ps[x]-m)*a);
  if (p)
    m1 = min(m1,(m-p)*b+f(x));
  d[xx] = m1;
  return m1;
}

int main() {
  ios::sync_with_stdio(0);
  int t; cin >> t;
  while (t--) {
    cin >> n >> m >> a >> b;
    for (int i = 0; i < n; i++) {
      cin >> s[i];
      d[i] = -1;
    }
    ll ss = 0;
    for (int i = n-1; i >= 0; i--) {
      ss += s[i];
      ps[i] = ss;
    }
    cout << f(0) << "\n";
  }
  return 0;
}
