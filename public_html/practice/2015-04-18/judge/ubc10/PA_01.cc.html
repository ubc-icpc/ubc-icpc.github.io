#include <iostream>

using namespace std;

int a[100000];
int b[100000];
bool aa[100001];
bool bb[100001];

int main() {
	int n;
	cin >> n;
	for (int cs = 0; cs < n; cs++) {
		int len;
		cin >> len;
		for (int i = 0; i < len; i++) {
			cin >> a[i];
			aa[a[i]] = false;
		}
		for (int i = 0; i < len; i++) {
			cin >> b[i];
			bb[b[i]] = false;
		}
		int left = 0;
		int c = 0;
		for (int i = 0; i < len; i++) {
			int aaa = a[i];
			int bbb = b[i];
			aa[aaa] = true;
			if (!bb[aaa]) {
				c++;
			} else {
				c--;
			}
			bb[bbb] = true;
			if (!aa[bbb]) {
				c++;
			} else {
				c--;
			}
			if (c == 0) {
				if (left > 0) {
					cout << " ";
				}
				cout << (left+1) << "-" << (i+1);
				left = i+1;
			}
		}
		cout << endl;
	}
}
