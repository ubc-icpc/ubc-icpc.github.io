#include<iostream>
#include<vector>
#include<string>

using namespace std;

int main( void ) {
  int T; cin >> T;

  while( T-- ) {
    string str; cin >> str;
    string res = "";
    int pos = 0;
    //res = res.insert(0,1,'a');
    for( size_t i=0; i<str.size(); i++ ) {
      //cout << str[i] << "   "<< res << "   " << pos << endl;
      if( str[i] == '<') {
	if(pos > 0) pos--;
      }else if( str[i] == '>' ) {
	if( pos < res.size() ) pos++;
      }else if( str[i] == '-' ) {
	if( pos > 0 ) {
	res = res.erase(pos-1, 1);
	pos--;
	}/* else if ( pos == 0 ) {
	  res = res.erase(pos, 1);
	  //pos--;
	  }*/
      }else {
	res = res.insert( pos, 1, str[i] );
	pos++;
      }
    }
    cout << res << endl;
  }

  return 0;
}
