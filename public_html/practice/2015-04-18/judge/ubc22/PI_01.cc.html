#include<bits/stdtr1c++.h>
using namespace std;

int main () {
  int T; cin >> T;
  while (T--) {
    string op; cin >> op;
    list<char> pass;
    list<char>::iterator it = pass.begin();
    for (char c : op) {
      switch (c) {
      case '-':
	if (it != pass.begin()) it = pass.erase(--it); break;
      case '<':
	if (it != pass.begin()) --it; break;
      case '>':
	if (it != pass.end()) ++it; break;
      default:
	pass.insert(it, c); break;
      }
    }
    for (char c : pass) cout << c;
    cout << endl;
  }
  return 0;
}
