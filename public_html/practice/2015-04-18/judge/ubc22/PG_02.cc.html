#include<bits/stdtr1c++.h>
using namespace std;

int T, N, M, S[2], P, C, L;
bool p1;

int main () {
  cin >> T;
  while (T--) {
    S[0] = S[1] = P = 0;
    C = 15; L = 27;
    cin >> N;
    int res = 0;
    for (int i = 0; i < N; ++i) {
      cin >> M;
      if (M == 0) P = !P;
      else if (M == 1) ++S[P], --C;
      else {
	S[P] += M;
	if (!p1) L -= M;
      }
      if (M == 1) p1 = true; else p1 = false;
      if (!res && abs(S[P] + (p1 && S[!P] > S[P]) * 7 - S[!P]) > C * 8 + L) res = i + 1;
    }
    cout << res << endl;
  }
  return 0;
}
