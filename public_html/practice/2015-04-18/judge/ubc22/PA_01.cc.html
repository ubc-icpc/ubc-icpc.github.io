#include<bits/stdtr1c++.h>
using namespace std;

int T, N, A[100005], B[100005], AC[100005], BC[100005], AL, BL;

int main () {
  cin >> T;
  while (T--) {
    memset(AC, 0, sizeof AC);
    memset(BC, 0, sizeof BC);
    AL = BL = 0;
    cin >> N;
    for (int i = 1; i <= N; ++i) cin >> A[i];
    for (int i = 1; i <= N; ++i) cin >> B[i];
    int start = 1;
    for (int i = 1; i <= N; ++i) {
      if (BC[A[i]]) {
	AC[A[i]] = BC[A[i]] = 2; --BL;
      } else {
	AC[A[i]] = 1; ++AL;
      }
      if (AC[B[i]]) {
	BC[B[i]] = AC[B[i]] = 2; --AL;
      } else {
	BC[B[i]] = 1; ++BL;
      }
      if (AL == 0 && BL == 0) {
	cout << start << "-" << i << ((i == N)? "\n" : " ");
	start = i + 1;
      }
    }
  }
  return 0;
}
