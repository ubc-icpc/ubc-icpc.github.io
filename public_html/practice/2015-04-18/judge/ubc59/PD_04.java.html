import java.io.*;
public class collatz {
	public static void main(String[] args) throws IOException{
		BufferedReader kinput=new BufferedReader(new InputStreamReader(System.in));

		String user=kinput.readLine();
		int cases=0;
		try{
			cases=Integer.parseInt(user);
			}catch(NumberFormatException nfe){
			}
		int keep=0, copy=0, rope=0;
		boolean even=true;
		
		for(int c=0;c<cases;c++){
			user=kinput.readLine();
			try{
			keep=Integer.parseInt(user);
			}catch(NumberFormatException nfe){
				keep=0;
			}
			copy=keep;
			if(keep%2==0)even=true;
			if(keep%2!=0){
				even=false;
				copy++;
			}
			
			rope+=copy/2;
			copy=keep;
			if (even==true)copy--;
			for(int cc=keep;cc>1;cc--){
				if((cc-1)%3==0&&(cc-1)/3%2!=0){
					rope+=(copy-(cc-1)/3)/2;
					break;
				}
			}
			if(keep==3)rope+=2;
			if(copy==1)rope++;
			System.out.println(rope);
			rope=0;
		}
	}
}
