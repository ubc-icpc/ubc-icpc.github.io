#include<bits/stdtr1c++.h>
using namespace std;
int main() {
	int testcase; cin >> testcase;
	while (testcase--) {
		int n; cin >> n;
		vector < int > vec1(n), vec2(n);
		for (int i = 0; i < n; ++i) {
			cin >> vec1[i];
		}
		for (int i = 0; i < n; ++i) {
			cin >> vec2[i];
		}
		set < int > set1, set2;
		vector < pair < int, int > > solution;
		for (int i = 0; i < n;) {
			for (int j = i; j < n; ++j) {
				set1.insert(vec1[j]);
				set2.insert(vec2[j]);
				if (set1 == set2) {
					solution.push_back(make_pair(i + 1, j + 1));
					i = j + 1;
					set1.clear();
					set2.clear();
					break;
				}
			}
		}
		for (int i = 0; i < solution.size(); ++i) {
			if (i == 0) cout << solution[i].first << "-" << solution[i].second;
			else cout << " " << solution[i].first << "-" << solution[i].second;
		}
		cout << endl;
	}
	return 0;
}
