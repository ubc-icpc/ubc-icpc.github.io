#include<bits/stdtr1c++.h>
using namespace std;
int main() {
	int tc; cin >> tc;
	while (tc--) {
		string cyp; cin >> cyp;
		list < char > decode;
		list<char>::iterator index = decode.begin();
		for (int i = 0; i < cyp.size(); ++i) {
			if (cyp[i] == '<') {
				if (index != decode.begin()) {
					--index;
				}
			}
			else if (cyp[i] == '>') {
				if (index != decode.end()) {
					++index;
				}
			}
			else if (cyp[i] == '-') {
				if (index != decode.begin()) {
					index = decode.erase(--index);
				}
			}
			else {
				decode.insert(index, cyp[i]);
			}
		}
		
		for (list<char>::iterator temp = decode.begin(); temp != decode.end(); ++temp) {
			cout << *temp;
		}
		cout << endl;
	}
	return 0;
}
