import java.util.Scanner;

public class shuffle {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		int T = scan.nextInt();
		while (T-- > 0) {
			int size = scan.nextInt();
			scan.nextLine(); //consume \n
			int[] seencount = new int[size];
			int seggood = 0;
			int prev = 0;
			String gen1 = scan.nextLine();
			String gen2 = scan.nextLine();
			Scanner s1 = new Scanner(gen1);
			Scanner s2 = new Scanner(gen2);
			for (int i=0; i<size; i++) {
				int k1 = s1.nextInt() -1;
				int k2 = s2.nextInt() -1;
				seencount[k1]++;
				seggood = (seencount[k1] == 1) ? (seggood + 1): (seggood - 1);
				seencount[k2]++;
				seggood = (seencount[k2] == 1) ? (seggood + 1): (seggood - 1);
				if (seggood == 0) {
					System.out.print((prev+1) +"-"+(i+1) + " ");
					prev = i+1;
				}
			}
			System.out.println();
			s1.close();
			s2.close();
		}
		scan.close();
	}

}
