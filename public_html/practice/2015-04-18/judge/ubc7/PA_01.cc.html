#include <iostream>
#include <vector>
using namespace std;
#define MAXN 100000
int main() {
    int t; cin>>t;
    for(int ti=0;ti<t;++ti) {
        int n; cin>>n;
        int a[MAXN] = {}; 
        int b[MAXN] = {};
        int pb[MAXN] = {};

        for(int i=0;i<n;++i) {
            cin>>a[i]; 
        }
        for(int i=0;i<n;++i) {
            cin>>b[i]; pb[b[i]-1] = i;
        }
        int i=0, j=0, k=0;
        vector<int> idx;
        while(j<n) {
            //cerr << a[j] << ' ' << pb[a[j]-1] << endl;
            k = max(k, pb[a[j]-1]);
            if(j == k) {
                idx.push_back(j);
                i = j+1;
            }
            j++;
        }
        int last = 0;
        for(int i=0;i<idx.size();++i){
            cout << last+1 << '-' << idx[i] + 1;
            if(i < idx.size()-1) cout << ' ';
            last = idx[i]+1;
        }
        cout << endl;
    }
    return 0;
}
