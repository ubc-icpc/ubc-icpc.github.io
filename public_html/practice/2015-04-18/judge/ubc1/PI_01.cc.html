#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int t; cin >> t;
    while (t--) {
        list<char> lst;
        string s; cin >> s;
        
        auto it = lst.begin();
        for (char c : s) {
            if (c == '>') {
                if (it != lst.end()) it++;
            } else if (c == '<') {
                if (it != lst.begin()) it--;
            } else if (c == '-') {
                if (it != lst.begin()) {
                    auto t = it;
                    lst.erase(--t);
                }
            } else {
                lst.insert(it, c);
            }
        }
        
        for (char c : lst) cout << c;
        cout << endl;
    }
	return 0;
}
