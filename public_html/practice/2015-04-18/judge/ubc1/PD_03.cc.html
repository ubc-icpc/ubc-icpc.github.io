#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

//k connected to 2k
//k connected to 3k+1.
//want k such that there is a connection
//to something > N
//all ceil(N/2) ... N are affected.
//all odd k s.t. k > (N-1)/3 works...

ll f(ll n) {
    ll last = (n+2)/3;
    ll del = n - last + 1;
    cerr << "n: " << n << " " << last << " " << del << endl;
    return n-n/2 + (del+last%2)/2;
}

int main() {
    ll n, t;
    cin >> t;
    while (t--) {
        cin >> n;
        cout << f(n) << endl;
    }
	return 0;
}
