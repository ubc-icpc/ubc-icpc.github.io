#include <iostream>
#include <algorithm>

using namespace std;
typedef long long ll;

ll dp[101][101];
ll p[101][101];
int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int c, m; cin >> c >> m;
    for (int i = 0; i < m; i++) {
        p[i][0] = 0;
        for (int j = 1; j <= c; j++) {
            cin >> p[i][j];
        }
    }

    for (int j = 0; j <= c; j++) {
        dp[0][j] = p[0][j];
    }

    for (int i = 1; i < m; i++) {
        for (int j = 0; j <= c; j++) {
            dp[i][j] = dp[i-1][j];
            for (int k = 1; k <= j; k++) {
                dp[i][j] = max(dp[i][j], dp[i-1][j-k] + p[i][k]);
            }
        }
    }

    ll ans = 0;
    for (int j = 0; j <= c; j++) {
        ans = max(ans, dp[m-1][j]);
    }
    cout << ans << endl;
}
