#include <bits/stdtr1c++.h>

using namespace std;
typedef double ld;
typedef complex<ld> pt;

inline ld sq(ld x) { return x*x; }

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    int x1, y1, x2, y2, x3, y3; cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3;
    ld r; cin >> r;

    pt p1(x1, y1), p2(x2, y2), p3(x3, y3);

    ld a = abs(p2-p1), b = abs(p3-p2), c = abs(p1-p3);

    ld area = 1.0/4.0*sqrt(4*a*a*b*b-sq(a*a+b*b-c*c));
    ld r2 = area/((a+b+c)/2.0);

    cout << setprecision(16) << fixed << (r2-r)/r*100.0 << endl;
}
