#include <bits/stdtr1c++.h>
using namespace std;

int mx[101], mx2[101];

int main() {
  ios::sync_with_stdio(0);
  int c, m; cin >> c >> m;
  for (int i = 0; i < m; i++) {
    for (int j = 1; j <= c; j++) {
      int x; cin >> x;
      for (int k = c-j; k >= 0; k--)
        mx2[j+k] = max(mx2[j+k],mx[k]+x);
    }
    memcpy(mx, mx2, sizeof mx);
  }
  int mxx = 0;
  for (int i = 1; i <= c; i++)
    mxx = max(mxx, mx[i]);
  cout << mxx << "\n";
  return 0;
}
