#include <bits/stdtr1c++.h>
using namespace std;

typedef long double ld;
typedef complex<ld> pt;

pt p[3];

int main() {
  ios::sync_with_stdio(0);
  for (int i = 0; i < 3; i++) {
    ld x, y; cin >> x >> y;
    p[i] = pt(x,y);
  }
  ld r; cin >> r;
  ld a = abs(p[0]-p[1]);
  ld b = abs(p[1]-p[2]);
  ld c = abs(p[2]-p[0]);
  ld A = sqrt(4*a*a*b*b-(a*a+b*b-c*c)*(a*a+b*b-c*c))/4;
  ld rc = 2*A/(a+b+c);
  ld p = rc/r-1;
  cout << p*100 << "\n";
  return 0;
}
