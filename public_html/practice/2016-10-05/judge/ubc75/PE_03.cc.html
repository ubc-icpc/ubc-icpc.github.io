int main() {
	ios_base::sync_with_stdio(0);
	int n,m,l, a,b,c;
	cin >> n >> m >> l;
	int j = 0;
	int totalcost = 0;
	int mincost = 0;
	bool seen[n+2];
	rep(i,n+1){seen[i] = false;}
	priority_queue<  pair<double,int> , vector<pair<double,int>>, greater<pair<double,int>> > pq;
	vector<vector<pair<int,int>>> graph(n+2);


	rep(i,m){
		cin >> a >> b >> c;
		if(j<l){
			totalcost+=c;
		}
		graph[a].push_back(make_pair(c,b));
		graph[b].push_back(make_pair(c,a));

		j++;
	}

	//seen[1] = true;

	//pq.push();
	pair<int, int> temp;
	pair<int, int> tempn;
	pq.push(make_pair(0,1));
	while(!pq.empty()){
		temp = pq.top(); pq.pop();
		
		if(seen[temp.second])
			continue;

		if(!seen[temp.second]){
			seen[temp.second] = true;	
			mincost+=temp.first;
			//cout << "prim" << temp.first << temp.second << endl;
		}
		
		for(int i = 0; i<graph[temp.second].size(); i++){
			tempn = graph[temp.second][i];
			if(!seen[tempn.second]){
				pq.push(tempn);
			}
		}

	}
	bool conn = true;
	rep(i,n){
		if(!seen[i+1]){
			//cout << "NOT SEEN" << i+1 << endl;
			conn = false;
		}
	}
	if(mincost<=totalcost && conn){
		cout << "possible" << endl;
	}

	else
		cout << "impossible" << endl;


    
    return 0;
}
