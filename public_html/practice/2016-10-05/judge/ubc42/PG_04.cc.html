#include <bits/stdc++.h>
using namespace std;
using ll = long long int;

#define Ds(A, B) ((x##A - x##B) * (x##A - x##B) + (y##A - y##B) * (y##A - y##B))

int main()
{
    ll x0, y0, x1, y1, x2, y2; cin >> x0 >> y0 >> x1 >> y1 >> x2 >> y2;
	double r0; cin >> r0;

	ll as = Ds(0, 1);
	ll bs = Ds(0, 2);
	ll cs = Ds(1, 2);

	double A = sqrt(4 * as * bs - (as + bs - cs) * (as + bs - cs)) / 4.0;
	double r = A * 2 / (sqrt(as) + sqrt(bs) + sqrt(cs));

	cout << fixed << setprecision(10) << (r - r0) / r0 * 100 << endl;
}
