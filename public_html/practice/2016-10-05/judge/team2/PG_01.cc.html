#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;
typedef long double ld;

ll X1, Y1, X2, Y2, X3, Y3, a, b, c;
ld r, ir;

int main () {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> X1 >> Y1 >> X2 >> Y2 >> X3 >> Y3 >> ir;

  a = (X1 - X2) * (X1 - X2) + (Y1 - Y2) * (Y1 - Y2);
  b = (X2 - X3) * (X2 - X3) + (Y2 - Y3) * (Y2 - Y3);
  c = (X1 - X3) * (X1 - X3) + (Y1 - Y3) * (Y1 - Y3);

  r = sqrt(4 * a * b - (a + b - c) * (a + b - c)) / 2 / (sqrt(a) + sqrt(b) + sqrt(c));

  cout << (r - ir) / ir * 100 << endl;

  return 0;
}
