#include<bits/stdc++.h>

using namespace std;

int c, m, p[104][103];

int dp[104][104];

int main(){
  ios_base::sync_with_stdio(false);cin.tie(0);
  cin >> c >> m;
  for(int i=0;i<m;i++){
    for(int j=0;j<c;j++){
      cin >>p[i+1][j+1];
    }
  }
  for(int i=1;i<m+1;i++){
    for(int j=0;j<=c;j++){
    for(int k=j;k<=c;k++){
      dp[i][k-j] = max(max(dp[i][k-j],dp[i-1][k-j]),dp[i-1][k]+p[i][j]);
      //cout << i << " " << j << " " << k << " " << k-j << " " << dp[i][k-j]<< endl;
    }
    }
  }
  int maxs= 0;
  for(int i=0;i<c;i++){
    //cout << m-1 << endl;
    //cout << dp[m-1][i] <<  " " << i << endl;
    maxs = max(maxs,dp[m][i]);
  }
  cout << maxs << endl;
}
