#include<bits/stdc++.h>

#define ss(a)   scanf("%d", &a)
#define pb(a)   push_back(a)

using namespace std;
typedef long long ll;

int n, a, b, c, xi, xc;

map<string, int> ma;

vector<int> adj[30004];

int getval(string s){
    if(ma.count(s)) return ma[s];
    else {
        ma[s] = c++;
        return ma[s];
    }
}

bool vis[33333];
bool ancvis[33333];

bool dfs(int a){
   // cout <<"at"<<  a << " " << x << endl;
    ancvis[a] = true;
    vis[a] = 1;
    for(auto i: adj[a]){
    //    cout << i << " " << vis[i] <<  endl;
        if(ancvis[i]) return true;
        if(vis[i]) continue;
        if(dfs(i)) return true;
    }
    ancvis[a] = false;
    return false;
}

int main(){
    ios_base::sync_with_stdio(false);cin.tie(0);
    c = 0;
    cin >> n;
    string s1, sign, s2;
    for(int i=0;i<n;i++){
        cin >> s1 >> sign >> s2;
        a = getval(s1); b = getval(s2);
 //       cout << a << " " << b << endl;
        if(sign[0]=='>')
            adj[a].pb(b);
        else
            adj[b].pb(a);
    }
  //  cout << endl;
    bool failed = false;
    for(int i=0;i<c;i++){
        xc = xi+1;
        if (!vis[i])
            if (dfs(i)) failed = true;
    }
    if (failed) cout << "impossible";
    else cout << "possible";
}
