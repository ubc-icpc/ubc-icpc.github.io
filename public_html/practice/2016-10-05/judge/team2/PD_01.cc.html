#include <bits/stdc++.h>
using namespace std;

void getinput(__int128& i) {
  i = 0;
  string s; cin >> s;
  for (char c : s) {
    i *= 10;
    i += c - '0';
  }
}

int main() {
  __int128 a11, a12, a21, a22;
  getinput(a11);
  getinput(a12);
  getinput(a21);
  getinput(a22);
  string bitstring = "";
  while(!(a11 == 1 && a12 == 0 && a21 == 0 && a22 == 1)) {
    if(a11 > a12 || a21 > a22) {
      a11 -= a12;
      a21 -= a22;
      bitstring.append("0");
    } else {
      a12 -= a11;
      a22 -= a21;
      bitstring.append("1");
    }
  }

  reverse(bitstring.begin(), bitstring.end());
  cout << bitstring << endl;
  return 0;
}
