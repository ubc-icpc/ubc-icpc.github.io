one = raw_input()
two = raw_input()
a = int(one.split()[0])
b = int(one.split()[1])
c = int(two.split()[0])
d = int(two.split()[1])
res = ''

while a != 1 or b != 0 or c != 0 or d != 1:
    if a >= b and c >= d:
        a -= b
        c -= d
        res += '0'
    else:
        b -= a
        d -= c
        res += '1'
print res[::-1]
