#include <bits/stdc++.h>

using namespace std;

int dp[200][200];
int val[200][200];

int main() {
  int C, M; cin >> C >> M;
  for (int i = 0; i < M; ++i) {
    for (int j = 0; j < C; ++j) {
      int num; cin >> num;
      val[i+1][j+1] = num;
    }
  }
  for (int i = 1; i <= M; ++i) {
    for (int j = 1; j <= C; ++j) {
      int msf = -1;
      for (int k = 0; k <= j; ++k) {
        msf = max(msf, dp[i-1][k] + val[i][j-k]);
      }
      dp[i][j] = max(dp[i][j-1], msf);
      //cout << dp[i][j] << " ";
    }
    //cout << endl;
  }
  cout << dp[M][C] << endl;
  return 0;
}
