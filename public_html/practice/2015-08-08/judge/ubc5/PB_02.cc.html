#include <iostream>
#include <cmath>
#include <complex>
#include <algorithm>
using namespace std;

typedef long double ld;
typedef complex<ld> pt;

pt c[1000];
int n[1000];

int main() {
  int t; cin >> t;
  while (t--) {
    int h; cin >> h;
    for (int i = 0; i < h; i++) {
      ld x, y;
      cin >> x >> y;
      c[i] = pt(x,y);
      cin >> n[i];
    }
    int mx = 0;
    for (int i = 0; i < h; i++) {
      if (c[i].imag() > 1000)
        continue;
      ld x = sqrt(1000000-c[i].imag()*c[i].imag());
      pt c1(c[i].real()-x,0);
      pt c2(c[i].real()+x,0);
      int mx1 = 0, mx2 = 0;
      for (int j = 0; j < h; j++) {
        if (abs(c1-c[j]) <= 1000)
          mx1 += n[j];
        if (abs(c2-c[j]) <= 1000)
          mx2 += n[j];
      }
      mx = max(mx, mx1);
      mx = max(mx, mx2);
    }
    cout << mx << "\n";
  }
  return 0;
}
