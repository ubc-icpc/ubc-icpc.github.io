#include <iostream>
#include <queue>
#include <utility>
using namespace std;

typedef pair<int,int> pii;

const int dx[4] = {1,-1,0,0};
const int dy[4] = {0,0,1,-1};

int l[100][100];

int main() {
  int t; cin >> t;
  while (t--) {
    int y, x, n, h;
    cin >> y >> x >> n >> h;
    for (int i = 0; i < y; i++) {
      for (int j = 0; j < x; j++) {
        char c; cin >> c;
        l[i][j] = c-'0';
      }
    }
    for (int i = 0; i < n; i++) {
      int yi, xi;
      cin >> yi >> xi;
      yi--; xi--;
      queue<pii> q;
      q.push(make_pair(yi,xi));
      while (q.size()) {
        int yf = q.front().first, xf = q.front().second;
        q.pop();
        l[yf][xf]++;
        if (l[yf][xf] > h) {
          l[yf][xf] -= 4;
          for (int j = 0; j < 4; j++) {
            int yy = yf+dy[j], xx = xf+dx[j];
            if (0 <= yy && yy < y && 0 <= xx && xx < x)
              q.push(make_pair(yy,xx));
          }
        }
      }
    }
    for (int i = 0; i < y; i++) {
      for (int j = 0; j < x; j++)
        cout << l[i][j];
      cout << "\n";
    }
  }
  return 0;
}
