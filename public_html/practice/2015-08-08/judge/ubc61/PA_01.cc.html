#include <bits/stdtr1c++.h>
using namespace std;

int board[105][105];

int ymax,xmax,n,h;

void addsand(int y,int x) {
	if (++board[y][x] > h) {
		board[y][x] -= 4;
		if (y>0) addsand(y-1,x);
		if (y<ymax-1) addsand(y+1,x);
		if (x>0) addsand(y,x-1);
		if (x<xmax-1) addsand(y,x+1);
	}
}

int main() {
	int t; cin >> t;
	while (t--) {
		cin >> ymax >> xmax >> n >> h;
		for (int j=0; j<ymax; j++) {
			string s; cin >> s;
			for (int i=0; i<xmax; i++) {
				board[j][i] = s[i] - '0';
			}
		}
		for (int i=0; i<n; i++) {
			int y1,x1; cin >> y1 >> x1;
			addsand(y1-1,x1-1);
		}
		for (int y=0; y<ymax; y++) {
			for (int x=0; x<xmax; x++) {
				cout << board[y][x];
			}
			cout << endl;
		}
	}
	return 0;
}
