#include <bits/stdtr1c++.h>
using namespace std;

typedef pair<int,int> pii;
typedef long double ld;

pii homes[1005];
int num[1005];

ld dist(pii p1, pii p2) {
	return (ld)(p1.first-p2.first)*(p1.first-p2.first) + (ld)(p1.second-p2.second)*(p1.second-p2.second);
}

int main() {
	int t; cin >> t;
	while (t--) {
		int h; cin >> h;
		for (int i=0; i<h; i++) {
			int x,y,n; cin >> x >> y >> n;
			homes[i] = make_pair(x,y);
			num[i] = n;
		}
		int ans = 0;
		for (int i=0; i<h; i++) {
			if (homes[i].second>1000 || homes[i].second <-1000)
				continue;
			ld xdist = sqrt(1000000 - homes[i].second*homes[i].second);
			int num1 = 0, num2 = 0;
			for (int j=0; j<h; j++) {
				if (dist(homes[j],make_pair(homes[i].first+xdist,0)) <= 1000000)
					num1+=num[j];
				if (dist(homes[j],make_pair(homes[i].first-xdist,0)) <= 1000000)
					num2+=num[j];
			}
			ans = max(ans,num1);
			ans = max(ans,num2);
		}
		cout << ans << endl;
	}
	return 0;
}
