#include <bits/stdtr1c++.h>

using namespace std;

const int maxn = 105;
int a[maxn][maxn];
int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};
int main() {
    int t; cin >> t;
    while (t--) {
        int r, c, n, h; cin >> r >> c >> n >> h;
        memset(a, 0xc0, sizeof a);
        for (int i = 1; i <= r; i++) {
            for (int j = 1; j <= c; j++) {
                char x; cin >> x;
                a[i][j] = x-'0';
            }
        }
        
        for (int i = 0; i < n; i++) {
            int x, y; cin >> x >> y;
            queue<pair<int, int>> q({{x, y}});
            while (!q.empty()) {
                x = q.front().first, y = q.front().second;
                q.pop();
                a[x][y]++;
                if (a[x][y] > h) {
                    a[x][y] -= 4;
                    for (int d = 0; d < 4; d++) {
                        q.push({x+dx[d], y+dy[d]});
                    }
                }
            }
        }
        
        for (int i = 1; i <= r; i++) {
            for (int j = 1; j <= c; j++) {
                cout << a[i][j];
            }
            cout << '\n';
        }
    }
	return 0;
}
