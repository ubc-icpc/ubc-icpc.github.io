#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
struct evt {
    ld x; int c;
    evt() {}
    evt(ld x, int c) : x(x), c(c) {}
    bool operator<(const evt& o) const {
        return x < o.x || (x == o.x && c > o.c);
    }
} e[105];

const ld R = 1000;
int main() {
    int t; cin >> t;
    while (t--) {
        int n, m = 0; cin >> n;
        for (int i = 0; i < n; i++) {
            ld x, y; int c; cin >> x >> y >> c;
            if (abs(y) > R) continue;
            e[m++] = evt(x-sqrt(R*R-y*y), +c);
            e[m++] = evt(x+sqrt(R*R-y*y), -c);
        }
        sort(e, e+m);
        
        int ans = 0, cur = 0;
        for (int i = 0; i < m; i++) {
            cur += e[i].c;
            ans = max(ans, cur);
        }
        cout << ans << endl;
    }
	return 0;
}
