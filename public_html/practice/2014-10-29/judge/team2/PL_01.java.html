import java.util.*;

public class L {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int n = s.nextInt();

        s.nextLine();

        Map<String, List<String>> dictionary = new HashMap<>();
        for (int i = 0; i < n; ++i) {
            String curr = s.nextLine();
            String[] lang = curr.split("[,.!;?() ]+");
            List<String> words = new LinkedList<>();
            for (int j = 1; j < lang.length; ++j) {
                words.add(lang[j].toLowerCase());
            }
            dictionary.put(lang[0], words);
        }

        s.nextLine();

        while (s.hasNextLine()) {
            String line = s.nextLine();
            if (null == line || "" == line) break;
            String[] words = line.split("[,.!;?() ]+");
            boolean matched = false;
            for (int i = 0; i < words.length; ++i) {
                for (Map.Entry<String, List<String>> e : dictionary.entrySet()) {
                    if (e.getValue().contains(words[i].toLowerCase())) {
                        System.out.println(e.getKey());
                        matched = true;
                        break;
                    }
                }
                if (matched) break;
            }
        }
    }
}
