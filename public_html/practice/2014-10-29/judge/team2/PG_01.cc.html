#include <stdio.h>

int main() {
    unsigned long long gen[68];
    gen[0] = 1;
    gen[1] = 1;
    gen[2] = 2;
    gen[3] = 4;
    for (int i = 4; i < 68; ++i) {
        gen[i] = gen[i-1] + gen[i-2] + gen[i-3] + gen[i-4];
    }

    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        printf("%llu\n", gen[n]);
    }

    return 0;
}
