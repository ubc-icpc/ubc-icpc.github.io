#include <stdio.h>

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        long long d;
        scanf("%d %lld", &n, &d);
        int count = 0;
        while (n--) {
            long long v, f, c;
            scanf("%lld %lld %lld", &v, &f, &c);
            if (f * v >= d * c) ++count;
        }
        printf("%d\n", count);
    }
    return 0;
}
