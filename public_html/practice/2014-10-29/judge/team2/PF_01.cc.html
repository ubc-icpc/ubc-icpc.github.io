#include <cstdio>
#include <cmath>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
    int n;
    while(true) {
        scanf("%d", &n);
        if (n == -1) return 0;
        int upper = (int)sqrt(n);
        vector<int> all;
        int sum = 1;
        for (int i = 2; i <= upper; ++i) {
            if (!(n % i)) {
                all.push_back(i);
                all.push_back(n / i);
                sum += i;
                sum += (n / i);
            }
        }
        if (n == sum) {
            sort(all.begin(), all.end());
            printf("%d = 1", n);
            for (int i = 0; i < all.size(); ++i) {
                printf(" + %d", all[i]);
            }
            printf("\n");
        } else {
            printf("%d is NOT perfect.\n", n);
        }
    }
}
