#include<iostream>
#include <vector>
#include <string>
using namespace std;
string name[101];
vector<string> voc[101];
bool found;
int n;
void search(string word) {
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < voc[i].size(); j++) {
            if(word == voc[i][j]) {
                found = true;
                cout<<name[i]<<endl;
                return ;
            }
        }
    }
}
int main() {
    cin>>n;
    string line;
    string word;
    for(int i = 0; i < n; i++) {
        cin>>name[i];
        getline(cin, line);
        word.clear();
        for(int j = 1; j < line.length(); j++) {
            if(line[j] == ' ' && !word.empty()) {
                voc[i].push_back(word);
                word.clear();
            }
            else {
                if(line[j] >= 'A' && line[j] <='Z')
                    line[j] = line[j] - 'A' + 'a';
                word.push_back(line[j]);
            }
        }
        if(!word.empty()) {
            //cout<<word<<endl;
            voc[i].push_back(word);
        }
    }
 
    /*for(int i= 0;i < n; i++) {
        cout<<name[i]<<": ";
        for(int j = 0; j < voc[i].size(); j++) {
            cout<<voc[i][j]<<" ";
        }
        cout<<endl;
    }*/
    getline(cin, line);
    while(getline(cin, line)) {
        word.clear();
        found = false;
        //cout<< "line is :"<<line<<endl;
        for(int i = 0; i < line.length(); i++) {
            if(found)
                break;
            if(line[i] >= 'A' && line[i] <= 'Z')
                line[i] = line[i] + 'a' - 'A';

            if(line[i] == ',' || line[i] == '.' || line[i] == ' ' || line[i] == '!' || line[i] == ';'
              || line[i] == '?' || line[i] == ')' || line[i] == '(') {
                if(!word.empty()) {
                    //cout<<"word is:"<<word<<endl;
                    search(word);
                    word.clear();
                    if(found)
                        break;
                }
            }
            else {
                word.push_back(line[i]);
            }
        }
        if(!found && !word.empty()) {
       //     cout<<"word is:"<<word<<endl;
            search(word);
        }
    }
}
