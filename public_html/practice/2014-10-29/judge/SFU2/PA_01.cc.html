#include<iostream>
using namespace std;

int main() {
    int test;
    cin>>test;
    int n,d;
    int v,f,c;
    while(test --) {
        cin>>n>>d;
        int ans = 0;
        for(int i = 0; i < n; i++) {
            cin>>v>>f>>c;
            if(d * 1.0/ v * c <= f * 1.0)
                ans++;
        }
        cout<<ans<<endl;

    }
}
