#include<iostream>
#include <vector>
#include <cmath>
#include <algorithm>
using namespace std;
int n;
vector<int> stack;
bool comp(int a, int b) {
    return a < b;
}
void solve(int n) {
    stack.clear();
    stack.push_back(1);
    for(int i = 2; i <= sqrt(n); i++) {
        if(n % i == 0){
            int temp = n / i;
            stack.push_back(i);
            if(temp > 1 && temp != i) {
                stack.push_back(temp);
            }
        }
        
    }
    sort(stack.begin(), stack.end(), comp);
    int sum = 0;
    for(int i = 0; i < stack.size(); i++) {
        sum += stack[i];
    }
    if(sum != n) {
        cout<<n<<" is NOT perfect."<<endl;
        return ;
    }
    cout<<n<<" = 1";
    for(int i = 1; i < stack.size(); i++) {
        cout<<" + "<<stack[i];
    }
    cout<<endl;
}
int main() {
    int n;
    while(true) {
        cin>>n;
        if(n < 0) {
            break;
        }
        solve(n);
    }
}
