#include <iostream>
using namespace std;

int main(){
	int co;
	cin >> co;
	int n, d, total;
	long long v, f, c;
	for (int i = 0; i < co; ++i)
	{
		total = 0;
		cin >> n >> d;
		for(int j = 0; j< n; j++){
			cin >> v >> f >> c;
			if (f*1.0/c * v >= d)
			{
				total ++;
			}
		}
		cout << total << endl;

	}

}
