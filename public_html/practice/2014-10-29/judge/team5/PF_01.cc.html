#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <algorithm>
#include <cmath>
#include <map>
#include <sstream>
#include <complex>
#include <vector>
#include <set>

using namespace std;

typedef long long ll;

int main()
{
  int n;
  cin >> n;
  while (n!= -1) {
    int factors[100];
    int count = 0;
    int sum = 1;
    
    for (int i=2; i<=sqrt(n); i++) {
      if (i*i == n) {
        factors[count] = i;
        count++;
        sum += i;
        continue;
      }
      if (n%i == 0) {
        factors[count] = i;
        count++;
        sum += i;
        factors[count] = n/i;
        count++;
        sum += n/i;
      }
    }
    
    sort(factors, factors + count);
    
    if (sum != n) {
      cout << n << " is NOT perfect."<< endl;
      cin>>n;
      continue;
    }
    
    else {
      cout << n << " = 1";
      for (int j=0; j<count; j++) {
        cout << " + " << factors[j];
      }
      cout << endl;
    }
    cin>>n;
  }
}
