#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>

int main() {

	int T;
	cin>>T;
	for (int t=0;t<T;t++) {
		int N;
		cin>>N;
		vector<long long> planets;
		for (int i=0;i<N;i++) {
			long long x;
			cin>>x;
			planets.push_back(x);
		}
		sort(planets.begin(),planets.end());
		long long ans=10000000000;
		for (int i=0;i<N-1;i++) {
		
			long long dist = max(planets[N-1]-planets[i+1],planets[i]-planets[0]);
			if (dist<ans) ans=dist;
		}
				

		cout<<ans<<endl;
	}
	
	return 0;
}
