using namespace std;

typedef long long ll;

int main()
{
  ll ar[68];
  
  ar[0] = 1;
  ar[1] = 1;
  ar[2] = 2;
  ar[3] = 4;
  
  for (int i=4; i<68; i++) {
    ar[i] = ar[i-1] + ar[i-2] + ar[i-3] + ar[i-4];
  }
  
  int t;
  cin >> t;
  while (t>0) {
    int s;
    cin >> s;
    
    cout << ar[s] << endl;
    
    t--;
  }
}
