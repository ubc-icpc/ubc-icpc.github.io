#include <iostream>
using namespace std;

int main() {

	int T;
	cin>>T;
	for (int t=0;t<T;t++) {
		int N,D;
		cin>>N>>D;
		int ans=0;
		for (int i=0;i<N;i++) {
			int v,f,c;
			cin>>v>>f>>c;
			if (f*v>=D*c) ans++;
		}
	cout<<ans<<endl;
	}
	
	return 0;
}
