#include <iostream>

using namespace std;
typedef long long ll;

int main() {
	int nc; cin >> nc;
	for (int cs = 1; cs <= nc; cs++) {
		int n; ll d; cin >> n >> d;
		int count = 0;
		for (int i = 0; i < n; i++) {
			ll v, f, c; cin >> v >> f >> c;
			ll fuelneeded = d * c / v;
			if (d*c % v != 0) fuelneeded++;
			if (fuelneeded <= f) count++;
		}
		cout << count << endl;
	}
}
