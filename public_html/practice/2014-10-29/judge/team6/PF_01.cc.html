#include <bits/stdc++.h>

using namespace std;

int main() {
	int n;
	while (cin >> n) {
		if (n==-1) break;

		long long ans = 0;
		vector<int> f;
		for (int i = 1; i*i <= n; i++) {
			if (n%i == 0) {
				ans += i;
				f.push_back(i);
				if (n/i != i) {
					ans += n/i;
					f.push_back(n/i);
				}
			}
		}
		if (ans != 2*n) {
			cout << n << " is NOT perfect." << endl;
		} else {
			cout << n << " =";
			sort(f.begin(), f.end());
			for (int i = 0; i < f.size()-2; i++) {
				cout << " " << f[i] << " + ";
			}
			cout << f[f.size()-2] << endl;
		}
	}
}
