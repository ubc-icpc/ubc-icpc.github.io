#include <iostream>
#include <algorithm>

using namespace std;

typedef long long ll;
ll a[5006];
int main() {
	int nc; cin >> nc;
	for (int cs = 1; cs <= nc; cs++) {
		int n; cin >> n;
		for (int i = 0; i < n; i++) {
			cin >> a[i];
			a[i] *= 2;
		}
		
		sort(a, a+n);
		ll ans = (a[n-1]-a[0])/2;
		for (int i = 0; i < n-1; i++) {
			ll a0 = (-a[i]-a[0] + a[n-1]+a[i+1])/2;
			ll a1 = a[i]-a[0];
			ll a2 = a[n-1]-a[i+1];
			ans = min(ans, min(a0,max(a1,a2)));
		}

		cout << (ans+1)/2 << endl;
	}
}
