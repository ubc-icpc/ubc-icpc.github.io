import java.util.*;
public class L {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		sc.nextLine();
		ArrayList<String>[] langs = new ArrayList[n];
		String[] langnames = new String[n];
		for (int i = 0; i < n; i++) {
			langs[i] = new ArrayList<String>();
			String[] tokens = sc.nextLine().split("\\s+");
			langnames[i] = tokens[0].trim();
			for (int j = 1; j < tokens.length; j++) {
				tokens[j] = tokens[j].trim();
				if (tokens[j].length() > 0) {
					langs[i].add(tokens[j].toLowerCase());
				}
			}
		}
		sc.nextLine();
		while (sc.hasNext()) {
			String[] tokens = sc.nextLine().split("(\\s|\\,|\\.|!|;|\\?|\\(|\\))+");
			boolean found = false;
			for (int i = 0; i < tokens.length && !found; i++) {
				tokens[i] = tokens[i].trim().toLowerCase();
				for (int j = 0; j < n && !found; j++) {
					for (String word : langs[j]) {
						if (word.equals(tokens[i])) {
							found = true;
							System.out.println(langnames[j]);
						}
					}
				}
			}
		}
	}
}
