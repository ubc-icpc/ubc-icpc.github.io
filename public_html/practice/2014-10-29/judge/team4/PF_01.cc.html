#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
    int n;
    while(cin>>n){
        if(n == -1) break;
        vector<int> v;
        int sum=1;
        for(int i = 2 ; i*i< n; i++){
            if(n%i == 0){
                sum += n/i;
                if(n/i != i){
                    sum += i;
                    v.push_back(i);
                }
                
                v.push_back(n/i);
                //cout<< i <<" "<< n/i<<endl;

        
            }
        }
        if(sum != n){
            cout<<n<<" is NOT perfect."<<endl;
        }
        else{
            sort(v.begin(), v.end());
            cout<<n<<" = 1";
            for(int i = 0; i< v.size() ; i++){
                cout<< " + "<< v[i];
            }
            cout<<endl;
        }
    }
    return 0;
}
