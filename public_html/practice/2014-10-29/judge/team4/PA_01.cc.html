#include<iostream>

using namespace std;

int main(){
    int t, n, d,  count;
    int v, f, c;
    cin>>t;
    while(t--){
        cin>>n>>d;
        count = 0;
        while(n--){
            cin>>v>>f>>c;
            if(d*c <= f*v){
                count++;
            }
        }
        cout<<count<<endl;
    }
    return 0;
}
