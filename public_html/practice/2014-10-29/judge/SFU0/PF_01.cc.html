#include <cstdio>

const int MAXN = 100005;

int div[MAXN];

int is_Perfect(int n){
	int sum=0;
	div[0] = 0;
	for(int i=1;i*2<=n;i++)
		if(n%i == 0){
			div[++div[0]] = i;
			sum += i;
		}
	if(sum == n) return 1;
	return 0;
}

int main(){
	int n;

	scanf("%d",&n);
	while(n != -1){
		if(is_Perfect(n)){
			printf("%d = %d",n,div[1]);
			for(int i=2;i<=div[0];i++)
				printf(" + %d",div[i]);
			printf("\n");
		}
		else printf("%d is NOT perfect.\n",n);

		scanf("%d",&n);
	}

	return 0;
}
