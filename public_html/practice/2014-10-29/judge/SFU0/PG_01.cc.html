#include <iostream>

using namespace std;

const int MAXN = 68;

long long T[MAXN];

int main(){
	T[0] = 1;
	T[1] = 1;
	T[2] = 2;
	T[3] = 4;
	for(int i=4;i<MAXN;i++) 
		T[i] = T[i-1] + T[i-2] + T[i-3] + T[i-4];

	int t;
	int n;

	cin >> t;
	for(int i=0;i<t;i++){
		cin >> n;
		cout << T[n] << endl;
	}

	return 0;
}
