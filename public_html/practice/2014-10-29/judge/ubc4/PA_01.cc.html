#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <vector>
#include <string>
#include <cstring>
#include <set>
#include <map>
#include <sstream>
#include <cmath>
using namespace std;

int main() {
	int T;
	int n, d;
	int ans;
	scanf("%d", &T);
	while (T-- > 0) {
		scanf("%d %d", &n, &d);
		ans = 0;
		int v, f, c;
		for (int i = 0; i < n; i++) {
			scanf("%d %d %d", &v, &f, &c);
			if (double(f) / c * v >= d)
				ans++;
		}
		printf("%d\n", ans);
	}
}
