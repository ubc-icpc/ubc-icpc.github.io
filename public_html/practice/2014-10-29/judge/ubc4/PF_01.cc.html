#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <vector>
#include <string>
#include <cstring>
#include <set>
#include <map>
#include <sstream>
#include <cmath>
using namespace std;

int main() {
	int n;
	while (scanf("%d", &n), n > 0) {
		vector<int> tv;
		tv.push_back(1);
		for (int i = 2; i * i <= n; i++) {
			if (n % i == 0) {
				tv.push_back(i);
				if (i * i != n) tv.push_back(n / i);
			}
		}
		int sum = 0;
		for (int i = 0; i < tv.size(); i++)
			sum += tv[i];
		if (sum != n) {
			printf("%d is NOT perfect.\n", n);
		} else {
			sort(tv.begin(), tv.end());
			printf("%d = ", n);
			for (int i = 0; i < tv.size(); i++) {
				if (i == 0)
					printf("%d", tv[i]);
				else printf(" + %d", tv[i]);
			}			
			printf("\n");
		}
	}
}
