#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <vector>
#include <string>
#include <cstring>
#include <set>
#include <map>
#include <sstream>
#include <cmath>
using namespace std;

const int maxn = 5000;
int n;
int a[maxn];

int main() {
	int T;
	scanf("%d", &T);
	while (T--) {
		scanf("%d", &n);
		for (int i = 0; i < n; i++) {
			scanf("%d", a + i);
		}
		sort(a, a + n);
		double ans = 1e20;
		for (int i = 0; i < n; i++) 
			for (int j = i + 1; j < n; j++) {
				double dis = (a[i] - a[0]) / 2.0 + (a[n - 1] - a[j]) / 2;
				dis = max(dis, (double)(a[n - 1] - a[i + 1]));
				dis = max(dis, (double)(a[j - 1] - a[0]));
				ans = min(ans, dis);
			}
		printf("%d\n", (int)(ans + 0.6));
	}
}
