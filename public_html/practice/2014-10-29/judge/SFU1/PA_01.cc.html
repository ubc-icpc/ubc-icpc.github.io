#include <cstdio>

using namespace std;

int main() {
	int ntests;

	scanf("%d",&ntests);

	while(ntests--) {
		int d,nships;

		scanf("%d %d",&nships,&d);

		int nok = 0;

		for(int i = 0;i < nships;i++) {
			double v,f,c;

			scanf("%lf %lf %lf",&v,&f,&c);

			double t = v * (f / c);

			if(t >= d) {
				nok++;
			}
		}

		printf("%d\n",nok);
	}

	return 0;
}
