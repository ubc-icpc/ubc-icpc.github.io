//package oct15Pratice;

import java.util.Scanner;


public class Favorites {
	public static void main(String args[]) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		while (n != -1) {
			System.out.println(output(n));
			n = sc.nextInt();
		}
		sc.close();
	}
	
	public static String output(int n) {
		int[] fact = factors(n);
		String out;
		if (perfect(n, fact)) {
			out = n + " = ";
			for (int i=0; i<fact.length-2; i++) {
				if (fact[i] != 0)
					out = out + fact[i] + " + ";
			}
			if (fact[fact.length-2] != 0)
				out = out + fact[fact.length-2];
		}
		else
			out = n + " is NOT perfect.";
		
		
		
		return out;
	}
	
	public static int[] factors (int n) {
		int[] ret = new int[n];
		int back = 0;
		int front = n-1;
		
		for (int i=1; i*i < n; i++) {
			if (n%i == 0) {
				ret[back] = i;
				ret[front] = n/i;
				back++;
				front--;
			}
		}
		return ret;
	}
	
	public static boolean perfect (int n, int[] fact) {
		int sum = sum(fact);
		
		return (sum == n);
		
	}
	
	public static int sum(int[] arr) {
		int sum = 0;
		for (int i=0; i<arr.length-1; i++) {
			sum += arr[i];
		}
		return sum;
	}
}
