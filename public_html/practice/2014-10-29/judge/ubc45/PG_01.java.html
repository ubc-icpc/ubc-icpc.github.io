//package oct15Pratice;

import java.util.Scanner;

public class Tribbles {
	static long[] store;
	
	public static void main(String args[]) {
		Scanner sc = new Scanner(System.in);
		int numberOfTests = sc.nextInt();
		
		for (int i=0; i<numberOfTests; i++) {
			int n = sc.nextInt();
			store = new long[n+1];
			System.out.println(output(n));
		}
		sc.close();
	}
	
	public static long output(int n) {
		if (store[n] != 0)
			return store[n];
		
		if (n < 2) 
			store[n] = 1;
		else if (n == 2)
			store[n] = 2;
		else if (n == 3)
			store[n] = 4;
		else {
			store[n-1] = output(n-1);
			store[n-2] = output(n-2);
			store[n-3] = output(n-3);
			store[n-4] = output(n-4);
			store[n] = store[n-1] + store[n-2] + 
					   store[n-3] + store[n-4];
		}
			
		return store[n];
	}
	
}
