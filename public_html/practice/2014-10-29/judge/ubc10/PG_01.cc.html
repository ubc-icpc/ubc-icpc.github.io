#include <iostream>
using namespace std;
typedef long long ll;

int main() {
	int nc; cin >> nc;
	for (int cs =1 ; cs <= nc; cs++) {
		int g; cin >> g;
		ll dp[100];
		dp[0] = 1;
		dp[1] = 1;
		dp[2] = 2;
		dp[3] = 4;
		for (int i = 4; i <= g; i++) {
			dp[i] = dp[i-1] + dp[i-2] + dp[i-3] + dp[i-4];
		}
		cout << dp[g] << endl;
	}
}
