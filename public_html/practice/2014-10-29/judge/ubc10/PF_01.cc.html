#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	int n; cin >> n;
	while (n != -1) {
		int sum = 0;
		vector<int> blah;
		for (int i = 1; i*i <= n; i++) {
			if (n % i == 0) {
				sum += i;
				blah.push_back(i);
				int x = n / i;
				if (x != i && x != n) {
					blah.push_back(x);
					sum += x;
				}
				if (sum > n) break;
			}
		}
		if (sum == n) {
			sort(blah.begin(), blah.end());
			cout << n << " = " << blah[0];
			for (int i = 1; i < blah.size(); i++) {
				cout << " + " << blah[i];
			}
			cout << endl;
		} else {
			cout << n << " is NOT perfect." << endl;
		}
		cin >> n;
	}
}
