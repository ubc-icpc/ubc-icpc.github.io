#include <iostream>
using namespace std;
typedef long long ll;
int main() {
	int nc; cin >> nc;
	for (int cs = 1; cs <= nc; cs++) {
		int n; ll d; cin >> n >> d;
		int ans = 0;
		for (int i = 0; i < n; i++) {
			ll v, f, c; cin >> v >> f >> c;
			ll fuel = d * c / v;
			if ((d*c)%v != 0) fuel++;
			if (fuel <= f) ans++;
		}
		cout << ans << endl;
	}
}
