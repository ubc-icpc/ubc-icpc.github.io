import sys

lut = {
    'R': { 'R':(0,0), 'P':(0,1), 'S':(1,0) },
    'P': { 'R':(1,0), 'P':(0,0), 'S':(0,1) },
    'S': { 'R':(0,1), 'P':(1,0), 'S':(0,0) }
}

n = int(sys.stdin.next())

for _ in xrange(n):
  j = int(sys.stdin.next())
  s1, s2 = 0, 0
  for __ in xrange(j):
    a, b = sys.stdin.next().split(' ')
    d1, d2 = lut[a[0]][b[0]]
    s1 += d1; s2 += d2;
  if s1 == s2: print "TIE"
  elif s1 > s2: print "Player 1"
  else: print "Player 2"
