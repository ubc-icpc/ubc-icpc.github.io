import sys

def inp(): return sys.stdin.next().rstrip('\n')

def from_b(x):
    aa = map(int, x[::-1])
    bb = map(lambda i: (-2)**i, xrange(len(x)))
    return sum(map(lambda a,b: a*b, aa, bb))
    
def to_b_it(x):
    if x == 0: yield 0
    while x != 0:
        q = -(x/2)
        r = x - (-2*q)
        yield r
        x = q
 
def to_b(x):
    return list(reversed(list(to_b_it(x))))
