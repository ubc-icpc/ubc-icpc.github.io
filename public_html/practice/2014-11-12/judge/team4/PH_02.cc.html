#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int T;
    cin >> T;
    while (T--) {
        map<string,int> m;
        vector<string> v;
        string names;
        getline(cin, names);
        getline(cin, names);
        stringstream line(names);
        int i = 0;
        string s;
        while (line >> s) {
            m[s] = i;
            v.push_back(s);
            i++;
        }
        string start; cin >> start;
        int n; cin >> n;
        cout << v[(m[start] + n - 1) % i] << endl;
    }
    return 0;
}
