#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	for (int i=0;i<t;i++) {
		int n;
		cin>>n;
		int a=0;
		int b=0;
		for (int j=0;j<n;j++) {
			string s,t;
			cin>>s>>t;
			if (s=="R") {
				if (t=="P") b++;
				if (t=="S") a++;
			}
			if (s=="P") {
				if (t=="S") b++;
				if (t=="R") a++;
			}
			if (s=="S") {
				if (t=="R") b++;
				if (t=="P") a++;
			}
		}
		if (a>b) cout<<"Player 1"<<endl;
		if (b>a) cout<<"Player 2"<<endl;
		if (a==b) cout<<"TIE"<<endl;
	}
	

	return 0;
}
