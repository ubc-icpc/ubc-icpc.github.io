#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int t; cin >> t;
    string s;
    while (t--)
    {
        getline(cin, s);
        getline(cin, s);
        int ind = 0, ind2 = 0;
        vector<string> p;
        while (ind2 < s.size())
        {
            while (ind2 < s.size() && s[ind2] != ' ')
                ind2++;
            p.push_back(s.substr(ind,ind2-ind));
            ind2++;
            ind = ind2;
        }
        string c; int n;
        cin >> c >> n;
        ind = 0;
        while (p[ind] != c)
            ind++;
        cout << p[(ind+n-1)%p.size()] << "\n";
    }
    return 0;
}
