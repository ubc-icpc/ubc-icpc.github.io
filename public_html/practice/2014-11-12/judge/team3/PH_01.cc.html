#include <iostream>
#include <sstream>
#include <vector>
using namespace std;

int main(){
	int t; cin >> t; string trash;
	while (t--){
		getline(cin, trash);
		stringstream ss; string s;
		getline(cin, s);
		ss << s;
		string name;


		vector<string> ppl;
		while (ss >> name){
			ppl.push_back(name);
		}


		string start; cin >> start; int n; cin >> n;
		int stt;
		for (int i = 0; i < ppl.size(); i++){
			if (ppl[i]==start) stt = i;
		}
		int ans = (stt + n - 1)%(ppl.size());
		cout << ppl[ans] << endl;
	}
	return 0;
}
