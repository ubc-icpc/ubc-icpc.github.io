#include <iostream>

using namespace std;
int first_win(char a, char b){
	if (a=='R' && b=='S') return 1;
	if (a=='S' && b=='P') return 1;
	if (a=='P' && b =='R') return 1;
	if (a==b) return 0;
	return -1;
}


int main(){
	int test; cin >> test;
	for (int t = 0; t < test; t++){
		int n; cin >> n;
		int f = 0, s = 0;
		for (int r = 0; r < n; r++){
			char a, b; cin >> a >> b;
			int res = first_win(a, b);
			if (res == 1) f++;
			else if (res==-1) s++;
		}
		if (f > s) cout << "Player 1" << endl;
		else if (f==s) cout << "TIE" << endl;
		else cout << "Player 2" << endl;
	}


	return 0;
}
