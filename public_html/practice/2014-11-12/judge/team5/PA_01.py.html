import sys

stdin = sys.stdin

t = int(stdin.readline().strip())

for _ in range(t):
    n = int(stdin.readline().strip())
    p1 = 0
    p2 = 0
    for _ in range(n):
        h1, h2 = stdin.readline().strip().split()
        if h1 != h2:
            if h1 == 'R':
                if h2 == 'S':
                    p1 += 1
                    continue
                p2 += 1
            elif h1 == 'S':
                if h2 == 'P':
                    p1 += 1
                    continue
                p2 += 1
            elif h1 == 'P':
                if h2 == 'R':
                    p1 += 1
                    continue
                p2 += 1
    if p1 > p2:
        print("Player 1")
    elif p1 == p2:
        print("TIE")
    elif p1 < p2:
        print("Player 2")
