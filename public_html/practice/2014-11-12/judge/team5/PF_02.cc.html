#include <bits/stdc++.h>
using namespace std;
int main() {
    int t; cin >> t;
    while (t--) {
        char s; cin >> s;
        
        if (s == 'b') {
            string b; cin >> b;
            int d = 0;
            for (int i = b.size() - 1; i >= 0; i--)
                d += ((b[i] == '0' ? 0 : 1) << (b.size() - i - 1)) * (((b.size() - i) % 2 == 0) ? -1 : 1);
                
            cout << "From binary: " << b << " is " << d << endl;
        }
        else {
            int d; cin >> d;
            int od = d;
            string b = "";
            while (d) {
                if (!(d & 1)) b = "0" + b;
                else { 
                    b = "1" + b;
                    d -= 1;
                }
                d /= -2;
            }
            cout << "From decimal: " << od << " is " << (b == "" ? "0" : b) << endl;
        }
    }
}
