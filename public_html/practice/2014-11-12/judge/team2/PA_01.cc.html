#include <iostream>
#include <map>

using namespace std;

int main() {
    map<string, map<string, int> > m;
    map<string, int> a, b, c;
    a["R"] = 0;
    a["P"] = 1;
    a["S"] = -1;
    b["R"] = -1;
    b["P"] = 0;
    b["S"] = 1;
    c["R"] = 1;
    c["P"] = -1;
    c["S"] = 0;
    m["R"] = a;
    m["P"] = b;
    m["S"] = c;

    int n; cin >> n;
    while (n--) {
        int k, sum = 0;
        cin >> k;
        while (k--) {
            string x, y; cin >> x >> y;
            sum += m[x][y];
        }
        cout << ((sum > 0) ? "Player 2" : (sum == 0) ? "TIE" : "Player 1") << endl;
    }
    return 0;
}
