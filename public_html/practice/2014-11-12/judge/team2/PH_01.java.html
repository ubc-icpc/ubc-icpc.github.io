import java.util.*;

public class H {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int t = s.nextInt();
        for (int i = 0; i < t; ++i) {
            s.nextLine();
            String curr = s.nextLine();
            String[] names = curr.split(" ");
            String start = s.nextLine();
            int q = 0;
            for (int j = 0; j < names.length; ++j) {
                if (names[j].equals(start)) {
                    q = j;
                    break;
                }
            }
            int p = s.nextInt();
            System.out.println(names[(p + q - 1) % names.length]);
        }
    }
}
