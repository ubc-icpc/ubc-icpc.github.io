#include <iostream>
#include <sstream>
#include <vector>

using namespace std;

int main() {
	int nc; cin >> nc;
	for (int cs = 1; cs <= nc; cs++) {
		string s;
		getline(cin, s);
		getline(cin, s);
		string p; cin >> p;
		int rot; cin >> rot;

		vector<string> sss;
		stringstream ss;
		ss << s;
		string tmp;
		while (ss >> tmp) {
			sss.push_back(tmp);
		}
		
		int n = sss.size();
		int start = 0;
		while (sss[start] != p) start++;
		start = (start + rot - 1) % n;
		cout << sss[start] << endl;
	}
}
