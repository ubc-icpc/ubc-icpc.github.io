#include <iostream>
#include <vector>

int main() {
	int T;
	std::cin >> T;
	while(T--) {
		int N;
		std::cin >> N;
		int w1 = 0, w2 = 0;
		for(int i = 0; i < N; i ++) {
			char a, b;
			std::cin >> a >> b;
			if(a == 'R' && b == 'P') w2 ++;
			if(a == 'R' && b == 'S') w1 ++;
			if(a == 'S' && b == 'R') w2 ++;
			if(a == 'S' && b == 'P') w1 ++;
			if(a == 'P' && b == 'R') w1 ++;
			if(a == 'P' && b == 'S') w2 ++;
		}
		if(w1 == w2) std::cout << "TIE" << std::endl;
		else if(w1>w2) std::cout << "Player 1" << std::endl;
		else std::cout << "Player 2" << std::endl;
	}
	return 0;
}
