#include<iostream>
#include<vector>
#include<string>
#include<sstream>

using namespace std;
stringstream ss;
string str;

int main() {
  getline(cin, str);
  ss << str;
  int N;
  ss >> N;
  ss.clear();
  while (N--) {
    getline(cin, str);
    ss << str;
    string s;
    vector<string> list;
    while (ss >> s) {
      list.push_back(s);
    }
    ss.clear();
    getline(cin, str);
    ss << str;
    string start;
    ss >> start;
    ss.clear();
    int startindex = -10000;
    for (unsigned int i = 0; i < list.size(); i++)
      if (start == list[i]) {
	startindex = i;
	break;
      }
    getline(cin, str);
    ss << str;
    int count;
    ss >> count;
    ss.clear();
    int res = (count + startindex - 1) % (int)list.size();
    cout << list[res] << endl;
  }

  return 0;
}
