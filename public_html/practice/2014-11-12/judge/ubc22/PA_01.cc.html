#include<iostream>
using namespace std;

int main() {
  int N;
  cin >> N;

  while (N--) {
    int T;
    cin >> T;
    int total = 0;

    string astr, bstr;
    while (T--) {
      cin >> astr >> bstr;
      switch (bstr[0] - astr[0]) {
      case 2:
      case -3:
      case 1:
	total++;
	break;
      case -2:
      case 3:
      case -1:
	total--;
	break;
      }
    }

    if (total < 0)
      cout << "Player 2" << endl;
    else if (total > 0)
      cout << "Player 1" << endl;
    else
      cout << "TIE" << endl;

  }

  return 0;
}
