//package contest;

import java.util.Scanner;

public class RockPaperScissors {
	public static void main(String args[]) {
		Scanner sc = new Scanner(System.in);
		int numberOfTests = sc.nextInt();
		
		for (int i=0; i<numberOfTests; i++) {
			int rounds = sc.nextInt();
			int count = 0;
			
			for (int j=0; j<rounds; j++) {
				String a = sc.next();
				String b = sc.next();
				count += output(a, b);
			}
			
			if (count == 0)
				System.out.println("TIE");
			else if (count > 0)
				System.out.println("Player 1");
			else System.out.println("Player 2");
		}
		sc.close();
	}
	
	public static int output(String a, String b) {
		if (a.equals(b))
			return 0;
		if (a.equals("R")) {
			if (b.equals("P"))
				return -1;
			if (b.equals("S"))
			return 1;
		}
		
		if (a.equals("P")) {
			if (b.equals("R"))
				return 1;
			if (b.equals("S"))
			return -1;
		}
		
		if (a.equals("S")) {
			if (b.equals("P"))
				return 1;
			if (b.equals("R"))
			return -1;
		}
		
		return 0;
	}
}
