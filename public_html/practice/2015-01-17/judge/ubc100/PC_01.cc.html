#include<iostream>
#include<set>
#include<vector>

using namespace std;

int main (void) {

  int N;
  int firsti = -1;
  int secondi = -1;
  vector<int> first;
  vector<int> second;

  cin >> N;
  multiset<int> myset;
  set<int> mys;
  while (N--) {
    int num;
    cin >> num;
    myset.insert(num);
    mys.insert(num);
  }

  multiset<int>::iterator it;

  for(it = mys.begin() ; it != mys.end() ; it++) {
    int num = *it;
    int count = myset.count(num);
    if(count > firsti) {
      secondi = firsti;
      second = first;
      firsti = count;
      first.clear();
      first.push_back(num);
    } else if ( count == firsti) {
      first.push_back(num);
    } else if( count > secondi) {
      secondi = count;
      second.clear();
      second.push_back(num);
    } else if( count == secondi) {
      second.push_back(num);
    }
  }
  /*
  for(int i = 0; i < first.size(); i++) {
    cout << first[i] << endl;
  }
  for(int i = 0; i < second.size(); i++) {
    cout << second[i] << endl;
    }*/
  if(first.size() > 1) {
    int max = 0;
    for(int k = 0; k < first.size(); k++) {
      for(int j = k+1; j < first.size(); j++) {
	int temp = first[k]-first[j];
	if(temp < 0) {
	  temp = -temp;
	}
	if(temp>max) {
	  max = temp;
	}
      }
    }
    cout << max << endl;
  } else {
    if(second.size() > 1) {
      int max = 0;
      for (int i = 0; i < second.size(); i++) {
	int temp = first[0] - second[i];
	if(temp < 0) {
	  temp = -temp;
	}
	if(temp > max) {
	  max = temp;
	}
      }
      cout << max << endl;
    } else {
      int temp = second[0] - first[0];
      if(temp < 0) {
	temp = -temp;
      }
      cout << temp << endl ;
    }
  }
  return 0;
}
