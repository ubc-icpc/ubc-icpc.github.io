#include<iostream>

using namespace std;
int main(void) {
  int scorer;
  cin >> scorer;
  int possibilities = 0;
  for( int i = scorer-1; i > 0; i--) {
    for( int k = i-1; k > 0; k--) {
      for( int l = k-1; l > 0; l--) {
	possibilities++;
      }
    }
  }
  cout << possibilities<<endl;
  return 0;
}
