#include <iostream>

int combination(int n) {
	int output;
	if (n < 4) {output = 0;}
	else if (n == 4) {output = 1;}
	else {
		output = (n-3)*(n-2)*(n-1)/6; 
	}

	return output;
}

int main(void) {
	int input;
	std::cout << std::endl;
	std::cin >> input;
	std::cout << combination(input) << std::endl;
}
