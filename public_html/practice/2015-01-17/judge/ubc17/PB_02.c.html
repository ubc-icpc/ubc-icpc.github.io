#include <stdio.h>
#include <stdlib.h>

int getValue(char x) {
    switch (x) {
        case 'I':
            return 1;
        case 'V':
            return 5;
        case 'X':
            return 10;
        case 'L':
            return 50;
        case 'C':
            return 100;
        case 'D':
            return 500;
        case 'M':
            return 1000;
    }
    exit(-1);
}

int main() {
    char arom[24];
    scanf("%s", arom);
    int digits = 0;
    int i;
    for (i = 0; i < 24; ++i) {
        if (arom[i] == '\0') {
            digits = i;
            break;
        }
    }
    long long value = 0;
    int prv = getValue(arom[digits-1]);
    for (i = digits-2; i >= 0; i -= 2) {
        int a = (int)(arom[i] - 48);
        char r = arom[i+1];
        int rv = getValue(r);
        if (rv < prv) {
            a *= -1;
        }
        value += (a * rv);
        prv = rv;
    }
    printf("%lld\n", value);
    return 0;
}
