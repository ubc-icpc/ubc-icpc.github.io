#include <bits/stdtr1c++.h>

using namespace std;

set<int> d;
bool dfs(int v, int w) {
    if (w < 0) return false;
    if (w == 0) return d.count(v);
    for (int x : d) {
        if (x != 0 && v%x == 0) {
            if (dfs(v/x, w-1)) return true;
        } else if (v == 0 && x == 0) {
            if (dfs(0, w-1)) return true;
        }
        if (dfs(v-x, w-1)) return true;
    }
    return false;
}

int main() {
    int n, w; cin >> w >> n;
    for (int i = 0; i < n; i++) {
        int x; cin >> x;
        d.insert(x);
    }
    int v; cin >> v;
    while (v--) {
        int x; cin >> x;
        if (dfs(x, w)) {
            cout << 'Y' << endl;
        } else {
            cout << 'N' << endl;
        }
    }
	return 0;
}
