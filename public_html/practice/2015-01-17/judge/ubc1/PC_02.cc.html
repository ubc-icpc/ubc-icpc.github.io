#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int main() {
    map<int, int> freq;
    int n; cin >> n;
    for (int i = 0; i < n; i++) {
        int x; cin >> x;
        freq[x]++;
    }
    
    map<int, vector<int> > v;
    for (auto p : freq) {
        v[p.second].push_back(p.first);
    }
    
    for (auto &p : v) {
        sort(p.second.begin(), p.second.end());
    }
    
    auto it = v.end(); --it;
    if (it->second.size() >= 2) {
        cout << it->second.back() - it->second[0] << endl;
    } else {
        auto pit = it; pit--;
        int l = *(pit->second.begin()), r = pit->second.back();
        int ans = max(abs(l-it->second[0]), abs(r-it->second[0]));
        cout << ans << endl;
    }
	return 0;
}
