#include <iostream>
#include <set>
#include <iomanip>
#include <map>
#include <algorithm>
using namespace std;

typedef map<int,int> mii;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  int n, r;
  while (cin >> n) {
    mii m;
    set<int> s1, s2;
    for (int i = 0; i < n; i++) {
      cin >> r;
      m[r]++;
    }
    int p1 = -1, p2 = -1;
    for (mii::iterator it = m.begin(); it != m.end(); it++) {
      if (it->second >= p1) {
        p2 = p1;
        p1 = it->second;
      }
      else if (it->second > p2)
        p2 = it->second;
    }
    for (mii::iterator it = m.begin(); it != m.end(); it++)
      if (it->second == p1)
        s1.insert(it->first);
      else if (it->second == p2)
        s2.insert(it->first);
    int mx = 0;
    if (s1.size() == 1) {
      for (set<int>::iterator it = s2.begin(); it != s2.end(); it++) {
        mx = max(mx, abs(*s1.begin()-*it));
      }
      cout << mx << "\n";
    }
    else
      cout << *(--s1.end())-*s1.begin() << "\n";
  }
  return 0;
}
