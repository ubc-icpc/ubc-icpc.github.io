#include <iostream>
#include <map>
#include <cmath>
using namespace std;

map<char,int> m;

int main() {
  string S = "IVXLCDM";
  for (int i = 0; i < 7; i++)
    m[S[i]] = pow(10,i/2)*(1|((i%2)<<2));
  string s;
  while (cin >> s) {
    int t = 0, l = 0;
    char lc = 0;
    for (int i = 0; i < s.size()/2; i++) {
      int a = s[2*i]-'0';
      char r = s[2*i+1];
      if (i && m[r] > m[lc])
        t -= 2*l;
      t += (l=m[r]*a);
      lc = r;
    }
    cout << t << "\n";
  }
  return 0;
}
