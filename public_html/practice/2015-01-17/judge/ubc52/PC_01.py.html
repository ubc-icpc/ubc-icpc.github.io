N=int(raw_input())

def maxdiff(l1,l2):
	return max(max(l1)-min(l2),max(l2)-min(l1))

freq=[[i,0] for i in range(1,1001)]

for sensor in xrange(N):
	R=int(raw_input())
	freq[R-1][1]+=1

freq.sort(key=lambda a:a[1],reverse=True)

highest=[freq[i][0] for i in range(1000) if freq[i][1]==freq[0][1]]

if len(highest)>1:
	print max(highest)-min(highest)
else:
	for i in range(1000):
		if freq[i][1]<freq[0][1]:
			lasthighest=i
			break
	secondhighest=[freq[i][0] for i in range(lasthighest,1000) if freq[i][1]==freq[lasthighest][1]]
	print maxdiff(highest,secondhighest)
