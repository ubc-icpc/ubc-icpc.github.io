R,C=map(int,raw_input().split(" "))
K=int(raw_input())
cats=[map(int,raw_input().split(" ")) for i in range(K)]
cats.append([1,1])

grid=[[0 for y in range(R)] for x in range(C)]

grid[0][0]=1

for x in range(C):
	for y in range(R):
		if [y+1,x+1] in cats:
			pass	#leave at zero
		elif x==0:
			grid[x][y]=grid[x][y-1]
		elif y==0:
			grid[x][y]=grid[x-1][y]
		else:
			grid[x][y]=grid[x-1][y]+grid[x][y-1]
print grid[C-1][R-1]
