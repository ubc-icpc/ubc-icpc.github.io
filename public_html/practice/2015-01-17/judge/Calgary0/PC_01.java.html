import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Scanner;

public class C {

	private void work() {
		Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(
				System.in)));
		int n = sc.nextInt();
		int[] a = new int[1001];
		while (n-- > 0) {
			a[sc.nextInt()]++;
		}
		int max = 0;
		int min = 0;
		int maxVal = 0;
		for (int i = 1000; i > 0; i--) {
			if (a[i] > maxVal) {
				maxVal = a[i];
				max = min = i;
			} else if (a[i] == maxVal) {
				min = i;
			}
		}
		if (max != min) {
			System.out.println(max - min);
		} else {
			int t = max;
			max = min = 0;
			maxVal = 0;
			for (int i = 1000; i > 0; i--) {
				if (a[i] == 0 || a[i] == a[t])
					continue;
				if (a[i] > maxVal) {
					maxVal = a[i];
					max = min = i;
				} else if (a[i] == maxVal) {
					min = i;
				}
			}
			int res = Math.max(Math.abs(t - max), Math.abs(t - min));
			System.out.println(res);
		}
		sc.close();
	}

	public static void main(String[] args) {
		C problem = new C();
		problem.work();
	}

}
