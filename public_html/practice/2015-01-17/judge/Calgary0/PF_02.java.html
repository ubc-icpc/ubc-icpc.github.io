import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.HashSet;
import java.util.Scanner;
import java.util.Set;

public class F {

	private void work() {
		Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(
				System.in)));
		int w = sc.nextInt();
		int dn = sc.nextInt();
		int[] ds = new int[dn];
		Set<Integer> fst = new HashSet<Integer>();
		Set<Integer> snd = new HashSet<Integer>();
		for (int i = 0; i < dn; i++) {
			ds[i] = sc.nextInt();
			fst.add(ds[i]);
		}
		while (w-- > 0) {
			snd.clear();
			for (int x : fst) {
				for (int i = 0; i < dn; i++) {
					snd.add(x + ds[i]);
					snd.add(x * ds[i]);
				}
			}
			fst.clear();
			fst.addAll(snd);
		}
		int kk = sc.nextInt();
		while (kk-- > 0) {
			int x = sc.nextInt();
			System.out.println(fst.contains(x) ? "Y" : "N");
		}
		sc.close();
	}

	public static void main(String[] args) {
		F problem = new F();
		problem.work();
	}

}
