import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Scanner;

public class E {

	private int[][] memo;

	private void work() {
		Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(
				System.in)));
		int m = sc.nextInt();
		int n = sc.nextInt();
		memo = new int[m + 1][n + 1];
		for (int i = 1; i <= m; i++) {
			for (int j = 1; j <= n; j++) {
				memo[i][j] = -1;
			}
		}
		memo[1][1] = 1;
		int k = sc.nextInt();
		while (k-- > 0) {
			int r = sc.nextInt();
			int c = sc.nextInt();
			memo[r][c] = 0;
		}
		System.out.println(go(m, n));
		sc.close();
	}

	private int go(int r, int c) {
		if (memo[r][c] >= 0) {
			return memo[r][c];
		}
		return memo[r][c] = go(r - 1, c) + go(r, c - 1);
	}

	public static void main(String[] args) {
		E problem = new E();
		problem.work();
	}

}
