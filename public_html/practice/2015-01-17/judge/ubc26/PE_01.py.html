import sys

r, c = map(int, sys.stdin.next().split(' '))

grid = [ [ -1 for j in xrange(c+1) ] for i in xrange(r+1) ]
grid[1][1] = 1

k = int(sys.stdin.next())
for i in range(k):
    cr, cc = map(int, sys.stdin.next().split(' '))
    grid[cr][cc] = 0

def f(i, j):
    if i == 0 or j == 0 or i > r or j > c: return 0
    if grid[i][j] == -1: grid[i][j] = f(i-1,j) + f(i,j-1)
    return grid[i][j]

print f(r,c)
