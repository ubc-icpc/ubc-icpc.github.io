import sys

dp = [ [ 1 for i in range(100) ] for n in range(5) ]

for k in range(5):
    for n in range(100):
        c = 0
        if k == 0 or n == 0: c = 0
        elif k == 1: c = 1
        else: c = sum(dp[k - 1][m] for m in range(n))
        dp[k][n] = c

n = int(sys.stdin.next())

print dp[4][n]
