#include<iostream>
#include<vector>
#include<cmath>
#include<algorithm>
using namespace std;

int main () {

  int N;
  cin >> N;

  int map[1001] = {};
  {
    int temp;
    for (int i = 0; i < N; i++) {
      cin >> temp;
      map[temp]++;
    }
  }
  int highest_val = -1;
  int second_highest_val = -1;
  vector<int> highest;
  vector<int> second_highest;
  
  for (int i = 1; i <= 1000; i++) {
    if (map[i] > highest_val && map[i] != 0) {
      highest_val = map[i];
      highest.clear();
      highest.push_back(i);
    } else if (map[i] == highest_val) {
      highest.push_back(i);
    }
  }
  for (int i = 1; i <= 1000; i++) {
    if (map[i] > second_highest_val && map[i] != 0 && map[i] != highest_val) {
      second_highest_val = map[i];
      second_highest.clear();
      second_highest.push_back(i);
    } else if (map[i] == second_highest_val) {
      second_highest.push_back(i);
    }
  }

  int max;
  
  if (highest.size() > 1)
    max = highest.back() - highest.front();
  else
    max = std::max(abs(highest.back() - second_highest.back()), abs(highest.back() - second_highest.front()));  

  cout << max << endl;

  return 0;
}
