#include <iostream>
#include <string>
#include <map>
using namespace std;

int main () {

  string s;
  cin >> s;

  map<char,int> m;
  m['I'] = 1;
  m['V'] = 5;
  m['X'] = 10;
  m['L'] = 50;
  m['C'] = 100;
  m['D'] = 500;
  m['M'] = 1000;
  
  int total = 0;

  for (int i = 0; i < s.length() / 2 - 1; i++) {
    int subtotal = s[i * 2] - '0';
    subtotal *= m[s[i * 2 + 1]];
    if (m[s[(i + 1) * 2 + 1]] > m[s[i * 2 + 1]])
      total -= subtotal;
    else
      total += subtotal;
  }
  total += (s[s.length() - 2] - '0') * m[s[s.length() - 1]];
  cout << total << endl;

  return 0;
}
