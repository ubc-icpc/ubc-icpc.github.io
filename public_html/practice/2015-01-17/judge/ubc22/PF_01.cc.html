#include<iostream>
#include<vector>
#include<set>
using namespace std;

int main () {
  int N, T;
  cin >> N >> T;
  set<int> num;
  while (T--) {
    int t;
    cin >> t;
    num.insert(t);
  }
  vector<set<int> > s(N + 1);
  s[0] = num;
  for (int i = 1; i <= N; ++i) {
    for (int j : s[i-1]) {
      for (int k : num) {
	s[i].insert(j + k);
	s[i].insert(j * k);
      }
    }
  }
  cin >> T;
  while (T--) {
    int t;
    cin >> t;
    if (s[N].count(t) == 1) cout << "Y" << endl;
    else cout << "N" << endl;
  }
  return 0;
}
