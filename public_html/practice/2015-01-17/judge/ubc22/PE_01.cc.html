#include<iostream>
#include<vector>
using namespace std;

int main () {
  int R, C, K;
  cin >> R >> C >> K;
  vector<vector<bool> > map(R + 1, vector<bool>(C + 1, false));
  while (K--) {
    int r, c;
    cin >> r >> c;
    map[r][c] = true;
  }
  vector<vector<int> > dp(R + 1, vector<int>(C + 1, 0));
  dp[0][1] = 1;
  for (int i = 1; i <= R; ++i)
    for (int j = 1; j <= C; ++j) {
      if (!map[i][j]) dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
    }
  cout << dp[R][C] << endl;
  return 0;
}
