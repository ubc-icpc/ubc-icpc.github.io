import java.util.Scanner;


public class B150117 {
	public static void main(String args[]){
		int v = 0;
		int f1=0, f2=0, s1=0, s2=0;
		String a;
		Scanner sc = new Scanner(System.in);
		a = sc.next();
		for(int i = 0; i< a.length(); i+=2){
			f1 = a.charAt(i)- '0';
			switch (a.charAt(i+1)){
				case 'I' : f2 = 1;
				break;
				case 'V' : f2 = 5;
				break;
				case 'X' : f2 = 10;
				break;
				case 'L' : f2 = 50;
				break;
				case 'C' : f2 = 100;
				break;
				case 'D' : f2 = 500;
				break;
				case 'M' : f2 = 1000;
				break;
			}
			if(i+2<a.length()){
				s1 = a.charAt(i+2)- '0';
				switch (a.charAt(i+3)){
					case 'I' : s2 = 1;
					break;
					case 'V' : s2 = 5;
					break;
					case 'X' : s2 = 10;
					break;
					case 'L' : s2 = 50;
					break;
					case 'C' : s2 = 100;
					break;
					case 'D' : s2 = 500;
					break;
					case 'M' : s2 = 1000;
					break;
				}
				if(s2> f2){
					v += -f2*f1;
				}
				else {
					v += f2*f1;
				}
			}else{
				v+= f2*f1;
			}	
			//System.out.println(v + " " + f1 + " " + f2+ " "  + s1+ " "  + s2);

		}
		System.out.println(v);
	}
}
