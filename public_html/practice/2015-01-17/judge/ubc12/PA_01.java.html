import java.util.Scanner;



public class A150117 {
	public static void main(String args[]){
		int j, count = 0;
		Scanner sc = new Scanner(System.in);
		j = sc.nextInt();
		for(int i = 1; i< j; i++){
			for(int k = i+1; k< j; k++){
				for(int l = k+1; l< j; l++){
					count ++;
				}	
			}	
		}
		System.out.println(count);
	}
}
