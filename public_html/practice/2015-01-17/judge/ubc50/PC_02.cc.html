#include<iostream>
#include<vector>
#include<list>
#include<cstdlib>
#include<map>
#include<algorithm>

using namespace std;

int main()
{
  long n;
  cin >> n;
  map<int,long> m;
  int r;
  int ans = 0;
  for(int i=0;i<n;i++)
    {
      cin >> r;
      m[r]+=1;
    }
  map<int,vector<int> >v;
  for(auto e : m)
    {
      v[e.second].push_back(e.first);
    }
  for(auto e : v)
    sort(e.second.begin(),e.second.end());
  auto it = v.end();
  --it;
  if((it->second).size() >=2)
    {
      ans = abs((it->second).front()-(it->second).back());
    }
  else
    {
      int h1 = (it->second).back();
      --it;
      ans =max(abs(h1-(it->second).front()), abs(h1-(it->second).back()));
    }
  cout << ans << endl;
}
