#include<iostream>
#include<string>

using namespace std;

int return_val(char r)
{
  int rval = 0;
  if(r=='I')
    rval = 1;
  else if(r=='V')
    rval = 5;
  else if(r=='X')
    rval = 10;
  else if(r=='L')
    rval = 50;
  else if(r=='C')
    rval = 100;
  else if(r=='D')
    rval = 500;
  else if(r=='M')
    rval = 1000;
  return rval;
}

int main()
{
  string::size_type sz;
  string s;
  cin >> s;
  int len = s.length();
  long ans = 0;
  for(int i = 0; i < len/2; i++)
    {
      string x = s.substr(2*i,2*i+1);
      int a = std::stoi(x,&sz);
      char r = s[2*i+1];
      int rval = 0;
      if(r=='I')
	rval = 1;
      else if(r=='V')
	rval = 5;
      else if(r=='X')
	rval = 10;
      else if(r=='L')
	rval = 50;
      else if(r=='C')
	rval = 100;
      else if(r=='D')
	rval = 500;
      else if(r=='M')
	rval = 1000;
      if(len!=2 || i!= ((len/2-1)-1))
	{
	  if(return_val(r) < return_val(s[2*i+3]))
	    ans-=(a*rval);
	  else
	    ans+=(a*rval);
	}
      else if(len==2)
	ans=(a*rval);
    }
  cout << ans;
}
