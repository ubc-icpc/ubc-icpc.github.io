#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

int main() {
	int W, D, V, T, dgt[10];
	cin >> W >> D;
	for (int d = 0; d < D; d++) cin >> dgt[d];
	cin >> V;
	while (V--) {
		cin >> T;
		set<LL> cur;
		for (int d = 0; d < D; d++) cur.insert(dgt[d]);
		for (int w = 0; w < W; w++) {
			set<LL> nxt;
			for (LL x : cur) {
				for (int d = 0; d < D; d++) {
					if (x*dgt[d] <= T) nxt.insert(x*dgt[d]);
					if (x+dgt[d] <= T) nxt.insert(x+dgt[d]);
				}
			}
			cur = move(nxt);
		}
		cout << (cur.count(T) ? "Y\n" : "N\n");
	}
}
