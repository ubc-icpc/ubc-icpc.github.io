#include <bits/stdc++.h>
using namespace std;

int main() {
	int N, rds[1001] = {}, x;
	cin >> N;
	while (N--) {
		cin >> x;
		rds[x]++;
	}
	vector<pair<int,int>> st;
	for (int i = 1; i <= 1000; i++)
		if (rds[i]) st.emplace_back(-rds[i], i);
	sort(st.begin(), st.end());

	int i = 1;
	while (i+1<st.size() && st[i+1].first == st[i].first) i++;
	int a = abs(st[0].second - st[1].second);
	a = max(a, abs(st[0].second - st[i].second));
	cout << a << endl;
}
