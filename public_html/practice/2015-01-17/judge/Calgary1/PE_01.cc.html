#include <bits/stdc++.h>
using namespace std;

int main() {
	int R, C, K, r, c, dp[25][25] = {};
	bool cat[25][25] = {};
	cin >> R >> C >> K;
	while (K--) {
		cin >> r >> c;
		cat[r-1][c-1] = true;
	}
	for (r = 0; r < R; r++) for (c = 0; c < C; c++) {
		if (!r && !c) dp[r][c] = 1;
		else if (!cat[r][c]) {
			if (r) dp[r][c] += dp[r-1][c];
			if (c) dp[r][c] += dp[r][c-1];
		}
	}
	cout << dp[R-1][C-1] << endl;
}
