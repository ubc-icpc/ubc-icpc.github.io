#include<iostream>
#include<string>

using namespace std;

int roman(char c){
	switch (c)
	{
	case 'I':
		return 1;
	case 'V':
		return 5;
	case 'X':
		return 10;
	case 'L':
		return 50;
	case 'C':
		return 100;
	case 'D':
		return 500;
	case 'M':
		return 1000;
	default:
		break;
	}
	return 0;
}

int main(){
	
	string input;
	cin >> input;
	int n = input.length();
	int result = 0;
	for (size_t i = 0; i < n/2 - 1; i++)
	{
		if (roman(input[2*i + 1]) < roman(input[2*i + 3]))
		{
			result = result - (input[2 * i] - '0')*roman(input[2 * i + 1]);
		}
		else
		{
			result = result + (input[2 * i] - '0')*roman(input[2 * i + 1]);
		}
	}

	result = result + (input[n-2] - '0')*roman(input[n-1]);
	cout << result;


	return 0;
}
