#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main() {
	long long n; cin >> n;
	vector < int > r(1001, 0);
	int mfn = 0, ml = 0, ms = 2147483647, sfn = 0, sl = 0, ss = 2147483647;
	for(int i = 0; i < n; ++i){
		int gar; cin >> gar;
		r[gar] ++;
	}
	for(int i = 1; i < r.size(); ++i){
		if(r[i] > mfn){
			mfn = r[i];
			ml = i;
			ms = i;
		}else if(r[i] == mfn){
			if(i > ml){
				ml = i;
			}
			if(i < ms){
				ms = i;
			}
		}
	}
	for(int i = 1; i < r.size(); ++i){
		if(r[i] > sfn && r[i] < mfn){
			sfn = r[i];
			sl = i;
			ss = i;
		}else if(r[i] == sfn){
			if(i > sl){
				sl = i;
			}
			if(i < ss){
				ss = i;
			}
		}
	}
	if(ml != ms){
		cout << abs(ml - ms) << endl;
	}else{
		if(abs(ml - ss) > abs(ml - sl)){
			cout << abs(ml - ss) << endl;
		}else{
			cout << abs(ml - sl) << endl;
		}
	}
	return 0;
}
