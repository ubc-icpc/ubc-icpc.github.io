#include <iostream>
#include <vector>
#include <map>
#include <sstream>
using namespace std;
int main(){
  string n; cin >> n;
  map < char, int > m;
  m['I'] = 1;
  m['V'] = 5;
  m['X'] = 10;
  m['L'] = 50;
  m['C'] = 100;
  m['D'] = 500;
  m['M'] = 1000;

  int total = 0;
  for(int i = 0; i < n.length()/2 - 1; ++i){
    int num = n[i * 2] - '0';
    int num2 = m[n[i * 2 + 1]];
    if(m[n[i * 2 + 1]] >= m[n[(i + 1) * 2 + 1]]){
      total += num * num2;
    }else{
      total -= num * num2;
    }
  }
  total += (n[n.length() - 2] - '0') * (m[n[n.length() - 1]]);
  cout << total << endl;
  return 0;
}
