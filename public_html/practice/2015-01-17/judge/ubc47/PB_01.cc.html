#include<iostream>
#include<map>
#include<string>

using namespace std;

int main()
{
	string::size_type sz;
	map<char,int> legend;
	legend.insert(make_pair('I',1));
	legend.insert(make_pair('V',5));
	legend.insert(make_pair('X',10));
	legend.insert(make_pair('L',50));
	legend.insert(make_pair('C',100));
	legend.insert(make_pair('D',500));
	legend.insert(make_pair('M',1000));
	
	string input;
	cin>>input;
	int number = 0;
	for(int i =0;i<input.length()-1;i+=2)
	{
		string s = input.substr(i,i+1);
		if(i+3 < input.length())
		{
			if(legend[input[i+1]] < legend[input[i+3]])
			{
				number -= stoi(s,&sz)*legend[input[i+1]];
			}
			else
			{
				number += stoi(s,&sz)*legend[input[i+1]];
			}
		}
		else
		{
			number += stoi(s,&sz)*legend[input[i+1]];
		}
	}	
	cout<<number<<"\n";
	return 0;
}
