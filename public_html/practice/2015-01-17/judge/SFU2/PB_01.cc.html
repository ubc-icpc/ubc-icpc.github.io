#include<iostream>
#include<string>
#include <map>
using namespace std;
map<char,int> mymap = {{'I',1},{'V',5},{'X',10},{'L',50},{'C',100},{'D',500},{'M',1000}};
string s;
int main() {
    cin>>s;
    int ans = 0;
    for(int i = 0; i < s.length(); i++) {
        int value;
        if(i % 2){
            value = (s[i-1] - '0') * mymap[s[i]];
if(i + 2 < s.length()) {
            if(mymap[s[i+2]] > mymap[s[i]]) {
                ans -= value;
            }
            else {
                ans += value;
            }
        }
        else {
            ans += value;
        }
        }
        
    }
    cout<<ans<<endl;
}
