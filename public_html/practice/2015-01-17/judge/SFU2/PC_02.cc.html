#include<iostream>
#include<algorithm>
using namespace std;
struct sensor {
    int r;
    int fre;
};
int n;
sensor set[1001];
bool comp(sensor a, sensor b) {
    return a.fre > b.fre;
}
int abs(int x, int y) {
    if(x > y)
        return x - y;
    else 
        return y - x;
}
int main() {
    cin>>n;
    int reading;
    for(int i = 0; i < n; i++) {
        cin>>reading;
        set[reading].r = reading;
        set[reading].fre++;
    }
    sort(set, set + 1001, comp);
    //for(int i = 0; i < n; i++) {
    //    cout<<set[i].r<<":"<<set[i].fre<<" ";
    //}
    //cout<<endl;
    int ans = 0;
    if(set[0].fre == set[1].fre) {
        int size = 1;
        while(set[size].fre == set[0].fre)
            size++;
        for(int i = 0; i < size; i++) {
            for(int j = 0; j < size; j++) {
                int diff = abs(set[i].r, set[j].r);
                if(diff > ans)
                    ans = diff;
            }
        }
    }
    else {
     //   cout<<"a"<<endl;
        int size = 1;
        while(set[size].fre == set[1].fre)
            size++;
        for(int i = 1; i < size; i++) {
            int diff = abs(set[0].r, set[i].r);
            if(diff > ans)
                ans = diff;
        }
    }
    cout<<ans<<endl;
}
