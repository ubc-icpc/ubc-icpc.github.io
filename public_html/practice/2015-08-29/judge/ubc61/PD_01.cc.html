#include <bits/stdtr1c++.h>
using namespace std;

struct route {
	int start;
	int end;
	string trial;
	route(){};
	route(int s, int e, string t) {
		start = s;
		end = e;
		trial = t;
	}
};

vector<int> adj[35];
int v[35];

void dfs(int start) {
	v[start] = 0;
	for (int i=0; i<adj[start].size(); i++) {
		if (v[adj[start][i]] == -1) {
			dfs(adj[start][i]);
		}
	}
}

int main() {
	int m; cin >> m;
	vector<string> trials;
	while(m--) {
		int t; cin >> t;
		string trials[t];
		for (int i=0; i<t; i++) {
			cin >> trials[i];
		}
		int n,e; cin >> n >> e;
		for (int i=0; i<n; i++) {
			adj[i].clear();
			v[i] = -1;
		}
		vector<route> routes;
		for (int i=0; i<e; i++) {
			int start, end; string trial;
			cin >> start >> end >> trial;
			if (find(trials,trials+t,trial) == trials+t) {
				adj[start].push_back(end);
			}
		}
		dfs(0);
		(v[n-1]) ? cout << 0 : cout << 1;
		cout << endl;
	}
	
	return 0;
}
