#include <iostream>
#include <algorithm>
using namespace std;

typedef long long ll;

int main() {
  int n; cin >> n;
  while (n--) {
    ll x1, y1, x2, y2;
    cin >> x1 >> y1 >> x2 >> y2;
    if (!x1 && !y1) {
      if (x2 || y2)
        cout << "0\n";
      else
        cout << "1\n";
    }
    else {
      if (abs(__gcd(x1,y1)) == abs(__gcd(x2,y2)))
        cout << "1\n";
      else
        cout << "0\n";
    }
  }
  return 0;
}
