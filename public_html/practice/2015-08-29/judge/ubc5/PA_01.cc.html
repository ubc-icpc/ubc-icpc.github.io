#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

typedef long double ld;

const ld pi = 3.1415926535897932384626433832795L;
const ld f = pi*5.0L/12.0L;

int main() {
  cout << fixed << setprecision(5);
  ld r, s, a;
  int n; cin >> n;
  while (n--) {
    cin >> r;
    s = 2*r*(1+2*sin(f));
    cout << s*s << "\n";
  }
  return 0;
}
