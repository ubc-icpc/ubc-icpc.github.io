#include <iostream>
#include <set>
using namespace std;

int main() {
  int t; cin >> t;
  while (t--) {
    int a, b, c;
    cin >> a >> b >> c;
    set<int> s;
    for (int i = b; i < a; i += b)
      s.insert(i);
    for (int i = c; i < a; i += c)
      s.insert(i);
    cout << a << " " << b << " " << c << "\n" << s.size() << "\n";
  }
  return 0;
}
