#include <iostream>
#include <vector>
#include <set>
using namespace std;

vector<int> adj[30];
set<string> ts;
bool vis[30];

void dfs(int x) {
  vis[x] = true;
  for (int i = 0; i < adj[x].size(); i++)
    if (!vis[adj[x][i]])
      dfs(adj[x][i]);
}

int main() {
  int m; cin >> m;
  while (m--) {
    for (int i = 0; i < 30; i++) {
      adj[i].clear();
      vis[i] = false;
    }
    ts.clear();
    int t; cin >> t;
    while (t--) {
      string s; cin >> s;
      ts.insert(s);
    }
    int n, e;
    cin >> n >> e;
    for (int i = 0; i < e; i++) {
      int a, b; string q;
      cin >> a >> b >> q;
      if (!ts.count(q)) {
        adj[a].push_back(b);
        adj[b].push_back(a);
      }
    }
    dfs(0);
    cout << vis[n-1] << "\n";
  }
  return 0;
}
