#include<iostream>
#include<iomanip>
#include<cmath>
using namespace std;
typedef long double ld;

int main() {
  ld pi = 4*atan(1.0);
  int test; cin >> test;
  while (test--) {
    ld radius; cin >> radius;
    ld side = 4*radius*cos(pi/12)+2*radius;
    ld area = side*side;
    cout << setprecision(5) << fixed << area << '\n';
  }
}
