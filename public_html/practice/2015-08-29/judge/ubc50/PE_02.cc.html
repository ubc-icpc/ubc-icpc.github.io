#include<iostream>
#include<cmath>
using namespace std;
typedef long long ll;

ll gcd(ll a, ll b) {
  while (b!=0) {
    ll t=a;
    a=b;
    b=t%b;
  }
  return a;
}

int main() {
  int test; cin >> test;
  while (test--) {
    ll x1, x2, y1, y2;
    cin >> x1 >> y1 >> x2 >> y2;
    if (gcd(abs(x1),abs(y1))==gcd(abs(x2),abs(y2))) cout << 1;
    else cout << 0;
    cout << '\n';
  }
}
