#include<iostream>
#include<queue>
#include<vector>
#include<set>
#include<cstring>
#include<string>
using namespace std;

int main() {
  int maps; cin >> maps;
  while (maps--) {
    int t; cin >> t;
    set<string> s;
    bool seen[35];
    vector<int>adj[35];
    memset(seen, 0, sizeof seen);
    for (int i=-0; i<t; i++) {
      string tm; cin >> tm;
      s.insert(tm);
    }
    int n, e; cin >> n >> e;
    for (int i=0; i<e; i++) {
      int a, b; string q;
      cin >> a >> b >> q;
      if (s.count(q)==0) {
	adj[a].push_back(b);
	adj[b].push_back(a);
      }
    }
    seen[0]=true;
    queue<int> q;
    q.push(0);
    while(!q.empty()) {
      int f=q.front();
      q.pop();
      for (int v : adj[f]) {
	if (!seen[v]) {
	  seen[v]=true;
	  q.push(v);
	}
      }
    }
    if (seen[n-1]) cout << 1;
    else cout << 0;
    cout << '\n';
  }
}
