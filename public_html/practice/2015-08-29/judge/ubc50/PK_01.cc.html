#include<iostream>
#include<algorithm>
using namespace std;

int gcd(int a, int b) {
  while (b!=0) {
    int t=a;
    a=b;
    b=t%b;
  }
  return a;
}

int main() {
  int test; cin >> test;
  while (test--) {
    int miles, gas, food; cin >> miles >> gas >> food;
    int ans=0;
    ans += (miles/gas);
    ans += (miles/food);
    if (miles%gas==0) ans--;
    if (miles%food==0) ans--;
    int a=max(gas, food);
    int b=min(gas, food);
    int g = gcd(a,b);
    int l = (a*b)/g;
    ans -= (miles/l);
    if (miles%l==0) ans++;
    cout << miles << " " << gas << " " << food << endl;
    cout << ans << endl;
  }
}
