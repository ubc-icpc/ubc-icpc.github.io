#include <iostream>

using namespace std;

int gcd(int a, int b) {
  
  return b == 0 ? a : gcd(b, a % b);
  
}

int main() {
  int t; cin >> t;
  for (int i=1; i<=t; i++) {
    int x1, y1, x2, y2; cin >> x1 >> y1 >> x2 >> y2;
    if ((gcd(x1,y1) == 0 || gcd(x2,y2) == 0) && !(gcd(x1,y1) == 0 && gcd(x2,y2) == 0)) cout << 0 << endl;
    else if (abs(gcd(x1,y1)) == abs(gcd(x2,y2))) cout << 1 << endl;
    else cout << 0 << endl;
  }
}
