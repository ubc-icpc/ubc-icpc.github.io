#include <iostream>

using namespace std;

int gcd(int a, int b) {
  return b == 0 ? a : gcd(b, a % b);
}

int lcm(int a, int b) {
  return a*b/gcd(a,b);
}

int main() {
  int t; cin >> t;
  for (int i=1; i<=t; i++) {
    int miles, gas, food; cin >> miles >> gas >> food;
    miles--;
    int ans = miles/gas + miles/food - miles/lcm(gas, food);
    cout << miles + 1 << " " << gas << " " << food << endl;
    cout << ans << endl;
    
  }
}
