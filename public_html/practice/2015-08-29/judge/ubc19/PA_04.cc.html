#include <iostream>
#include <iomanip>
#include <math.h>

using namespace std;

int main() {
  int t; cin >> t;
  for (int i=1; i<=t; i++) {
    double x; cin >> x;
    double sin15 = .25*sqrt(2) * (sqrt(3)-1);
    double y = (4*sin15+2*sqrt(2)+2);
    cout << setprecision(5) << fixed << y*y*x*x << endl;
  }
}
