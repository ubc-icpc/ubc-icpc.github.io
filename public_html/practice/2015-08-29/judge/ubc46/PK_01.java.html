import java.util.HashSet;
import java.util.Scanner;

/**
 * Created by Peter on 2015-09-03.
 */
public class Trip {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);

        int noOfCases = Integer.parseInt(s.nextLine());

        for(int i = 0; i < noOfCases; i++) {
            String line = s.nextLine();
            System.out.println(line);
            String[] theCase = line.split("\\s+");
            int distance = Integer.parseInt(theCase[0]);
            int gasFrequency = Integer.parseInt(theCase[1]);
            int foodFrequency = Integer.parseInt(theCase[2]);

            HashSet<Integer> stops = new HashSet<Integer>();

            for(int j = 1; j * gasFrequency < distance; j++){
                stops.add(j*gasFrequency);
            }

            for(int k = 1; k * foodFrequency < distance; k++){
                stops.add(k * foodFrequency);
            }
            System.out.println(stops.size());
        }
    }
}
