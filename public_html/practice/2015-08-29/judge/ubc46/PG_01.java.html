import java.util.Scanner;

/**
 * Created by Peter on 2015-09-02.
 */
public class Bulk {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);

        int noOfCases = Integer.parseInt(s.nextLine());

        for(int i = 0; i < noOfCases; i++){
            String str = s.nextLine();
            System.out.println(str);
            String[] line = str.split("\\s+");
            int noOfItems = Integer.parseInt(line[0]);
            int costPerItem = Integer.parseInt(line[1]);

            System.out.println(costPerItem + (noOfItems-1) * (costPerItem-2));

        }
    }
}
