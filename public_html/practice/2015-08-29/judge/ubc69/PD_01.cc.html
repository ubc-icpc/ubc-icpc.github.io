#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

string trials[50];
int t;

bool included(string s){
	for(int i=0;i<t;i++){
		if(s.compare(trials[i])==0)
			return true;
	}
	return false;
}

int main() {
	int m;
	cin>>m;
	for(int i=1;i<=m;i++){
		cin>>t;
		for(int j=0;j<t;j++){
			cin>>trials[j];
		}
		int n,e;
		cin>>n>>e;
		bool map[n][n];
		for(int j=0;j<n;j++)
			for(int k=0;k<n;k++){
				map[j][k]=false;
			}
		for(int j=1;j<=e;j++){
			int a,b; string s;
			cin>>a>>b>>s;
			if(included(s)) continue;
			map[a][b]=map[b][a]=true;;
		}
		
		for(int j=0;j<n;j++)
			for(int k=0;k<n;k++)
				for(int l=0;l<n;l++){
					if(map[j][k]&&map[k][l])
						map[j][l]=map[l][j]=true;
				}
		if(map[0][n-1])
			cout<<1<<endl;
		else
			cout<<0<<endl;
	}
    return 0;
}
