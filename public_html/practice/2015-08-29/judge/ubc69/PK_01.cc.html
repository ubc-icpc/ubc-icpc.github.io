#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int lcm(int a,int b){
	for(int i=1;i<=b;i++){
		if((a*i)%b==0)
			return a*i;
	}
	return a*b;
}

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		int a,b,c;
		cin>>a>>b>>c;
		cout<<a<<' '<<b<<' '<<c<<endl;
		cout<<(a-1)/b+(a-1)/c-(a-1)/lcm(b,c)<<endl;
	}
    return 0;
}
