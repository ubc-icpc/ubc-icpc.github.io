#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

const double pi = 3.14159265358979323846263228327950;

int T;

double solve(double r) {
    double base = sin(150.0 / 360.00 * 2 * pi) * 2 * r / sin(15.0 / 360.0 * 2 * pi);
    double side = base + 2 * r;
    return side * side;
}

int main() {
    scanf("%d", &T);

    for (int t = 1; t <= T; ++t) {
        double r;
        scanf("%lf", &r);
        printf("%0.9lf\n", solve(r));
    }

    return 0;
}
