#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	while (n--){
		int t; cin >> t;
		map < string, bool > trial;
		for (int i = 0; i < t; ++i){
			string s; cin >> s;
			trial[s] = 1;
		}
		int a, b; cin >> a >> b;
		vector < vector < pair < int, string > > > l(a);
		for (int i = 0; i < b; ++i){
			int x, y; string temp; cin >> x >> y >> temp;
			l[x].push_back(make_pair(y, temp));
		}

		queue < int > s; s.push(0);
		bool ans = false;
		while (s.size()){
			for (int i = 0; i < l[s.front()].size(); ++i){
				if (!trial[l[s.front()][i].second]){
					if (l[s.front()][i].first == a - 1){
						ans = true;
						break;
					}
					s.push(l[s.front()][i].first);
				}
			}
			if (ans){
				cout << 1 << endl;
				break;
			}
			s.pop();
		}
		if (!ans){
			cout << 0 << endl;
		}
	}
	return 0;
}
