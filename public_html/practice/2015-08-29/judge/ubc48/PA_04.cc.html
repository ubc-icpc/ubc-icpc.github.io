#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	cout << fixed << setprecision(5);
	double pi = 3.141592653589793238462643383279502884197169399375105820974944592307816406286;
	while (n--){
		double t, ans; cin >> t;
		ans = pow(cos(pi/12) * 4 * t + 2 * t, 2);
		cout << ans << endl;
	}
	return 0;
}
