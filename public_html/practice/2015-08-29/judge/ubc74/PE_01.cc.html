#include<iostream>
#include<algorithm>
#include<cmath>

using namespace std;

int gcd(int a, int b) {
	return b == 0 ? a : gcd(b, a % b);
}

int main() {
	int T; cin >> T;
	for (int i = 0; i < T; ++i) {
		int x1, y1, x2, y2; cin >> x1 >> y1 >> x2 >> y2;
		if (x1 == 0 && y1 == 0) {
			if (x2 == 0 && y2 == 0)
				cout << "1\n";
			else
				cout << "0\n";
		}
		else if (gcd(abs(x1), abs(y1)) == gcd(abs(x2), abs(y2)))
			cout << "1\n";
		else
			cout << "0\n";
	}
}
