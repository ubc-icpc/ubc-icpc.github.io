#include<iostream>
#include<math.h>

using namespace std;

int gcd(int a, int b) {
	return b == 0 ? a : gcd(b, a % b);
}

int lcm(int a, int b) {
	int temp = gcd(a, b);
	return temp ? (a / temp * b) : 0;
}

int main() {
	int T; cin >> T;
	for (int i = 0; i < T; ++i) {
		int t, g, f; cin >> t >> g >> f;
		cout << t << " " << g << " " << f << endl;
		t -= 1;
		cout << floor(t / g) + floor(t / f) - floor(t / lcm(g, f)) << endl;
	}
	return 0;
}
