#include<bits/stdtr1c++.h>
using namespace std;

int T, N, M, A, B;
string str;

bool rec(int at, vector<vector<int> >& p, vector<bool>& v) {
    if (v[at]) return false;
    v[at] = true;
    if (at == N-1) return true;
    bool ret = false;
    for (int i : p[at])
        ret |= rec(i, p, v);
    return ret;
}

int main () {
    cin >> T;
    while (T--) {
        map<string, bool> m;
        cin >> N;
        for (int i = 0; i < N; ++i) {
            cin >> str;
            m[str] = true;
        }
        cin >> N >> M;
        vector<vector<int> > P = vector<vector<int> >(N);
        vector<bool> V = vector<bool>(N);
        while (M--) {
            cin >> A >> B >> str;
            if (m[str]) continue;
            P[A].push_back(B);
            P[B].push_back(A);
        }
        cout << rec(0, P, V) << endl;
    }
    return 0;
}
