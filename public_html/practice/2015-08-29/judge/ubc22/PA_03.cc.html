#include<bits/stdtr1c++.h>
using namespace std;

#define PI 3.1415926535897932L

int N;
long double R;

int main () {
    cin >> N;
    while (N--) {
        cin >> R;
        R = (2 * R * cos(15 * PI / 180) + R) * 2;
        R = R * R;
        cout << fixed << setprecision(5) << R << endl;
    }
    return 0;
}
