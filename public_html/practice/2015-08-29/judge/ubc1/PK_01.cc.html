#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int t; cin >> t;
    while (t--) {
        int a, b, c;
        cin >> a >> b >> c;
        set<int> s;
        for (int i = 1; i*b < a; i++) {
            s.insert(i*b);
        }
        
        for (int i = 1; i*c < a; i++) {
            s.insert(i*c);
        }
        
        cout << a << " " << b << " " << c << endl;
        cout << s.size() << endl;
    }
	return 0;
}
