#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

//(ax+by, cx+dy) is reachable if
//((a-kc)x + (b-kd)y, cx+dy) is reacheable or
//(ax + by, (c-ka)x + (d-kb)y) is reacheable
//canonize all points by their gcd, since if two pairs (a,b), (c,d) can reach (g,0), they can reach
//each other

int main() {
    int t; cin >> t;
    while (t--) {
        ll x0, y0, x1, y1; 
        cin >> x0 >> y0 >> x1 >> y1;
        
        if (x0 == 0 && y0 == 0) cout << (x1 == 0 && y1 == 0) << endl;
        else if (abs(__gcd(x0, y0)) == abs(__gcd(x1, y1))) cout << 1 << endl;
        else cout << 0 << endl;
    }
	return 0;
}
