#include <bits/stdtr1c++.h>

using namespace std;

vector<int> g[35];
bool seen[35];
int main() {
    int t; cin >> t;
    while (t--) {
        int nw; cin >> nw;
        set<string> st;
        for (int i = 0; i < nw; i++) {
            string s; cin >> s;
            st.insert(s);
        }
        
        int n, m; cin >> n >> m;
        for (int i = 0; i < n; i++) {
            g[i].clear();
        }
        
        for (int i = 0; i < m; i++) {
            int a, b; string c; cin >> a >> b >> c;
            if (st.count(c)) continue;
            g[a].push_back(b);
            g[b].push_back(a);
        }
        
        memset(seen, 0, sizeof seen);
        seen[0] = true;
        queue<int> q({0});
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : g[u]) {
                if (!seen[v]) {
                    seen[v] = true;
                    q.push(v);
                }
            }
        }
        
        cout << seen[n-1] << endl;
    }
	return 0;
}
