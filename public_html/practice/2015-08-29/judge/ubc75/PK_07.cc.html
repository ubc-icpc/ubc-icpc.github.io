#include <bits/stdtr1c++.h>
 
using namespace std;


int main(){
	int total; cin >> total;
	while(total--){
		int distance, numGas, numFood; cin >> distance >> numGas >> numFood;
		set<int> stops;
		
		for(int i=1; i*numGas<distance;i++){
			stops.insert(i*numGas);
		}
		
		for(int i=1; i*numFood<distance;i++){
			stops.insert(i*numFood);
		}
		
		cout << distance << " " << numGas << " " << numFood << endl;
		cout << stops.size() << endl;
		
		
	}
	return 0;
}
