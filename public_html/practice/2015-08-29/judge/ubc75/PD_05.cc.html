#include <bits/stdtr1c++.h>
 
using namespace std;

vector<int> vertex[34];
bool seen[34];

int main(){
	int numMap; cin >> numMap;
	while (numMap--){
		int numTrials; cin >> numTrials;
		set<string> trials;
		while(numTrials--){
			string trial; cin>>trial;
			trials.insert(trial);
		}
		int numVertices, numEdges; cin >> numVertices >> numEdges;
		for (int i=0; i<numVertices; i++){
			vertex[i].clear();
		}
		
		for (int i=0; i<numEdges; i++){
			int a,b; string c; cin >> a >> b >> c;
			if (trials.count(c)) continue;
			vertex[a].push_back(b);
			vertex[b].push_back(a);
			
		}
	
		memset(seen, 0, sizeof seen);
		seen[0]=true;
		queue<int> q({0});
		while(!q.empty()){
			int u = q.front(); q.pop();
			for (int v:vertex[u]){
				if(!seen[v]){
					q.push(v);
					seen[v]=true;
				}
			}
		}
	
		cout << seen[numVertices-1] << endl;
	}	
	
	
	return 0;
	
}
