#include<iostream>
#include<set>
#include<string>
#include<queue>

using namespace std;

bool g[31][31];
set<string> s;
bool v[31];

bool bfs(int n) {
  queue<int> q;
  q.push(0);
  v[0]=true;
  while(!q.empty()) {
	int t=q.front();
	q.pop();
	if(t==n-1) return true;
	for(int i=0; i<n; i++) {
	  if(g[t][i]==true && v[i]==false) {
		q.push(i);
		v[i]=true;
	  }
	}
  }
  return false;
}

int main(void) {
  int t; cin>>t;
  while(t--) {
	int trials; cin>>trials;
	s.clear();
	for(int i=0; i<trials; i++) {
	  string str; cin>>str;
	  s.insert(str);
	}
	int n,e; cin>>n>>e;
	for(int i=0;i<n;i++) {
	  v[i]=false;
	  for(int j=0;j<n;j++) {
		g[i][j]=0;
	  }
	}

	for(int i=0;i<e;i++) {
	  int n1,n2; cin>>n1>>n2;
	  string str; cin>>str;
	  if(s.find(str)==s.end()){
		g[n1][n2]=true;
		g[n2][n1]=true;
	  }
	}

	cout << bfs(n) << endl;
  }
  return 0;
}
