#include<iostream>

using namespace std;
int gcd (int a, int b) {
  while(true) {
	if(a==0) return b;
	b%=a;
	if(b==0) return a;
	a%=b;
  }
}
int lcm (int a, int b) {
  if(a==0 || b==0) return 0;
  int t=gcd(a,b);
  return (a/t * b);
}
int main (void) {
  int T; cin >> T;
  while(T--) {
	int m, f, e; cin >> m >> f >> e;
	cout << m << " " << f << " " << e << endl;
	int t = (f>0 ? m/f : 0) + (e>0 ? m/e : 0);
	if(f>0 && m%f==0)t--;
	if(e>0 && m%e==0)t--;
	int l=lcm(f,e);
	if(l>0) {
	  t-=m/l;
	  if(m%l==0)t++;
	}
	cout << t << endl;
  }
  return 0;
}
