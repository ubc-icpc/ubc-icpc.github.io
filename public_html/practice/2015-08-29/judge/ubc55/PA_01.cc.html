#define _USE_MATH_DEFINES // for C++
#include <cmath>
#include <iostream>


typedef long double ld;
using namespace std;

int main()
{
	int n; 
	cin >> n;
	while (n--)
	{
		ld r;
		cin >> r;
		cout.precision(5);
		ld length = 2 * r + 4 * cos(M_PI / 12) * r;
		cout << fixed << length * length << endl;
	}
	return 0;
}
