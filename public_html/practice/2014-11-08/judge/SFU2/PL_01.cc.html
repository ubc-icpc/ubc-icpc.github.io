#include <iostream>
#include <stdio.h>
#include <vector>
#include <string>
using namespace std;

char cipher[101];
char text[101];
char vowel[] = {'a', 'i', 'y', 'e', 'o', 'u'};
char consonant[] = {'b', 'k', 'x', 'z', 'n', 'h', 'd', 'c', 'w', 'g', 'p', 'v', 'j', 'q', 't', 's', 'r', 'l', 'm','f'};
string s;
int main() {

	while(getline(cin,s)) {
		for(int i = 0; i < s.length(); i++) {
			bool upper = false;
			if(s[i] >= 'A' && s[i] <= 'Z') {
				upper = true;
				s[i] = s[i] - 'A'+ 'a';
			}
			bool vowel_f = false;
			int index = 0;
			for(int j = 0; j < 6; j++) {
				if(s[i] == vowel[j]){
					vowel_f = true;
					index = j;
					break;
				}
			}
			if(vowel_f) {
				s[i] = vowel[( index + 3) % 6];
				if(upper) 
					s[i] = s[i] + 'A' - 'a';
			}
			if(!vowel_f) {
				bool c_f = false;
				for(int j = 0; j < 20; j++) {
					if(consonant[j] == s[i]) {
						c_f = true;
						index = j;
						break;
					}
				}
				if(c_f) {
					s[i] = consonant[(index + 10) % 20];
					if(upper)
						s[i] = s[i] + 'A' - 'a';
				}
			}
		}
		cout<<s<<endl;
	}
	
}
