#include <iostream>
#include <stdio.h>
#include <vector>
using namespace std;
typedef long long ll;
bool mark[10];
int cnt;

vector<int> multi(vector<int> &v, int k) {
	int value;
	int carry = 0;
	vector<int> ans;
	for(int i = 0; i < v.size(); i++) {
		value = v[i] * k + carry;
		carry = value / 10;
		value = value % 10;
		ans.push_back(value);
	}

	if(carry != 0)
		ans.push_back(carry);
	// cout<<"ans: ";
	// for(int i = ans.size() - 1; i >= 0; i--)
	// 	cout<<ans[i];
	// cout<<endl;
	return ans;
}
int main() {
	int n;
	vector<int> temp;
	while(cin>>n) {
		cnt = 0;
		int k = 0;
		vector<int> v;
		v.clear();
		while(n != 0){
			v.push_back( n % 10);
			n /= 10;
		}
		// v = multi(v, 5);
		// // for(int i = v.size() - 1; i >= 0; i--){
		// // 	cout<<v[i];
		// // } 
		// // cout<<endl;
		for(int i = 0; i < 10; i++) {
			mark[i] = false;
		}
		while(cnt < 10) {
			k++;
			temp = multi(v, k);
			for(int i = 0; i < temp.size(); i++) {
				if(!mark[temp[i]]){
					mark[temp[i]] = true;
					cnt++;
				}
			}
			
		}
		cout<<k<<endl;
	}
}
