#include <cstdio>
#include <cstring>

const int MAXN = 100;
const int MAXL = 1000;

int GD[6] = {1,2,3,3,4,10};
int ED[7] = {1,2,2,2,3,5,11};

int G[MAXN],E[MAXN];
int cG=6,cE=7;

void read(int s[],int c){
	for(int i=0;i<c;i++)
		scanf("%d",&s[i]);
}

int main(){
	int n;



	scanf("%d",&n);

	for(int i=0;i<n;i++){
		read(G,cG);
		read(E,cE);
		int sG=0,sE=0;
		for(int j=0;j<cG;j++) sG += G[j]*GD[j];
		for(int j=0;j<cE;j++) sE += E[j]*ED[j];
		if(sG>sE)
			printf("Battle %d: Good triumphs over Evil\n",i+1);
		else if(sG<sE)
			printf("Battle %d: Evil eradicates all trace of Good\n",i+1);
		else 
			printf("Battle %d: No victor on this battle field\n",i+1);
	}

	return 0;
}
