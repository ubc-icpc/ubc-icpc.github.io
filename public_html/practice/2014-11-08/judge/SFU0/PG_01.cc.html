#include <iostream>
#include <cstdio>

using namespace std;

long long gao(long long k) { 
    long long ans = 0;
    while(k) {
        if (k % 2) ans++;
        k = k >> 1;
    }
    return ans;
}


long long gcd(long long a, long long b) {
    if (a == 0) return b;
    if (b == 0) return a;
    return gcd(b % a, a);
}


long long work(long long n, long long m) { 
    long long ans = 1;
    long long temp, temp2;
    if (m > (n-m)) m = n-m;
    for (long long i = 1; i <= m; i++) {
        temp = (n-i+1);
        temp2 = gcd(temp, i);
        ans = ans / (i/temp2) * (temp/temp2);
    }
    return ans;
}

long long gao2(long long n) {
    long long ans = 0, temp = 0,k = ((long long)1) << 60, r = 60;

    while(k) {
        if (k & n) {
            for (long long i = 0; i <= r; i++) {
                if (((i + temp) > 0) && ((i + temp) % 3 == 0)) {
                    ans += work(r, i);
                }
            }
            temp++;
        }
        k = k >> 1;
        r--;
    }
    if ((temp > 0) && (temp % 3) == 0) ans++;
    return ans;
}

int main () {
    long long n;
    while(cin >> n) {
        cout << "Day " << n << ": Level = " << gao2(n) << endl;
    }
    return 0;
}
