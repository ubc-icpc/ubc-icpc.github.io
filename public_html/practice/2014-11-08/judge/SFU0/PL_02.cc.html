#include <cstdio>
#include <cstring>

const int MAXL = 110;

char s[MAXL];

int ly = 6,lf = 20;

char yuan0[6+1] = "aiyeou";
char yuan1[6+1] = "AIYEOU";
char fu0[20+1] = "bkxznhdcwgpvjqtsrlmf";
char fu1[20+1] = "BKXZNHDCWGPVJQTSRLMF";

int ty = -3,tf=-10;

int find(char s[],char c){
	for(int i=0;s[i];i++)
		if(s[i]==c) return i;
	return -1;
}

int main(){
	while(gets(s)!=NULL){
		//printf("%s\n",s);
		for(int i=0;s[i];i++){
			int p = find(yuan0,s[i]);
			if(p!=-1) { s[i] = yuan0[(p+ty+ly)%ly]; continue; }
			
			p = find(yuan1,s[i]);
			if(p!=-1) { s[i] = yuan1[(p+ty+ly)%ly]; continue; }

			p = find(fu0,s[i]);
			if(p!=-1) { s[i] = fu0[(p+tf+lf)%lf]; continue; }

			p = find(fu1,s[i]);
			if(p!=-1) { s[i] = fu1[(p+tf+lf)%lf]; continue; }

			//printf("%c ",s[i]);
		}
		//printf("\n");
		printf("%s\n",s);
	}

	return 0;
}
