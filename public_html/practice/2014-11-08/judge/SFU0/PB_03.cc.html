#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

bool tryit[10];
int count;

void record(long long x){
	while(x!=0){
		long long  p = x %10;
		if(!tryit[p]){
			count++;
			tryit[p] = 1;
		}
		x = x / 10;
	}
}

int main(){
	long long x;
	char s[20];

	while(cin >> x){

		for(int i=0; i<10;i++) tryit[i] = 0;
		count = 0;

		int j=0;
		while(1){
			long long t = x * (++j);
			record(t);
			if(count == 10) {
				printf("%d\n",j);
				break;
			}
		}
	}

	return 0;
}
