#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<cstring>
#include<string>
#include<cmath>
#include<map>
#include<utility>
#include<algorithm>
using  namespace std;
typedef long long LL;

map<string, int> num;
int a[100010];
LL c[200000];
LL ans;
int n;

void add(int x){
	int p = x;
	while (p <= n){
		c[p] ++;
		p = p + (p& -p);
	}
}

LL sum(int x){
	LL tmp = 0;
	int p = x;
	while (p > 0){
		tmp += c[p];
		p = p - (p&-p);
	}
	return tmp;
}

int main(){
	while (cin >> n){
	//	cout << n << endl;
		if (n == 0) break;
		/* init */
		num.clear();
		memset(c, 0, sizeof(c));
		ans = 0;
		string s;
		for (int i = 1; i <= n; ++i){
			cin >> s;
		//	cerr << s <<endl;
			num[s] = i;
		}

		for (int i = 1; i <= n; ++i){
			cin >> s;
		//	cerr << s << endl;
			int x = num[s];
		//	cout << x << endl;
			ans = ans + sum(n) - sum(x);
			add(x);
		}
		cout << ans << endl;
	}
	return 0;

}
