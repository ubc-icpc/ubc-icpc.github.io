#include <iostream>
#include <set>
#include <map>

using namespace std;

char getNewC(char c, map<char, int> &mp, string &list, int addv) {
    int ind = mp[c];
    int newInd = ind - addv;
    if (newInd < 0) newInd = list.size() + newInd;
    return list[newInd];
}

int main() {
    map<char, int> vo;
    map<char, int> c_vo;
    map<char, int> con;
    map<char, int> c_con;
    
    string voList = "aiyeou";
    // a i y e o u
    string c_voList = "AIYEOU";
    
    string conList = "bkxznhdcwgpvjqtsrlmf";
    // b k x z n h d c w g p v j q t s r l m f
    string c_conList = "BKXZNHDCWGPVJQTSRLMF";
    
    for (int i=0; i<voList.size(); i++) {
        vo[ voList[i] ] = i;
        c_vo[ c_voList[i] ] = i;   
    }
    for (int i=0; i<conList.size(); i++) {
        con[ conList[i] ] = i;
        c_con[ c_conList[i] ] = i;
    }
    
    string line;
    while (getline(cin, line)) {
        for (int i=0; i<line.size(); i++) {
            char c = line[i];
            char newC;
            if (vo.find(c) != vo.end()) {
                newC = getNewC(c, vo, voList, 3);
            }
            else if (con.find(c) != con.end()) {
                newC = getNewC(c, con, conList, 10);
            }
            else if (c_vo.find(c) != c_vo.end()) {
                newC = getNewC(c, c_vo, c_voList, 3);
            }
            else if (c_con.find(c) != c_con.end()) {
                newC = getNewC(c, c_con, c_conList, 10);
            }
            else {
                newC = c;
            }
            cout << newC;
        }
        cout << endl;
    }
    
    
    return 0;
}
