#include <iostream>
#include <map>
#include <vector>

using namespace std;

bool mark[10];
int count;

void scan(long long n) {
    long long cur = n;
    while (cur != 0) {
        if (mark[cur%10] == false) {
            mark[cur%10] = true;
            count++;
        }
        cur /= 10;
    }
}

int main() {
    long long n;
    while (cin >> n) {
        for (int i=0; i<10; i++) mark[i] = false;
        count = 0;
        long long k=1;
        while (true) {
            scan(k*n);
            if (count == 10) break;
            else k++;
        }
        cout << k << endl;
    }
    
    return 0;
}
