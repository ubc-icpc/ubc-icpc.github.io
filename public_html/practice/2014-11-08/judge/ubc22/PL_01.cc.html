	case 'I': return 'O';
	case 'y': return 'u';

	case 'O': return 'I';
	case 'u': return 'y';
	case 'U': return 'Y';
	
	case 'b': return 'p';
	case 'B': return 'P';
	case 'k': return 'v';
	case 'K': return 'V';
	case 'x': return 'j';
	case 'X': return 'J';
	case 'z': return 'q';
	case 'Z': return 'Q';
	case 'n': return 't';
	case 'N': return 'T';
	case 'h': return 's';
	case 'H': return 'S';
	case 'd': return 'r';
	case 'D': return 'R';
	case 'c': return 'l';
	case 'C': return 'L';
	case 'w': return 'm';
	case 'W': return 'M';
	case 'g': return 'f';
	case 'G': return 'F';
	case 'p': return 'b';

int main() {
	
	string line;
	while (getline(cin, line)) {
		for (string::iterator it = line.begin(); it != line.end(); ++it) {
			*it = getchar(*it);
		}
		cout << line << endl;
	}
	
	return 0;
}
