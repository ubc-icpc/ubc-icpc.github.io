#include<iostream>
using namespace std;

int main() {
	
	int N;
	while (cin >> N) {
		bool b[10] = {};
		int counter = 1;
		unsigned long long val = N;
		while (true) {
			unsigned long long temp = val;
			
			while (temp != 0) {
				b[temp % 10] = true;
				temp /= 10;
			}
			
			bool f = true;
			for (int i = 0; i < 10; i++) f = f && b[i];
			if (f) break;
			
			val += N;
			counter++;
		}
		cout << counter << endl;
	}
	
	return 0;
}
