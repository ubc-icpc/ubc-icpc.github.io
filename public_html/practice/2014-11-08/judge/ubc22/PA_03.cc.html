#include<iostream>
using namespace std;

int main() {
	
	int g[6] = {1,2,3,3,4,10};
	int e[7] = {1,2,2,2,3,5,11};
	
	int N;
	cin >> N;
	
	for (int i = 1; i <= N; i++) {
		unsigned int good = 0;
		unsigned int evil = 0;
		unsigned int temp;
		for  (int j = 0; j < 6; j++) {
			cin >> temp;
			good += temp * g[j];
		}
		for  (int j = 0; j < 7; j++) {
			cin >> temp;
			evil += temp * e[j];
		}
		cout << "Battle " << i << ": ";
		if (good > evil)
			cout << "Good triumphs over Evil" << endl;
		else if (good < evil)
			cout << "Evil eradicates all trace of Good" << endl;
		else
			cout << "No victor on this battle field" << endl;
	}
	return 0;
}
