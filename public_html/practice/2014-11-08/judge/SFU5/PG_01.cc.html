#include <stdio.h>
#include <string.h>

long long C(int n, int m) {
	long long ret = 1;
	for (int i = 1; i <= m; i++) {
		ret *= n - i + 1;
		ret /= i;
	}
	return ret;
}

int main() {
	long long n;
	while (scanf("%lld", &n) != EOF) {
		int i = 0;
		long long ans = 0, l = 1;
		while (l <= n) {
			l <<= 1;
			i++;
		}
		l >>= 1;
		i--;
		//printf("%lld %d\n", l, i);
		int d = 0;
		for (; l > 0; l >>= 1, i--) {
			if (!(l & n)) continue;
			for (int j = (3 - d) % 3; j <= i; j += 3) {
				ans += C(i, j);
				//printf("%d,%d %lld\n", i, j, ans);
			}
			d = (d + 1) % 3;
		}
		if (d == 0) ans++;
		printf("Day %lld: Level = %lld\n", n, ans - 1);
	}
}
