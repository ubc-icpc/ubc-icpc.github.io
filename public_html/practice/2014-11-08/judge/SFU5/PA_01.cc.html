#include <stdio.h>
#include <string.h>

int main() {
	int cas;
	long long a[6] = {1, 2, 3, 3, 4, 10}, b[7] = {1, 2, 2, 2, 3, 5, 11};
	int A[6], B[7];
	scanf("%d", &cas);
	for (int re = 1; re <= cas; re++) {
		long long x = 0, y = 0;
		for (int i = 0; i < 6; i++) {
			scanf("%d", A + i);
			x += A[i] * a[i];
		}
		for (int i = 0; i < 7; i++) {
			scanf("%d", B + i);
			y += B[i] * b[i];
		}
		printf("Battle %d: ", re);
		if (x < y) {
			puts("Evil eradicates all trace of Good");
		} else if (x > y) {
			puts("Good triumphs over Evil");
		} else {
			puts("No victor on this battle field");
		}
	}
}
