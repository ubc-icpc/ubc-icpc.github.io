#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>

using namespace std;

long long num,i,j,k,l,tmp,sum,tmp2;
long long a[10];

int main()
{
	long long mark;
	mark = scanf("%lld", &num);
	while (mark != EOF){
		memset(a,0,sizeof(a));
		sum = 0;
		k = 0;
		while (sum<10)
		{
			k++;
			tmp = k*num;
			while (tmp)	
			{
				tmp2 = tmp%10;
				tmp = tmp/10;
				if (sum<10 && a[tmp2]==0)
				{
					a[tmp2] = 1;
					sum++;
				}
			}
		}	
		cout<<k<<endl;		
		mark = scanf("%lld",&num);
	}
	return 0;
}
