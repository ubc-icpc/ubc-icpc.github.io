#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>

using namespace std;

long long n,i,t,l,k,num,ans;

long long f[100][3];

long long getNum(long long num,long long l)
{
	if (num==0)
	{
		if (l==0)	
			return 1;
		else 
			return 0;
	}
	long long i = 0,tmp = 1;
	while (tmp<=num)
	{
		i++;
		tmp*=2;
	}
	i--;
	long long ans,t;
	t = num-tmp/2;	
//	cout<<num<<' '<<t<<' '<<i<<' '<<l<<' '<<f[i][l]<<endl;
	ans = f[i][l]+getNum(t,(l+2)%3);
//	cout<<num<<"   "<<ans<<endl;
	return ans;
}

int main()
{
	memset(f,0,sizeof(f));
	f[0][0] = 1;
	for (i=1;i<100;i++)
		for (k=0;k<3;k++)
			f[i][k] = f[i-1][(k+2)%3]+f[i-1][k];
	//for (i=0;i<100;i++)
	//	f[i][0]--;
	int mark;
	mark = scanf("%lld",&num);
	while (mark != EOF)
	{
		ans = getNum(num,0);
		//cout<<getNum(num,0)<<' '<<getNum(num,1)<<' '<<getNum(num,2)<<endl;
		cout<<"Day "<<num<<": Level = "<<ans-1<<endl;
		mark = scanf("%lld",&num);
	}
	return 0;
}
