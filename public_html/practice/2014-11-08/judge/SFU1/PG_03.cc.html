using namespace std;

long long a[100][3];
int main()
{
	long long n;
	while (scanf("%lld",&n) == 1)
	{
		a[0][0] = 1;
		a[0][1] = 1;
		a[0][2] = 0;
		long long index = 0;
		long long i=0;
		for (i = 2 ; i <= n; i = (i << 1))
		{
			++index;
			a[index][0] = a[index-1][0] + a[index-1][2];
			a[index][1] = a[index-1][1] + a[index-1][0];
			a[index][2] = a[index-1][2] + a[index-1][1];
		}
		i = (i >> 1);
		long long remain = 0;
		long long sum = 0;
		for (index; index >0; --index)
		{
			if (i & n)
			{
				long long d = (3-remain)%3;
				if (d == 0)
				{
					sum += a[index-1][0]-1;
				}
				else
				{
					sum += a[index-1][d];
				}
				++remain;
			}
			if (remain == 3)
			{
				sum++;
				remain = 0;
			}
			i = (i >> 1);
		}
		if (remain == 2 && (n & 1)) sum++;
		cout<<"Day "<<n<<": Level = "<<sum<<endl;
	}
	return 0;
}
