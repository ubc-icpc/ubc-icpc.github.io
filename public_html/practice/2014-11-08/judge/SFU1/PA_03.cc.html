// problem A
// SFU Royal

#include <cstdio>

using namespace std;

int main() {
	int ntests;
	int good[6];
	int evil[7];

	scanf("%d",&ntests);

	for(int ct = 1;ct <= ntests;ct++) {
		long long goodsum = 0;
		long long badsum = 0;

		for(int c = 0;c < 6;c++) {
			scanf("%d",good+c);
		}

		for(int c = 0;c < 7;c++) {
			scanf("%d",evil+c);
		}

		goodsum += good[0];
		goodsum += good[1] * 2;
		goodsum += good[2] * 3;
		goodsum += good[3] * 3;
		goodsum += good[4] * 4;
		goodsum += good[5] * 10;

		badsum += evil[0] * 1;
		badsum += evil[1] * 2;
		badsum += evil[2] * 2;
		badsum += evil[3] * 2;
		badsum += evil[4] * 3;
		badsum += evil[5] * 5;
		badsum += evil[6] * 11;

		printf("Battle %d: ",ct);

		if(goodsum > badsum) {
			puts("Good triumphs over Evil");
		} else if(badsum > goodsum) {
			puts("Evil eradicates all trace of Good");
		} else {
			puts("No victor on this battle field");
		}
	}

	return 0;
}
