#include <stdio.h>
#include <string.h>

#define ASCII 48
#define ull unsigned long long

inline bool allseen(bool * pSeen) {
    for (int i = 0; i < 10; ++i) {
        if (!pSeen[i]) return false;
    }
    return true;
}

int main() {
    ull n;
    bool seen[10];
    while (scanf("%llu", &n) != EOF) {
        memset(seen, 0, sizeof(seen));
        char str[22];
        for (ull k = 1;; ++k) {
            int l = sprintf(str, "%llu", n * k);
            for (int i = 0; i < l; ++i) {
                int index = (int)(str[i] - ASCII);
                seen[index] = true;
            }
            if (allseen(seen)) {
                printf("%llu\n", k);
                break;
            }
        }
    }
    return 0;
}
