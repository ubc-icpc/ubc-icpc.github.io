import fileinput

vowels = "aiyeou"
vowels_upper = vowels.upper()
vowels = dict(zip(vowels, vowels[3:] + vowels[:3]))
vowels_upper = dict(zip(vowels_upper, vowels_upper[3:] + vowels_upper[:3]))
consonants = "bkxznhdcwgpvjqtsrlmf"
consonants_upper = consonants.upper()
consonants = dict(zip(consonants, consonants[10:] + consonants[:10]))
consonants_upper = dict(zip(consonants_upper, consonants_upper[10:] + consonants_upper[:10]))
translation = dict(vowels.items() + vowels_upper.items() + consonants.items() + consonants_upper.items())

for line in fileinput.input():
    line = line.strip()
    answer = ""
    for char in line:
        if char in translation:
            answer += translation[char]
        else:
            answer += char
    print(answer)
