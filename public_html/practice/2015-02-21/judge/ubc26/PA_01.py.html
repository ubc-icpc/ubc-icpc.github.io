import sys
def N(): return [ int(s) for s in sys.stdin.next().split(' ') if s != '' ]


def K():
    #print "----------------"

    n, c = N()
    values = sorted(N(), key=lambda n:-n)
    others = sorted(N())
    others.append(sum(others) + 10000)
    prizes = sorted(N(), key=lambda n:-n)
    prizes.reverse()

    #print "n={0} c={1}".format(n, c)
    #print "values={0}".format(values)
    #print "prizes={0}".format(prizes)
    #print "others={0}".format(others)

    best = -1000000000
    for bribes in xrange(n+1):
        cost = 420 * bribes
        #print "{0} Bribes, Cost={1}".format(bribes, cost)
        listPoints = values[0:bribes]
        points = sum(listPoints)
        #print "Points: sum({0}) = {1}".format(listPoints, points)

        rank = 0
        for other in others:
            #print "---{0} {1}---".format(rank, other)
            if points <= other: break
            rank += 1

        assert rank >= 0

        behind = others[rank]
        #print "Rank: {0}, behind: {1}".format(rank, behind)

        prize = prizes[rank]
        result = prize - cost
        #print "Prize: {0}, Result: {1}".format(prize, result)
        #print ""
        
        if result > best:
            best = result

    print best

t, = N()
for i in xrange(t):
    K()
