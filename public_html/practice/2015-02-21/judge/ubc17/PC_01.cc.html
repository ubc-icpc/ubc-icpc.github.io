#include <stdio.h>

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int th, tm, ts, mm, ms;
        scanf("%d%*c%d%*c%d %d%*c%d", &th, &tm, &ts, &mm, &ms);

        int micro = 60 * mm + ms;
        bool ok = false;

        if (mm == th && ms == tm) {
            ok = true;
            goto end;
        }

        while (micro--) {
            ++ts;
            if (ts >= 60) {
                ts -= 60;
                ++tm;
            }
            if (tm >= 60) {
                tm -= 60;
                ++th;
            }
            if (th >= 24) {
                th -= 24;
            }
            --ms;
            if (ms < 0) {
                ms += 60;
                --mm;
            }
            if (th == mm && tm == ms) {
                ok = true;
                goto end;
            }
        }

end:
        printf(ok ? "Yes\n" : "No\n");
    }
}
