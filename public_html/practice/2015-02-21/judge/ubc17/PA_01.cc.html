#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

struct comp {
    bool operator() (int a, int b) {
        return a > b;
    }
} compt;

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n, c;
        scanf("%d %d", &n, &c);
        vector<int> s, v, p;
        for (int i = 0; i < n; ++i) {
            int temp;
            scanf("%d", &temp);
            v.push_back(temp);
        }
        for (int i = 0; i < c; ++i) {
            int temp;
            scanf("%d", &temp);
            s.push_back(temp);
        }
        for (int i = 0; i <= c; ++i) {
            int temp;
            scanf("%d", &temp);
            p.push_back(temp);
        }
        sort(s.begin(), s.end());
        sort(v.begin(), v.end(), compt);
        sort(p.begin(), p.end());

        int cash = 0;
        int maxcash = p[0];
        int points = 0;
        int place = c;
        for (int i = 0; i < n; ++i) {
            cash = (i + 1) * (-420);
            points += v[i];
            place = c;
            for (int j = 0; j < c; ++j) {
                if (s[j] == points) {
                    place = j;
                    break;
                }
                if (s[j] > points) {
                    place = j;
                    break;
                }
            }
            cash += p[place];
            //printf("q %d cash %d place %d points %d\n", i, cash, place + 1, points);
            if (cash > maxcash) {
                maxcash = cash;
            }
        }
        printf("%d\n", maxcash);
    }
    return 0;
}
