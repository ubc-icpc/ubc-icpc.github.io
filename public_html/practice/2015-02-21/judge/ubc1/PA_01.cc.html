#include <bits/stdc++.h>

using namespace std;

int main() {
	int t; cin >> t;
	while (t--) {
		int n, c; cin >> n >> c;
		vector<int> p(n), s(c), val(c+1);
		for (int i = 0; i < n; i++) {
			cin >> p[i];
		}

		for (int i = 0; i < c; i++) {
			cin >> s[i];
		}

		for (int i = 0; i < c+1; i++) {
			cin >> val[i];
		}

		sort(p.rbegin(), p.rend());
		sort(s.begin(), s.end());

		int ans = 0;
		for (int i = 0; i <= n; i++) {
			int cost = 420*i;
			int sc = 0;
			for (int j = 0; j < i; j++) {
				sc += p[j];
			}

			int k = 0;
			while (k < c && sc > s[k]) k++;

			int place = c-k;
			int profit = val[place] - cost;
			ans = max(ans, profit);
		}
		cout << ans << endl;
	}
	return 0;
}
