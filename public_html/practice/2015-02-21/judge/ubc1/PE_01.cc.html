#include <bits/stdc++.h>

using namespace std;

typedef long double ld;

int a[100005];
int main() {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	while (t--) {
		multiset<int, greater<int> > s;
		int n, u; cin >> n >> u;
		
		for (int i = 0; i < n; i++) {
			cin >> a[i];
			s.insert(a[i]);
		}

		ld tot = 0, cnt = 0;
		for (int i = 0; i < n; i++) {
			int x; cin >> x;
			u -= x;
			if (x) {
				s.erase(s.find(a[i]));
				tot += a[i];
				cnt++;
			}
		}

		while (u > 0 || *s.begin() > tot/cnt) {
			u--;
			tot += *s.begin();
			cnt++;
			s.erase(s.begin());
		}

		cout << setprecision(5) << fixed << tot/cnt << endl;
	}
}
