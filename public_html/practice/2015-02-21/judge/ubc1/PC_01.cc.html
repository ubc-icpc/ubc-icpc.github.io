#include <bits/stdc++.h>

using namespace std;

int to_sec(int h, int m, int s) {
	return 3600*h + 60*m + s;
}

vector<int> to_clock(int t, bool mode) {
	vector<int> v;
	if (mode) {
		v.push_back(t/3600);
	}
	v.push_back((t%3600)/60);
	if (!mode) {
		v.push_back(t%60);
	}
	return v;
}

int main() {
	int t; cin >> t;
	while (t--) {
		int h, m, s; char c;
		cin >> h >> c >> m >> c >> s;
		int t0 = to_sec(h,m,s);
		cin >> m >> c >> s;
		int t1 = to_sec(0,m,s);

		bool yes = false;
		while (t1 >= 0) {
			vector<int> v1 = to_clock(t1, 0), v0 = to_clock(t0, 1);

			if (v0 == v1) {
				yes = true;
				break;
			}

			t1--;
			t0 = (t0+1)%(24*3600);
		}

		cout << (yes ? "Yes" : "No") << endl;
	}
	return 0;
}
