import sys

line = sys.stdin.readline()

d =  {1: 5, 2: 5, 3: 5, 4: 5, 5: 5, 6: 5, 7: 5, 8: 5, 9: 5, 10: 5, 11: 4, 12: 4, 13: 4, 14: 4, 15: 4, 16: 3, 17: 3, 18: 3, 19: 3, 20: 3, 21: 2, 22: 2, 23: 2, 24: 2, 25: 2}
rank = 25
stars = 0
lastCount = 0

for g in line[:-1]:
    if g == 'L':
        lastCount = 0
        if rank <= 20:
            if rank < 20 and rank > 0 and stars == 0:
                rank += 1
                stars = d[rank] - 1
            else:
                stars = max(0, stars - 1)
    else:
        lastCount += 1
        if rank > 0 and stars == d[rank]:
            if lastCount >= 3 and rank >= 6:
                stars = 2
            else:
                stars = 1
            rank -= 1
        elif rank > 0 and stars == d[rank] - 1 and rank >= 6 and lastCount >= 3:
            rank -= 1
            stars = 1
        else:
            stars += 1 + int(lastCount >= 3 and rank >= 6)

if rank == 0:
    print "Legend"
else:
    print rank
