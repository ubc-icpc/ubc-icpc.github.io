import sys

def solve(arr):
    stocks = 0
    money = 100
    for i in range(len(arr)):
        # if have stocks
        if stocks:
            # should sell? only if this is a local maxima
            if i == len(arr) - 1 or arr[i] > arr[i+1]:
                money += arr[i]*stocks
                stocks = 0
        else:
            if i < len(arr) - 1 and money >= arr[i] and arr[i] < arr[i+1]:
                stocks = min(100000, int(money/arr[i]))
                money -= stocks * arr[i]
    return money

d = int(sys.stdin.readline())
arr = []
for i in range(d):
    arr.append(int(sys.stdin.readline()))

print solve(arr)
