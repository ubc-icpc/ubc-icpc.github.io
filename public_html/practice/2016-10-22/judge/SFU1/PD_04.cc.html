#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	int d;
	long long money = 100;
	long long stock = 0;

	cin >> d;
	vector<int> price(d);
	for (int i = 0; i < d; i++) {
		cin >> price[i];
	}

	for (int i = 0; i < d - 1; i++) {
		if (price[i] < price[i + 1] && stock == 0) {
			stock = min(money / price[i], 100000LL);
			money -= stock * price[i];
		}
		else if (price[i] > price[i + 1]) {
			money += stock * price[i];
			stock = 0;
		}
	}
	money += stock * price[d-1];
	stock = 0;
	cout << money << endl;
}
