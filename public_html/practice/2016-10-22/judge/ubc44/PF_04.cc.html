#include <iostream>
#include <cmath>
//#include <bits/stdc++.h>

using namespace std;

int main() {
	double n, p; cin >> n >> p;
	double prob = 0;
	double t=floor(n/(p-1)); double times = 0;

	prob = p*t/(n+t);
	times = p-1;
	for (int i=n; i>n-times; i--) {
		prob *= i;
		prob /= i+t-1;
	}

	cout << prob << endl;
	return 0;
}
