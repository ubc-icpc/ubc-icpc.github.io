import java.util.*;
 
public class JumbledCompass {
 
   
    static int assignQuadrant(int num) {
        if (num <= 90)
            return 1;
        else if (num <=180)
            return 2;
        else if (num <= 270)
            return 3 ;
        else
            return 4;
    }
    public static void main(String args[])  {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();
        sc.close();
        int output =0;
        int aQuad = assignQuadrant(a);
        int bQuad = assignQuadrant(b);
        int quadDiff = aQuad - bQuad;
        if (quadDiff == 0) {
            output = a-b;
        }
        else {
            output = getDistanceToAxis(aQuad, a, bQuad, b);
        }
        if (output == 180)
            output = -1 *output;
        System.out.println(-output);   
    }
   
    static int getDistanceToAxis(int aQuad, int a, int bQuad, int b) {
        int aDist, bDist = 0;
        if (aQuad ==1 & bQuad ==2) {
            aDist = 90 -a;
            bDist = b-90;
            return (-(aDist+ bDist));
        }
        else if (aQuad == 2 & bQuad ==1) {
            aDist = 90 -a;
            bDist = b-90;
            return (-aDist- bDist);
        }
        else if (aQuad ==2 & bQuad ==3) {
            aDist = 180 -a;
            bDist = b-180;
            return (-aDist- bDist);
        }
        else if ( aQuad == 3 & bQuad == 2 ) {
            aDist = a - 180;
            bDist = 180 - b;
            return (aDist +  bDist);
        }
        else if (aQuad == 3 & bQuad ==4) {
            aDist = 270 -a;
            bDist = b-270;
            return (-aDist- bDist);
        }
        else if(aQuad == 4 & bQuad == 3) {
            aDist = a - 270;
            bDist = 270 - b;
            return (aDist + bDist);
        }
        else if (aQuad ==1 & bQuad == 4) {
            aDist = a;
            bDist = 360-b;
            return (aDist + bDist);
        }
        else if (aQuad == 4 & bQuad == 1){
            aDist = 360 -a;
            bDist = b;
            return (-aDist - bDist);
        }
        else if( (aQuad == 1 & bQuad == 3)) {
            int clock = a + 90 + (270 - b);
            int counterClock = (90-a) + 90 + (b-180);
            if (clock <= counterClock)
                return clock;
            else
                return (-counterClock);
        }
        else if( aQuad == 3 & bQuad == 1) {
            int clock = (a-180) + 90 + (90 -b);
            int counterClock = (270-a) + 90 + b;
            if (clock <= counterClock)
                return clock;
            else
                return (-counterClock);
        }
        else if( aQuad == 2 & bQuad == 4) {
            int clock = (a-90) + 90 + (360-b);
            int counterClock = (180-a) + 90 + (b-270);
            if (clock <= counterClock)
                return clock;
            else
                return (-counterClock);
        }
        else {
            int clock = (a-270) + 90 + (180-b);
            int counterClock = (360-a) + 90 + (b-90);
            if (clock <= counterClock)
                return clock;
            else
                return (-counterClock);
        }
    }
 
}
