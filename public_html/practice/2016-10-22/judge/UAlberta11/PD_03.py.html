def buy (price):
	global money
	global stock
	#print(money)
	amount = money // price
	if(amount > 100000):
		amount = 100000
	# cout << "BUYING: " << amount << " at " << price << endl;
	#print("BUYING: {} at {}".format(amount,price))
	stock += amount
	money -= amount * price

def sell (price):
	# cout << "SELLING: " << stock << " at " << price << endl;
	global money
	global stock
	#print("SELLING: {} at {}".format(stock,price))
	money += stock*price
	stock = 0
