import java.util.ArrayList;
import java.util.Scanner;

public class StockBroker {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int foo = Integer.parseInt(sc.nextLine());
		long cash = 100;
		int stocks = 0;
		ArrayList<Integer> days = new ArrayList<>();
		for(int i = 0; i<foo; i++){
			days.add(Integer.parseInt(sc.nextLine()));
		}
		for(int i = 0; i< days.size(); i++){
			if(i == days.size()-1){
				cash += stocks*days.get(i);
			} else{
				if(days.get(i) <= days.get(i+1)){
					if(stocks == 100000){
						continue;
					}
					long newStocks =  cash/days.get(i);
					
					if(stocks + newStocks > 100000){
						newStocks = 100000 - stocks;
					}
					stocks += newStocks;
					cash  -=  days.get(i) * newStocks;
				} else{
					cash += stocks * days.get(i);
					stocks = 0;
				}
			}
		}
		
		System.out.println(cash);
		
	}
}
