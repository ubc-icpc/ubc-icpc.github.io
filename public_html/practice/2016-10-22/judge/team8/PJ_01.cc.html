#include <bits/stdc++.h>

using namespace std;

int main() {
	int n1, n2; cin >> n1 >> n2;
	int ans;
	if (n1 > n2) {
		if (n1 - n2 < n2 + 360 - n1) {
			ans = n2 - n1;
		} else {
			ans = n2 + 360 - n1;
		}
	} else {
		if (n2 - n1 < n1 + 360 - n2) {
			ans = n2 - n1;
		} else {
			ans = n2 - n1 - 360;
		}
	}
	if (ans == -180) cout << 180 << endl;
	else cout << ans << endl;
	return 0;
}
