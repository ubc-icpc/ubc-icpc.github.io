#include <bits/stdc++.h>

#define endl '\n'
using namespace std;

vector<int> minStars = vector<int>(26);

void pre() {
	int curStars = 0;
	for (int i = 25; i >= 0; i--) {
		minStars[i] = curStars;
		if (i >= 21) curStars += 2;
		else if (i >= 16) curStars += 3;
		else if (i >= 11) curStars += 4;
		else curStars += 5;
	}
}

int main() {
	pre();
	char c;
	int stars = 0;
	bool wprev = false;
	bool wprev2 = false;
	bool legend = false;
	int rank = 25;
	while (cin >> c) {
		if (c == 'W') {
			stars++;
			if (rank >= 6 && wprev && wprev2) stars++;
			if (stars > minStars[rank-1]) {
				rank--;
				if (rank == 0) {
					legend = true;
					break;
				}
			}
			wprev2 = wprev;
			wprev = true;
		} else {
			if (rank <= 20 && stars != minStars[20]) {
				stars--;
			}
			if (stars < minStars[rank]) rank++;
			wprev2 = wprev;
			wprev = false;
		}
	}
	if (legend) {
		cout << "Legend" << endl;
	} else {
		cout << rank << endl;
	}
	return 0;
}
