#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;
char a[20000];
int t[30];
int Rank[30];
int star = 0;
int consecutive = 0;
int pos = 25;
bool Flag = false;
int main(){
    Rank[0]=1000000000000;
    for (int i = 1; i <= 10; i++) Rank[i] = 5;
    for (int i = 11; i <= 15; i++) Rank[i] = 4;
    for (int i = 16; i <= 20 ; i++) Rank[i] = 3;
    for (int i = 21; i <= 25 ; i++) Rank[i] = 2;
    for (int i = 1; i <= 25 ; i++) t[i]=Rank[i];

    gets(a);
    for (int i=0; i < strlen(a); i++){
        if (a[i]=='W'){
            consecutive++;
            if ((consecutive>=3)&&(6<=pos)&&(pos<=25)) {
                if (Rank[pos]==0) pos--;
                    Rank[pos]--;
            }
            if (Rank[pos]==0) pos--;
            Rank[pos]--;
        }
        if (pos <= 0) {
            Flag = true;
            pos = 0;
        }
        if (a[i]=='L'){
            consecutive = 0;
            if ((1<=pos)&&(pos<20)){
                Rank[pos]++;
                if (Rank[pos]>t[pos]){
                    Rank[pos]=t[pos];
                    pos++;
                    Rank[pos]=1;
                }
            }
            else
            if (pos == 20){
                Rank[pos]++;
                if (Rank[pos]>t[pos]) Rank[pos] = t[pos];
            }
        }
    }
    if (Flag) {
        cout<<"Legend";
    }
    else
        cout<<pos;
    return 0;
}
