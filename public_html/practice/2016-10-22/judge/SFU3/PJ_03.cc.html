#include <iostream>
#include <cmath>

using namespace std;

int main(){
	int beg, end;
	cin >> beg >>  end;
		
		
		int spin;
		if (end>beg){
			if (end-beg>180)
				spin=(360-end+beg)*-1;
			else
				spin=end-beg;
		}
		else if (end==beg)
			spin=0;
		else{
			if (beg-end>180)
				spin=(360-beg+end);
			else
				spin=(beg-end)*-1; 
	}
	if (spin==-180)
		spin=abs(spin);

	cout << spin <<endl;
	
	return 0;
}
