#include <iostream>
#include <stdio.h>
using namespace std;

int n;
long a[400];
long minn;
long profit=100;
long many=0;
int main()
{
    scanf("%d",&n);
    for (int i=1; i <= n; i++){
        cin>>a[i];
    }
    for (int i=2; i <= n; i++){
        if (a[i] > a[i-1]){
            profit = profit + many*a[i-1];
            if (profit/a[i-1]>100000)
                many = 100000;
            else
                many = profit/a[i-1];
            profit-= many*a[i-1];
        }
        else
        if (a[i] < a[i-1]){
            profit = profit + many*a[i-1];
            if (profit/a[i-1]>100000)
                many = 100000;
            else
                many = profit/a[i-1];
            profit -= many*a[i];
        }
    }
    profit = profit + many*a[n];
    cout<<profit;
    return 0;
}
