import java.util.*;
public class Main{
    public static void main (String[] args){
        Scanner input = new Scanner(System.in);

        int d = input.nextInt();

        int[] prices = new int[d];
        for (int i = 0; i < d; i++){
            prices[i] = input.nextInt();
        }

        long purse = 100;
        int sharesOwned = 0;
        for (int i = 0; i < d-1; i++){
            if (prices[i+1] < prices[i]){
                if (sharesOwned > 0){
                    purse+= (sharesOwned)*prices[i];
                    sharesOwned = 0;
                }
            }else{
                if (sharesOwned < 100000){
                    if (purse >= prices[i]){
                        long amountToBuy = purse / prices[i];
                        if (amountToBuy + sharesOwned > 100000){
                            amountToBuy = 100000 - sharesOwned;
                        }
                        sharesOwned += amountToBuy;
                        purse -= amountToBuy*prices[i];
                    }
                }
            }
        }
        if (sharesOwned > 0){
            purse+= sharesOwned * prices[d-1];
        }
        System.out.println(purse);
    }
}
