import java.util.*;
import java.math.*;
public class Main{
    public static void main (String[] args){
        Scanner input = new Scanner(System.in);
        int n = input.nextInt();
        int n2 = input.nextInt();

        int ans1 = 0;
        int ans2 = 0;
        int ans = 0;
        if (n2 < n){
            ans1 = 360 - n;
            ans1 += n2;
            ans2 = n2 - n;
            ans = Math.min(ans1, Math.abs(ans2));
            if (ans == Math.abs(ans2)) ans = ans2;
        }else{
            ans1 = n2 - n;
            ans2 = -n;
            ans2+= -(360-n2);
            ans = Math.min(ans1, Math.abs(ans2));
            if (ans == Math.abs(ans2)) ans = (ans2);
        }
        if (ans == -180) ans = 180;
        System.out.println(ans);
    }
   
}
