#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>

int main()
{
	int d;
	scanf("%d",&d);
	long long m=100;
	int last;
	scanf("%d",&last);
	for (int i=1;i<d;i++){
		int price;
		scanf("%d",&price);
		if (price>last){
			long long share=(m/last);
			if (share>100000) share=100000;
			m=m+share*(price-last);
		}
		last=price;
	}
	printf("%lld\n",m);

	return 0;
}
