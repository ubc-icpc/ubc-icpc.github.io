#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>

int main()
{
	int t1, t2;
	scanf("%d", &t1);
	scanf("%d", &t2);
	int a1 = (t2 - t1) % 360;
	int a2 = -(t1 + 360 - t2) % 360;
	if (a1 <= -180) a1 = 360 + a1;
	if (a2 <= -180) a2 = 360 + a2;
	if (a1 > 180) a1 = a1-360;
	if (a2 > 180) a2 = a2-360;
	if (abs(a1) < abs(a2)) printf("%d\n",a1);
	else printf("%d\n", a2);
	return 0;
}
