double rec[1000005];

double cm(int a)
{
    if(a==1)return 1.0/(n+1);
    if(rec[a]!=0)return rec[a];
    return rec[a]=1.0*(n+a-p)/(n+a)*cm(a-1);
}

int main()
{
    cin>>n>>p;
    int a=1;
    double lt=-1,tt;
    //memset(rec,-1,sizeof(rec));
    while(1)
    {
        //tt=i*c(n,p-1)/c(n+i,p);
        tt=a*p*cm(a);
        if(tt<lt)break;
        lt=tt,a++;
    }
    printf("%.8lf\n",lt);
    return 0;
}
