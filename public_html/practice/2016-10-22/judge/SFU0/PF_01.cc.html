#include <iostream>
#include <cstdio>


using namespace std;

double solve(long long n, long long y, long long p) {
    double result = y * p;
    for (int i = 0; i < p - 1; i++) {
        result = result * (n - i) / (n + y - i);
    }
    result /= (n + y - p + 1);
    return result;
}


int main() {
    int n, p;
    cin >> n >> p;
    
    int l = 1;
    int r = n;
    while (l < r) {
        double mid = solve(n, (l + r) / 2, p);
        double mid1 = solve(n, (l + r) / 2 + 1, p);
        if (mid1 > mid) {
            l = (l + r) / 2 + 1;
        }
        else {
            r = (l + r) / 2;
        }
    }
    //cout << solve(3, 2, 2) << endl;
    
    printf("%.10f\n", solve(n,l,p));
}
