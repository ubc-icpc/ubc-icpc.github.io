#include <map>
#include <iostream>
using namespace std;
map <int,int> stark = {
{1,5},{2,5},{3,5},{4,5},{5,5},{6,5},{7,5},{8,5},{9,5},{10,5},{11,4},{12,4},{13,4},{14,4},{15,4},{16,3},{17,3},{18,3},{19,3},{20,3},{21,2},{22,2},{23,2},{24,2},{25,2}};

bool win_game(int &rk,int& st,int& wstreak){
    wstreak++;st++;
    if(wstreak>2 && rk>5) st++;
    if(st>stark[rk]){st-=stark[rk];rk--;}
    if(rk==0) return 1;
    return 0;
}
void lose_game(int &rk,int& st,int& wstreak){
    wstreak=0;
    if(rk<21)st--;
    if(rk==20&&st<0)st=0;
    if(st<0){rk++;st+=stark[rk];}
    return;
}
int main(){
    int wstreak(0),rk(25),st(0);
    char g;
    bool leg = false;
    while(!(leg) && cin>>g){
        if(g=='W'){leg=win_game(rk,st,wstreak);}
        else lose_game(rk,st,wstreak);
    }
    if(leg)cout << "Legend"<<endl;
    else cout << rk << endl;
    return 0;
}
