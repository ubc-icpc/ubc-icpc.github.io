#include <iostream>


using namespace std;

long long betmin(long long mon){
    if(mon<100000) return mon;
    return 100000;
}

int main()
{
    int D,last,nw;
    long long money(100);
    cin >> D;
    cin >> last;
    for(int i=1;i<D;i++){
        cin >> nw;
        if(nw>last) money += betmin(money/last) * (nw-last);
        last = nw;
    }
    cout << money<<endl;
    return 0;
}
