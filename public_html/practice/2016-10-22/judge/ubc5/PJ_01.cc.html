#include <bits/stdtr1c++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  int n1, n2;
  while (cin >> n1 >> n2) {
    int N = n1, cw = 0, ccw = 0;
    while (N != n2) {
      N = (N+1)%360;
      cw++;
    }
    N = n1;
    while (N != n2) {
      N = (N+359)%360;
      ccw++;
    }
    if (ccw < cw)
      cout << -ccw << "\n";
    else
      cout << cw << "\n";
  }
  return 0;
}
