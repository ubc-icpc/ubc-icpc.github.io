d = int(raw_input())
p = []
s, m = 0, 100
for i in xrange(d):
  p.append(int(raw_input()))
ind = 0
low = False
while ind < d:
  if ind == d-1:
    m += s*p[ind]
    break
  if not low:
    if p[ind+1] > p[ind]:
      sb = min(100000,m/p[ind])
      s = sb
      m -= sb*p[ind]
      low = True
  else:
    if p[ind+1] < p[ind]:
      m += p[ind]*s
      s = 0
      low = False
  ind += 1
print m
