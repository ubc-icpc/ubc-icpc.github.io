def w(n,p,y):
  t = 1.0
  for i in xrange(p-1):
    t *= float(n-i)/float(n+y-i)
  return t*y*p/float(n+y-p+1)

n, p = [int(i) for i in raw_input().split()]
cp = 1.0
for i in xrange(p-1):
  cp *= float(n-i)/(n-i+1)
cp *= float(p)/(n-p+2)
P = cp
y = 1
while True:
  cp *= float(n+y-p+2)*(n+y-p+1)*(y+1)/(n+y+1)/(n+y-p+2)/y
  if cp > P:
    P = cp
  else:
    break
  y += 1
print P
