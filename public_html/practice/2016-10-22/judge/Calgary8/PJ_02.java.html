import java.util.*;
import java.io.*;

public class J {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n1 = sc.nextInt();
        int n2 = sc.nextInt();

        int ans = Integer.MAX_VALUE;

        if (n1 >= n2) {
            int clockwise = 360 - n1 + n2;
            int cclockwise = n1 - n2;
            ans = Math.min(clockwise, cclockwise);
            if (ans == cclockwise) ans = -ans;
        } else {
            int clockwise = n2 - n1;
            int cclockwise = 360 - n2 + n1;
            ans = Math.min(clockwise, cclockwise);
            if (ans == cclockwise) ans = -ans;
        }

        if (ans == -180) ans = 180;
        System.out.println(ans);
    }
}
