import java.util.*;
import java.io.*;

public class D {
    public static void main(String[] args) {
        Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        int d = sc.nextInt();
        long[] p = new long[d];
        for (int i = 0; i < d; i++) {
            p[i] = sc.nextLong(); 
        }

        long money = 100;
        long buy = 0;
        long buyCount = 0;
        int i = 0;
        while (i < d && p[i] > money) i++;

        for (; i < d && i + 1 < d; i++) {
            if (buy > 0 && buy < p[i]) {
                money += p[i] * buyCount;
                buy = 0;
                buyCount = 0;
            }
            if (buy == 0 && money >= p[i] && p[i] < p[i+1]) {
                buy = p[i];
                buyCount = Math.min(money / buy, 100000);
                money -= buy * buyCount;
            }
        }
        money += Math.max(buy * buyCount, p[d-1] * buyCount);

        System.out.println(money);
    }
}
