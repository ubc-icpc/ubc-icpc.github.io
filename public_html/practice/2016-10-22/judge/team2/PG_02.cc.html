#include <cstdio>

int size(int n) {
	if (n >= 21) return 2;
	else if (n >= 16) return 3;
	else if (n >= 11) return 4;
	else return 5;
}

int main() {
	int rank = 25, star = 0, streak = 0;
	char s[10001]; scanf("%s", s);
	for (int i = 0; s[i] != 0; i++) {
		char c = s[i];
		if (rank == 0)
			continue;
		if (c == 'W') {
			streak += 1;
			star += (streak >= 3 && rank >= 6) ? 2 : 1;
			if (star > size(rank)) {
				star -= size(rank);
				rank -= 1;
			}
		} else {
			streak = 0;
			if (rank <= 20) {
				star -= 1;
				if (star == -1) {
					if (rank == 20) {
						star = 0;
					} else {
						rank += 1;
						star = size(rank) - 1;
					}
				}
			}
		}
	}
	if (rank == 0)
		printf("Legend\n");
	else
		printf("%d\n", rank);
	return 0;
}
