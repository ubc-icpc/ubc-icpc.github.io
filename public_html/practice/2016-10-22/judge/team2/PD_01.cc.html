#include <bits/stdc++.h>

using namespace std;

#define endl '\n'

int main() {
	int d; cin >> d;
	vector<int> prices;
	for(int i = 0; i < d; i++) {
		int curr; cin >> curr; prices.push_back(curr);
	}
	long long money = 100;
	int shares = 0;
	for(int i = 0; i < d-1; i++) {
		if(prices[i+1] > prices[i] && shares == 0) {
			shares = min(money / prices[i], (long long) 100000);
			money -= shares*prices[i];
		} else if(prices[i+1] < prices[i] && shares > 0) {
			money += shares*prices[i];
			shares = 0;
		}
	}

	if(shares > 0) money += shares * prices[d-1];
	cout << money << endl;
}
