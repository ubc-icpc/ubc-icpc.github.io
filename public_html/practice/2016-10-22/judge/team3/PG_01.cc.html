#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

int main() {
	ios::sync_with_stdio(0);
	string inp; cin >> inp;
	int rank = 25;
	int stars = 0;
	int wins = 0;
	for (char& c : inp) {
		if (c == 'W') {
			wins++;
			stars++;
		} else {
			wins = 0;
			if (rank <= 20) stars--;
		}
		if (wins >= 3 && rank >= 6) {
			stars++;
		}
		if (rank >= 21) {
			if (stars > 2) {
				rank--;
				stars -= 2;
			}
		} else if (rank >= 16) {
			if (stars > 3) {
				rank--;
				stars -= 3;
			} else if (stars < 0) {
				if (rank == 20) {
					stars = 0;
				} else {
					rank++;
					stars = 2;
				}
			}
		} else if (rank >= 11) {
			if (stars > 4) {
				rank--;
				stars -= 4;
			} else if (stars < 0) {
				rank++;
				stars = 3;
				if (rank == 16) stars = 2;
			}
		} else if (rank >= 1) {
			if (stars > 5) {
				rank--;
				stars -= 5;
			} else if (stars < 0) {
				rank++;
				stars = 4;
				if (rank == 11) stars = 3;
			}
		}
		if (rank == 0) break;
	}
	if (rank == 0) {
		cout << "Legend" << endl;
	} else {
		cout << rank << endl;
	}
	return 0;
}
