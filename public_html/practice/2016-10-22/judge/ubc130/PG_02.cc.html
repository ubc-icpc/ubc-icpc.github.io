#include <iostream>

using namespace std;

int main(void) {
    std::string line; cin >> line;

    int rank = 25;
    int stars = 0;
    int streak = 0;

    for(int i = 0; i < line.length(); i++) {
        char c = line[i];
        if (c == 'W') {
            streak++;
            if (rank == 1 && stars == 5) {
                cout << "Legend";
                return 0;
            }
            if (rank <= 5) {
                stars++;
                if (stars > 5) {
                    stars = 1;
                    rank--;
                }
            } else if (rank <= 10) {
                if (streak >= 3) {
                    stars += 2;
                } else {
                    stars++;
                }
                if (stars > 5) {
                    stars-= 5;
                    rank--;
                }
            } else if (rank <= 15) {
                if (streak >= 3) {
                    stars += 2;
                } else {
                    stars++;
                }
                if (stars > 4) {
                    stars-= 4;
                    rank--;
                }
            } else if (rank <= 20) {
                if (streak >= 3) {
                    stars += 2;
                } else {
                    stars++;
                }
                if (stars > 3) {
                    stars-= 3;
                    rank--;
                }

            } else if (rank <= 25) {
                if (streak >= 3) {
                    stars += 2;
                } else {
                    stars++;
                }
                if (stars > 2) {
                    stars-= 2;
                    rank--;
                }
            }
        } else {
            streak = 0;
            if (rank <= 9) {
                if (stars == 0) {
                    rank++;
                    stars = 4;
                } else {
                    stars--;
                }
            } else if (rank <= 14) {
                if (stars == 0) {
                    rank++;
                    stars = 3;
                } else {
                    stars--;
                }
            } else if (rank <= 19) {
                if (stars == 0) {
                    rank++;
                    stars = 2;
                } else {
                    stars--;
                }
            } else if (rank <= 20) {
                if (stars > 0) {
                    stars--;
                }

            }
        }
    }

    cout << rank;
    return 0;
}
