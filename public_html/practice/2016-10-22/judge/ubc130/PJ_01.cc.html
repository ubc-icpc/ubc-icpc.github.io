#include <iostream>

using namespace std;

int main(void) {
    int a1, a2; cin >> a1; cin >> a2;

    if (a1-a2 == 180 || a1-a2 == -180) {
        cout << 180;
        return 0;
    }

    if (a1 == a2) {
        cout << 0;
        return 0;
    } else if (a1 < a2) {
        if (abs(a2 - a1) > abs(a2 - a1 - 360)) {
            cout << a2 - a1 - 360;
        } else {
            cout << a2 - a1;
        }
        return 0;
    } else if (a2 < a1) {
        if (abs(a2 - a1) > abs(a2 - a1 + 360)) {
            cout << a2 - a1 + 360;
        } else {
            cout << a2 - a1;
        }
        return 0;
    }
}
