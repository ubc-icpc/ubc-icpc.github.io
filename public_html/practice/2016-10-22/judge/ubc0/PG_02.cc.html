int main(){
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    string s;
    cin >> s;
    int rank = 25, stars = 0;
    
    int consecW = 0;
    for(int i = 0; i < s.size(); i++){
        if(s[i] == 'W')
            consecW++;
        else
            consecW = 0;
        bool gotBonus = false;
        if(rank < 6)
            gotBonus = true;
    bonus:
        if(s[i] == 'W'){
            if((rank >= 21 && stars == 2) || (rank >= 16 && stars == 3) || (rank >= 11 && stars == 4) ||
                (rank >= 1 && stars == 5)){
                rank--;
                stars = 1;
            }
            else
                stars++;
            
            if(rank == 0){
                cout << "Legend" << endl;
                return 0;
            }
            
            if(!gotBonus && consecW >= 3){
                gotBonus = true;
                goto bonus;
            }
        }
        else{
            if(rank > 20)
                continue;
            else if(stars != 0)
                stars--;
            else if(rank >= 20)
                continue;
            else if(rank >= 15){
                rank++;
                stars = 2;
            }
            else if(rank >= 10){
                rank++;
                stars = 3;
            }
            else{
                rank++;
                stars = 4;
            }
        }
    }
    
    cout << rank << endl;
    
    return 0;
}
