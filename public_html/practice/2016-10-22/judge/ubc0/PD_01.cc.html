typedef vector<int> VI;

typedef double T;

typedef vector<VT> VVT;

int main(){
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    int d;
    cin >> d;
    
    VI prices(d);
    
    for(int i = 0; i < d; i++)
        cin >> prices[i];
    
    long long money = 100;
    for(int i = 1; i < d; i++){
        if(prices[i] > prices[i - 1])
            money += min(100000LL, money / prices[i - 1]) * (prices[i] - prices[i - 1]);
    }
    
    cout << money << endl;
    
    return 0;
}
