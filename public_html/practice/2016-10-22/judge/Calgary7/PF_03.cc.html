using namespace std;

typedef long long ll;

double prob(ll n, ll m, ll p) {
	double ret = 1.0 * n * p / (m + 1.0);
	while (m + n - p > 0 && m + n > m + 1) {
		ret *= (1.0 * m + n - p) / (m + n);
		n--;
	}
	return ret;
}

int main() {
	ll m, p;
	cin >> m >> p;
	double ans = 0.0;
	ll l = 1, r = 1e6;
	while (r - l > 20ll) {
		ll one_third = l + (r - l) / 3, two_third = l + 2 * (r - l) / 3;
		double x = prob(one_third, m, p), y = prob(two_third, m, p);
		if (x < y) l = one_third;
		else r = two_third;
	}
	for (ll i = l; i <= r; i++) {
		ans = max(ans, prob(i, m, p));
	}
	printf("%.10lf\n", ans);
	return 0;
}
