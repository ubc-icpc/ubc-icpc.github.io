	cin >> a >> b;
	if (a < b) {
		int x = -a - (360 - b), y = b - a;
		if (abs(x) == 180) cout << 180 << endl;
		else cout << (abs(x) < abs(y) ? x : y) << endl;
	}
	else {
		int x = 360 - a + b, y = b - a;
		if (abs(x) == 180) cout << 180 << endl;
		else cout << (abs(x) < abs(y) ? x : y) << endl;
	}
	return 0;
}
