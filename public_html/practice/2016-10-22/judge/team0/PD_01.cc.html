#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;
int D, A[370];
ll M[370];

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> D;
  for (int i = 0; i < D; ++i) {
    cin >> A[i];
  }
  M[0] = 100;

  for (int i = 1; i < D; ++i) {
    for (int j = 0; j < i; ++j) {
      M[i] = max(M[i], M[j]);
      ll s = min(M[j] / A[j], 100000LL);
      M[i] = max(M[i], M[j] - s * A[j] + s * A[i]);
    }
  }
  cout << M[D-1] << endl;

  return 0;
}
