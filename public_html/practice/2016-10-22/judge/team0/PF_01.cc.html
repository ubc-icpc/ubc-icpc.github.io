#include<bits/stdtr1c++.h>
using namespace std;

typedef long double ld;

int N, P;

ld calc(int k) {
  ld res = (ld)k / (N + k);
  for (int i = N; i >= N - P + 2; --i) {
    res *= (ld)i / (k - 1 + i);
  }
  res *= P;
  return res;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> N >> P;
  //for (int i = 0; i < N + 10; ++i) {
  //  cout << "K = " << i << " " <<  calc(i) << endl;
  //}
  cout << fixed << setprecision(10) << calc(N / (P - 1)) << endl;

  return 0;
}
