#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    int a,b; cin>>a>>b;
    if (abs(a-b)<=180) {
        if ((a>b)&&(abs(a-b)!=180)) cout<<"-";
        cout<<abs(a-b)<<endl;
    }
    else{
        if ((a<b)&&(abs(a-b)!=180)) cout<<"-";
        cout<<360-abs(a-b)<<endl;
    }
    return 0;
}
