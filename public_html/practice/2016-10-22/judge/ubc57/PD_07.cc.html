#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll xx[400];
int main(){
    ios::sync_with_stdio(0);
    ll money=100,shares=0;
    int d; cin>>d;
    ll least=505,most=0;
    for (int i=0;i<d;i++) {
        cin>>xx[i];
        least=min(least,xx[i]);
        most=max(most,xx[i]);
    }
    for (int i=0;i<d;i++){
        ll x=xx[i];
        bool least=true,most=true;
        if ((i==d-1)||(xx[i]>xx[i+1])){
            money+=x*shares; shares=0;
        }
        if (xx[i]<xx[i+1]){
            ll buy=money/x; 
            if (shares+buy>100000) buy=100000-shares;
            money-=buy*x; shares+=buy;
        }
        //cout<<money<<' '<<shares<<' '<<least<<endl;
    }
    cout<<money<<endl;
    return 0;
}
