#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int needed[30];
int main(){
    ios::sync_with_stdio(0);
    needed[0]=1000000;
    for (int i=1;i<=10;i++) needed[i]=5;
    for (int i=11;i<=15;i++) needed[i]=4;
    for (int i=16;i<=20;i++) needed[i]=3;
    for (int i=21;i<=25;i++) needed[i]=2;
    int rank=25,stars=0,winning=0;
    string s; cin>>s;
    for (int i=0;i<s.size();i++){
        if (s[i]=='W'){
            winning++;
            stars++; 
            if ((winning>=3)&&(rank>=6)) stars++; 
            if (stars>needed[rank]){
                stars-=needed[rank];
                rank--;
            }
        }
        else{
            winning=0;
            if (rank<=20){
                if ((stars==0)&&(rank!=0)){
                    //drop rank
                    if (rank!=20){
                        rank++; 
                        stars=needed[rank]-1;
                    }
                }
                else stars--;
            }
        }
        //cout<<rank<<' '<<stars<<endl;
    }
    if (rank==0) cout<<"Legend"<<endl;
    else cout<<rank<<endl;
    return 0;
}
