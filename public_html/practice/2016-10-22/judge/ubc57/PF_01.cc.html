#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef long double ld;

int main(){
    ios::sync_with_stdio(0);
    ld n,p; cin>>n>>p;
    ld ans=0;
    ld tent=1.0/(n+1.0);
    ans=max(ans,p*tent);
    for (int x=2;x<=1000000;x++){
        tent=tent*(n-p+ld(x))/(n+ld(x));
        ans=max(ans,p*ld(x)*tent);
    }
    cout<<fixed<<setprecision(10)<<ans<<endl;
    return 0;
}
