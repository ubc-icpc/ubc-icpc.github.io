#include <bits/stdc++.h>
using namespace std;

inline int getstar(int n) {
        return min(5, 6 - (n - 1) / 5);
}

int main() {
        string s;
        cin >> s;
        int r = 25;
        int stars = 0; 
        int countc = 0;
        for (int i = 0; i < s.size(); i++) {
                if (s[i] == 'W') {
                        stars++;
                        countc++;
                        if (countc >= 3 && r >= 6) {
                                stars++;
                        }
                } else {
                        countc = 0;
                        if (r <= 20 && r > 0)
                                stars--;
                }
                if (stars > getstar(r)) {
                        r--;
                        stars = stars - getstar(r + 1);
                }
                if (stars < 0) {
                        if (r < 20) {
                                r++;
                                stars = getstar(r) - 1;
                        } else if (r == 20) {
                                stars = 0;
                        }
                }
        }
        if (r <= 0) {
                cout << "Legend\n";
        } else { 
                cout << r << endl;
        }
        return 0;
}
