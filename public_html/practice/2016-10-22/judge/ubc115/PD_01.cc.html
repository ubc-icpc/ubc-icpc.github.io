#include <bits/stdc++.h>
using namespace std;

vector<long long> pp;
int main() {
        int d;
        long long p;
        cin >> d;
        for (int i = 0; i < d; i++) {
                cin >> p;
                pp.push_back(p);
        }
        int q = 0;
        while (pp[q] != 100) {
                q++;
        }
        long long M = 100;
        int N = 0;
        for (int i = 0; i < d - 1; i++) {
                if (pp[i + 1] > pp[i]) {
                        N = min((long long) 100000, M / pp[i]);
                        M = M - N * pp[i];
                        M += N * pp[i + 1];
                }
        }
        cout << M << endl;
        return 0;
}
