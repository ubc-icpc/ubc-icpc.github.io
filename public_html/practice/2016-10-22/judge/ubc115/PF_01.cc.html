#include <bits/stdc++.h>
using namespace std;
typedef long double ld;
int main() {
        int n, p;
        cin >> n >> p;
        ld r = (ld) p / (n + 1.0) * (n / (p - 1));
        for (int i = 2; i <= n / (p - 1); i++) {
                r *= (ld) (n + i - p) / (n + i);
        }
        cout << fixed << setprecision(7) << r << endl;
        return 0;
}
