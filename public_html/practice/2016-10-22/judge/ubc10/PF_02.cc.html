#include <iostream>
#include <iomanip>

using namespace std;

double f(int x, int y, int p) {
    double ans = (double)y/(x+y);
    double prod = (double)x/(x+y);
    for (int i = 1; i < p; i++) {
        ans = ans * (double)(x-i+1)/(x-i+y) + prod * (double)y/(x-i+y);
        prod *= (double)(x-i)/(x-i+y);
    }
    return ans;
}

int main() {
    int n, p; cin >> n >> p;
    int y = 1;
    double fy = f(n, y, p);
    while (true) {
        double fy2 = fy * (double)(n-p+2+y-1)/(n+y+1)*(double)(y+1)/y;
        if (fy2 > fy) {
            fy = fy2;
            y++;
        } else {
            break;
        }
    }
    cout << setprecision(16) << fixed << fy << endl;
}
