#include <iostream>
#include <algorithm>

using namespace std;
typedef long long ll;

int main(){
    int n; cin >> n;
    ll money = 100;
    ll prev; cin >> prev;
    for (int i = 1; i < n; i++) {
        ll curr; cin >> curr;
        if (curr > prev) {
            ll buy = min(100000LL, money/prev);
            money += buy*(curr-prev);
        }
        prev = curr;
    }
    cout << money << endl;
}
