#include <iostream>
#include <string>
#include <iomanip>

using namespace std;

int main() {
	long double n, p;
	cin >> n >> p;
	long double max = 0, prev = 1;
	prev = p/(1.0+n);
	max = prev;
	for(long double i = 2; i <= n+p; i++) {
		prev *= i/(i-1);
		prev *= (n-p+i)/(i+n);
		if(prev > max) max = prev;
	}
	cout << setprecision(10) << max << endl;
}
