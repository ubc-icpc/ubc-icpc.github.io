#include <iostream>
using namespace std;

int main() {
  int d;
  int p[365];
  long long curr_money = 100;
  long long curr_stock = 0;
  cin >> d;
  for (int i = 0; i < d; i++)
    cin >> p[i];

  int min_price = p[0];
  for (int i = 1; i < d; i++) {
    while (p[i] <= min_price) {
      if (i == d - 1) goto end;
      min_price = p[i];
      i++;
    }
    curr_stock = curr_money / min_price;
    if (curr_stock > 100000) curr_stock = 100000;
    curr_money -= curr_stock * min_price;
    curr_money += curr_stock * p[i];
    min_price = p[i];
  }
end:
  cout << curr_money << '\n';
}
