#include <iostream>
#include <cmath>
using namespace std;

int main() {
  int n1, n2;
  cin >> n1 >> n2;
  if (n2 >= n1) {
    if (n2 - n1 <= 180) {
      cout << n2 - n1 << '\n';
    } else {
      cout << n2 - n1 - 360 << '\n';
    }
  } else {
    if (n1 - n2 < 180) {
      cout << n2 - n1 << '\n';
    } else {
      cout << n2 - n1 + 360 << '\n';
    }
  }
  return 0;
}
