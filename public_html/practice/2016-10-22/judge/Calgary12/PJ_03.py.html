n1 = int(input())
n2 = int(input())

if n1 == n2:
    print(0)
else:
    if n1 > n2:
        adjn2 = n2 + 360
        clockwise_dist = adjn2 - n1
        cc_dist = -(n1 - n2)
    elif n2 > n1:
        adjn2 = n2 - 360
        clockwise_dist = n2 - n1
        cc_dist = adjn2 - n1

    if abs(clockwise_dist) > abs(cc_dist):
        print(cc_dist)
    elif abs(clockwise_dist) < abs(cc_dist):
        print(clockwise_dist)
    else:
        print(180)
