import java.util.Scanner;

public class outlets {
	
	public static void main(String[] args) {
		Scanner s = new Scanner(System.in);
		int N = s.nextInt();
		while(N-->0) {
			int K = s.nextInt();
			int sum = 0;
			for(int i=0; i<K; i++)
				sum += (s.nextInt() - 1);
			sum++;
			System.out.println(sum);
		}
		s.close();
	}

}
