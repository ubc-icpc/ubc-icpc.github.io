#include<iostream>
using namespace std;
int main(int argc, char* argv[]) {
    int n;
    cin >> n;
    while (n--) {
        int strips, totalStrips;
        int totalOutlets = 1;
        cin >> strips;
        totalStrips = strips;
        while (strips--) {
            int outlets;
            cin >> outlets;
            totalOutlets += outlets;
        }
        cout << totalOutlets - totalStrips << "\n";
    }
    return 0;
}
