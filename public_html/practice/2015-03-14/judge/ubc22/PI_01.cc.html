#include<bits/stdtr1c++.h>
using namespace std;

int main () {
  int N; cin >> N;
  while (N) {
    long double wires[22], total = 0;
    for (int i = 0; i < N; ++i) cin >> wires[i];
    sort(wires, wires + N);
    bool good = false;
    total = wires[0];
    for (int i = 1; !good && i < N; ++i) {
      if (total >= wires[i]) good = true;
      total += wires[i];
    }
    if (good) cout << "YES" << endl;
    else cout << "NO" << endl;
    cin >> N;
  }
  return 0;
}
