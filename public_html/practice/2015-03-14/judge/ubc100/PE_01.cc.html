#include<iostream>

using namespace std;

int main (void) {
  int T; cin >> T;

  while (T--) {
    int N; cin >> N;
    int arr[N];
    for(int i=0; i<N; i++) {
      int t; cin >> t;
      arr[i] = t;
    }

    int res = arr[0];
    for( int i=1; i<N; i++) {
      res+=arr[i]-1;
    }
    cout << res << endl;
  }
}
