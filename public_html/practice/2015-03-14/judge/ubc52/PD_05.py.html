def generate(x): 
	if x== 2: return [2]
	elif x<2: return []
	s=range(3,x+1,2)
	yroot = x ** 0.5
	half=(x+1)/2-1
	i=0
	y=3
	while y <= yroot:
		if s[i]:
			j=(y*y-3)/2
			s[j]=0
			while j<half:
				s[j]=0
				j+=y
		i=i+1
		y=2*i+3
	return [2]+[x for x in s if x]

primes=generate(1000000)

while True:
	K,L=map(int,raw_input().split())
	index=0
	if K==0:
		break
	else:
		try:
			while True:
				prime=primes[index]
				if prime>=L:
					print "GOOD"
					break
				elif (K/prime)*prime==K:
					print "BAD",prime
					break
				index+=1
		except IndexError:
			print "GOOD"
