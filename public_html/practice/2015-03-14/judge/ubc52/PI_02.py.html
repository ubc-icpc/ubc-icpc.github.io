while True:
	K=int(raw_input())
	if K==0:
		break
	else:
		L=map(float,raw_input().split())
		L.sort()
		i=0
		tot=0
		found=False
		for i in range(K-1):
			tot+=L[i]
			if tot>=L[i+1]:
				found=True
			
		if found:
			print "YES"
		else:
			print "NO"
