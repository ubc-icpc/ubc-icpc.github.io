s = [True]*1000001
for i in xrange(2,1000001):
  if s[i]:
    for j in xrange(i*i,1000001,i):
      s[j] = False

while True:
  k, l = [int(i) for i in raw_input().split()]
  if k == l == 0:
    break
  ok = True
  p = 2
  while p < l:
    while p < l and not s[p]:
      p += 1
    if p < l and k % p == 0:
      ok = False
      print "BAD", p
      break
    p += 1
  if ok:
    print "GOOD"
