#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

const ld PI = acos(-1.0L);
int main() {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	while (t--) {
		int q; cin >> q;
		pt c(0, 0), d(1, 0);
		while (q--) {
			string s; ld x;
			cin >> s >> x;
			if (s == "fd") {
				c += x * d;
			} else if (s == "lt") {
				d *= polar(1.0L, x / 180.0 * PI);
			} else if (s == "bk") {
				c -= x * d;
			} else {
				d *= polar(1.0L, -x / 180.0 * PI);
			}
		}
		cout << (long long)(abs(c) + 0.5) << endl;
	}
	return 0;
}
