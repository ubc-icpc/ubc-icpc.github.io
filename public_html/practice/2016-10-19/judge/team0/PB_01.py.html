import sys
sys.setrecursionlimit(4000000)

n, m = [int(x) for x in raw_input().split()]
n = n/2 + 1

memo = {}
def solve(i, j):
	if i < j or i >= n or j >= n: 
		return 0
	if i == n-1 and j == n-1:
		return 1
	if (i, j) in memo:
		return memo[i, j]
	
	memo[i, j] = solve(i+1, j) + solve(i, j+1)
	return memo[i, j]

i, j = 0, 0
ans = []
while i != n-1 or j != n-1:
	if solve(i+1, j) >= m:
		ans.append('(')
		i += 1
	else:
		m -= solve(i+1, j)
		ans.append(')')
		j += 1
		
print "".join(ans)
