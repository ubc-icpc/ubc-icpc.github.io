#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

char b[205][205];
int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};
// N, E, S, W
int main() {
	ios::sync_with_stdio(0);
	int r, c; cin >> r >> c;
	for (int i = 0; i < r; i++) {
		for (int j = 0; j < c; j++) {
			cin >> b[i][j];
		}
	}
	
	map<char, int> m({{'N', 0}, {'E', 1}, {'S', 2}, {'W', 3}});
	int i = 0, j = 0, d = 0;
	while (b[i][j] != 'T' && d <= r * c) {
		//cerr << i << " " << j << endl;
		int z = m[b[i][j]];
		i += dx[z];
		j += dy[z];
		d++;
		if (i >= r || j >= c || i < 0 || j < 0) {
			cout << "Out" << endl;
			return 0;
		}
	}
	if (d >= r*c) cout << "Lost" << endl;
	else cout << d << endl;
	return 0;
}
