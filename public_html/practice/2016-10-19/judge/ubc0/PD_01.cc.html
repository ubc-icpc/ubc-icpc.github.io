typedef vector<int> VI;
typedef vector<VI> VVI;
typedef bool T;
typedef vector<T> VT;
typedef vector<VT> VVT;


void search(vector<string> maze, int cr, int cc, int depth){
    VT temp(maze[0].size(), false);
    VVT visited(maze.size(), temp);
    
    while(true){
        visited[cr][cc] = true;
        if(maze[cr][cc] == 'T'){
            cout << depth << endl;
            return;
        }
        else if(maze[cr][cc] == 'N'){
            if(cr == 0){
                cout << "Out" << endl;
                return;
            }
            if(visited[cr - 1][cc]){
                cout << "Lost" << endl;
                return;
            }
            cr--;
        }
        else if(maze[cr][cc] == 'S'){
            if(cr == maze.size() - 1){
                cout << "Out" << endl;
                return;
            }
            if(visited[cr + 1][cc]){
                cout << "Lost" << endl;
                return;
            }
            cr++;
        }
        else if(maze[cr][cc] == 'E'){
            if(cc == maze[0].size() - 1){
                cout << "Out" << endl;
                return;
            }
            if(visited[cr][cc + 1]){
                cout << "Lost" << endl;
                return;
            }
            cc++;
        }
        else if(maze[cr][cc] == 'W'){
            if(cc == 0){
                cout << "Out" << endl;
                return;
            }
            if(visited[cr][cc - 1]){
                cout << "Lost" << endl;
                return;
            }
            cc--;
        }
        depth++;
    }
    
    return;
}

int main(){
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    int r, c;
    cin >> r >> c;
    vector<string> maze(r);
    for(int i = 0; i < r; i++)
        cin >> maze[i];
    
    search(maze, 0, 0, 0);
    
    return 0;
}
