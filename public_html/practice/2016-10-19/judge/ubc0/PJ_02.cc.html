int main(){
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    
    int t;
    cin >> t;
    
    for(int test = 0; test < t; test++){
        int n;
        cin >> n;
        double x = 0, y = 0;
        int degree = 0;
        
        for(int i = 0; i < n; i++){
            string c;
            cin >> c;
            int value;
            cin >> value;
            
            if(c == "lt")
                degree = (degree + value) % 360;
            else if(c == "rt")
                degree = (degree + 360 - value) % 360;
            else if(c == "fd"){
                x += cos((double) degree / 180.0 * PI) * value;
                y += sin((double) degree / 180.0 * PI) * value;
            }
            else if(c == "bk"){
                x -= cos((double) degree / 180.0 * PI) * value;
                y -= sin((double) degree / 180.0 * PI) * value;
            }
        }
        
        cout << (int) round(sqrt(x * x + y * y)) << endl;
    }

}
