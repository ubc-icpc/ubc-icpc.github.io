#include <cstdio>
#include <cstring>
#include <cmath>

int main() {
    int T; scanf("%d", &T);
    while (T--) {
        double x = 0, y = 0;
        int theta = 0;
        int C; scanf("%d", &C);
        while (C--) {
            char cmd[10];
            int val;
            scanf("%s %d", cmd, &val);
            if (strcmp(cmd, "bk") == 0) {
                strcpy(cmd, "fd");
                val = -val;
            } else if (strcmp(cmd, "rt") == 0) {
                strcpy(cmd, "lt");
                val = -val;
            }
            if (strcmp(cmd, "lt") == 0) {
                theta = (theta + val) % 360;
            }
            if (strcmp(cmd, "fd") == 0) {
                x += val*cos(theta / 180.0 * M_PI);
                y += val*sin(theta / 180.0 * M_PI);
            }
            /* printf("%.2f %.2f %d\n", x, y, theta); */
        }
        printf("%.0f\n", round(sqrt(x*x+y*y)));
    }
    return 0;
}
