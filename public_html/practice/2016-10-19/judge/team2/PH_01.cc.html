#include<bits/stdc++.h>

#define ss(a)   scanf("%d", &a)
#define pb(a)   push_back(a)

using namespace std;
typedef long long ll;

int n, T, a, b;

map<string, int> ma;

int par[100001];

int parent(int a) { return (par[a]<0 ? a:(par[a] = parent(par[a]))); }

int merge(int a, int b){
    int x = parent(a);
    int y = parent(b);
    if (x==y) return -par[x];
    if (x>y) swap(x,y);
    par[x] +=par[y];
    par[y] = x;
    return -par[x];
}

int main(){
    ios_base::sync_with_stdio(false);cin.tie(0);
    cin >> T;
    while(T--){
        cin >> n;
        ma.clear();
        memset(par,-1,sizeof par);
        int c = 0;
        for(int i=0;i<n;i++){
            string sa, sb;
            cin >> sa >> sb;
            if(!ma.count(sa)) ma[sa]=c++;
            if(!ma.count(sb)) ma[sb]=c++;
            cout << merge(ma[sa],ma[sb]) << '\n';
        }
    }
}
