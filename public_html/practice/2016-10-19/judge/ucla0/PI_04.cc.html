#define ll long long
#define N 200005
#define INF 1000000000000LL
#define EPS (1e-7)
#define PI (3.14159265358979323846)
#define ld long double
#define MOD (int)(1e9+7)

vector<ll> g[N], gb[N], v;
ll deg[N], degO[N];
bool b[N];
ll cnt;

void dfs(ll cur) {
    b[cur] = true;
    for (ll i = 0; i < g[cur].size(); i++) {
        ll next = g[cur][i];
        if (b[next])
            continue;
        dfs(next);
    }
    for (ll i = 0; i < gb[cur].size(); i++) {
        ll next = gb[cur][i];
        if (b[next])
            continue;
        dfs(next);
    }
}

void solve() {
    for (ll i = 0; i < N; i++) {
        g[i].clear();
        gb[i].clear();
        deg[i] = degO[i] = 0;
        b[i] = false;
        v.clear();
    }
    cnt = 0;
    ll n, m;
    cin >> n >> m;
    
    for (ll i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        a--; b--;
        g[a].push_back(b);
        gb[b].push_back(a);
        deg[b]++;
    }
    
    for (ll i = 0; i < n; i++) {
        if (deg[i] == 0)
            v.push_back(i);
    }
    
    for (ll i = 0; i < v.size(); i++) {
        cnt++;
        dfs(v[i]);
    }
    for (ll i = 0; i < n; i++) {
        if (!b[i]) {
            cnt++;
            dfs(i);
        }
    }
    cout << cnt << endl;
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    ll z;
    cin >> z;
    for (ll i = 0; i < z; i++)
        solve();
}
