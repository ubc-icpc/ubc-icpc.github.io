#define ll long long

#define EPS (1e-9)

#define ld long double
#define MOD (int)(1e9+7)

ll dp[N][N], res[N];

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    ll lim = (ll)(1e18+3);
    dp[1][1] = 1;
    for (ll i = 2; i <= 2000; i++) {
        for (ll j = 1; j < i; j++) {
            dp[i][j] = dp[i][j-1] + dp[i-1][j];
        }
        dp[i][i] = dp[i][i-1] + dp[i-1][i-1];
    }
    

    ll n, m;
    cin >> n >> m;
    ll cur = n/2;
    for (ll i = 0; i < n/2; i++) {
        for (ll j = 0; j <= n/2; j++) {
            if (dp[cur][j] >= m || dp[cur][j] < dp[cur][j-1]) {
                res[i] = n/2-(j-1);
                m -= dp[cur][j-1];
                break;
            }
        }
        cur--;
    }
//    for (ll i = 0; i < n/2; i++) {
//        cout << res[i] << " ";
//    }
    
    cur = 0;
    for (ll i = 0; i < n/2; i++) {
        while (cur < res[i]) {
            cur++;
            cout << '(';
        }
        cout << ')';
    }
    
    cout << endl;
    
}
