#define ll long long
#define N 205
#define INF 1000000000000LL
#define EPS (1e-9)
#define PI (3.14159265358979323846)
#define ld long double
#define MOD (int)(1e9+7)

char ch[N][N];
bool b[N][N];

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    ll n, m;
    cin >> n >> m;
    for (ll i = 0; i < n; i++) {
        for (ll j = 0; j < m; j++) {
            cin >> ch[i][j];
        }
    }
    ll r, c;
    r = c = 0;
    
    ll cnt = 0;
    while (ch[r][c] != 'T') {
        if (r < 0 || c < 0 || r >= n || c >= m) {
            cout << "Out" << endl;
            return 0;
        }
        if (b[r][c]) {
            cout << "Lost" << endl;
            return 0;
        }
        b[r][c] = true;
        switch (ch[r][c]) {
            case 'E':
                c++;
                break;
            case 'S':
                r++;
                break;
            case 'W':
                c--;
                break;
            case 'N':
                r--;
                break;
            case 'T':
                break;
        }
        cnt++;

    }
    cout << cnt << endl;
    
}
