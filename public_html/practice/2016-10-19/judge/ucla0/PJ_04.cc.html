void solve() {
    double x, y, ang;
    x = y = ang = 0;
    ll n;
    cin >> n;
    for (ll i = 0; i < n; i++) {
        string s;
        int mag;
        cin >> s >> mag;
        if (s == "fd") {
            x += mag*cos(ang);
            y += mag*sin(ang);
        }
        else if (s == "bk") {
            x -= mag*cos(ang);
            y -= mag*sin(ang);
        }
        else if (s == "lt") {
            ang += PI*mag/180;
        }
        else {
            ang -= PI*mag/180;
        }
        //cout << x << " " << y << endl;
    }
    cout << (int)(round(sqrt(x*x+y*y))) << endl;
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    ll z;
    cin >> z;
    for (ll i = 0; i < z; i++)
        solve();
}
