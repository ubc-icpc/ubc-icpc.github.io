#define ll long long
#define N 200005
#define INF 1000000000000LL
#define EPS (1e-9)
#define PI (3.14159265358979323846)
#define ld long double
#define MOD (int)(1e9+7)

ll size[N];
ll par[N];

ll up(ll u) {return par[u] == u? u : par[u] = up(par[u]);}

void solve() {
    for (ll i = 0; i < N; i++) {
        par[i] = i;
        size[i] = 1;
    }
    ll n;
    cin >> n;
    ll num = 0;
    map<string, ll> ma;
    for (ll i = 0; i < n; i++) {
        string s1, s2;
        cin >> s1 >> s2;
        if (ma.find(s1) == ma.end()) {
            ma[s1] = num++;
        }
        ll x, y;
        if (ma.find(s2) == ma.end()) {
            ma[s2] = num++;
        }
        if ((x = up(ma[s1])) == (y = up(ma[s2]))) {
            cout << size[x] << endl;
            continue;
        }
        if (size[y] < size[x])
            swap(x, y);
        
        size[x] += size[y];
        par[y] = x;
        cout << size[x] << endl;
    }
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    ll z;
    cin >> z;
    for (ll i = 0; i < z; i++)
        solve();
}
