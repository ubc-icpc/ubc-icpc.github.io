#include <bits/stdc++.h>

using namespace std;

int main() {
	int R, C; cin >> R >> C;
	vector<vector<char> > board;
	for (int i = 0; i < R; i++) {
		vector<char> row;
		for (int j = 0; j < C; j++) {
			char c; cin >> c;
			row.push_back(c);
		}
		board.push_back(row);
	}
	vector<vector<bool> > visited(R, vector<bool>(C, false));
	int x = 0; int y = 0;
	int steps = 0;
	while (true) {
		if (x < 0 || x >= R || y < 0 || y >= C) {
			cout << "Out" << endl;
			break;
		}
		if (visited[x][y]) {
			cout << "Lost" << endl;
			break;
		}
		if (board[x][y] == 'T') {
			cout << steps << endl;
			break;
		}
		visited[x][y] = true;
		switch (board[x][y]) {
			case 'E': y++; break;
			case 'W': y--; break;
			case 'N': x--; break;
			case 'S': x++; break;
		}
		steps++;
	}
}
