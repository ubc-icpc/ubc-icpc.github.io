import sys
import math
input = sys.stdin

t = int(input.readline())

def fd(n, c, a):
	c['y'] = c['y'] + math.sin(math.radians(a)) * n
	c['x'] = c['x'] + math.cos(math.radians(a)) * n
	return c

for ti in range(t):
	m = int(input.readline())

	a = 0.0
	c = {'x': 0.0, 'y': 0.0}

	for mi in range(m):
		[cmd, n] = input.readline().split()
		n = float(n)

		if (cmd == 'lt'):
			a = (a + n) % 360
		elif (cmd == 'rt'):
			a = (a - n) % 360
		elif (cmd == 'fd'):
			c = fd(n, c, a)
		elif (cmd == 'bk'):
			c = fd(n, c, a - 180)

	print(int(round(math.sqrt(c['x']**2 + c['y']**2))))
