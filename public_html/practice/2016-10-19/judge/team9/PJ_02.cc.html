using namespace std;
typedef complex<long double> point;
const long double PI = acos(-1.0L);

int main() {
	ios::sync_with_stdio(0); 
	
	int T, C;
	cin >> T;

	for (int i = 0; i < T; i++) {
		cin >> C;
		string command;
		long double argument;

		tuple<double, double, int> res;
		point c(0, 0), d(1, 0);

		// for each command
		for (int j = 0; j < C; j++) {
			cin >> command >> argument;
			
			if (command == "lt") {
				d *= polar(1.0L, argument / 180.0 * PI);
			}
			else if (command == "rt") {
				d *= polar(1.0L, -argument / 180.0 * PI);
			}
			// the following lines kinda make sense. You travel x points in d direction
			else if (command == "fd") {
				c += argument * d;
			}
			else {// bk
				c -= argument * d; 
			}


		}

		cout << (long long)(abs(c) + 0.5) << endl;
	}

	return 0;
}
