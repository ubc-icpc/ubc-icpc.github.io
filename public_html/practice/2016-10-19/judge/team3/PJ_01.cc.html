#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

#define PI 3.1415926535

int main() {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	while(t--) {
		int n; cin >> n;
		ld x = 0;
		ld y = 0;
		ld angle = 0;
		while(n--) {
			string cmd; ld val;
			cin >> cmd >> val;
			if (cmd == "fd") {
				x += val*cos(angle*PI/180);
				y += val*sin(angle*PI/180);
			} else if (cmd == "bk") {
				x -= val*cos(angle*PI/180);
				y -= val*sin(angle*PI/180);
			} else if (cmd == "lt") {
				angle += val;
			} else {
				angle -= val;
			}
		}
		ld ans = sqrt(x*x + y*y);
		cout << (int)round(ans) << endl;
	}
	return 0;
}
