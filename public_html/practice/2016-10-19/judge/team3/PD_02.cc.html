#include <iostream>

using namespace std;

char grid[201][201];
bool visited[201][201];

int main() {
  int R, C; cin >> R >> C;
  for (int i = 0; i < R; ++i) {
    string s; cin >> s;
    for (int j = 0; j < C; ++j) {
      grid[i][j] = s[j];
    }
  }
  int steps = 0;
  int r_curr = 0, c_curr = 0;
  while (grid[r_curr][c_curr] != 'T') {
    visited[r_curr][c_curr] = true;
    if (grid[r_curr][c_curr] == 'N') {
      r_curr--;
    }
    else if (grid[r_curr][c_curr] == 'S') {
      r_curr++;
    }
    else if (grid[r_curr][c_curr] == 'E') {
      c_curr++;
    }
    else {
      c_curr--;
    }
    if (r_curr < 0 || c_curr < 0 || r_curr >= R || c_curr >= C) {
      cout << "Out" << endl;
      return 0;
    }
    if (visited[r_curr][c_curr]) {
      cout << "Lost" << endl;
      return 0;
    }
    steps++;
  }
  cout << steps << endl;
  return 0;
}
