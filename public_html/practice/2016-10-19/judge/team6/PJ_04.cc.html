#include <iostream>
#include <cmath>
#define PI 3.14159265358979323
using namespace std;

int main() {
	int t;cin>>t;
	while(t--){
		int n;cin>>n;
		double x = 0, y = 0;
		int deg = 0;
		while(n--){
			string cmd;int v;cin>>cmd>>v;
			switch(cmd[0]) {
				case 'f':
					x += cos(deg * PI / 180.0) * v;
					y += sin(deg * PI / 180.0) * v;
					break;
				case 'b':
					x -= cos(deg * PI / 180.0) * v;
					y -= sin(deg * PI / 180.0) * v;
					break;
				case 'l':
					deg += (v + 360) % 360;
					deg = (deg + 360) % 360;
					break;
				case 'r':
					deg -= (v + 360) % 360;
					deg = (deg + 360) % 360;
					break;
			}
		}

		cout << (int)(sqrt(x*x+y*y) + 0.5) << endl;
	}
}
