#include <bits/stdtr1c++.h>

using namespace std;

int a[1005];
int main() {
    int t; cin >> t;
    while (t--) {
        double x; cin >> x;
        
        int amt = 100*x + 0.5;
        int de = 100;
        int g = __gcd(amt, de);
        amt /= g;
        de /= g;
        
        int n; cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> a[i];
            while (a[i]%2 == 0) a[i] /= 2;
        }
        
        if (__builtin_popcount(de) != 1) {
            cout << "no" << endl;
        } else {
            
            
            int f = a[0];
            for (int i = 0; i < n; i++) f = __gcd(f, a[i]);
            
            if (amt % f == 0) cout << "yes" << endl;
            else cout << "no" << endl;
        }
    }
	return 0;
}
