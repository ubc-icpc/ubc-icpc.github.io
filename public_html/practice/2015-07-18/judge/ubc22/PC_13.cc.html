#include <bits/stdtr1c++.h>
using namespace std;

int T, V, N, M[1005];
double D;

int main() {
	cin >> T;
	while (T--) {
		cin >> D;
		V = D = D * 100 + 0.5;
		V /= __gcd(V, 100);
		cin >> N;
		for (int i = 0; i < N; ++i) {
            cin >> M[i];
            while (M[i] % 2 == 0) M[i] /= 2;
            M[0] = __gcd(M[0], M[i]);
		}
		if ((int)D % 25 == 0 && V % M[0] == 0) cout << "yes" << endl;
		else cout << "no" << endl;
	}
	return 0;
}
