#include <bits/stdtr1c++.h>
using namespace std;
int main() {
	int t; cin >> t;
	while (t--) {
		int b[1000];
		float x; cin >> x;
		int amt = x*100;
		int g = amt/__gcd(amt,100);
		int n; cin >> n;
		for (int i=0; i<n; i++) {
			cin >> b[i];
			while (b[i] % 2 == 0) b[i] /=2;
			b[0] = __gcd(b[0],b[i]);
		}
		if (amt % 25 == 0 && g % b[0] == 0) {
			cout << "yes" << endl;
		} else {
			cout << "no" << endl;
		}
	}
	return 0;
}
