#include <bits/stdtr1c++.h>
using namespace std;

typedef long long ll;

ll memo[51][6];
int numlen[11];

ll dp(int l, int n) {
	if (n<0) return 0;
	if (memo[l][n] != -1) return memo[l][n];
	memo[l][n] = 0;
	for (int i=3; i<=min(10,l); i++) {
		if (numlen[i] != 0) {
			memo[l][n] += dp(l-i,n-1)*numlen[i];
		}
	}
	return memo[l][n];
}

int main() {
	int t; cin >> t;
	while (t--) {
		int m, n, q; cin >> m >> n >> q;
		memset(numlen,0,sizeof numlen);
		while (m--) {
			string s; cin >> s;
			numlen[s.length()] = numlen[s.length()] +1;
		}
		while (q--) {
			memset(memo, -1, sizeof memo);
			memo[0][0]=1;
			int pwl; cin >> pwl;
			cout << dp(pwl,n) << endl;
		}
	}
	return 0;
}
