int possible (char word[]){
	int numB = 0, numW = 0, i;
	if(strlen(word)%2 == 1){
		return 0;
	}
	for (i = 0; i< strlen(word); i++){
		if(word[i] == 'W'){
			numW++;
		}
		if (word[i] == 'B'){
			numB++;
		}
	}
	//printf("%d %d\n", numB, numW);	
	if(numB != numW){
		return 0;
	}

}
