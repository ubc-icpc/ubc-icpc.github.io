#include <bits/stdc++.h>
using namespace std;

int main() {
	string s;
	cin >> s;
	int offset = 0;
	for (int i = 0; i < s.size(); i++) {
		if (s[i] == 'W')
			offset++;
		else
			offset--;
	}
	if (offset == 0)
		cout << 1 << endl;
	else
		cout << 0 << endl;
	return 0;
}
