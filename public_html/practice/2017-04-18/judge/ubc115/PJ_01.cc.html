#include <bits/stdc++.h>
using namespace std;

int main() {
  string s;
  cin >> s;
  stack<int> k;
  for (int i = 0; i < s.size(); i++) {
    if (k.empty() || k.top() == s[i]) {
      k.push(s[i]);
    } else {
      k.pop();
    }
  }
  if (k.size() == 0) {
    cout << 1;
  } else {
    cout << 0;
  }
  return 0;
}
