#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

int main() {
	int n; cin >> n;
	string ans;
	if (n <= 25) {
		ans += 'a';
		ans += 'a'+n;
	} else {
		ans += 'a';
		while(n > 0) {
			if (n >= 25) {
				if (ans[ans.length()-1] != 'a') ans += 'a';
				else ans += 'z';
				n -= 25;
			} else {
				if (ans[ans.length()-1] == 'z') ans += 'z'-n;
				else ans += 'a'+n;
				break;
			}
		}
		if (ans[ans.length()-2] == 'a') {
			while(ans[ans.length()-1] < 'y') {
				ans[1] -= 1;
				ans[ans.length()-1] += 2;
			}
		} else {
			while(ans[ans.length()-1] > 'b') {
				ans[1] -= 1;
				ans[ans.length()-1] -= 2;
			}
		}
		
	}
	cout << ans << endl;
	return 0;
}
