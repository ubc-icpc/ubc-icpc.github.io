#include <bits/stdc++.h>
using namespace std;

int main() {
	int k;
	while (cin >> k) {
		cout << "a";
		if (k < 26) {
			cout << (char)('a'+k) << "\n";
			continue;
		}
		int r = k%25;
		if (!r) r += 25;
		int l = (k+49)/25;
		if (l % 2) {
			cout << (char)('a'+13+r/2);
			for (int i = 2; i+2 < l; i += 2)
				cout << "az";
			cout << (char)('a'+((r+1)%2)) << "\n";
		}
		else {
			cout << (char)('a'+13+r/2) << "a";
			for (int i = 3; i+2 < l; i += 2)
				cout << "za";
			cout << (char)('y'+(r%2)) << "\n";
		}
	}
	return 0;
}
