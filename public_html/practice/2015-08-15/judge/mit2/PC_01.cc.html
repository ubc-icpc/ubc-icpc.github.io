#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

const int MAX = 555;

int T, ncells, jump;
char cells[MAX];

int main() {
    scanf("%d", &T);
    for (int t = 1; t <= T; ++t) {
        scanf("%d %d", &ncells, &jump);
        scanf("%s", cells);
        int cur = 0;
        int answer = 0;
        while (cur != ncells - 1) {
            answer++;
            int nextcur = cur;
            bool jumped = false;
            for (int j = 1; j <= jump + 1 && cur + j < ncells; ++j) {
                if (cells[cur + j] == '.') {
                    nextcur = cur + j;
                    jumped = true;
                }
            }
            if (!jumped) {
                answer = 0;
                break;
            }
            cur = nextcur;
        }

        printf("Day #%d\n", t);
        printf("%d %d\n%s\n", ncells, jump, cells);
        printf("%d\n\n", answer);
    }

    return 0;
}
