#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <cstring>

using namespace std;

const int MAXN = 22;

int T;
char input[MAXN];

int main() {
    scanf("%d", &T);

    for (int t = 0; t < T; ++t) {
        scanf("%s", input);
        int n = 0;
        for (int i = 0; i < strlen(input); ++i) {
            if (input[i] == 'a' || input[i] == 'e' || input[i] == 'i' || input[i] == 'o' || input[i] == 'u') {
                n++;
            }
        }
        printf("%s\n%d\n", input, n > strlen(input) / 2 ? 1 : 0);
    }

    return 0;
}
