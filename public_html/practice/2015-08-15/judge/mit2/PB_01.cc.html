#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

int T, games, points;

int main() {
    scanf("%d", &T);
    for (int t = 1; t <= T; ++t) {
        scanf("%d %d", &games, &points);

        printf("Team #%d\n", t);
        printf("Games: %d\n", games);
        printf("Points: %d\n", points);
        printf("Possible records:\n");
        for (int wins = points / 3; wins >= 0; --wins) {
            int ties = points - wins * 3;
            if (ties + wins <= games) {
                printf("%d-%d-%d\n", wins, ties, games - wins - ties);
            }
        }
        printf("\n");
    }

    return 0;
}
