#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

const int MAX = 222;

int T, n, drive[MAX], eat[MAX];

int main() {
    scanf("%d", &T);

    for (int t = 1; t <= T; ++t) {
        scanf("%d", &n);
        for (int i = 0; i < 4 * n; i += 4) {
            scanf("%d", &drive[i]);
            drive[i + 1] = drive[i + 2] = drive[i + 3] = drive[i];
        }
        for (int i = 0; i < 4 * n; ++i) {
            scanf("%d", &eat[i]);
        }

        sort(drive, drive + 4 * n);
        sort(eat, eat + 4 * n);

        int answer = 0;
        for (int i = 0; i < 4 * n; ++i) {
            answer = max(answer, drive[i] + eat[4 * n - 1 - i]);
        }

        printf("Trip #%d: %d\n", t, answer);
    }

    return 0;
}
