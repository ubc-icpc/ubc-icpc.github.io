#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <algorithm>
#include <cmath>
#include <map>
#include <sstream>
#include <complex>
#include <vector>
#include <set>

using namespace std;

typedef long long ll;

int main()
{
  int n; cin >> n;
  
  for (int i=0; i<n; i++) {
    int c, d; cin >> c >> d;
    string s; cin >> s;
    cout << "Day #" << i+1 << endl << c << " " << d << endl << s << endl;
    int p=0;
    int jumps = 0;
    bool cant = false;
    while (p<c-1) {
      int j = d+1;
      while (j>0) {
        if (s[p+j] == '.') {
          p += j;
          jumps++;
          break;
        }
        j--;
      }
      if (j==0) {
        cant = true;
        cout << 0 << endl << endl;
        break;
      }
    }
    if (!cant)
      cout << jumps << endl << endl;
  }
}
