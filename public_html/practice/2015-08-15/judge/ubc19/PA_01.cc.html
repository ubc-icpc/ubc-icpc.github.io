using namespace std;

typedef long long ll;

int main()
{
  string vowels = "aeiou";
  int n; cin >> n;
  
  for (int i=0; i<n; i++) {
    string s; cin >> s;
    cout << s << endl;
    int v = 0;
    for (int j=0; j<s.size(); j++) {
      if (vowels.find(s[j]) != std::string::npos) v++;
      else v--;
    }
    bool morevowls = v >0;
    cout << morevowls << endl;
  }
}
