#include <iostream>
#include <stdio.h>
#include <string.h>

#include <cmath>

#include <set>

using namespace std;

typedef long long ll;

int main()
{
  int n; cin >> n;
  
  for (int i=0; i<n; i++) {
    int c; cin >> c;
    int ctime [c]; int ttime [4*c];
    for (int j=0; j<c; j++)
      cin >> ctime[j];
    for (int j=0; j<4*c; j++)
      cin >> ttime[j];
    sort(ctime, ctime+c);
    sort(ttime, ttime+4*c);
    
    int max=0;
    for (int j=0; j<c; j++) {
      max = (max > ctime[c-j-1] + ttime[4*j+3]) ? max : ctime[c-j-1] + ttime[4*j+3];
    }
    cout << "Trip #" << i+1 << ": " << max << endl;
  }



}
