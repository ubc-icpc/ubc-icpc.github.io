#include <bits/stdtr1c++.h>

using namespace std;


//dp[n][k] := deciding item n with k chosen
int memo[505][505], val[505], n;
int dp(int i, int k) {
    if (i == n) return 0;
    if (i != 0 && k+1 > (i+1)/2) return dp(i+1, k);
    if (memo[i][k] != -1) return memo[i][k];
    return memo[i][k] = max(val[i] + dp(i+1, k+1), dp(i+1, k));
}

int main() {
    int t; cin >> t;
    for (int c = 1; c <= t; c++) {
        cout << "Spree #" << c << ": ";
        cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> val[i];
        }
        
        memset(memo, -1, sizeof memo);
        cout << dp(0,0) << endl;
    }
	return 0;
}
