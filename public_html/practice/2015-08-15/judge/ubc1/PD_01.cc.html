#include <bits/stdtr1c++.h>

using namespace std;

int ct[205], et[205];
int main() {
    int t; cin >> t;
    for (int c = 1; c <= t; c++) {
        cout << "Trip #" << c << ": ";
        
        int nc; cin >> nc;
        for (int i = 0; i < nc; i++) {
            cin >> ct[i];
        }
        sort(ct, ct+nc);
        
        for (int i = 0; i < 4*nc; i++) {
            cin >> et[i];
        }
        sort(et, et+4*nc);
        reverse(et, et+4*nc);
        
        int ans = 0;
        for (int i = 0; i < nc; i++) {
            ans = max(ans, ct[i] + et[4*i]);
        }
        cout << ans << endl;
    }
	return 0;
}
