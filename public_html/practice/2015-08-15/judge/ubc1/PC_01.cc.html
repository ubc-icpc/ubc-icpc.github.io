#include <bits/stdtr1c++.h>

using namespace std;

bool bad[105];
int dist[105];
int main() {
    int t; cin >> t;
    for (int c = 1; c <= t; c++) {
        cout << "Day #" << c << endl;
        
        memset(bad, 0, sizeof bad);
        int nc, d; cin >> nc >> d;
        string s; cin >> s;
        for (int i = 0; i < nc; i++) {
            bad[i] = (s[i] == 'X');
        }
        
        memset(dist, -1, sizeof dist);
        queue<int> q({0});
        dist[0] = 0;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int del = 1; del <= d+1; del++) {
                int nu = u + del;
                if (dist[nu] != -1 || bad[nu]) continue;
                dist[nu] = dist[u] + 1;
                q.push(nu);
            }
        }
        
        cout << nc << " " << d << endl;
        cout << s << endl;
        cout << max(0, dist[nc-1]) << endl;
        cout << endl;
    }
	return 0;
}
