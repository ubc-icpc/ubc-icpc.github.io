#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;

const int maxn = 2505;
ld lfsum[2505], eps = 1e-6;
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    //log dem factorials and compare
    for (int i = 2; i < maxn; i++) {
        lfsum[i] = lfsum[i-1] + log(i);
    }
    
    int t; cin >> t;
    for (int c = 1; c <= t; c++) {
        cout << "Case #" << c << ": ";
        int n[3]; cin >> n[0] >> n[1] >> n[2];
        ld v[3] = {0, 0, 0};
        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < n[j]; k++) {
                int x; cin >> x;
                v[j] += lfsum[x];
            }
        }
        
        int ans = 0;
        for (int j = 0; j < 3; j++) {
            if (v[j] > v[ans]) {
                ans = j;
            }
        }
        
        //check tie
        bool tie = false;
        for (int j = 0; j < 3; j++) {
            if (ans != j && abs(v[j] - v[ans]) < eps) {
                tie = true;
            }
        }
        
        if (tie) cout << "TIE" << endl;
        else cout << "ABC"[ans] << endl;
    }
	return 0;
}
