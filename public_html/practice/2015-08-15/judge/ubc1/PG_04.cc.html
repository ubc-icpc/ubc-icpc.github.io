#include <bits/stdtr1c++.h>

using namespace std;

//every turn one surface gets eaten and one surface is stained
//with the exception of the first turn if there are no dirty two sided dishes
//and an even number of 1sided dirty dishes

int main() {
    int t; cin >> t;
    for (int c = 1; c <= t; c++) {
        cout << "Scenario #" << c << ": ";
        int C, S, D; cin >> C >> S >> D;
        if (S >= 2) D = 0;
        
        int ans = 0;
        C += S/2;
        S %= 2;
        if (S == 0) {
            if (D == 0) ans = (C > 0 ? C+1 : 0);
            else ans = C;
        } else if (S == 1) {
            ans = C + 1;
        }
        cout << ans << endl;
    }
	return 0;
}
