#include <bits/stdtr1c++.h>

using namespace std;

set<char> vowels({'a', 'e', 'i', 'o', 'u'});
int main() {
    int n; cin >> n;
    for (int i = 0; i < n; i++) {
        string s; cin >> s;
        int cnt = 0;
        for (char c : s) {
            if (vowels.count(c)) cnt++;
            else cnt--;
        }
        
        cout << s << endl;
        if (cnt > 0) cout << 1 << endl;
        else cout << 0 << endl;
    }
	return 0;
}
