#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int n; cin >> n;
    for (int i = 1; i <= n; i++) {
        int g, p; cin >> g >> p;
        cout << "Team #" << i << endl;
        cout << "Games: " << g << endl;
        cout << "Points: " << p << endl;
        cout << "Possible records:" << endl;
        for (int w = g; w >= 0; w--) {
            int pw = 3 * w, pt = p - pw;
            if (pt >= 0 && pt + w <= g) {
                cout << w << "-" << pt << "-" << g-w-pt << endl;
            }
        }
        cout << endl;
    }
	return 0;
}
