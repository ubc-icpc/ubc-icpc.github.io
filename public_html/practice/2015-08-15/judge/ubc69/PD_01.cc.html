#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <stdlib.h> 
#define INF 1000000

using namespace std;

int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		int c, ans=0;
		cin>>c;
		int d[c], t[4*c];
		for(int j=0;j<c;j++){
			cin>>d[j];
		}
		for(int j=0;j<4*c;j++){
			cin>>t[j];
		}
		qsort(d,c,sizeof(int),compare);
		qsort(t,4*c,sizeof(int),compare);
		
		for(int j=0;j<c;j++){
			if(d[j]+t[4*c-4*j-1]>ans)
				ans=d[j]+t[4*c-4*j-1];
		}
		
		cout<<"Trip #"<<i<<": "<<ans<<endl;
	}
    return 0;
}
