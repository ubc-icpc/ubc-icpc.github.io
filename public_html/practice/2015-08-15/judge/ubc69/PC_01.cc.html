#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <algorithm>
#define INF 1000000

using namespace std;

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		int c,d; string s;
		cin>>c>>d>>s;
		cout<<"Day #"<<i<<endl;
		cout<<c<<' '<<d<<endl;
		cout<<s<<endl;
		
		int count=0, pos=0;
		while(pos<c-1){
			int next = pos + d + 1;
			while(next>c||s[next]=='X') next--;
			if(next<=pos){
				count=0;
				break;
			}
			count++;
			pos=next;
		}
		cout<<count<<endl;
		cout<<endl;
	}
    return 0;
}
