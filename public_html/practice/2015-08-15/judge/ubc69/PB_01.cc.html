#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <algorithm>
#define INF 1000000

using namespace std;

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		int g, p;
		cin>>g>>p;
		cout<<"Team #"<<i<<endl;
		cout<<"Games: "<<g<<endl;
		cout<<"Points: "<<p<<endl;
		cout<<"Possible records: "<<endl;
		
		for(int win=p/3;win>=0;win--){
			int tie = p-3*win;
			int lose = g-win-tie;
			if(tie>=0&&lose>=0)
				cout<<win<<'-'<<tie<<'-'<<lose<<endl;
		}
		cout<<endl;
	}
    return 0;
}
