#include <bits/stdtr1c++.h>
using namespace std;
int main() {
	int n; cin >> n;
	for (int t=1; t<=n; t++) {
		int c; cin >> c;
		int d[4*c];
		int e[4*c];
		for (int i=0; i<4*c; i+=4) {
			int d1; cin >> d1;
			d[i] = d1;
			d[i+1] = d1;
			d[i+2] = d1;
			d[i+3] = d1;
		}
		for (int i=0; i<4*c; i++) {
			cin >> e[i];
		}

		sort(d,d+4*c);
		sort(e,e+4*c);

		int ans = 0;
		for (int i=0; i<4*c; i++) {
			int time = d[i] + e[4*c-i-1];
			ans = max(time,ans);
		}

		cout << "Trip #" << t << ": " << ans << endl;
	}
}
