#include <bits/stdtr1c++.h>
using namespace std;

long memo[501][501], items[501];
int s;
long dp(int i, int j) {
	if (i==s) return 0;
	if (memo[i][j] != -1) return memo[i][j];
	if (i!=0 && j+1 > (i+1)/2) return dp(i+1,j);
	memo[i][j] = max(items[i]+dp(i+1,j+1),dp(i+1,j));
	
	return memo[i][j];
}

int main() {
	int n; cin >> n;
	for (int d=1; d<=n; d++) {
		cin >> s;
		for (int i=0; i<s; i++) {
			cin >> items[i];
		}
		memset(memo,-1,sizeof memo);
		cout << "Spree #" << d << ": " << dp(0,0) << endl;
	}
	return 0;
}
