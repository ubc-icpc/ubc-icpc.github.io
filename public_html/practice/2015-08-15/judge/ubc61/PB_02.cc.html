#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int t; cin >> t;
	for (int i=1; i<=t; i++) {
		int g; cin >> g;
		int p; cin >> p;

		cout << "Team # " << i << endl;
		cout << "Games: " << g << endl;
		cout << "Points: " << p << endl;
		cout << "Possible records:" << endl;

		int maxw = p/3;
		for (int w=maxw; w>=0; w--) {
			int t = (p-(w*3));
			if ((w+t) <= g) 
				cout << w << '-' << t << '-' << (g-w-t) << endl;
		}
		cout << endl;
	}
}
