#include <bits/stdtr1c++.h>
using namespace std;
int main() {
	int t; cin >> t;
	for (int T=1; T<=t; T++) {
		cout << "Scenario #" << T << ": ";

		int c,s,d; cin >> c >> s >> d;
		int ans=0;
		
		if (s%2) {
			ans = c+1+s/2;
		} else {
			if (d && s<2)
				ans = c+s/2;
			else if (c+s/2)
				ans = c+1+s/2;
		}
		cout << ans << endl;
	}
	return 0;
}
