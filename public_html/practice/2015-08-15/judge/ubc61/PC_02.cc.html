#include <bits/stdtr1c++.h>
using namespace std;
int main() {
	int t; cin >> t;
	for (int d=1; d<=t; d++) {
		int c,j; cin >> c >> j;
		string s; cin >> s;

		int curr = 0;
		int ans = 0;
		
		while (curr < c-1) {
			int next = curr+j+1;
			while (next>c || s[next]=='X') next--;
			if (next <= curr) {
				ans=0;
				break;
			}
			ans++;
			curr = next;
		}
		

		cout << "Day #" << d << endl;
		cout << c << " " << j << endl;
		cout << s << endl;
		cout << ans << endl;
		cout << endl;
	}
}
