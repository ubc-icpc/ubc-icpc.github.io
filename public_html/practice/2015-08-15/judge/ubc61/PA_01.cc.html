#include <bits/stdtr1c++.h>
using namespace std;
int main() {
	int t; cin >> t;
	while (t--) {
		string s; cin >> s;
		int cons = 0;
		int v = 0;
		for (int i=0; i<s.length(); i++) {
			char c = s[i];
			if (c=='a' || c=='e' || c=='i' ||
				c=='o' || c=='u') {
				v++;
			} else {
				cons++;
			}
		}
		cout << s << endl;
		cout << (v>cons) << endl;
	}
}
