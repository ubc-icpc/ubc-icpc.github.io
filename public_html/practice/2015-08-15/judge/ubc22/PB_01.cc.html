#include <bits/stdtr1c++.h>
using namespace std;

int N, G, P;

int main () {
    cin >> N;
    for (int n = 1; n <= N; ++n) {
        cin >> G >> P;
        printf("Team #%d\n", n);
        printf("Games: %d\n", G);
        printf("Points: %d\n", P);
        printf("Possible records:\n");
        for (int i = G; i >= 0; --i)
            if (3 * i <= P && P - 3 * i + i <= G)
                printf("%d-%d-%d\n", i, P - 3 * i, G - (P - 3 * i + i));
        printf("\n");
    }
    return 0;
}
