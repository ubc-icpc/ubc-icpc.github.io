#include <bits/stdtr1c++.h>
using namespace std;

int N, V;
string S;

int main () {
    cin >> N;
    while (N--) {
        cin >> S;
        cout << S << endl;
        V = 0;
        for (char c : S)
            if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
                ++V;
        cout << (V > S.size() - V) << endl;
    }
    return 0;
}
