#include <bits/stdtr1c++.h>
using namespace std;

int N, C, R, D[55], T[205];

int main () {
    cin >> N;
    for (int n = 1; n <= N; ++n) {
        cin >> C;
        for (int i = 0; i < C; ++i) cin >> D[i];
        for (int i = 0; i < 4 * C; ++i) cin >> T[i];
        sort(D, D + C);
        sort(T, T + 4 * C);
        reverse(T, T + 4 * C);
        R = 0;
        for (int i = 0; i < 4 * C; ++i)
            R = max(R, D[i / 4] + T[i]);
        cout << "Trip #" << n << ": " << R << endl;
    }
    return 0;
}
