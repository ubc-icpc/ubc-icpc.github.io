#include <bits/stdtr1c++.h>
using namespace std;

int T, N, D;
string P;

int main () {
    cin >> T;
    for (int t = 1; t <= T; ++t) {
        cin >> N >> D >> P;
        cout << "Day #" << t << endl << N << " " << D << endl << P << endl;
        for (int i = 0, c = 1, b = 0;; ++c, b = 0) {
            for (int j = min(i + D + 1, N - 1); j > i; --j)
                if (P[j] == '.') {
                    b = i = j; break;
                }
            if (!b) { cout << 0 << endl; break; }
            if (i == N - 1) { cout << c << endl; break; }
        }
        cout << endl;
    }
    return 0;
}
