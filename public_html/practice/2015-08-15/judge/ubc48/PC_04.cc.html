#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	for (int i = 1; i <= n; ++i){
		int c, d, jump = 0; cin >> c >> d;
		string s; cin >> s;
		cout << "Day #" << i << endl << c << " " << d << endl << s << endl;
		s.append(d, '.');
		int pos = 0;
		while (true){
			//see if at destination
			if (pos >= c - 1){
				cout << jump << endl << endl;
				break;
			}
			//see if destination smaller than jump
			else if (pos + d + 1 < s.length()){
				//check max jump
				if (s[pos + d + 1] == 'X'){
					bool pass = false;
					//check max accessible jump
					for (int j = pos + d; j > pos; --j){
						if (s[j] == '.'){
							pos = j;
							++jump;
							pass = true;
						}
					}
					//no possible jump found, output 0
					if (pass == false){
						cout << 0 << endl << endl;
						break;
					}
				}
				//jump d
				else{
					++jump;
					pos = pos + d + 1;
				}
			}
		}
	}
	return 0;
}
