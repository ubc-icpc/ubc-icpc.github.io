#include<bits/stdtr1c++.h>
using namespace std;

bool big(int i, int j){ return(i > j); };

int main(){
	int n; cin >> n;
	for (int i = 1; i <= n; ++i){
		int c; cin >> c;
		vector < int > car(c), ppl(4*c), ans;

		

		for (int j = 0; j < c; ++j) cin >> car[j];
		for (int j = 0; j < 4 * c; ++j) cin >> ppl[j];
		sort(car.begin(), car.end()); sort(ppl.begin(), ppl.end(), big);

		for (int j = 0; j < c; ++j){
			ans.push_back(car[j] + ppl[j * 4]);
		}

		sort(ans.begin(), ans.end(), big);
		cout << "Trip #" << i << ": " << ans[0] << endl;
	}
	return 0;
}
