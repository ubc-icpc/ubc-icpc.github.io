#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	for (int i = 1; i <= n; ++i){
		int g, p; cin >> g >> p;
		cout << "Team #" << i << endl << "Games: " << g << endl << "Points: " << p << endl << "Possible records:" << endl;
		for (int w = p/3; w >= 0; --w){
			for (int t = 0; t <= p; ++t){
				if (w * 3 + t == p && w + t <= g){
					cout << w << "-" << t << "-" << (g - w - t) << endl;
				}
			}
		}
		cout << endl;
	}
	return 0;
}
