#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	char v[5] = { 'a', 'e', 'i', 'o', 'u' };
	int n; cin >> n;
	while (n--){
		string s; cin >> s;
		int nov = 0, noc = 0;
		for (int i = 0; i < s.length(); ++i){
			bool c = true;
			for (int j = 0; j < 5; ++j){
				if (s[i] == v[j]){
					++nov;
					c = false;
					break;
				}
			}
			if (c) ++noc;
		}
		cout << s << endl;
		if (nov > noc) cout << 1 << endl; else cout << 0 << endl;
	}
	return 0;
}
