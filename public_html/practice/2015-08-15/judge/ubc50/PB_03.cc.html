#include<iostream>

using namespace std;

int main() {
  int n; cin >> n;
  for (int t=1; t<=n; t++) {
    cout << "Team #" << t << endl;
    int g, p; cin >> g >> p;
    cout << "Games: " << g << endl << "Points: " << p << endl << "Possible records: " << endl;
    for (int i = g; i >= 0; i--) {
      for (int j = 0; j <= g; j++) {
        if (i+j > g)
          break;
        if (3*i+j == p)
          cout << i << "-" << j << "-" << g-i-j << "\n";
      }
    }
    cout << endl;
  }
}
