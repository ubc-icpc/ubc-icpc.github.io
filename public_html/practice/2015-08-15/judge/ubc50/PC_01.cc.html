#include<iostream>
#include<string>

using namespace std;

int main() {
  int n; cin >> n;
  for (int a=1; a<=n; a++) {
    cout << "Day #" << a << endl;
    int c,d; cin >> c >> d;
    cout << c << " " << d << endl;
    string s; cin >> s;
    cout << s << endl;
    int pos=0, jmp=0;
    bool flag = true;
    while (pos < c) {
      pos += d+1;
      jmp++;
      if (pos >= c-1) break;
      flag = false;
      pos++;
      for (int x=0; x<=d; x++) {
	pos--;
	if (s[pos]=='.') {
	  flag = true; break;
	}
      }
      if (flag == false) {
	cout << 0 << endl;
	break;
      }
    }
    if (flag)
      cout << jmp << endl;
  }
}
