#include<iostream>
#include<string>

using namespace std;

int main() {
  int n; cin >>n;
  for  (int i=0; i<n; i++) {
    string s; cin >> s;
    int vow=0, cons=0;
    for (int j=0; j<s.size(); j++) {
      if(s[j]=='a' || s[j]=='e' || s[j]=='i' || s[j]=='o' || s[j]=='u')
	vow++;
      else cons++;
    }
    cout << s << '\n';
    if (vow > cons)
      cout << 1 << '\n';
    else cout << 0 << '\n';
  }
}
