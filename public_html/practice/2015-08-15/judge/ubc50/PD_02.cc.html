#include<iostream>
#include<vector>
#include<queue>
#include<algorithm>
#include<functional>

using namespace std;

vector<int> cars;
vector<int> eating;

int main() {
  int n; cin >> n;
  for (int i=1; i<=n; i++) {
    cars.clear(); eating.clear();
    int c; cin >> c;
    for (int i=0; i<c; i++) {
      int d; cin >> d; cars.push_back(d);
    }
    for (int i=0; i<4*c; i++) {
      int t; cin >> t; eating.push_back(t);
    }
    sort(cars.begin(), cars.end());
    sort(eating.begin(), eating.end(), greater<int>());
    int ans = 0;
    for (int i=0; i<c; i++) {
      ans = max(ans, cars[i]+eating[4*i]);
    }
    cout << "Trip #" << i << ": "<< ans << endl;
  }
}
