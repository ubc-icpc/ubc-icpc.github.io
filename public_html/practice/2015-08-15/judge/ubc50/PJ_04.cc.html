#include<iostream>
#include<vector>
#include<string>
#include<cmath>

using namespace std;

typedef long long ll;
typedef long double ld;
ld memo[2503];
ld eps=1e-6;

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  for(int i=2; i<2503; i++) {
    memo[i] = memo[i-1]+log(i);
  }
  int n; cin>>n;
  for (int t=1; t<=n; t++) {
    int sz[3]; cin >> sz[0] >> sz[1] >> sz[2];
    ld prod[] = {0,0,0};
    for (int i=0; i<3; i++) {
      for (int j=0; j<sz[i]; j++) {
	int x; cin >> x; prod[i] += memo[x];
      }
    }
    bool tie = false;
    int idx = 0;
    ld tmp = 0;
    string res = "ABC";
    for (int i=0; i<3; i++) {
      if(prod[i]>prod[idx]) idx=i;
    }
    for (int i=0; i<3; i++) {
      if (i != idx && abs(prod[i]-prod[idx])<eps) {
	tie = true;
      }
    }
    cout << "Case #" << t << ": ";
    if (tie) cout << "TIE" << endl;
    else cout << res[idx] << endl;
  }
}
