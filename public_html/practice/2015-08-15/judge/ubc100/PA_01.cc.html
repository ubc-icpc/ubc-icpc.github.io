#include<iostream>
#include<string>

using namespace std;

int main (void) {
  int n; cin >> n;

  while(n--) {
    string str; cin>>str;
    cout << str << endl;
    int c=0;
    for( int i=0; i<str.size(); i++ ) {
      if(str[i]=='a'
	 || str[i]=='e'
	 || str[i]=='i'
	 || str[i]=='o'
	 || str[i]=='u') {
	c++;
      } else {
	c--;
      }
    }
    if(c>0) cout << 1 << endl;
    else cout << 0 << endl;
  }
  return 0;
}
