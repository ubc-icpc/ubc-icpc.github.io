#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;


int main (void) {
  
  int t; cin >> t;

  for(int k=1; k<=t; k++) {
    int c; cin >> c;
    vector<int> cars, team;
    for(int i=0; i<c; i++) {
      int tmp; cin >> tmp; cars.push_back(tmp);
    }
    for(int i=0; i<4*c; i++) {
      int tmp; cin >> tmp; team.push_back(tmp);
    }
    sort(cars.begin(), cars.end());
    sort(team.begin(), team.end());
    int ans = -1;
    for(int i=0; i<c; i++) {
      ans = max(cars[i]+team[4*(c-i)-1],ans);
    }
    
    //int ans = max(cars[0]+team[4*c-1], cars[c-1]+team[3]);
    cout << "Trip #" << k << ": " << ans << endl;
  }

  return 0;
}
