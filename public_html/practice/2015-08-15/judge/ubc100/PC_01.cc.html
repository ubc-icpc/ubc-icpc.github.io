#include<iostream>
#include<string>
#include<string.h>
#include<algorithm>

using namespace std;


int dp[51];

int main (void) {
  
  int n; cin >> n;
  

  for(int i=1; i<=n; i++) {
    cout << "Day #" << i << endl;
    int cells, jump; cin >> cells >> jump;
    cout << cells << " " << jump << endl;
    string str; cin >> str;
    cout << str << endl;
    for(int t=1; t<cells; t++) dp[t]=-1;
    for(int j=1; j<cells; j++) {
      bool flag=true;
      if(str[j] == 'X')continue;

      for(int k=j-1; k>=0; k--) {
	if( (j-k) > jump+1) break;
	
	if(str[k]=='.' && dp[k]>=0) {
	  if(flag) {
	    dp[j] = dp[k]+1;
	    flag = false;
	  } else {
	    dp[j] = min(dp[k]+1,dp[j]);
	  }
	}
      }

     
    }
    if(dp[cells-1]>=0)
      cout << dp[cells-1] << endl;
    else 
      cout << 0 << endl;
    if(i!=n) cout << endl;
  }
  
  return 0;
}
