#include<iostream>

using namespace std;


int main (void) {
  
  int t; cin>>t;

  for(int k=0; k<t; k++) {
    int games, points; cin >> games >> points;
    int m_wins = points/3;
    cout << "Team #" << k+1 << endl;
    cout << "Games: " << games << endl;
    cout << "Points: " << points << endl;
    cout << "Possible records:" << endl;
    for(int i=m_wins; i>=0; i--) {
      if( i + (points-(i*3)) <= games ) {
	cout << i << " - " << points-(i*3) << " - " << games - i - (points-(i*3)) << endl;
      }
    }
    if(k!=t-1)
    cout << endl;
  }

  return 0;
}
