#include <iostream>	
#include <string>
using namespace std;
int main()
{
	int n;
	cin >> n;
	while (n--)
	{
		string name;
		int count = 0;
		cin >> name;
		for (int i = 0; i < name.length(); i++)
		{
			if (name[i] == 'a' || name[i] == 'e' || name[i] == 'i' || name[i] == 'o' || name[i] == 'u')
			{
				count++;
			}
		}
		cout << name << endl;
		if (count > name.length() - count)
		{
			cout << 1 << endl;
		}
		else
		{
			cout << 0 << endl;
		}
	}
}
