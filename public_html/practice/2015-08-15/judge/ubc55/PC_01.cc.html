#include <string>
#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int day = 1;
	while (n--)
	{
		bool error = false;
		int length;
		cin >> length;
		int max;
		cin >> max;	
		int remaining = length - 1;
		int count = 0;
		int current = 0;
		string path;
		cin >> path;
		while (true)
		{
			if (remaining - 1 <= max)
			{
				count++;
				break;
			}

			for (int i = max; i >= -1; i--)
			{
				if (i == -1)
				{
					error = true;
					goto done;
				}
				if (path[current + i + 1] == '.')
				{
					count++;
					current = current + i + 1;
					remaining = remaining - i - 1;
					break;
				}
			}
		}
	done:
		cout << "Day #" << day << endl;
		cout << length << " " << max << endl;
		cout << path << endl;
		day++;
		if (error)
		{
			cout << 0 << endl;
		}
		else
		{
			cout << count << endl;
		}
	}
}
