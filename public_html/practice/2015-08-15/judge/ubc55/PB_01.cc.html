#include <iostream>


using namespace std;

int main()
{
	int n;
	cin >> n;
	int total = n;
	while (n)
	{
		cout << "Team #" << total - n + 1<< endl;
		int g; cin >> g;
		int p; cin >> p;
		cout << "Games: " << g << endl;
		cout << "Points: " << p << endl;
		cout << "Possible records:" << endl;
		int win, tie, loss;
		for (int i = g; i >= 0; i--)
		{
			if (i * 3 > p)
			{
				continue;
			}
			if (i * 3 + (g - i) < p)
			{
				continue;
			}
			cout << i << " - " << (p - 3 * i) << " - " << (g - i - (p - 3 * i)) << endl;
		}
		n--;
	}
}
