#include <iostream>
using namespace std;

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    cout << "Scenario #" << T << ": ";
    int c, s, d;
    cin >> c >> s >> d;
    bool tc = true;
    int m = 0;
    while (c || s) {
      m++;
      if (tc) {
        if (s && s % 2 == 0) {
          if (c) {
            c--;
            s++;
          }
          else {
            s--;
            d++;
          }
        }
        else if (s % 2) {
          if (c) {
            c--;
            s++;
          }
          else {
            s--;
            d++;
          }
          tc = false;
        }
        else {
          if (d) {
            if (c == 1) {
              c--;
              d++;
            }
            else {
              c -= 2;
              s += 2;
            }
            tc = false;
          }
          else {
            c--;
            s++;
          }
        }
      }
      else {
        if (s % 2) {
          if (c) {
            c--;
            s++;
          }
          else {
            s--;
            d++;
          }
        }
        else if (s && s % 2 == 0) {
          if (c) {
            c--;
            d++;
          }
          else {
            s -= 2;
            d += 2;
          }
        }
        else {
          if (c == 1) {
            c--;
            d++;
          }
          else {
            c -= 2;
            s += 2;
          }
        }
      }
    }
    cout << m << "\n";
  }
  return 0;
}
