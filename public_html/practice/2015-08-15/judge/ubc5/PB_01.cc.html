#include <iostream>
using namespace std;

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    cout << "Team #" << T << "\n";
    int g, p;
    cin >> g >> p;
    cout << "Games: " << g << "\nPoints: " << p << "\nPossible records:\n";
    for (int i = g; i >= 0; i--) {
      for (int j = 0; j <= g; j++) {
        if (i+j > g)
          break;
        if (3*i+j == p)
          cout << i << "-" << j << "-" << g-i-j << "\n";
      }
    }
    cout << "\n";
  }
  return 0;
}
