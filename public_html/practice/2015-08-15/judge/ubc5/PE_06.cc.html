#include <iostream>
#include <queue>
#include <set>
#include <utility>
#include <map>
using namespace std;

set<int> adj[50];
string S[50];
int v[50];

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    cout << "Chain Email #" << T << ":\n";
    int p, s;
    cin >> p >> s;
    s--;
    for (int i = 0; i < p; i++) {
      adj[i].clear();
      v[i] = 0;
      cin >> S[i];
    }
    for (int i = 0; i < p; i++) {
      int n; cin >> n;
      for (int j = 0; j < n; j++) {
        int x; cin >> x;
        adj[i].insert(x-1);
      }
    }
    queue<int> q;
    q.push(s);
    v[s] = 1;
    while (q.size()) {
      int f = q.front();
      q.pop();
      if (v[f] > 10000)
        continue;
      for (set<int>::iterator it = adj[f].begin(); it != adj[f].end(); it++) {
        v[*it]++;
        q.push(*it);
      }
    }
    bool first = true;
    for (int i = 0; i < p; i++) {
      if (v[i] > 1000) {
        if (first)
          first = false;
        cout << S[i] << " ";
      }
    }
    if (first)
      cout << "Safe chain email!";
    cout << "\n\n";
  }
  return 0;
}
