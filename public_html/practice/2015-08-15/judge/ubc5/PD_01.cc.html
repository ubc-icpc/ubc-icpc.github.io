#include <iostream>
#include <algorithm>
using namespace std;

int d[50], k[200];

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    cout << "Trip #" << T << ": ";
    int c; cin >> c;
    for (int i = 0; i < c; i++)
      cin >> d[i];
    sort(d,d+c);
    for (int i = 0; i < 4*c; i++) {
      cin >> k[i];
      k[i] *= -1;
    }
    sort(k,k+4*c);
    int mx = 0;
    for (int i = 0; i < c; i++)
      for (int j = i*4; j < (i+1)*4; j++)
        mx = max(mx,d[i]-k[j]);
    cout << mx << "\n";
  }
  return 0;
}
