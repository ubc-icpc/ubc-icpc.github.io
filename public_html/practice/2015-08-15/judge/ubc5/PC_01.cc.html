#include <iostream>
using namespace std;

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    cout << "Day #" << T << "\n";
    int c, d;
    string s;
    cin >> c >> d >> s;
    cout << c << " " << d << "\n" << s << "\n";
    int f = 0, j = 0;
    while (f < c) {
      f += d+1;
      j++;
      if (f >= c-1)
        break;
      bool p = false;
      f++;
      for (int i = 0; i <= d; i++) {
        f--;
        if (s[f] == '.') {
          p = true;
          break;
        }
      }
      if (!p) {
        j = 0;
        break;
      }
    }
    cout << j << "\n\n";
  }
  return 0;
}
