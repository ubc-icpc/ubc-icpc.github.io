#include <iostream>
using namespace std;

int main() {
  int t; cin >> t;
  while (t--) {
    string s; cin >> s;
    int v = 0;
    for (int i = 0; i < s.size(); i++) {
      if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u')
        v++;
    }
    cout << s << "\n" << (2*v > s.size() ? '1' : '0') << "\n";
  }
  return 0;
}
