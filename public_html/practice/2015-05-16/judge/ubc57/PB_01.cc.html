#include <iostream>
using namespace std;


int main()
{
	int n;
	cin>>n;
	int x,y,small_x,big_x,small_y,big_y;
	small_x=101;
	big_x=0;
	small_y=101;
	big_y=0;
	for (int i=1; i<=n;i++){
		cin>>x;
		cin>>y;
		if (x<small_x) small_x=x;
		if (x>big_x) big_x=x;
		if (y<small_y) small_y=y;
		if (y>big_y) big_y=y;
	}
	int l=max(big_x-small_x,big_y-small_y);
	cout<<l*l<<endl;
	return 0;
}
