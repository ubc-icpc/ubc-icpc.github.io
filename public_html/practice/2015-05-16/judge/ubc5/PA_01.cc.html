#include <iostream>
using namespace std;

typedef long long ll;

int main() {
  int n, b;
  while (cin >> n) {
    ll t = 0;
    for (int i = 0; i < n; i++) {
      cin >> b;
      if (i)
        cout << " ";
      ll a = (ll)(i+1)*b-t;
      cout << a;
      t += a;
    }
    cout << "\n";
  }
  return 0;
}
