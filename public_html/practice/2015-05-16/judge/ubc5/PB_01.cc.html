#include <iostream>
#include <algorithm>
using namespace std;

int main() {
  int n;
  while (cin >> n) {
    int minx = 4242, maxx = -4242, miny = 4242, maxy = -4242;
    for (int i = 0; i < n; i++) {
      int x, y;
      cin >> x >> y;
      minx = min(minx, x);
      maxx = max(maxx, x);
      miny = min(miny, y);
      maxy = max(maxy, y);
    }
    int d = max(maxx-minx,maxy-miny);
    cout << d*d << "\n";
  }
  return 0;
}
