import java.util.Scanner;

/**
 * Created by Peter on 2015-05-17.
 */
public class Klopka {
    public static void main(String[] args) {
        int minX = Integer.MAX_VALUE;
        int maxX = Integer.MIN_VALUE;
        int minY = Integer.MAX_VALUE;
        int maxY = Integer.MIN_VALUE;
        Scanner s = new Scanner(System.in);
        int noOfFlies = Integer.parseInt(s.nextLine());
        for(int i = 0; i < noOfFlies; i++){
            String[] ints = s.nextLine().split("\\s+");
            int x = Integer.parseInt(ints[0]);
            int y = Integer.parseInt(ints[1]);
            if(x < minX)
                minX = x;
            if(y < minY)
                minY = y;
            if(x > maxX)
                maxX = x;
            if(y > maxY)
                maxY = y;
        }
        System.out.println(Math.max((maxX-minX)*(maxX-minX), (maxY-minY)*(maxY-minY)));

    }
}
