import java.util.Scanner;

/**
 * Created by Peter on 2015-05-17.
 */
public class Prosjek {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);
        int noOfNumbers = Integer.parseInt(s.nextLine());
        String[] strs = s.nextLine()/*"3 2 3 5"*/.split("\\s+");
        int soFar = 0;

        for(int i = 0; i < noOfNumbers; i++){
            int j = Integer.parseInt(strs[i]);
            int k = j*(i+1)-soFar;
            String space = " ";
            if(i == noOfNumbers - 1)
                space = "";
            System.out.print(k + space);
            soFar += k;
        }
    }
}
