#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;
ll N, M, A, B;
int L[1000005][30];
char C;
string S;
ll G (char c, int x, int y) { return L[y][c - 'A'] - L[x][c - 'A']; }
int main () {
  ios::sync_with_stdio(0); cin.tie(0);
  cin >> N >> S;
  cin >> N;
  for (int i = 1; i <= S.size(); ++i) {
    for (int j = 0; j < 26; ++j) L[i][j] = L[i - 1][j];
    ++L[i][S[i - 1] - 'A'];
  }
  while (N--) {
    cin >> M >> C;
    A = M - 1; B = M;
    if (A % 2) B /= 2;
    else A /= 2;
    A = ((A % S.size()) * (B % S.size())) % S.size(); 
    B = G(C, A, min((ll)S.size(), A + M));
    if (A + M > S.size()) {
      B += G(C, 0, (A + M) % S.size());
      B += G(C, 0, S.size()) * ((M + A - S.size()) / S.size());
    }
    cout << B << "\n";
  }
  return 0;
}
