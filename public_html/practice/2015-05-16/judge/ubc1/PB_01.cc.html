#include <bits/stdtr1c++.h>

using namespace std;
  
int main() {
    int n; cin >> n;
    
    int xl = 100, xr = -100, yl = 100, yr = -100;
    for (int i = 0; i < n; i++) {
        int x, y; cin >> x >> y;
        xl = min(xl, x);
        yl = min(yl, y);
        xr = max(xr, x);
        yr = max(yr, y);
    }
    cout << max(xr-xl, yr-yl) * max(xr-xl, yr-yl) << endl;
    
	return 0;
}
