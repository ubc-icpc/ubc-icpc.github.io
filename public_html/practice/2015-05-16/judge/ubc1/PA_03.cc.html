#include <bits/stdtr1c++.h>

using namespace std;

// b[i] = (i*b[i-1] + a[i])/(i+1)
// (i+1)b[i] - i*b[i-1] = a[i]
int main() {
    int n; cin >> n;
    
    long long prev = 0, cur;
    for (int i = 0; i < n; i++) {
        cin >> cur;
        cout << (i+1)*cur - i*prev << " ";
        prev = cur;
    }
    cout << endl;
	return 0;
}
