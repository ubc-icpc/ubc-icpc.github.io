#include <iostream>
#include <algorithm>

using namespace std;

int main (void) {
  int n; cin>>n;
  int X,x;
  int Y,y;
  cin >> x >> y;
  X=x; Y=y;
  for( int i=1; i<n; i++ ) {
    int tx, ty; cin >> tx >> ty;
    x=min(x,tx);
    X=max(X,tx);
    y=min(y,ty);
    Y=max(Y,ty);
  }

  int r = max(Y-y,X-x);

  cout << r*r << endl;
  

  return 0;
}
