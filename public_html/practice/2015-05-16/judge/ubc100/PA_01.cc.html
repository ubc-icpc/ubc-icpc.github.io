#include <iostream>

using namespace std;

int main (void) {

  int n; cin >> n;
  int s[n];

  for( int i=0; i<n; i++ ) {
    int t; cin >> t;
    s[i] = t;
  }

  if( n == 1 ) {
    cout << s[0] << endl;
    return 0;
  }
  int r[n];

  for( int i=n-1; i>0; i-- ) {
    r[i] = s[i]*(i+1) - s[i-1]*i;
  }
  r[0]=s[0];

  for( int i=0; i<n; i++ ) cout << r[i] << " ";
  cout << endl;
  return 0;
}
