import java.io.*;
import java.util.*;

public class Klopka {
	
	public static void main(String args[]) throws IOException {
		
		BufferedReader kinput = new BufferedReader(new InputStreamReader(System.in));
		
		int m = Integer.parseInt(kinput.readLine());
		int[][] c = new int[m][2];
		
		for(int a = 0; a < m; a++) {
			
			StringTokenizer st = new StringTokenizer(kinput.readLine());
			c[a][0] = Integer.parseInt(st.nextToken());
			c[a][1] = Integer.parseInt(st.nextToken());
			
		}
		
		int min_x = c[0][0], max_x = c[0][0];
		
		for(int a = 0; a < m; a++) {
			
			if(max_x < c[a][0]) {
				
				max_x = c[a][0];
				
			}
			
			if(min_x > c[a][0]) {
				
				min_x = c[a][0];
				
			}
			
		}
		
		int min_y = c[0][1], max_y = c[0][1];
		
		for(int a = 0; a < m; a++) {
			
			if(max_y < c[a][1]) {
				
				max_y = c[a][1];
				
			}
			
			if(min_y > c[a][1]) {
				
				min_y = c[a][1];
				
			}
			
		}
		
		int x = max_x - min_x;
		int y = max_y - min_y;
		
		if(x > y) {
			
			System.out.println(x*x);
			
		}
		
		else {
			
			System.out.println(y*y);
			
		}
		
	}
	
}
