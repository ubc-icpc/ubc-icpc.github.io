import java.io.*;
import java.util.*;

public class PROSJEK {
	
	public static void main(String args[]) throws IOException {
		
		BufferedReader kinput = new BufferedReader(new InputStreamReader(System.in));
		
		int length = Integer.parseInt(kinput.readLine());
		StringTokenizer seq = new StringTokenizer(kinput.readLine());
		int[] seqB = new int[length];
		int[] seqA = new int[length];
		
		for(int a = 0; a < length; a++) {
			
			seqB[a] = Integer.parseInt(seq.nextToken());
			
		}
		
		seqA[0] = seqB[0];
		int temp = seqB[0];
		
		for(int a = 1; a < length; a++) {
			
			seqA[a] = (seqB[a] * (a + 1)) - temp;
			
			temp += seqA[a];
			
		}
		
		for(int a = 0; a < length; a++) {
			
			System.out.print(seqA[a] + " ");
			
		}
		
	}
	
}
