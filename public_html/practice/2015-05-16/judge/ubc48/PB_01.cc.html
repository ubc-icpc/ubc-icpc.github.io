#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	vector < int > x(n), y(n);
	for(int i = 0; i < n; ++i){
		cin >> x[i]; cin >> y[i];
	}
	sort(x.begin(), x.end()); sort(y.begin(), y.end());
	int big = ((x[x.size() - 1] - x[0]) > (y[y.size() - 1] - y[0])) ? (x[x.size() - 1] - x[0]) : (y[y.size() - 1] - y[0]);
	cout << big * big << endl;
	return 0;
}
