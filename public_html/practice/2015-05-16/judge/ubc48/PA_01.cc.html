#include<bits/stdtr1c++.h>
#include<string>
using namespace std;
int main(){
	int length; cin >> length;
	stringstream ss;
	long long total = 0;
	string a = "";
	for(int i = 0; i < length; ++i){
		
		string temp; cin >> temp;
		
		ss.str(temp);
		
		long long result; ss >> result; 
		ss.clear(); ss.str("");
		result = result * (i + 1) - total;
		total += result;
		
		ss << result;
		string garb; ss >> garb;
		ss.clear(); ss.str("");
		a.append(garb + " ");
		
	}
	a.erase(a.size() - 1, 1);
	cout << a << endl;
	return 0;
}
