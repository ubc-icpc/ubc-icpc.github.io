#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
typedef vector<int> vi;

int main() {
  int N; cin >> N;
  vi x(N), y(N);
  for (int i = 0;i < N; ++i)
    cin >> x[i] >> y[i];

  int min_x = 100, min_y = 100, max_x = 0, max_y = 0;
  for (int i = 0; i < N; ++i) {
    min_x = min(min_x, x[i]);
    min_y = min(min_y, y[i]);
    max_x = max(max_x, x[i]);
    max_y = max(max_y, y[i]);
  }
  int side = max(max_x - min_x, max_y - min_y);
  cout << side * side << endl;

  return 0;
}
