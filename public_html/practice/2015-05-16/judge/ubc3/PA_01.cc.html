#include <iostream>
#include <vector>
using namespace std;
typedef vector<int> vi;

int main() {
  int N;
  cin >> N; 
  vi B(N);
  for (int i = 0; i < N; ++i) {
    cin >> B[i];
  }
  vi A(N);
  A[0] = B[0];
  for (int i = 1; i < N; ++i) {
    A[i] = B[i] * (i + 1) - B[i-1] * i;
  }
  for (int k : A) {
    cout << k << ' ';
  }
  cout << endl;

  return 0;
}
