import java.util.Scanner;

public class mosquito {

	public static void main(String[] args) {
		Scanner s= new Scanner(System.in);
		int T = s.nextInt();
		int minX = 100;
		int maxX = 1;
		int minY = 100;
		int maxY = 1;
		while(T-->0) {
			int thisX = s.nextInt();
			int thisY = s.nextInt();
			minX = (thisX < minX) ? thisX: minX;
			minY = (thisY < minY) ? thisY: minY;
			maxX = (thisX > maxX) ? thisX: maxX;
			maxY = (thisY > maxY) ? thisY: maxY;
		}
		int Xlen = maxX - minX;
		int Ylen = maxY - minY;
		int length = (Xlen < Ylen) ? Ylen : Xlen;
		int area = length*length;
		System.out.println(area);
		s.close();
	}

}
