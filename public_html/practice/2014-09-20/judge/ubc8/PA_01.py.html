import sys
import fileinput
from collections import Counter
from itertools import combinations

for line in fileinput.input():
    if line.strip() == "0":
        sys.exit(0)

    n, cheeses = line.split()[0], line.split()[1:]
    cnt = Counter()

    for cheese in cheeses:
        cnt[cheese] += 1

    cheddar, gouda, havarti = cnt['Cheddar'], cnt['Gouda'], cnt['Havarti']

    print max(cheddar, gouda, havarti,
              (min(cheddar, gouda) * 2),
              (min(gouda, havarti) * 2),
              (min(cheddar, havarti) * 2),
              (min(cheddar, gouda, havarti) * 3),
              )
