import fileinput

f = fileinput.input()

for case in xrange(0, int(f.readline())):
    # Sort enemies by ratio of HP / DPS
    enemies = sorted(
        zip([float(x) for x in f.readline().split()],  # enemy HP
            [float(x) for x in f.readline().split()]), # enemy DPS
        key=lambda x: x[0] / x[1])

    teamDPS = sum(float(x) for x in f.readline().split())

    print "%.5f" % sum(
        sum(dps * (enemy[0] / teamDPS) for _, dps in enemies[i:])
        for i, enemy in enumerate(enemies))
