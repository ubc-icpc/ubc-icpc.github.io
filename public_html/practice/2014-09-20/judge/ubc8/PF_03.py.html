import fileinput

for line in fileinput.input():
    lineArray = line.split()
    n, boats = int(lineArray[0]), sorted(min(420, int(x)) for x in lineArray[1:])

    running_tab = 0

    while boats:
        if len(boats) <= 2:
            running_tab += boats[len(boats) - 1]
            boats = []
        elif len(boats) == 3:
            running_tab += boats[0] + boats[1] + boats[2]
            boats = []
        else:
            option_a = (boats[0] * 2) + boats[-1] + boats[-2]
            option_b = (boats[1] * 2) + boats[-1] + boats[0]

            running_tab += min(option_a, option_b)
            boats = boats[:-2]

    print running_tab
