import fileinput
import math

def fn(c, m_0, b, r, i, l):
    m = m_0 - c
    times = (m - r * l) / (c - b - i * l)
    return math.floor(times) + 1

input = fileinput.input()
cases = int(input.readline())

for i in range(cases):
    data = input.readline().split()
    c = float(data[0])
    m = float(data[1])
    b = float(data[2])
    r = float(data[3])
    i = float(data[4])

    min_l = fn(c, m, b, r, i, 0)
    max_l = fn(c, m, b, r, i, 100000)
    print int(max(min_l, max_l))

input.close()
