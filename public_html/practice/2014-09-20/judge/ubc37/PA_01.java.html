import java.util.*;

public class A {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		while (sc.hasNext()) {
			int blocks = sc.nextInt();
			HashMap<String, Integer> map = new HashMap<String,Integer>(blocks);
			if (blocks == 0) {
				return;
			}
			for (int i = 0; i < blocks; i++) {
				String block = sc.next();
				if (map.containsKey(block)) {
					map.put(block, map.get(block) + 1);
				} else {
					map.put(block, 1);
				}
			}
			int max = -1;
			for (int i = 1; i <= Collections.max(map.values()); i++) {
				int curr = 0;
				for (int num : map.values()) {
					if (num >= i)
						curr += i;
				}
				if (curr > max)
					max = curr;
			}
			System.out.println(max);
		}
	}
}
