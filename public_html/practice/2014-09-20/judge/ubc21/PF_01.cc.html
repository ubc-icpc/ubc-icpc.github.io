#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    int n;
    while(scanf("%d", &n) != EOF)
    {
        vector<int> x;
        int q;
        for (int i = 0 ; i < n; i++)
        {
            scanf("%d", &q);
            x.push_back(min(q, 420));
        }
        sort(x.begin(), x.end());
        int sum = 0;
        if (n == 1)
        {
            sum = x[0];
        }
        else
        {
            int i = n - 1;
            int diff1 = x[1] - x[0];
            while(i > 2)
            {
                if (x[i-1] - x[1] < diff1)
                {
                    sum += x[i--];
                    sum += x[0];
                }
                else
                {
                    sum += x[1];
                    sum += x[0];
                    sum += x[i--];
                    sum += x[1];
                    i--;
                }
            }
            if (i == 1)
            {
                sum += x[1];
            }
            else if (i == 2)
            {
                sum += x[0] + x[1] + x[2];
            }
        }
        printf("%d\n", sum);
    }
}
