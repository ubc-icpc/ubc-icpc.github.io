#include <stdio.h>
#include <math.h>
#include <algorithm>

#define ll long long

ll clamp(ll x)
{
    return (x < 0L) ? 0L : (x > 100000L) ? 100000L : x;
}

int main()
{
    int n;
    scanf("%d", &n);
    while (n--)
    {
        ll cost, max, regen, res, boost;
        scanf("%lld %lld %lld %lld %lld", &cost, &max, &regen, &res, &boost);
        double vasymp = (cost - regen) / (boost);
        ll above = clamp((ll)floor(vasymp + (double)1.0f));
        ll below = clamp((ll)ceil(vasymp - (double)1.0f));
        ll a = (max - regen - (boost + res) * above) / (cost - regen - boost * above);
        ll b = (max - regen - (boost + res) * below) / (cost - regen - boost * below);
        ll c = (max - regen - (boost + res) * 0L) / (cost - regen - boost * 0L);
        ll d = (max - regen - (boost + res) * 100000L) / (cost - regen - boost * 100000L);
        printf("%lld\n", std::max(std::max(a, b), std::max(c, d)));
    }
}
