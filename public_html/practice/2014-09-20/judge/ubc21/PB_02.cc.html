#include <stdio.h>
#include <stdlib.h>
#include <algorithm>

struct Node
{
    int d;
    int h;
};

int compare(const void * a, const void * b)
{
    int x = (((Node *)a)->h * ((Node *)b)->d) - (((Node *)b)->h * ((Node *)a)->d);
    if (x != 0) return x;
    return ((Node *)a)->h - ((Node *)b)->h;
}

int main()
{
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
    {
        double total = 0.0f;
        double damage = 0.0f;
        Node arr[5];
        for (int j = 0; j < 5; j++)
        {
            int x;
            scanf("%d", &x);
            arr[j].h = x;
        }
        for (int j = 0; j < 5; j++)
        {
            int x;
            scanf("%d", &x);
            arr[j].d = x;
        }
        int power = 0;
        for (int j = 0; j < 5; j++)
        {
            int x;
            scanf("%d", &x);
            power += x;
        }
        qsort(arr, 5, sizeof(Node), compare);
        for (int j = 0; j < 5; j++)
        {
            double killtime = (double)(arr[j].h) / (double)(power);
            total += killtime;
            damage += (total * arr[j].d);
        }
        printf("%.5f\n", damage);
    }
}
