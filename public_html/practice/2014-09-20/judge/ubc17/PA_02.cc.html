#include <iostream>
#include <string>
#include <map>
#include <algorithm>
#include <vector>

using namespace std;

typedef pair<string, int>	psi;
typedef map<string, int>	msi;
typedef vector<int>			vi;

struct cmpV {
	bool operator() (const int &x, const int &y) const {
		return x < y;
	}
};

int main() {
	int n;
	cin >> n;
	while (n > 0) {
		msi cheeses;
		for (int i=0; i<n; i++) {
			string cname;
			cin >> cname;
			cheeses[cname]++;
		}
		vi cc;
		for (msi::iterator it = cheeses.begin(); it != cheeses.end(); it++) {
			cc.push_back(it->second);
		}
		sort (cc.begin(), cc.end(), cmpV());
		int max = 0;
		int cSi = cheeses.size();
		int i = 1;
		for (vi::iterator it = cc.begin(); it != cc.end(); it++) {
			if (max < *it*(cSi-i+1))
				max = *it*(cSi-i+1);
			i++;
		}
		cout << max << endl;
		cin >> n;
	}
	return 0;
}
