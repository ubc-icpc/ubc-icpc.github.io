#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair<double, double>	pii;
typedef vector<pii>		vpii;

#define ins(a) push_back(a)

struct cmp {
	bool operator() (const pii &x, const pii &y) const {
		return x.first*y.second < y.first*x.second;
	}
};

int main() {
	int T; cin >> T;
	cout.setf(ios::fixed,ios::floatfield);
	cout.precision(5);
	while (T--) {
		vpii oppTeam;
		for (int i=0; i<5; i++) {
			double hi; cin >> hi;
			oppTeam.ins(pii(hi, 0));
			// cout << oppTeam[i].first << endl;
		}
		double dmgLeft=0;
		for (int i=0; i<5; i++) {
			int ei; cin >> ei;
			oppTeam[i].second = ei;
			dmgLeft+=ei;
		}
		sort (oppTeam.begin(), oppTeam.end(), cmp());
		for (int i =0; i< 5; i++) {
			// cout << "hp of: " << oppTeam[i].first << "; dps of: " << oppTeam[i].second << "." << endl;
		}
		double myDps=0;
		for (int i=0; i<5; i++) {
			double tm; cin >> tm;
			myDps += tm;
		}
		// cout << myDps << endl;
		double total=0;
		for (int i=0; i<5; i++) {
			double ttk = (oppTeam[i].first/myDps);
			total+= (ttk*dmgLeft);
			dmgLeft-=oppTeam[i].second;
		}
		cout << total << endl;
	}
	return 0;
}
