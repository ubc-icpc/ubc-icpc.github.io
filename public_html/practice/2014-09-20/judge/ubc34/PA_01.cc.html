#include <iostream>
#include <vector>
#include <algorithm>
#include <map>

using namespace std;

int main() {
	int n;
	cin >> n;
	while (n > 0) {
		map<string, int> cheese;
		for (int i = 0; i < n; i++) {
			string s; cin >> s;
			if (cheese.count(s) > 0) {
				cheese[s]++;
			} else {
				cheese[s] = 1;
			}
		}
		vector<int> num;
		for (map<string, int>::iterator it = cheese.begin();
				it != cheese.end(); ++it) {
			num.push_back(it->second);
		}
		sort(num.begin(), num.end());
		int nn = num.size();
		int best = 0;
		int count = 0;
		for (int i = nn-1; i >= 0; i--) {
			count++;
			if (best < count * num[i]) {
				best = count*num[i];
			}
		}
		cout << best << endl;
		cin >> n;
	}
}
