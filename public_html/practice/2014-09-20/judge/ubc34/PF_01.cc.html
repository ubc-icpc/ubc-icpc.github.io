#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	int n;
	while (cin >> n) {
		vector<int> boats;
		for (int i = 0; i < n; i++) {
			int j; cin >> j;
			boats.push_back(j);
		}
		sort(boats.begin(), boats.end());
		int cost = 0;
		int size = boats.size();
		while (size > 3) {
			int m1 = boats[0];
			int m2 = boats[1];
			int M1 = boats[size - 1];
			int M2 = boats[size - 2];
			int c1 = min(420, M1) + 2 * min(420, m1) + min(420, M2);
			int c2 = min(420, m2) + min(420, m1) + min(420, M1) + min(420, m2);
			cost += min(c1, c2);
			size -= 2;
		}
		if (size == 3) {
			cost += min(boats[0], 420) + min(boats[2], 420);
			size--;
		}
		if (size == 2) {
			cost += min(boats[1], 420);
			size -= 2;
		}
		if (size == 1) {
			cost += min(boats[0], 420);
			size--;
		}
		cout << cost << endl;
	}
}
