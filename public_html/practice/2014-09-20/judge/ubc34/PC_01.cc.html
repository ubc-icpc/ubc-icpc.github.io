#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;
typedef long long ll;

ll c, m, b, r, ex;
ll calc(int i) {
	ll maxmana = m - r*i - c;
	ll rate = c - (b + i*ex);
	if (maxmana < 0) return 0;
	return maxmana / rate + 1;
}

int main() {
	int ncase;
	cin >> ncase;
	for (int csnum = 1; csnum <= ncase; csnum++) {
		// mana cost, max mana, base regen
		cin >> c >> m >> b;
		// reserve, extra rate
		cin >> r >> ex;
		int vleft = calc(0);
		int vright = calc(100000);
		cout << max(vleft, vright) << endl;
	}
}
