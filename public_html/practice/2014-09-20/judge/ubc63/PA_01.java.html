import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.Scanner;


public class Main {
	public static void main(String[] args) {
		
		Scanner sc = new Scanner(System.in);
		while (true) {
			String line = sc.nextLine();
			String[] ins = line.split(" ");
			if (Integer.parseInt(ins[0]) == 0) {
				break;
			}
			System.out.println(maxCheese(ins));
		}
	}
	
	public static int maxCheese(String[] args) {
		int cCount = 0;
		int gCount = 0;
		int hCount = 0;
		
		for (int i = 0; i<Integer.parseInt(args[0]); i++) {
			if (args[1+i].equals("Cheddar")) {
				cCount++;
			} else if (args[1+i].equals("Gouda")) {
				gCount++;
			} else if (args[1+i].equals("Havarti")) {
				hCount++;
			}
		}
		
		ArrayList<Integer> cheeseCount = new ArrayList<Integer>();
		cheeseCount.add(fillCheese(cCount, gCount, hCount));
		cheeseCount.add(fillCheese(gCount, hCount, cCount));
		cheeseCount.add(fillCheese(hCount, gCount, cCount));
		
		return Collections.max(cheeseCount, new Comparator<Integer>() {

			@Override
			public int compare(Integer o1, Integer o2) {
				return o1.compareTo(o2);
			}
			
		});
	}

	private static Integer fillCheese(int a, int b,
			int c) {
		int max = a;
		
		if (a <= b) {		
			max += a;
		}
		if (a <= c) {
			max += a;
		}
		return max;
	}
}
