#include <bits/stdtr1c++.h>

using namespace std;

int main() {
	int n;
	while (cin >> n) {
		if (!n) break;
		map<string, int> m;
		for (int i = 0; i < n; i++) {
			string s; cin >> s;
			m[s]++;
		}
		
		int ans = 0;
		for (int i = 1; i <= n; i++) {
			int tans = 0;
			for (const auto& it : m) {
				if (i <= it.second) {
					tans += i;
				}
			}
			ans = max(ans, tans);
		}
		cout << ans << endl;
	}
	return 0;
}
