#include <bits/stdtr1c++.h>

using namespace std;

typedef int ll;
ll C, M, B, R, I;
inline ll solve(ll c, ll m, ll regen) {
	ll res = m/(c-regen);
	return res;
}

inline ll f(ll l) {
	if (M-R*l < C || (C-B-I*l) <= 0) return 0;
	if (l > 100000) return 0;
	return 1+solve(C, M-R*l-C, B+I*l);
}

int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	int t; cin >> t;
	while (t--) {
		cin >> C >> M >> B >> R >> I;
		ll ans1 = f(0);
		ll ans2 = f(100000);
		ll ans = max(ans1, ans2);
		for (int i = (C-B)/I-100; i <= (C-B)/I+100; i++) {
			ans = max(ans, f(i));
		}
		cout << ans << '\n';
	}	
	return 0;
}
