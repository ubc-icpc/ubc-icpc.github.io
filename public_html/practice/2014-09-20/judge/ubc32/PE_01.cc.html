#include <bits/stdtr1c++.h>

using namespace std;

long double expt(long double p_) {
	long double p = 1.0;
	long double res = 0;
	for (int k = 1; ; k++) {
		if (k*p_ >= 1) {
			res += p*k;
			break;
		}
		res += p*p_*k*k;
		p *= (1-k*p_);
	}
	return res;
}

int main() {
	long double d;
	while (cin >> d) {
		long double l = 0.00001, r = 1.0;
		while (abs(r-l) > 1e-8) {
			long double m = (l+r)/2;
			if (expt(m) >= 1/d) {
				l = m;
			} else {
				r = m;
			}
		}
		cout << fixed << setprecision(3) << (l+r)/2 << endl;
	}
	return 0;
}
