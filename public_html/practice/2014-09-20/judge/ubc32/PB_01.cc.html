#include <bits/stdtr1c++.h>

using namespace std;

typedef pair<double, double> pdd;

double D[5];
pdd HE[5];

int main() {
	int t; cin >> t;
	while (t--) {
		for (int i = 0; i < 5; i++) {
			cin >> HE[i].first;
		}
		
		double tot_ed = 0;
		for (int i = 0; i < 5; i++) {
			cin >> HE[i].second;
			tot_ed += HE[i].second;
		}
		
		double tot_d = 0;
		for (int i = 0; i < 5; i++) {
			cin >> D[i];
			tot_d += D[i];
		}
		
		double dmg = 1e99;
		vector<int> p({0,1,2,3,4});
		do {
			double ttot_ed = tot_ed;
			double tdmg = 0;
			for (int i = 0; i < 5; i++) {
				double dt = HE[p[i]].first/tot_d;
				tdmg += ttot_ed*dt;
				ttot_ed -= HE[p[i]].second;
			}
			dmg = min(dmg, tdmg);
		} while (next_permutation(p.begin(), p.end()));
		
		cout << fixed << setprecision(5) << dmg << endl;
	}
	return 0;
}
