import java.text.DecimalFormat;
import java.util.Arrays;
import java.util.Scanner;

public class B {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		int T = sc.nextInt();
		sc.nextLine();
		for (int t = 0; t < T; t++) {
			int[] h = new int[5];
			int[] e = new int[5];
			int[] d = new int[5];
			for (int i = 0; i < 5; i++)
				h[i] = sc.nextInt();
			sc.nextLine();
			for (int i = 0; i < 5; i++)
				e[i] = sc.nextInt();
			sc.nextLine();
			for (int i = 0; i < 5; i++)
				d[i] = sc.nextInt();
			sc.nextLine();
			n[] w = new n[5];
			for (int i = 0; i < 5; i++) {
				w[i] = new n(h[i], e[i]);
			}
			Arrays.sort(w);
			double totalD = 0;
			for (int i = 0; i < 5; i++)
				totalD += d[i];
			double time = 0;
			for (int i = 0; i < 5; i++) {
				time += w[i].h / totalD;
				w[i].t = time;
			}
			double total = 0;
			for (int i = 0; i < 5; i++) {
				total += w[i].d * w[i].t;
			}
			System.out.println(new DecimalFormat("0.00000").format(total));
		}
		sc.close();
	}

}

class n implements Comparable {
	int h;
	int d;
	double w;
	public double t;
	n (int h, int d) {
		this.h = h;
		this.d = d;
		this.w = (double)h / d;
	}
	@Override
	public int compareTo(Object arg0) {
		n temp = (n) arg0;
		if (this.w < temp.w) return -1;
		if (this.w > temp.w) return 1;
		return 0;
	}
	
}
