def countCheese(cheeseList):
	cheeseNum = [0,0,0] # Cheddar Gouda and Havarti
	for item in cheeseList:
		if item == "Cheddar":
			cheeseNum[0] = cheeseNum[0]+1
		elif item == "Gouda":
			cheeseNum[1] = cheeseNum[1]+1
		else:
			cheeseNum[2] = cheeseNum[2]+1

	cheeseNew = []
	for i in range(0,3):
		if cheeseNum[i] != 0:
			cheeseNew.append(cheeseNum[i]);

	mini = min(cheeseNew)
	maxi = max(cheeseNew)

	res = 0
	for i in range(mini,maxi+1):
		sum = 0
		for j in range(0,len(cheeseNew)):
			if(cheeseNew[j] >= i):
				sum = sum+i
		if sum > res:
			res = sum

	return res
