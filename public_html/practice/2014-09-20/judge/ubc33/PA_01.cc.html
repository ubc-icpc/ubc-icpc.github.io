#include <iostream>
#include <string>
#include <cmath>

using namespace std;

int main() {
	
	int N;
	while (cin >> N, N) {
		int num_c = 0;
		int num_g = 0;
		int num_h = 0;
		string name;
		for (int i=0; i<N; ++i) {
			cin >> name;
			if (name == "Cheddar") num_c++;
			if (name == "Gouda") num_g++;
			if (name == "Havarti") num_h++;
		}
		int min_all = min(num_c, min(num_g, num_h));
		int max_all = max(num_c, max(num_g, num_h));
		int min_c_g = min(num_c, num_g);
		int min_c_h = min(num_c, num_h);
		int min_g_h = min(num_g, num_h);
		int max_pairs = max(min_c_g, max(min_c_h, min_g_h));

		// cout << "min all " << min_all << " max_pairs " << max_pairs << " max all " << max_all << endl;

		cout << max(3*min_all, max(2*max_pairs, max_all)) << endl;


	}

	return 0;
}
