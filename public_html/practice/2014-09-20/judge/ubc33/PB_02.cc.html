#include <iostream>
#include <iomanip>
#include <vector>
#include <algorithm>

using namespace std;

// have all guys attack one guy
// rank by attack strength / seconds to kill

int D[5];
double tot_damage;

struct enemy {
	int H, E;
	double ratio() const {
		return ((double)E)/((double)H);
	}
	bool operator<(const enemy& other) const {
		return this->ratio() > other.ratio();
	}
};


int main() {

	int T;
	cin >> T;
	for (int i=0; i<T; ++i) {

		vector<enemy> e(5);

		for (int j1=0; j1<5; ++j1) cin >> e[j1].H;
		for (int j2=0; j2<5; ++j2) cin >> e[j2].E;
		for (int j3=0; j3<5; ++j3) cin >> D[j3];

		tot_damage = 0;
		sort(e.begin(), e.end());

		double tot_attack = D[0]+D[1]+D[2]+D[3]+D[4];
		double tot_time = 0;
		for (int i=0; i<5; ++i) {
			tot_time += (e[i].H / tot_attack);
			tot_damage += tot_time*e[i].E;
		}


		cout << setprecision(5) << fixed << tot_damage << endl;



	}

	return 0;
}
