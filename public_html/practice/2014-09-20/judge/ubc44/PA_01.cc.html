#include <algorithm>
#include <complex>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <string>
#include <vector>

typedef std::pair<int, int> II;
typedef std::vector<int> VI;
typedef std::vector<II> VII;
typedef std::vector<VI> VVI;

typedef long double ld;

typedef std::complex<ld> Pt;

int main() {
    int N;
    while(std::cin >> N && N) {
        std::map<std::string, int> counts;
        for(int i = 0; i < N; i ++) { std::string s; std::cin >> s; counts[s] ++; }

        VI C;
        for(auto e : counts) C.push_back(e.second);
        std::sort(C.begin(), C.end());
        int best = 0;
        for(int i = 0; i < C.size(); i ++) {
            //std::cout << C[i] << std::endl;
            best = std::max(best, (int)(C[i]*(C.size()-i)));
        }
        std::cout << best << std::endl;
    }
    return 0;
}
