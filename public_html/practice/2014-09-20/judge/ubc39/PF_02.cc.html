#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

#define N 500

int a[N];

int main() {
    int n;
    while(cin>>n) {
        for(int i=0; i< n;++i) {
            cin >> a[i];
            a[i] = min(a[i], 420);
        }
        sort(a , a+n);
        int cost = 0;
        while(n > 0) {
            if(n == 1)  {
                cost += a[0] ;
            } else if (n == 2) {
                cost += a[1];
            } else{
                int c1 = a[n-1] + a[0] + a[n-2];
                if(n > 3) c1 += a[0];
                
                int c2 = a[1] + a[0] + a[n-1];
                
                if(n > 3) {
                    c2 += a[1];
                }

                if(c1 < c2) {
                    cost += c1;
                } else {
                    cost += c2;
                }
            }
            n -= 2;
            if(n == 1) n -= 1;
        }

        cout << cost << endl;
    }
    return 0;
}
