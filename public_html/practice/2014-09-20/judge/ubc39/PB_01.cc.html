#include <iostream>
#include <algorithm>

using namespace std;
#define N 5
typedef long double ld;
int h[N], e[N], d[N], o[N], t[N];


int main() {
    int T;
    cin>> T;
    while(T > 0) {
        T--;
        for(int i=0; i< N;++i) cin>> h[i];
        for(int i=0; i< N;++i) cin>> e[i];
        for(int i=0; i< N;++i) cin>> d[i];


        int dt = 0;

        int et = 0;
        for(int i=0; i< N;++i) {
            dt += d[i]; 
            o[i] = i;
            et += e[i];
        }

        int mn = -1;
        do {
            int dam = 0;
            int tt = 0;
            int left = et;
            
            for(int i=0; i< N; ++i) {
                int j = o[i];
                dam += h[j] * left;
                left -= e[j];
            }

            if(mn < 0 || dam < mn) {
                mn = dam;
                //cerr << ' ' << dam << endl;
                //for(int i=0; i< N; ++i) {
                //    cerr << o[i] << ' ';
                //}
                //cerr << endl;
                /*for(int i=0; i< N; ++i) {
                    cerr << e[o[i]] << ' ' << h[o[i]] << endl;
                }
                cerr << endl;*/
            }
        } while( next_permutation(o, o +N) );
        
        ld ans =  (ld)mn / (ld)dt;
        cout << fixed << setprecision(5) << ans << endl;    
    }
    return 0;
}
