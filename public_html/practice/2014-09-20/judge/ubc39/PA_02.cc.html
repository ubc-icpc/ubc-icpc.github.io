#include <iostream>
#include <map>
#include <utility>
using namespace std;
typedef pair<string, int> si;

int main() {
    int n;
    while(cin>>n && n) {
        string s;
        map<string, int> mp;
        int mx = 0;
        for(int i=0; i< n;++i) {
            cin >> s;
            if(mp.find(s) == mp.end()) {
                mp.insert(si(s, 1));
            } else {
                mp[s]++;
            }
            mx = max(mp[s], mx);
        }
        int ans = 0;
        for(int i= 1; i<= mx; ++i) {
            int sc = 0;
            for(map<string, int>::iterator it = mp.begin();
                    it != mp.end(); it++ ) {
                if(it->second >= i)sc++;
            }
            ans = max(ans, sc* i);
        }
        cout << ans << endl;

    }
    return 0;
}
