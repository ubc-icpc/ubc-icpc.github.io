#include <string>
#include "stdio.h"
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdlib>
#include <cmath>

using namespace std;

int main()
{
    int n;
    while(true)
    {
        scanf("%d", &n);
        if(n == 0) break;
        int c, g, h;
        c = g = h = 0;
        for(int i = 0; i < n; i ++)
        {
            string temp;
            cin >> temp;
            if(temp[0] == 'C')
                c++;
            else if(temp[0] == 'G')
                g++;
            else
                h ++;
            
        }
        int maxC = max(c, max(g, h));
        int minC = min(c, min(g, h));
        int midC = c+g+h-maxC-minC;
        int res = max(maxC, max(midC*2, minC*3));
        printf("%d\n", res);
    }
    
    return 0;
}
