#include <string>
#include "stdio.h"
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdlib>
#include <cmath>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t --)
    {
        int c, m, b, r, j;
        scanf("%d %d %d %d %d", &c, &m, &b, &r, &j);
        int diff = b*r - r*c - j*c + j*m;
        int res;
        if(diff > 0)
        {
            res = 100000;
        }
        else
        {
            res = 0;
        }
        printf("%d\n", (m-res*r-b-j*res)/(c-b-j*res));
    }
    
    return 0;
}
