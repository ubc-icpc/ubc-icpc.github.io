#include <string>
#include "stdio.h"
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdlib>
#include <cmath>

using namespace std;

struct node{
    double dps;
    double hp;
    double dpshp;
    bool operator < (const node &other) const{
        return dpshp < other.dpshp;
    }
};
int main()
{
    int t;
    scanf("%d", &t);
    while(t --)
    {
        node enemy[5];
        double totaldps = 0.0;
        for(int i = 0; i < 5; i ++)
        {
            scanf("%lf", &enemy[i].hp);
            
        }
        for(int i = 0; i < 5; i ++)
        {
            scanf("%lf", &enemy[i].dps);
            totaldps += enemy[i].dps;
            enemy[i].dpshp = enemy[i].dps/enemy[i].hp;
        }
        double mine = 0;
        for(int i = 0; i < 5; i ++)
        {
            double temp;
            scanf("%lf", &temp);
            mine += temp;
        }
        sort(enemy, enemy+5);
        double damage = 0;
        for(int i = 4; i >= 0; i --)
        {
            damage += totaldps*(enemy[i].hp/mine);
            totaldps -= enemy[i].dps;
        }
        printf("%.5lf\n", damage);
    }
    
    return 0;
}
