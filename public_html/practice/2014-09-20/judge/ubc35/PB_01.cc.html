#include <iostream>
#include <iomanip>
#include <algorithm>
using namespace std;

int H[5], E[5], D[5], d;
double solve(vector<int> &v, int i) {
  if (i == 5) return 0;
  int id = v[i];
  double t = H[id]*1.0/d;
  int dt = 0;
  for (int j=i; j < 5; j++) dt += E[v[j]];
  return solve(v, i+1) + dt*t;
}
int main() {
    int T; cin >> T;
    while(T--) {
        for (int i=0; i < 5; i++) cin >> H[i];
        for (int i=0; i < 5; i++) cin >> E[i];
        for (int i=0; i < 5; i++) cin >> D[i];
        d=0;
        for (int i=0; i < 5; i++) d += D[i];
        vector<int> v(5); for (int i=0; i < 5; i++) v[i] = i;
        double m = 500*500*500.;
        do {
          m = min(m, solve(v, 0));
        } while(next_permutation(v.begin(), v.end()));
        cout << fixed << setprecision(5) << m << endl;
    }
    return 0;
}
