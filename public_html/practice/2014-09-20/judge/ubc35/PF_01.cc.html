#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int N; while(cin >> N) {
        int a[N]; for (int i=0; i < N; i++) cin >> a[i];
        for (int i=0; i < N; i++) a[i] = min(a[i], 420);
        sort(a, a+N);
        if (N == 1) cout << a[0] << endl;
        else {
          int res = 0;
          int end = N-1;
          while(end >= 3 && a[end-1] + a[0] >= 2*a[1]) {
              res += 2*a[1] + a[end] + a[0];
              end -= 2;
          }
          while(end > 1) {
              res += a[end] + a[0];
              end--;
          }
          cout << res + a[end] << endl;
        }
    }
    return 0;
}
