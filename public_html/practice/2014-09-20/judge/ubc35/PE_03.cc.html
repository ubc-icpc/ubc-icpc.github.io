#include <iostream>
#include <cmath>
#include <iomanip>
#include <cstring>
using namespace std;
typedef long double ld;
ld dp[2][500][500];
ld solve(ld X) {
    ld res = 0;
    ld diff = 0;
    ld loss = 1;
    ld cur;
    for (int i=1; ; i++) {
        diff = min(1.L, diff+X);
        cur = loss * diff;
        loss *= 1-diff;
        ld delta = i * cur;
        res += delta;
        if (delta <= 1e-6) break;
    }
    return 1/res;
}
int main() {
    ld P; while(cin >> P) {
      ld lo = 0, hi = P;
      while(hi-lo > 1e-5) {
        ld m = (hi+lo)/2;
        if (solve(m) > P) hi = m;
        else lo = m;
      }
      cout << fixed << setprecision(3) << lo << endl;
    }
    return 0;
}
