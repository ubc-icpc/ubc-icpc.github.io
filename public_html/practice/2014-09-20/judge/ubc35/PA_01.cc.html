#include <iostream>
#include <algorithm>
#include <map>
#include <vector>
using namespace std;

int main() {
    int N;
    while(cin >> N, N) {
        map<string, int> m;
        for (int i=0; i < N; i++) {
            string s; cin >> s;
            m[s]++;
        }
        vector<int> v;
        for (auto p : m) {
            v.push_back(p.second);
        }
        while(v.size() < 3) v.push_back(0);
        sort(v.rbegin(), v.rend());
        cout << max(v[0], max(2*v[1], 3*v[2])) << endl;
    }
    return 0;
}
