#include <bits/stdc++.h>
using namespace std;

int main() {
	while(true) {
		int n;
		int cheese[3];
		for(int i = 0; i < 3; i++) {
			cheese[i] = 0;
		}
		cin >> n;
		if(!n)
			break;
		while(n--) {
			string s;
			cin >> s;
			if(s == "Cheddar")
				cheese[0]++;
			else if(s == "Gouda")
				cheese[1]++;
			else if(s == "Havarti")
				cheese[2]++;
		}

		int max = 0;
		int test = 0;
		test = min(min(cheese[0], cheese[1]), cheese[2]) * 3;
		if(test > max)
			max = test;
		test = min(cheese[0], cheese[1]) * 2;
		if(test > max)
			max = test;
		test = min(cheese[0], cheese[2]) * 2;
		if(test > max)
			max = test;
		test = min(cheese[2], cheese[1]) * 2;
		if(test > max)
			max = test;
		test = cheese[0];
		if(test > max)
			max = test;
		test = cheese[1];
		if(test > max)
			max = test;
		test = cheese[2];
		if(test > max)
			max = test;

		cout << max << endl;
	}
}
