#include <bits/stdc++.h>
using namespace std;

int data[3][5];
double dph[5];
int main() {
	int t;
	cin >> t;
	while(t--) {
		double totaldamage = 0.0;
		int totalenemydps = 0;
		int totaldps = 0;
		for(int i = 0; i < 15; i++) {
			int n;
			cin >> n;
			data[i / 5][i % 5] = n;
		}
		for(int i = 0; i < 5; i++) {
			dph[i] = data[1][i] / (double)data[0][i]; 
			totaldps += data[2][i];
			totalenemydps += data[1][i];
		}
		for(int i = 0; i < 5; i++) {
			int maxdphid = 0;
			double maxdph = dph[0];
			for(int o = 1; o < 5; o++) {
				if(dph[o] > maxdph) {
					maxdph = dph[o];
					maxdphid = o;
				}
			}
			totaldamage += (double)totalenemydps * data[0][maxdphid] / totaldps;
			totalenemydps -= data[1][maxdphid];
			dph[maxdphid] = 0.0;
		}
		printf("%.5f\n", totaldamage);
	}
}
