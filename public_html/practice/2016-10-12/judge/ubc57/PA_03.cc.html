#include <bits/stdc++.h>
using namespace std;

string rules[100];
int main(){
    ios::sync_with_stdio(0);
    string s;
    int n; cin>>n; getline(cin,s);
    for (int i=1;i<=n;i++) getline(cin,rules[i]);
    int m; cin>>m;
    while (m--){
        int x; cin>>x; 
        cout<<"Rule "<<x<<": ";
        if ((x<=0)||(x>n)) cout<<"No such rule"<<endl;
        else cout<<rules[x]<<endl;
    }
    return 0;
}
