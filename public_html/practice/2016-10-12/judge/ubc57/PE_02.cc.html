#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll Fib[100];
int main(){
    ios::sync_with_stdio(0);
    Fib[0]=0,Fib[1]=1;
    int now=2;
    while (true){
        Fib[now]=Fib[now-1]+Fib[now-2];
        if (Fib[now]<0) break;
        now++;
    }
    now--;
    ll lo,hi; 
    while (cin>>hex>>lo>>hi){
        if (lo>=hi) break; 
        cout<<"Range "<<lo<<" to "<<hi<<":"<<endl;
        bool printed=false;
        for (int i=0;i<=92;i++) if ((lo<=Fib[i])&&(Fib[i]<=hi)){
            printed=true;
            cout<<"Fib"<<'('<<i<<") = "<<Fib[i]<<", "; 
            if (Fib[i]==0) cout<<"lg does not exist"<<endl;
            else cout<<"lg is "<<fixed<<setprecision(6)<<log2(Fib[i])<<endl;
            vector<ll> v; 
            ll now=Fib[i];
            while (now>1){
                if (now%2==0){
                    v.push_back(2);
                    now/=2;
                }
                else{
                    bool isPrime=true; 
                    for (ll i=3;i*i<=now;i+=2) if (now%i==0){
                        isPrime=false;
                        v.push_back(i);
                        ll res=now/i; 
                        now=res;
                        break;
                    }
                    if (isPrime){
                        v.push_back(now); break;
                    }
                }
            }
            if (v.size()==0) cout<<"No prime factors\n";
            else{
                cout<<"Prime factors:";
                for (ll x:v) cout<<" "<<x;
                cout<<endl;
            }
        }
        if (!printed) cout<<"No Fibonacci numbers in the range"<<endl;
        cout<<endl;
    }
    return 0;
}
