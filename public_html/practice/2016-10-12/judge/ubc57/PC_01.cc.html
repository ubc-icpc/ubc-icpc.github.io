#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    ll n; 
    //freopen("readable.txt","w",stdout);
    while (cin>>n){
        if (n==0) break;
        vector<pair<int,pair<int,int>>> sols;
        for (ll b=1;b<3998;b++){
            ll tot=n*b*b*b; 
            for (ll a=1;a+2*b<=4000LL;a++){
                ll c=pow(tot-a*a*a,1.0/3.0); 
                if (((a*a*a+c*c*c)==tot)&&(c>0)) if ((a>=c)&&(a+b+c+b<4000)) {
                    sols.push_back({b,{a,c}});
                    //cout<<b<<' '<<a<<' '<<c<<endl;
                }
                c++;
                if (((a*a*a+c*c*c)==tot)&&(c>0)) if ((a>=c)&&(a+b+c+b<4000)) {
                    sols.push_back({b,{a,c}});
                    //cout<<b<<' '<<a<<' '<<c<<endl;
                }
            }
        }
        if (sols.size()==0) cout<<"No value."<<endl;
        else{
            pair<int,pair<int,int>> ans=sols[0];
            //cout<<'('<<ans.second.first<<'/'<<ans.first<<')'<<"^3 + ("<<ans.second.second<<'/'<<ans.first<<')'<<"^3 = "<<n<<endl;
            for (int i=1;i<sols.size();i++) if (sols[i].first+sols[i].second.first+sols[i].second.second<ans.first+ans.second.first+ans.second.second) {
                ans=sols[i];
            }
            cout<<'('<<ans.second.first<<'/'<<ans.first<<')'<<"^3 + ("<<ans.second.second<<'/'<<ans.first<<')'<<"^3 = "<<n<<endl;
        }
    }
    return 0;
}
