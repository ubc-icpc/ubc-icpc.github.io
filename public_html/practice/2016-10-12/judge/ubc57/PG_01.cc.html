#include <bits/stdc++.h>
using namespace std;

double r(int a, int m, int &s){
 double val = (s%m) / double(m);
 s = ( a * s + 1 )%m;
 return val;
}
int main(){
    ios::sync_with_stdio(0);
    int a,m,s; cin>>a>>m>>s;
    string out;
    getline(cin,out);
    while (getline(cin,out)){
        //cout<<out<<endl;
        string ans=""; 
        for (int i=0;i<out.size();i++){
            double v=r(a,m,s); char c=out[i]; 
            if ((c<32)||(c>126)) ans+=c;
            else{
                for (int k=32;k<=126;k++){
                     int asd=int((k-32)+ceil(95-v*95))%95+32;
                     if (asd==c){
                         ans+=char(k); break;
                     }
                } 
            }
        }
        cout<<ans<<endl; 
    }
    return 0;
}
