#include <bits/stdtr1c++.h>

using namespace std;

int a, m, s;
string line, tmp;

double r() {
  double val = (s % m) / double(m);
  s = (a * s + 1) % m;
  return val;
}

int main() {
  ios_base::sync_with_stdio(0);
  cin >> a >> m >> s;
  getline(cin, tmp);
  while (getline(cin, line)) {
    for (char c : line) {
      if (c >= 0x20 && c <= 0x7e) {
        char d = (char)(((c-32) - (int)ceil(95-r()*95) + 95) %95 + 32);
        cout << d;
      } else {
        cout << c;
      }
    }
    cout << endl;
  }
}
