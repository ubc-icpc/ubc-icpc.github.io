#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;
ll A, M, S;
string L;

double r() {
  double val = (double)(S % M) / M;
  S = (A * S + 1) % M;
  return val;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> A >> M >> S;
  getline(cin, L);
  while(getline(cin, L)) {
    for (char cc : L) {
      int c = cc;
      if (c < 0x20 || c > 0x7e) {
      } else {
        c = ((c-32) + 95 - int(ceil(95-r()*95 - 1e-6)+1e-3) % 95) % 95 + 32;
      }
      cout << (char)c;
    }
    cout << endl;
  }
  return 0;
}
