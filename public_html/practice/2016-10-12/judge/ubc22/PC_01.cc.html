#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

ll I;
unordered_map<ll, pii> M;

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  for (int i = 1; i <= 4000; ++i) {
    for (int j = i;  i+j <= 4000; ++j) {
      ll t = (ll)i * (ll)i * (ll)i;
      t += (ll)j * (ll)j * (ll)j;
      if (M.count(t)) {
        pii p = M[t];
        if (p.first + p.second < i + j) continue;
      }
      M[t] = pii(i, j);
    }
  }

  while (true) {
    cin >> I;
    if (!I) break;
    int a, b, c; a = -1;
    for (int i = 1; i <= 4000; ++i) {
      ll t = (ll)i * (ll)i * (ll)i;
      if (M.count(I * t)) {
        int sa = M[I * t].second;
        int sc = M[I * t].first;
        int sb = i;
        if (sa + sc + sb > 4000) continue;
        if (a == -1 || a + b + c > sa + sb + sc) {
          a = sa; b = sb; c = sc;
        }
      }
    }
    if (a == -1) {
      cout << "No value." << endl;
    } else {
      cout << "(" << a << "/" << b << ")^3 + (" << c << "/" << b << ")^3 = " << I << endl;
    }
  }

  return 0;
}
