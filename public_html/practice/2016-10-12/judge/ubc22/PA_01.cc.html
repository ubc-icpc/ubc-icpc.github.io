#include<bits/stdtr1c++.h>
using namespace std;

int N, R, I;
string A[55];

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> N;
  getline(cin, A[54]);
  for (int i = 0; i < N; ++i) {
    getline(cin, A[i]);
  }
  cin >> R;
  for (int i = 0; i < R; ++i) {
    cin >> I;
    if (I < 1 || I > N) {
      cout << "Rule " << I << ": No such rule" << endl;
    } else {
      cout << "Rule " << I << ": " << A[I-1] << endl;
    }
  }

  return 0;
}
