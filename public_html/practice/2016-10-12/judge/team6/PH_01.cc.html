#include <iostream>
#include <cstdio>
using namespace std;

bool scan() {
	char c;
	while(1){
		cin >> c;
		switch(c){
			case '!':
				return !scan();
			case '(': {
				bool l = scan();
				cin >> c;
				bool r = scan();
				bool v;
				if (c == '|') v = l || r;
				if (c == '&') v = l && r;
				cin >> c;
				return v;
					  }
			case 't':
				return true;
			case 'f':
				return false;
		}
	}
}


int main() {
	int x; cin >> x;
	int i = 1;
	while (x--) {
		bool v = scan();
		string e, o; cin >> e >> o;
		if (v == (o[0] == 't')) {
			printf("%d: Good brain\n", i);
		} else {
			printf("%d: Bad brain\n", i);
		}
		i++;
	}
}
