#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	while (true) {
		double x;
		cin >> x;

		if (x == 0) return 0;

		double a,b,c;
		bool found = false;
		for (b = 1; b < 2000; b++) {
			for (a = 1; a < min(4000.0,(x * b)); a++) {
				double ab = pow(a/b,3);
				c = cbrt((x - ab) * pow(b,3));
				// cout << c << ' ' << round(c) << '\n';
				if (abs(c - round(c)) < 0.000001 && c <= a && c > 0 && a + b + c + b <= 4000) {
					if (pow(a,3) + pow(round(c),3) - (x * pow(b,3)) == 0) {
						c = round(c);
						found = true;
						goto end;
					}
				}
			}
		}
end:
		if (found)
			printf("(%d/%d)^3 + (%d/%d)^3 = %d\n",(int)a,(int)b,(int)c,(int)b,(int)x);
		else
			cout << "No value.\n";
	}
}
