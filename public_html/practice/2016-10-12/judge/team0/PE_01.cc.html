ull conv(string str) {
    ull res = 0;
    for (int i = 2; i < str.length(); i++) {
        char c = str[i];
        ull d = 0;
        if ('0' <= c && c <= '9') d = (c - '0');
        else d = 10+(c - 'a');
        res = res * 16 + d;
    }
    return res;
