#include <iostream>
#include <cmath>

using namespace std;
int a, m, s;
double r() {
    double val = (s%m)/(double)m;
    s = (a*s+1)%m;
    return val;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    cin >> a >> m >> s;
    string line;
    getline(cin, line);
    while (getline(cin, line)) {
        for (int i = 0; i < line.length(); i++) {
            char c = line[i];
            if (c >= 0x20 && c <= 0x7e) {
                char d = (char)(((c-32) - (int)ceil(95-r()*95+1e-8) + 95) %95 + 32);
                cout << d;
            } else {
                cout << c;
            }
        }
        cout << endl;
    }
}
