#include<bits/stdc++.h>

using namespace std;

string ss[55];

int main(){
    int n,m,t;
    cin >> n;
    getline(cin,ss[0]);
    for(int i=0;i<n;i++){
        getline(cin,ss[i+1]);
    }
    cin >> m;
    for(int i=0;i<m;i++){
        cin >> t;
        cout << "Rule "<< t << ": ";
        if (t>0 && t<=n)
            cout << ss[t] << "\n";
        else
            cout << "No such rule" << "\n";
    }
}
