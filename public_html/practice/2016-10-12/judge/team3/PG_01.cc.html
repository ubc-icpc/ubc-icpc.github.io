#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

ld r(int a, int m, int* s) {
	ld val = *s%m / ld(m);
	*s = (a * (*s) + 1)%m;
	return val;
}

int main() {
	ios::sync_with_stdio(0);
	int a,m,s; cin >> a >> m >> s;
	string inp; getline(cin, inp);
	while(getline(cin, inp)) {
		string ans = "";
		// cout << inp << endl;
		for (char c : inp) {
			int c1 = int(c);
			if (c1 >=32 && c1 <= 126) {
				int r1 = ceil(95 - r(a,m,&s) * 95);
				c1 -= 32;
				c1 = (c1 + 95 - r1)%95;
				c1 = (c1 + 32);
				//cout << r1 << " " << c1 << endl;
			}
			ans += char(c1);
			
		}
		cout << ans << endl;
	}
	
	return 0;
}
