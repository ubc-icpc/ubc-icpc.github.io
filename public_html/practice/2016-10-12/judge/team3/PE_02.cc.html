#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

int main() {
	ios::sync_with_stdio(0);
	ll lo, hi;
	vector<ll> fib;
	fib.push_back(0);
	fib.push_back(1);
	for (int i=2; i<=92; i++) {
		fib.push_back(fib[i-1] + fib[i-2]);
	}
	while(cin >> hex >> lo >> hi) {
		if (lo >= hi) break;
		cout << "Range " << lo << " " << "to " << hi << ":" << endl;
		int i=0;
		bool any = false;
		while(i <= 92 && fib[i] < lo) i++;
		while(i <= 92 && fib[i] <= hi) {
			any = true;
			cout << "Fib(" << i << ") = " << fib[i] << ", lg ";
			if (fib[i] == 0) {
				cout << "does not exist";
			} else {
				cout << fixed << setprecision(6) << "is " << log2l(fib[i]);
			}
			cout << endl;
			if (fib[i] < 2) {
				cout << "No prime factors" << endl;
			} else {
				cout << "Prime factors: ";
				ll curr = fib[i];
				string primes = "";
				while(curr%2 == 0) {
					primes += to_string(2);
					primes += " ";
					curr = curr/2;
				}
				for (int j = 3; j <= sqrt(curr); j+=2) {
			        while (curr%j == 0) {
			        	primes += to_string(j);
			        	primes += " ";
			            curr = curr/j;
			        }
			    }
			    if (curr > 2) {
			    	primes += to_string(curr);
			    	primes += " ";
			    }
			    primes.pop_back();
			    cout << primes << endl;
			}
			i++;
		}
		if (!any) cout << "No Fibonacci numbers in the range" << endl;
		cout << endl;
	}
	return 0;
}
