n = raw_input()
n = int(n)

for i in range(1,n+1):
  line = raw_input()
  expr, eq, val = line.split()
  post = ""
  for c in expr:
    if c == 't':
      post += '1'
    elif c == 'f':
      post += '0'
    elif c == '!':
      post += ' not '
    elif c == '&':
      post += ' and '
    elif c == '|':
      post += ' or '
    else:
      post += c
  if eval(post) == (val == "t"):
    print str(i) + ": Good brain"
  else:
    print str(i) + ": Bad brain"
