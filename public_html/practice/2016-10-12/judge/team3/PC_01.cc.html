#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

int main() {
	ios::sync_with_stdio(0);
	ll n;
	unordered_map<ll,pair<ll,ll>> cubesum;
	for (ll i=1; i<4000; i++) {
		for (ll j=1; j<4000-i; j++) {
			cubesum[i*i*i+j*j*j] = {i,j};
		}
	}
	while(cin >> n) {
		if (n == 0) break;
		bool valid = false;
		for (ll b=1; b<2000; b++) {
			ll curr = b*b*b*n;
			if (cubesum.count(curr) != 0) {
				ll a = cubesum[curr].first;
				ll c = cubesum[curr].second;
				if (a + 2*b + c < 4000) {
					valid = true;
					cout << "(" << a << "/" << b << ")^3 + (" << c << "/" << b << ")^3 = " << n << endl;
					break;
				}
			}
		}
		if (!valid) cout << "No value." << endl;
	}
	return 0;
}
