#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

int main() {
	ios::sync_with_stdio(0);
	int q; cin >> q;
	string inp[55];
	getline(cin, inp[0]);
	for (int i=1; i<=q; i++) {
		getline(cin, inp[i]);
	}
	int r; cin >> r;
	for (int j=1; j<=r; j++) {
		int a; cin >> a;
		cout << "Rule " << a << ": ";
		if (a >= 1 && a <= q) {
			 cout << inp[a] << endl;
		} else {
			cout << "No such rule" << endl;
		}
	}
	return 0;
}
