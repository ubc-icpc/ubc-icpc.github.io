n = int(raw_input())
for i in range(0, n):
	s = raw_input()

	s = s.replace('t', 'True')
	s = s.replace('f', 'False')
	s = s.replace('=', '==')
	s = s.replace('!', ' not ')
	s = s.replace('&', ' and ')
	s = s.replace('|', ' or ')
	x = eval(s)

	if x:
		print(str(i+1) + ': Good brain')
	else:
		print(str(i+1) +': Bad brain')
