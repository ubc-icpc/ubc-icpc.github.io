#include <iostream>
#include <string>

using namespace std;

int main() {
	int n; cin >> n;
	string s; getline(cin,s);
	string q[55];
	for(int i = 1; i <= n; i++) {
		getline(cin, q[i]);
		//cout << q[i] << endl;;
	}
	int m; cin >> m;
	while(m--) {
		int x; cin >> x;
		if(x <= 0 || x > n) {
			cout << "Rule " << x << ": No such rule"<<endl;
		}
		else {
			cout << "Rule " << x << ": " << q[x]<<endl;
		}
	}
}
