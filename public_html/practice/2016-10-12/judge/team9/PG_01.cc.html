#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int, int> pii;
typedef vector<int> vi;

#define endl '\n'

int a, m, s;

double r() {
	double val = (s % m) / (double) m;
	s = (a * s + 1) % m;
	return val;
}

int main(){
	ios::sync_with_stdio(0);
	cin >> a >> m >> s;
	string line;
	getline(cin, line);	
	while (getline(cin, line)) {
//		cout << "in: " << line << endl;
//		cout << "out:";
		for (int i = 0; i < line.length(); i++) {
			char c = line[i];
//			cout << c << " " << endl;			
			if (c < 0x20 || c > 0x7e) {
				cout << c;
				continue;
			}
			
			int rand = ceil(95 - r()*95-1e-8);
	//		cout << rand << endl;
			int limit = 32 + rand;
/*			if (c == '|') {
				cout << rand << " " << (int) c << endl;
			}*/
			if (c < limit) {
				cout << (char) (c - rand + 95);
			} else {
				cout << (char) (c - rand);
			}
		}
		cout << endl;
	}
	return 0;
}
