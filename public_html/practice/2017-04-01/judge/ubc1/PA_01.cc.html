#include <bits/stdtr1c++.h>
using namespace std;

#define MAXN 105

int r, c, d;
int mat[MAXN][MAXN];

int main(){

  ios_base::sync_with_stdio(false);
  cin.tie(0);

  cin >> r >> c >> d;

  for(int i = 1;i <= r;i++)
  for(int j = 1;j <= c;j++)
    cin >> mat[i][j];

  for(int i = 1;i <= d;i++){
    int a, b, f, g;
    cin >> a >> b >> f >> g;

    if(!mat[a][b]) continue;

    mat[a][b]--;
    mat[f][g]++;
  }

  for(int i = 1;i <= r;i++){
    for(int j = 1;j <= c;j++) cout << mat[i][j] << (j==c?"\n":" ");
    //cout << endl;
  }

  return 0;
}
