#include <bits/stdtr1c++.h>
using namespace std;

#define MAXN 100100

int n, q;
int acc[MAXN];
int call[MAXN];

int main(){

  ios_base::sync_with_stdio(false);
  cin.tie(0);

  cin >> n >> q;
  for(int i = 1;i <= n;i++) cin >> call[i];

  for(int i = 1;i <= n;i++) acc[i] = acc[i-1] + call[i];

  for(int i = 1;i <= q;i++){

    int s, t;
    cin >> s >> t;

    cout << (acc[t] - acc[s-1]) << endl;
  }

  return 0;
}
