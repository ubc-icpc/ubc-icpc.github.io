#include<bits/stdtr1c++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  int a, b, c, d; cin >> a >> b >> c >> d;
  
  int s = -1;

  if (b == c && b == 0) {
    if (a && d) {
      s = -1;
    } else if (!a && !d) {
    } else if (a) {
      s = 0;
    } else {
      s = 1;
    }
  } else if (b - c == 1 || b == c) {
    s = 0;
  } else if (c - b == 1) {
    s = 1;
  } else {
    s = -1;
  }

  if (s == -1) {
    cout << "impossible";
  } else {
    cout << s;
    while (true) {
      if (s == 0) {
        if (a) {
          cout << 0;
          a--;
        } else if (b) {
          cout << 1;
          s = 1;
          b--;
        } else break;
      } else {
        if (d && !b) {
          cout << 1;
          d--;
        } else if (c) {
          cout << 0;
          s = 0;
          c--;
        } else break;
      }
    }
    assert(a == 0 && b == 0 && c == 0 && d == 0);
  }
  
  cout << endl;

  return 0;
}
