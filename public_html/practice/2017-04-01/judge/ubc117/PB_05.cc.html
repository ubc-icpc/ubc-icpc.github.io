#include <bits/stdc++.h>
using namespace std;

int main() {
	int a, b, c, d;
	cin >> a >> b >> c >> d;
	if (b == 0 && c == 0) {
		if (a*d != 0)
			cout << "impossible" << endl;
		else {
			if (a)
				for (int i = 0; i <= a; i++)
					cout << 0;
			else
				for (int i = 0; i <= d; i++)
					cout << 1;
			cout << endl;
		}
	}
	else if (abs(b-c) > 1)
		cout << "impossible" << endl;
	else {
		if (b > c) {
			for (int i = 0; i < a; i++)
				cout << 0;
			for (int i = 0; i < b; i++)
				cout << 0 << 1;
			for (int i = 0; i < d; i++)
				cout << 1;
		}
		else if (b == c) {
			for (int i = 0; i < a; i++)
				cout << 0;
			for (int i = 0; i < b; i++)
				cout << 0 << 1;
			for (int i = 0; i < d; i++)
				cout << 1;
			cout << 0;
		}
		else {
			if (b == 0) {
				for (int i = 0; i <= d; i++)
					cout << 1;
				for (int i = 0; i <= a; i++)
					cout << 0;
			}
			else {
				cout << 1;
				for (int i = 0; i <= a; i++)
					cout << 0;
				for (int i = 0; i < c-2; i++)
					cout << 1 << 0;
				for (int i = 0; i <= d; i++)
					cout << 1;
				cout << 0;
			}
		}
		cout << endl;
	}
	return 0;
}
