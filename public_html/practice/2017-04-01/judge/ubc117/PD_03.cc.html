#include <bits/stdc++.h>
#define endl '\n'
using namespace std;
typedef long long ll;
typedef long double ld;
typedef complex<ld> pt;
const int MOD = 1e9 + 7;

vector<int> h;
int n, r, T;

bool time(int s) {
	int cur = s;
	int ans = 0;
	for (int i = 0; i < n; i++) {
		if (h[i+1] - h[i] > cur) {
			ans += r;
			cur = s;
		}
		cur -= h[i+1] - h[i];
	}
	ans += h[n] - h[0];
	return ans <= T;
}

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	cin >> n >> r >> T;
	h = vector<int>(n+1);
	h[0] = 0;
	int least = 0;
	for (int i = 0; i < n; i++) {
		cin >> h[i+1];
		least = max(least, h[i+1] - h[i]);
	}
	int l = least, r = h[n]+1;
	int mid;
	while (r - l > 0) {
		mid = (l+r)/2;
		if (time(mid)) r = mid;
		else l = mid+1;
	}
	cout << (l+r)/2 << endl;
	return 0;
}
