#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <map>

using namespace std;
#define LLI long long int

int calls[86401];
LLI sums[86401];

int mOpen;

int main()
{
	int N;
	scanf("%d %d", &mOpen, &N);
	int p;
	sums[0] = 0;
	for(int i=1;i<=mOpen;++i)
	{
		scanf("%d", &p);
		calls[i] = p;
		sums[i] = sums[i-1]+calls[i];
	}
	int s, t;
	for(int i=0;i<N;++i)
	{
		scanf("%d %d", &s, &t);
		printf("%lld\n", (LLI) sums[t]-sums[s-1]);
	}
	


}
