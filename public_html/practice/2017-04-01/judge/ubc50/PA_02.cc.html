#include <bits/stdtr1c++.h>

using namespace std;

int arr[105][105];

int main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);

  int r, c, d;
  cin >> r >> c >> d;
  for (int i = 1; i <= r; i++)
    for (int j = 1; j <= c; j++)
      cin >> arr[i][j];
  for (int i = 0; i < d; i++) {
    int r1, c1, r2, c2;
    cin >> r1 >> c1 >> r2 >> c2;
    if (arr[r1][c1] == 0) continue;
    arr[r1][c1]--; arr[r2][c2]++;
  }
  for (int i = 1; i <= r; i++) {
    for (int j = 1; j <= c; j++) {
      cout << arr[i][j];
      if (j != c) cout << " ";
    }
    cout << endl;
  }
}
