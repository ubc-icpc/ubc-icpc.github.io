#include <bits/stdtr1c++.h>

using namespace std;

int main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);

  int n, q;
  cin >> n >> q;
  vector<int> v(n+1);
  for (int i = 0; i < n; i++)
    cin >> v[i+1];
  vector<int> s(n+1);
  s[1] = v[1];
  for (int i = 2; i <= n; i++)
    s[i] = s[i-1] + v[i];
  for (int i = 0; i < q; i++) {
    int as, t; cin >> as >> t;
    cout << s[t] - s[as-1] << endl;
  }
}
