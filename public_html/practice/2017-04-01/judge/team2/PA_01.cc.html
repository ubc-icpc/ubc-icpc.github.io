#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define debug printf
typedef long long ll;

int G[101][101];

int main() {
    int r, c, d; scanf("%d%d%d", &r, &c, &d);
    for (int y = 1; y <= r; y++)
        for (int x = 1; x <= c; x++)
            scanf("%d", &G[y][x]);
    for (int i = 0; i < d; i++) {
        int r1, c1, r2, c2; scanf("%d%d%d%d", &r1, &c1, &r2, &c2);
        if (G[r1][c1])
            G[r1][c1] -= 1, G[r2][c2] += 1;
    }
    for (int y = 1; y <= r; y++) {
        for (int x = 1; x <= c; x++)
            printf(x == c ? "%d\n" : "%d ", G[y][x]);
    }
    return 0;
}
