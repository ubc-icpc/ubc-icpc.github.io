#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n, q;
	cin >> n >> q;
	
	vector<int> callcounts;
	for (int i = 0; i < n; ++i)
	{
		int c;
		cin >> c;
		callcounts.push_back(c);
	}
	for (int i = 0; i < q; ++i)
	{
		int s, t;
		cin >> s >> t;
		int out = 0;
		for (int j = s; j <= t; j++)
		{
			out += callcounts.at(j-1);
		}
		cout << out << endl;
	}
}
