#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int r;
	int c;
	int d;
	cin >> r;
	cin >> c;
	cin >> d;
	vector<vector<int> > containers;

	for (int i = 0; i < r; ++i)
	{
		vector<int> line;
		for (int j = 0; j < c; ++j)
		{
			int that;
			cin >> that;
			line.push_back(that);
		}
		containers.push_back(line);
	}

	for (int i = 0; i < d; ++i)
	{
		int r1, r2, c1, c2;
		cin >> r1 >> c1 >> r2 >> c2;
		r1-=1;
		r2-=1;
		c1-=1;
		c2-=1;
		if (containers.at(r1).at(c1) > 0)
		{
			containers.at(r1).at(c1)--;
			containers.at(r2).at(c2)++;
		}
	}

	for (int i = 0; i < r; ++i)
	{
		for (int j = 0; j < c; ++j)
		{
			cout << containers.at(i).at(j);
			if (j < c-1)
			{
				cout << " ";
			}
		}
		cout << endl;
	}
}
