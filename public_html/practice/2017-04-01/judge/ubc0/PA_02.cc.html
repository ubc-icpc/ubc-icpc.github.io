#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int G[105][105];
int main(){
    ios::sync_with_stdio(0);
    int r,c,q; cin>>r>>c>>q;
    for (int i=1;i<=r;i++) for (int j=1;j<=c;j++) cin>>G[i][j];
    while (q--){
        int a,b,x,y; cin>>a>>b>>x>>y;
        if (G[a][b]>0){
            G[a][b]--; G[x][y]++;
        }
    }
    for (int i=1;i<=r;i++){
        cout<<G[i][1];
        for (int j=2;j<=c;j++) cout<<" "<<G[i][j];
        cout<<endl;
    }
    return 0;
}
