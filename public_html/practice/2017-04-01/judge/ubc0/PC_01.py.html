a, n, m = [int(x) for x in raw_input().split()]

def modexp(p):
    if p == 0: return 1
    if p == 1: return a
    t = modexp(p/2)
    if p%2: return (t*t*a)%m
    return (t*t)%m
    
def solve(p):
    if p == 0: return 1
    if p == 1: return (1+a)%m
    t = solve(p/2)*(1+modexp(p/2+1))
    t = t % m
    if p%2 == 0:
        t = (t - modexp(p+1))%m
        t = (t+m)%m
    return t
    
print solve(n)
