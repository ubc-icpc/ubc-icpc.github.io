#include <bits/stdtr1c++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  int s00, s01, s10, s11;
  while (cin >> s00 >> s01 >> s10 >> s11) {
    // 1
    if (!s01 && !s10 && !s11) {  // s00
      for (int i = 0; i <= s00; i++)
        cout << "0";
      cout << "\n";
      continue;
    }
    if (!s00 && !s10 && !s11) {  // s01
      if (s01 > 1)
        cout << "impossible\n";
      else
        cout << "01\n";
      continue;
    }
    if (!s00 && !s01 && !s11) {  // s10
      if (s10 > 1)
        cout << "impossible\n";
      else
        cout << "10\n";
      continue;
    }
    if (!s00 && !s01 && !s10) {  // s11
      for (int i = 0; i <= s11; i++)
        cout << "1";
      cout << "\n";
      continue;
    }
    // 2
    if (abs(s01-s10) > 1) {
      cout << "impossible\n";
      continue;
    }
    // 4
    if (!s01 && !s10) {
      cout << "impossible\n";
      continue;
    }
    // 5a
    if (!s01) {
      for (int i = 0; i <= s11; i++)
        cout << "1";
      for (int i = 0; i <= s00; i++)
        cout << "0";
      cout << "\n";
      continue;
    }
    // 5b
    if (!s10) {
      for (int i = 0; i <= s00; i++)
        cout << "0";
      for (int i = 0; i <= s11; i++)
        cout << "1";
      cout << "\n";
      continue;
    }
    // 3
    if (s01 < s10) {
      cout << "1";
      s10--;
    }
    for (int i = 0; i <= s00; i++) {
      cout << "0";
    }
    s00 = 0;
    bool p1 = true;
    while (s01 && s10) {
      if (p1) {
        cout << "1";
        s01--;
      } else {
        cout << "0";
        s10--;
      }
      p1 = !p1;
    }
    if (!s10) {
      for (int i = 0; i <= s11; i++)
        cout << "1";
      cout << "\n";
      continue;
    }
    for (int i = 0; i < s11; i++)
      cout << "1";
    cout << "0\n";
  }
  return 0;
}
