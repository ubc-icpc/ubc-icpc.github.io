#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll A[50005];
ll n,T,rest;
bool good(ll m){
    ll st=0; ll tot=0;
    for (int i=1;i<=n;i++){
        if (A[i]-st>m){
            tot+=A[i-1]-st+rest; st=A[i-1]; 
        }
    }
    tot+=A[n]-st;
    //cout<<tot<<endl;
    if (tot>T) return false;
    return true;
}
int main(){
    ios::sync_with_stdio(0);
    cin>>n>>rest>>T; 
    ll l=0;
    for (int i=1;i<=n;i++) {
        cin>>A[i];
        l=max(l,A[i]-A[i-1]);
    }
    
    ll r=T;
    ll ans;
    //cout<<l<<" "<<r<<endl;
    while (true){
        if (l+1==r){
            if (good(l)) ans=l;
            else ans=r;
            break;
        }
        if (l==r){
            ans=l; break;
        }
        ll m=(l+r)/2;
        if (good(m)) r=m;
        else l=m+1;
    }
    cout<<ans<<endl;
    return 0;
}
