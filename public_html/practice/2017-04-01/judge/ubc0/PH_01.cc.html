#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll A[100000];
ll d[100000];
int main(){
    ios::sync_with_stdio(0);
    int n,q; cin>>n>>q;
    for (int i=1;i<=n;i++) cin>>A[i];
    for (int i=1;i<=n;i++) d[i]=d[i-1]+A[i];
    while (q--){
        int i,j; cin>>i>>j;
        cout<<d[j]-d[i-1]<<endl;
    }
    return 0;
}
