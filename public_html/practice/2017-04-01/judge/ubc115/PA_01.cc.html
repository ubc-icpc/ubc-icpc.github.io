#include <bits/stdc++.h>
using namespace std;

int g[128][128];
int main() {
    int r, c, d;
    cin >> r >> c >> d;
    for (int j = 0; j < r; j++) {
        for (int i = 0; i < c; i++) {
            cin >> g[j][i];
        }
    }
    int r1, c1, r2, c2;
    for (int i = 0; i < d; i++) {
        cin >> r1 >> c1 >> r2 >> c2;
        r1--; c1--; r2--; c2--;
        if (g[r1][c1]) {
            g[r1][c1]--;
            g[r2][c2]++;
        }
    }
    for (int j = 0; j < r; j++) {
        for (int i = 0; i < c - 1; i++) {
            cout << g[j][i] << ' ';
        }
        cout << g[j][c - 1];
        cout << '\n';
    }
    return 0;
}
