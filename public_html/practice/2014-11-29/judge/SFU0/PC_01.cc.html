#include <iostream>
#include <stack>
using namespace std;

int main() {
    int T;
    cin >> T;
    while (T--) {
        int n;
        cin >> n;
        stack<int> hill, branch;
        for (int i = 0; i < n; ++i) {
            int t;
            cin >> t;
            hill.push(t);
        }

        bool isok = true;
        int c = 1;
        while (c <= n) {
            if (!hill.empty() && hill.top() == c) {
                hill.pop();
                ++c;
            } else if (!branch.empty() && branch.top() == c) {
                branch.pop();
                ++c;
            } else if (hill.empty()) {
                if (branch.empty()) {
                    //...
                    isok = true;
                    break;
                } else {
                    isok = false;
                    break;
                }
            } else {
                branch.push(hill.top());
                hill.pop();
            }
        }

        if (isok) {
            cout << "Y" << endl;
        } else {
            cout << "N" << endl;
        }
    }

    return 0;
}
