#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long ll;

int main () {
    int N, P;

    cin >> N >> P;

    vector<pair<ll, ll> > v;

    ll tot = 0;
    ll cv = 0;

    for (int i = 1; i <= N; ++i) {
        ll a, b;
        cin >> a >> b;
        if (i == P) {
            tot += a;
            ++cv;
        } else {
            if (a - b >= 0)
                v.push_back(make_pair(b, a - b));
        }
    }

    sort(v.begin(), v.end());

    for (int i = 0; i < v.size(); ++i) {
        if (tot < v[i].first) {
            break;
        } else {
            tot += v[i].second;
            cv++;
        }
    }

    cout << tot << endl;
    cout << cv << endl;

    return 0;
}
