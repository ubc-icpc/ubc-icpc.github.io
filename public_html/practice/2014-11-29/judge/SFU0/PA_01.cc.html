#include <iostream>
#include <vector>
using namespace std;

int main() {
    int K;
    cin >> K;
    vector<int> v;
    for (int i = 1; i <= K; ++i)
        v.push_back(i);

    int n;
    cin >> n;
    while (n--) {
        int r;
        cin >> r;
        vector<int> tv;
        for (int i = 0; i < v.size(); ++i) {
            if ((i+1) % r != 0) {
                tv.push_back(v[i]);
            }
        }
        v = tv;
    }

    for (int i = 0; i < v.size(); ++i) {
        cout << v[i] << endl;
    }

    return 0;
}
