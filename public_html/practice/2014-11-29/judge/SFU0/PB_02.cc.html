#include <iostream>
#include <map>
#include <string>
#include <vector>
using namespace std;

int main () {
    int N;
    vector<string> v;
    map<string, string> p;
    cin >> N;
    for (int i = 0; i < N; ++i) {
        string s;
        cin >> s;
        v.push_back(s);
    }


    for (int i = 0; i < N; ++i) {
        string s;
        cin >> s;
        if (s != v[i] && p.find(s) == p.end()) {
            p[s] = v[i];
            p[v[i]] = s;
        } else {
            if (s == v[i] || p[s] != v[i]) {
                cout << "bad" << endl;
                return 0;
            }
        }
    }
    cout << "good" << endl;
    return 0;
}
