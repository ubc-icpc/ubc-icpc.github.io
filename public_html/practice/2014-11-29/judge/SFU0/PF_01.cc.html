#include <iostream>
#include <algorithm>
using namespace std;

const int maxn = 2005;

typedef long long ll;

ll d[maxn][maxn];

int main () {
    int n;
    cin >> n;
    for (int j = 0; j < n; ++j) {
        for (int i = 0; i < n; ++i) {
            char c;
            cin >> c;
            if (c == '#') {
                d[j][i] = 1;
            } else {
                d[j][i] = 0;
            }
        }
    }


    for (int j = n-2; j >= 0; --j) {
        for (int i = 1; i < n; ++i) {
            if (d[j][i] >= 1)
                d[j][i] += min(min(d[j+1][i-1], d[j+1][i]), d[j+1][i+1]);
        }
    }

    ll tot = 0;
    for (int j = 0; j < n; ++j) {
        for (int i = 0; i < n; ++i) {
            tot += d[j][i];
        }
    }

    cout << tot << endl;

}
