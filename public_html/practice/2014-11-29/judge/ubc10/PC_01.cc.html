#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
	int nc; cin >> nc;
	for (int cs = 1; cs <= nc; cs++) {
		int n; cin >> n;
		vector<int> stuff;
		for (int i = 0; i < n; i++) {
			int x; cin >> x;
			stuff.push_back(x);
		}
		reverse(stuff.begin(), stuff.end());
		int i = 0;
		int next = 1;
		vector<int> morestuff;
		bool good = true;
		while (next < n) {
			if (i < n && stuff[i] == next) {
				i++; next++;
				continue;
			}
			int j = morestuff.size();
			if (j > 0 && morestuff[j-1] == next) {
				morestuff.pop_back();
				next++;
				continue;
			}
			if (i < n) {
				morestuff.push_back(stuff[i]);
				i++;
			} else {
				good = false;
				break;
			}
		}
		if (good) cout << "Y" << endl;
		else cout << "N" << endl;
	}
}
