#include <iostream>
#include <map>

using namespace std;

int main() {
	int n; cin >> n;
	map<string, string> m;
	string s1[n];
	string s2[n];
	for (int i = 0; i < n; i++) {
		cin >> s1[i];
	}
	for (int i = 0; i < n; i++) {
		cin >> s2[i];
	}
	for (int i = 0; i < n; i++) {
		m[s1[i]] = s2[i];
		if (s1[i] == s2[i] || (m.count(s2[i]) > 0 && m[s2[i]] != s1[i])) {
			cout << "bad" << endl;
			return 0;
		}
		m[s2[i]] = s1[i];
	}
	cout << "good" << endl;
	return 0;
}
