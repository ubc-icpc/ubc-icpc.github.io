#include <bits/stdc++.h>

using namespace std;

int n;
int a[2005][2005];
int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			char x; cin >> x;
			a[i][j] = x == '#';
		}
	}

	for (int i = n-2; i >= 0; i--) {
		for (int j = 1; j < n-1; j++) {
			if (a[i][j] == 0) continue;
			else {
				a[i][j] = 1+min(a[i+1][j-1], min(a[i+1][j], a[i+1][j+1]));
			}
		}
	}

	long long ans = 0;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			ans += a[i][j];
		}
	}
	cout << ans << endl;
}
