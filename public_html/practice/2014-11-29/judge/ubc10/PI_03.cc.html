#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;
typedef long long ll;
typedef pair<ll, ll> pll;

struct cmp {
	bool operator() (const pll &p1, const pll &p2) const {
		return p1.first < p2.first || (p1.first == p2.first && p1.second > p2.second);
	}
};

int main() {
	int n, p; cin >> n >> p; p--;
	vector<pll> blah;
	ll sum = 0;
	ll count = 0;
	for (int i = 0; i < n; i++) {
		ll a, b; cin >> a >> b;
		if (i == p) {
			sum += a;
			count++;
		} else if (a - b >= 0) {
			blah.push_back(pll(b, a - b));
		}
	}
	sort(blah.begin(), blah.end(), cmp());
	for (int i = 0; i < blah.size(); i++) {
		if (sum < blah[i].first) break;
		sum += blah[i].second;
		count++;
	}
	cout << sum << endl << count << endl;
}
