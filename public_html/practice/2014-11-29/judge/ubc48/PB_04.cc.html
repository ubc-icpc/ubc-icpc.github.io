#include <iostream>
using namespace std;

int main() {
	int student, tempind;
	cin >> student;
	string s1[student], s2[student], answer, temp;
	for(int a = 0; a < student; a++){
		cin >> s1[a];
	}
	for(int a = 0; a < student; a++){
		cin >> s2[a];
	}
	for(int a = 0; a < student; a++){
		temp = s2[a];
		for(int b = 0; b < student; b++){
			if(temp == s1[b]){
				tempind = b;
				break;
			}
		}
		if(s2[tempind] == s1[a]){
			if(s2[a] != s1[a]){
				answer = "good";
			}else{
				answer = "bad";
				break;
			}
		}else{
			answer = "bad";
			break;
		}
	}
	cout << answer;
	return 0;
}
