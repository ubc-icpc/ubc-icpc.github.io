#include <iostream>
#include <vector>
using namespace std;

int main() {
	int numoftest, numofnum, vectemp, order;
	cin >> numoftest;
	char answer = 'Y';
	vector< int >branch;
	vector< int >trail;
	for(int noft = 0; noft < numoftest; noft++){
		branch.clear();
		trail.clear();
		cin >> numofnum;
		for(int nofn = 0; nofn < numofnum; nofn++){
			cin >> vectemp;
			trail.push_back(vectemp);
		}
		order = 1;
			while(true){
				if(!trail.empty() && trail.back() == order){
					trail.pop_back();
					order++;
				}else if(!branch.empty() && branch.back() == order){
						branch.pop_back();
						order++;
				}else if(trail.empty() && branch.empty()){
					cout << "Y" << endl;
					break;
				}else if(trail.empty()){
					cout << "N" << endl;
					break;
				}else{
					branch.push_back(trail.back());
					trail.pop_back();
				}
			}
	}
	return 0;
}
