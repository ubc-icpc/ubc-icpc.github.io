#include <iostream>
using namespace std;


int main(){
    int k , m, i;
    cin>>k;
    cin>>m;

    int f[k+1];
    for(int j = 0; j<= k; j++){
        f[j] = j;
    }
    while(m--){
        cin>>i;
        int n = 1;
        for(int j = 1;j<=k;j++ ){
            if(n==i && f[j]!= 0){

                f[j] = 0;
                n = 1;
            }
            if(n<i&& f[j]!= 0){
                n++;
            }
        }
    }

    for(int j = 0; j<= k; j++){
        if(f[j]!= 0){
            cout<<	j<<endl;
        }
    }
    return 0;
}
