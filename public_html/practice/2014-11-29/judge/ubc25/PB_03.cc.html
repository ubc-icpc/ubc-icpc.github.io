#include <iostream>
#include <string>
#include <vector>
#include <map>

using namespace std;

int main() {
    int N;
    cin >> N;
    if (N % 2 == 1) {
        cout << "bad" << endl;
        return 0;
    }
    
    vector<string> names;
    
    for (int i = 0; i < N; i++) {
        string name;
        cin >> name;
        names.push_back(name);
    }
    
    vector<string> partners;
    
    for (int i = 0; i < N; i++) {
        string partner;
        cin >> partner;
        partners.push_back(partner);
    }
    
    map<string, string> pairings;
    
    for (int i = 0; i < N; i++) {
        if (names[i].compare(partners[i]) == 0) {
            cout << "bad" << endl;
            return 0;
        }
        pairings[names[i]] = partners[i];
    }
    
    for (int i = 0; i < N; i++) {
        if (names[i].compare(pairings[partners[i]])) {
            cout << "bad" << endl;
            return 0;
        }
    }
            
    cout << "good" << endl;
}
