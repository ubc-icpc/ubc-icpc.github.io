#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>

using namespace std;

typedef long long int ll;
typedef pair<ll, ll> planet_t;

ll N, P;
vector<planet_t> planets;
ll fuel, visited;

bool compare (planet_t a, planet_t b) {
    return (a.first < b.first) || (a.first == b.first && a.second < b.second);
}

int main() {
    cin >> N >> P;

    for (int i = 1; i < N+1; i++) {
        ll cells, cost;
        cin >> cells >> cost;
        if (i == P) {
            fuel += cells;
            visited++;
        } else {
            planets.push_back(planet_t(cost, cells-cost));
        }
    }
    
    // sort planets
    sort(planets.begin(), planets.end(), compare);
    
    // visit all the planets
    for (auto planet : planets) {
        if (fuel >= planet.first && planet.second >= 0) {
            fuel += planet.second;
            visited++;
        }
    }
    
    cout << fuel << endl;
    cout << visited << endl;
}
