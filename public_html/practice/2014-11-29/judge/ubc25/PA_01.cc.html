#include <iostream>

using namespace std;

int friends[101];
int K;
int m;
int r_i;

void init(int max) {
    for (int i = 1; i < max + 1; i++) {
        friends[i] = 1;
    }
}

void remove(int i) {
    int count = 0;
    for (int j = 1; j < K+1; j++) {
        if (friends[j]) {
            count = (count + 1) % i;
        }
        if (count == 0) {
            friends[j] = 0;
        }
    }
}

int main() {
    cin >> K;
    init(K);
    cin >> m;
    for (int i = 0; i < m; i++) {
        cin >> r_i;
        remove(r_i);
    }
    for (int i = 1; i < 101; i++) {
        if (friends[i]) {
            cout << i << endl;
        }
    }
}
