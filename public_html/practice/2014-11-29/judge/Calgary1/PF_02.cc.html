#include <bits/stdc++.h>
using namespace std;

int dp[2000][2000], ans = 0;
string grid[2000];

int main() {
	int N;
	cin >> N;
	for (int n = 0; n < N; n++) cin >> grid[n];
	for (int r = N-1; r >= 0; r--) {
		for (int c = 0; c < N; c++) {
			if (grid[r][c] == '#') {
				dp[r][c] = 1;
				if (r != N-1 && c != 0 && c != N-1) {
					dp[r][c] += min(dp[r+1][c-1], min(dp[r+1][c], dp[r+1][c+1]));
				}
				ans += dp[r][c];
				//cout << "dp[" << r << "][" << c << "] = " << dp[r][c] << endl;
			}
		}
	}
	cout << ans << endl;
}
