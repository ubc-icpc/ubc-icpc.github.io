import java.util.*;

public class S2 {

	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		boolean good = true;
		int pairs = in.nextInt();
		ArrayList<String> list1 = new ArrayList<String>(pairs);
		ArrayList<String> list2 = new ArrayList<String>(pairs);
		for (int i = 0; i < pairs; i++)
			list1.add(in.next());
		for (int i = 0; i < pairs; i++)
			list2.add(in.next());

		// case1 partner of itself
		for (int i = 0; i < list1.size(); i++)
			if (list1.get(i).equals(list2.get(i)))
				good = false;
		// case2 different partners
		for (int i = 0; i < list1.size(); i++)
			for (int j = 0; j < list2.size(); j++) {
				String onesPartner = list2.get(i);
				String twosPartner = list1.get(j);
				if (list2.get(j).equals(list1.get(i))
						&& (!(onesPartner.equals(twosPartner))))
					good = false;

			}

		if (good)
			System.out.println("good");
		else
			System.out.println("bad");
	}

}
