#include<iostream>
#include<vector>
#include<stack>
using namespace std;

int main() {
  int T;
  cin >> T;
  while (T--) {
    int N;
    cin >> N;
    stack<int> s1;
    stack<int> s2;
    while (N--) {
      int temp;
      cin >> temp;
      s1.push(temp);
    }
    int expect = 1;
    bool f = true;
    while (f) {
      if (!s2.empty() && s2.top() == expect) {
	s2.pop();
	expect++;
      }
      else if (!s1.empty() && s1.top() == expect) {
	s1.pop();
	expect++;
      }
      else if (s1.empty() && s2.empty())
	break;
      else if (s1.empty() && s2.top() != expect)
	f = false;
      else {
	int temp;
	temp = s1.top();
	if (!s2.empty() && temp > s2.top())
	  f = false;
	else
	  s2.push(temp);
	s1.pop();
      }
    }
    if (f) cout << "Y" << endl;
    else cout << "N" << endl;
  }
  return 0;
}
