#include<iostream>
#include<vector>
#include<map>
#include<string>
#include<sstream>
using namespace std;

string line;
stringstream ss1;
stringstream ss2;

int main () {
  
  map<string, string> m;
  int N;
  getline(cin, line);
  ss1 << line;
  ss1 >> N;
  ss1.clear();
  string l1, l2;
  getline(cin, l1);
  getline(cin, l2);
  ss1 << l1;
  ss2 << l2;
  while (N--) {
    string t1, t2;
    ss1 >> t1;
    ss2 >> t2;
    m[t1] = t2;
  }
  bool f = true;
  for (map<string, string>::iterator it = m.begin(); f && it != m.end(); ++it) {
    if (it->first != m[it->second] || it-> first == it ->second) {
      f = false;
    }
  }
  if (f) cout << "good" << endl;
  else cout << "bad" << endl;
  return 0;
}
