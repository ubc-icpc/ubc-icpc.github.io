#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

struct planet {
  int cost;
  int netprofit;
  bool operator<(const planet &b) const {
    return cost < b.cost;
  }
};

int main() {
  int N, P;
  cin >> N >> P;
  vector<planet> l;
  int fuel = 0;
  int planetcount = 1;
  for (int i = 1; i <= N; i++) {
    planet temp;
    cin >> temp.netprofit >> temp.cost;
    if (i == P) {
      fuel += temp.netprofit;
      continue;
    }
    temp.netprofit -= temp.cost;
    if (temp.netprofit >= 0)
      l.push_back(temp);
  }
  sort(l.begin(), l.end());
  for (planet p : l) {
    if (p.cost <= fuel) {
      fuel += p.netprofit;
      planetcount++;
    } else break;
  }
  cout << fuel << endl << planetcount << endl;
  return 0;
}
