#include <bits/stdc++.h>
using namespace std;

char p[1024];
int main() {
        string s;
        while (cin >> s && s.compare("0") != 0) {
                vector<int> v;
                for (int i = 0; i < s.length(); i++) {
                        p[i] = s[i] - '0';
                }
                for (int i = 1;  i < s.length(); i++) {
                        if (p[i] == 1) {
                                v.push_back(i);
                                for (int j = i; j < s.length(); j += i) {
                                        p[j] ^= 1;
                                }
                        }
                }
                for (int e : v) {
                        cout << e << ' ';
                }
                cout << '\n';
        }
        return 0;
}
