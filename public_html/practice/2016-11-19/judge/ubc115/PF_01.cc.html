#include <bits/stdc++.h>
using namespace std;

struct pt {
        long long x, y, z;
        bool operator<(const pt &other) const {
                return x < other.x;
        }
};


inline double dis(pt a, pt b) {
        return sqrt((double) ((a.x - b.x) * (a.x - b.x) + 
                        (a.y - b.y) * (a.y - b.y) + 
                        (a.z - b.z) * (a.z - b.z)));
}

int main() {
        ios::sync_with_stdio(0);
        int n, k;
        long long x, y, z;
        while (cin >> n >> k && n && k) {
                vector<pt> v;
                for (int i = 0; i < n; i++) {
                        cin >> x >> y >> z;
                        v.push_back({x, y, z});
                }
                sort(v.begin(), v.end());
                int count = 0;
                for (int i = 0; i < v.size(); i++) {
                        for (int j = i + 1; j < v.size(); j++) {
                                if (v[j].x - v[i].x > (double) k) {
                                        break;
                                }

                                if (dis(v[j], v[i]) < (double) k) {
                                        count++;
                                }
                        }

                }
                cout << count << '\n';
                
        }
        return 0;
}
