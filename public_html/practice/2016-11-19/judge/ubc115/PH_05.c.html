#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define a3(a, b, c) ((a.x - c.x) * (b.y - a.y) - (a.x - b.x) * (c.y - a.y) ) \
 
typedef struct pt {
        int x, y;
} pt;

pt v[2048];

int main() {
        int n, x, y;
        while (scanf("%d", &n) && n) {
                int i, j, k;
                for (i = 0; i < n; i++) {
                        scanf("%d %d", &x, &y);
                        v[i].x = x;
                        v[i].y = y;
                }
                int ma = 0;
                int mi = 0x3f3f3f3f;
                int a = 0;
                for (i = 0; i < n; i++) {
                        for (j = i + 1; j < n; j++) {
                                for (k = j + 1; k < n; k++) {
                                        a = a3(v[i], v[j], v[k]);
                                        if (a < 0) a *= -1;
                                        if (a > ma) {
                                                ma = a;
                                        }
                                        if (a < mi) {
                                                mi = a;
                                        }
                                }
                        }
                }
                printf("%.1f %.1f\n", mi / 2.0, ma / 2.0);
        }
        return 0;
}
