#include <algorithm>
#include <iostream>

using namespace std;

typedef long long ll;
struct pt {
  ll x, y, z;
  pt() {}
  pt(ll a, ll b, ll c) : x(a), y(b), z(c) {}
  bool operator<(const pt &other) const {
    if (x == other.x) {
      if (y == other.y) {
        return z < other.z;
      }
      return y < other.y;
    }
    return x < other.x;
  }
};

ll n, k;
vector<pt> p;

int main() {
  ios_base::sync_with_stdio(0);
  while (cin >> n >> k && n != 0 && k != 0) {
    p.clear();
    p.resize(n);
    for (int i = 0; i < n; i++) cin >> p[i].x >> p[i].y >> p[i].z;
    sort(p.begin(), p.end());
    ll ans = 0;
    for (int i = 0; i < n; i++) {
      for (int j = i+1; j < n && p[j].x <= p[i].x+k; j++) {
        if (abs(p[j].y-p[i].y <= k)
            && abs(p[j].y-p[i].y <= k)
            && abs(p[j].z-p[i].z <= k)
            && abs(p[j].z-p[i].z <= k)
            && (p[j].x-p[i].x)*(p[j].x-p[i].x)
            + (p[j].y-p[i].y)*(p[j].y-p[i].y)
            + (p[j].z-p[i].z)*(p[j].z-p[i].z) < k*k) ans++;
      }
    }
    cout << ans << endl;
  }
}
