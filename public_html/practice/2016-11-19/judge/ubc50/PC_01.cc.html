#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main() {
  ios_base::sync_with_stdio(0);
  string s;
  while (cin >> s && s != "0") {
    vector<int> tmp, ans;
    for (int i = 0; i < s.size(); i++)
      tmp.push_back(0);
    for (int i = 1; i < s.size(); i++) {
      char cur = s[i];
      if ((cur == '1' && tmp[i] == 0)
          || (cur == '0' && tmp[i] == 1)) {
        ans.push_back(i);
        for (int j = i; j < s.size(); j += i) {
          tmp[j] = (tmp[j]+1)%2;
        }
      }
    }
    for (int i = 0; i < ans.size() - 1; i++) {
      cout << ans[i] << " ";
    }
    cout << ans[ans.size() - 1] << endl;
  }
}
