#include <bits/stdc++.h>
using namespace std;

int main(){
	string a;
	bool oddNumberOfSatellites, ping, noSpaces;
	int i, j;
	std::vector<int> numbers;

	do{
		cin>>a;
		noSpaces = true;
		for(i = 0; i < a.length(); i += 1){
			if(i == 0){/*do nothing...*/}
			else{
				//set ping to 0 initially
				ping = false;

				//compute expected value for ping
				for(j = 0; j < numbers.size(); j += 1){
					if (i % numbers[j] == 0)
					{
						//flip value of ping 
						//(ie: change from 0 to 1 or vice-versa)
						ping = !ping;
					}
				}

				//check if actual != expected
				if((a[i] == '1') != ping){
					//then we have a new satellite
					numbers.push_back(i);
					
					if(noSpaces){
						cout<<i;
						noSpaces = false;
					}
					else{
						cout<<" "<<i;
					}
				}
			}
		}
		cout<<endl;	
		
		numbers.clear();
	}while(a.compare("0") != 0);

	return 0;
}
