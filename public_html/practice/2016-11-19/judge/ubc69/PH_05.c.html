#include <stdlib.h>
#include <stdio.h>
#include <math.h>

int main() {
    int points[2000][2];
    int area, maxi, mini;
	int n, i, j, k;
    scanf("%d", &n);
	while(n != 0) {
        maxi = 0;
        mini = 1e9;
		for(i = 0; i < n; i++)
            scanf("%d %d", &points[i][0], &points[i][1]);
		for(i = 0; i < n; i++) {
			for(j = i + 1; j < n; j++) {
				for(k = j + 1; k < n; k++) {
					area = (points[i][0] - points[k][0])*(points[j][1] - points[i][1]) - (points[i][0] - points[j][0])*(points[k][1] - points[i][1]);
                    if (area < 0) area *= -1;
                    if (area < mini) {
                        mini = area;
                    }
                    if (area > maxi) {
                        maxi = area;
                    }
				}
			}
		}
        printf("%.1f %.1f\n", mini / 2.0, maxi / 2.0);
        scanf("%d", &n);
	}
	return 0;
}
