#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main() {
	string s; cin >> s;
	while(!(s == "0")) {
		vector<int> pings;
		for(int i = 1; i < s.size(); i++) {
			int count = 0;
			for(int j = 0; j < pings.size(); j++) {
				if(i%pings[j] == 0) count++;
			}
			if(s[i] - '0' != count%2) {
				pings.push_back(i);
			}
		}
		for(int i = 0; i < pings.size() - 1; i++) {
			cout << pings[i] << ' ';
		}
		cout << pings[pings.size() - 1] << endl;
		cin >> s;
	}
}
