#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
typedef pair<pii, pii> piipii;
typedef pair<piipii, bool> piipii_O;
typedef vector<ll> vi;
typedef vector<vi > vvi;
typedef vector<bool> vb;
typedef vector<vb > vvb;

int main(){
	vb present(1001, false);
	string heard;
	while (true) {
		vi satellites;
		cin >> heard;
		if (heard == "0") 
			break;
		for (int i=1; i<heard.size(); i++) {
			int modstate = 0;			
			for (int j=1; j<i; j++) {
				if (i%j == 0 && present[j]) { //number of divisors
					modstate++;
					modstate %= 2;
				}
			}
			if ( modstate == (heard[i]-'0')) {
				present[i] = false;
			} else {
				present[i] = true;
				satellites.push_back(i);
			}
		}
		for (int satellite: satellites)
			cout << satellite << " ";
		cout << endl;
	}

	
}
