import java.util.LinkedList;
import java.util.List;
import java.util.Scanner;

public class Main {

    static List<Integer> answer = new LinkedList<Integer>();
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String nextS = sc.nextLine();

        while (!(nextS.equals("0"))) {
            List<Integer> l = new LinkedList<Integer>();
            for (int a = 0; a < nextS.length(); a++) {
                int c = Character.getNumericValue(nextS.charAt(a));
                l.add(c);
            }
                Solve(l);
                for (int b = 0; b < answer.size(); b++) {
                    if (b == answer.size() - 1) {
                        System.out.println(answer.get(b));
                    } else {
                        System.out.print(answer.get(b) + " ");
                    }
                }
            answer.clear();
            nextS = sc.nextLine();
        }
    }
    private static void Solve(List<Integer> l) {
        if (l.get(1) == 1) {
            answer.add(1);
        }
        for (int a = 2; a < l.size(); a++) {
            int tryy = l.get(a);
                for (int b : answer) {
                    if (a % b == 0) {
                        if (tryy == 0) {
                            tryy = 1;
                        }
                        else {
                            tryy = 0;
                        }
                    }
                }
                if (tryy == 1) {
                    answer.add(a);
                }
            }
        }
}
