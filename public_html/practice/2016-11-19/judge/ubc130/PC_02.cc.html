#include <bits/stdc++.h>
#define endl '\n'
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	string s;
	vector<bool> ping;
	vector<int> satellites;
	while (cin >> s) {
		if (s.length() == 1)
			break;
		ping = vector<bool>(s.length(), 0);
		for (int i = 1; i < s.length(); i++) {
			if (s[i] == '1' != ping[i]) {
				satellites.push_back(i);
				for (int j = i; j < s.length(); j += i) {
					ping[j] = !ping[j];
				}
			}
		}
/*		if (s[0] == '1')
			satellites.erase(satellites.begin());*/
		for (int i : satellites) {
			cout << i << " ";
		}
		cout << endl;	
		satellites.clear();
	}
	return 0;
}
