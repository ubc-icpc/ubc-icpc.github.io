#include <iostream>

using namespace std;

int main() {
    string s; cin >> s;
    while (s != "0") {
        int n = s.length();
        string lol(n, '0');
        int cnt = 0;
        for (int i = 1; i < n; i++) {
            if (s[i] != lol[i]) {
                if (cnt > 0) cout << ' ';
                cout << i;
                cnt++;

                for (int j = i; j < n; j += i) {
                    if (lol[j] == '0') lol[j] = '1';
                    else lol[j] = '0';
                }
            }
        }
        cout << endl;

        cin >> s;
    }
}
