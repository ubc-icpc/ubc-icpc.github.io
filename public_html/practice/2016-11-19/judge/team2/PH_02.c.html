#include<stdio.h>
#include<stdlib.h>
#define add_det(i,j) (x[i]*y[j]-x[j]*y[i])
int x[2001],y[2001],n;
int main(){
    scanf("%d",&n);
    while(n!=0){
        int i,j,k,a,mina,maxa;
        for(i=0;i<n;i++){
            scanf("%d%d",&x[i],&y[i]);
        }
        mina=maxa=abs(add_det(0,1)+add_det(1,2)+add_det(2,0));
        for(i=3;i<n;i++){
            for(j=1;j<i;j++){
                for(k=0;k<j;k++){
                a = abs(add_det(i,j)+add_det(j,k)+add_det(k,i));
                if(a<mina) mina = a;
                else if(a>maxa) maxa= a;
                }
            }
        }
        printf("%.1f %.1f\n", mina/2.0, maxa/2.0);
        scanf("%d",&n);
    }
}
