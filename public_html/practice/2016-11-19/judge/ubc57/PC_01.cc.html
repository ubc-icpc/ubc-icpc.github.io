#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    string s;
    while (cin>>s){
        if (s.size()==1) break;
        vector<int> v;
        for (int i=1;i<s.size();i++){
            int res=0; 
            for (int x:v) if (i%x==0) res=!res; 
            if (res!=(s[i]-'0')) v.push_back(i);
        }
        //if ((s[0]-'0')!=(v.size()%2))
        for (int i=0;i<v.size()-1;i++) cout<<v[i]<<' '; cout<<v[v.size()-1]<<endl;
    }
    return 0;
}
