import fileinput

def process(line):
    pings = map(int, line)
    pings[0] = 0
    output = []
    while True:
        try:
            index = pings.index(1)
        except:
            break
        output.append(index)
        for i in xrange(index, len(pings), index):
            pings[i] = pings[i] ^ 1
    return " ".join(map(str, output))

for line in fileinput.input():
    line = line.strip()
    if line == '0':
        break
    print process(line)
