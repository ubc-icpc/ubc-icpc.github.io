#include <stdio.h>
#include <stdlib.h>

typedef struct {
   long x;
   long y;
   long z;
} point;

int compar(const void *a, const void *b) {
   return ((point*)a)->x - ((point*)b)->x;
}

int main() {
   long n, k2, k, count;
   int i, j;
   while (1) {
      scanf("%d %d", &n, &k);
      if (n == 0)
         return 0;
      point p[n];
      count = 0;
      for (i = 0; i < n; i++)
         scanf("%ld %ld %ld", &p[i].x, &p[i].y, &p[i].z);
      k2 = k*k;
      qsort(p,n,sizeof(point),compar);
      for (i = 0; i < n; i++) {
         for (j = i+1; p[j].x-p[i].x < k && j < n; j++) {
            if (p[i].y-p[j].y < k && p[i].z-p[j].z < k)
               if ((p[i].x-p[j].x)*(p[i].x-p[j].x)+(p[i].y-p[j].y)*(p[i].y-p[j].y)+(p[i].z-p[j].z)*(p[i].z-p[j].z) < k2)
                  count++;
         }
      }
      printf("%ld\n", count);
   }
   return 0;
}
