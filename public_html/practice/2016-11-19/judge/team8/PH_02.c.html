#include <stdio.h>
#include <stdlib.h>

typedef struct {
   int x;
   int y;
} point;

int main() {
   int points, min, max, area;
   int i, j, k;
   while(1) {
      scanf("%d", &points);
      if (!points)
         return 0;
      point p[points];
      for (i = 0; i < points; i++)
         scanf("%d %d", &p[i].x, &p[i].y);
      min = max = abs((p[2].x*p[1].y + p[1].x*p[0].y + p[0].x*p[2].y)
                -(p[2].y*p[1].x + p[1].y*p[0].x + p[0].y*p[2].x));
      for (i = 3; i < points; i++) {
         for (j = 1; j < i; j++) {
            for (k = 0; k < j; k++) {
               area = abs((p[i].x*p[j].y + p[j].x*p[k].y + p[k].x*p[i].y)-(p[i].y*p[j].x + p[j].y*p[k].x + p[k].y*p[i].x));
               if (area > max)
                  max = area;
               else if (area < min)
                  min = area;
            }
         }
      }
      printf("%.1f %.1f \n", min/2.0, max/2.0);
   }
   return 0;
}
