#include <stdio.h>

int main() {
   char s[1001];
   int i, j, pos, ping;
   int satl[1001];
   while (1) {
      pos = 0;
      scanf("%s", s);
      if (s[1] == '\0' && s[0]-'0' == 0)
         return 0;
      for (i = 1; s[i] != '\0'; i++) {
         for (ping = j = 0; j < pos; j++) {
            if (i%satl[j] == 0) {
               ping++;
            }
         }
         if (ping%2 != (s[i]-'0'))
            satl[pos++] = i;
      }
      for (j = 0; j < pos; j++)
         printf("%d ", satl[j]);
      printf("\n");
   }
}
