#include <iostream>

using namespace std;

#define FOR(i, n) for(int i = 0; i < (n); i++)
#define SZ(A) ((int)(A).size())
#define PB push_back

#define show(x) cerr<<"XXXX----> "<<#x<<" = "<<x<<endl;

int movY[]={-1,0,1,0};
int movX[]={0,-1,0,1};

int n, m;
char gr[100][100];

void find(){
	int nx, ny;
	FOR(i, n)FOR(j, m)
		if (gr[i][j] == '*'){
			nx = i;
			ny = j;
		}

	int dir;
	if (nx == 0)
		dir = 3;
	if (nx == n-1)
		dir = 1;
	if (ny == 0)
		dir = 2;
	if (ny == m-1)
		dir = 0;

	while(true){
		nx += movX[dir];
		ny += movY[dir];
		if (gr[nx][ny] == 'x')	break;
		if (gr[nx][ny] == '.')	continue;
		if (gr[nx][ny] == '/'){
			if (dir == 0)
				dir = 3;
			else if (dir == 3)
				dir = 0;
			else if (dir == 1)
				dir = 2;
			else if (dir == 2)
				dir = 1;
		}
		if (gr[nx][ny] == '\\'){
			if (dir == 0)
				dir = 1;
			else if (dir == 1)
				dir = 0;
			else if (dir == 2)
				dir = 3;
			else if (dir == 3)
				dir = 2;
		}

	}

	gr[nx][ny] = '&';
}

int main(){
	int t = 0;
	while(cin >> m >> n && n){
		FOR(i, n)FOR(j, m)
			cin >> gr[i][j];

		find();

		cout << "HOUSE " << ++t << endl;
		FOR(i, n){
			FOR(j, m)
				cout << gr[i][j];
			cout << endl;
		}

	}
	return 0;
}
