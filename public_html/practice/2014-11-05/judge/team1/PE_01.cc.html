#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

int c[100], wl[100];

int main()
{
    int w, n, T = 1;
    while (cin >> w >> n && w)
    {
        string s;
        int t = 0, m = 0;
        for (int i = 0; i < n; i++)
        {
            cin >> s >> c[i];
            wl[i] = s.size();
            t += c[i];
            m = max(m, c[i]);
        }
        int h = 0, lw = 0, lh = 0;
        for (int i = 0; i < n; i++)
        {
            int p = 8 + ceil(40.0*(c[i]-4.0)/(m-4.0));
            int ww = ceil(9.0*wl[i]*p/16.0);
            if (lw+ww <= w)
            {
                lw += ww+10;
                lh = max(lh, p);
            }
            else
            {
                h += lh;
                lw = ww+10;
                lh = p;
            }
        }
        cout << "CLOUD " << T++ << ": " << h+lh << "\n";
    }
    return 0;
}
