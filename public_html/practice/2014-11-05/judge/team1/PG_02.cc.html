char toChar(int k){
	if (k < 26)
		return 'A' + k;
	if (k == 26)	return '_';
	if (k == 27)	return '.';
}

int toInt(char ch){
	if (ch == '_')	return 26;
	if (ch == '.')	return 27;
	return ch-'A';
}
