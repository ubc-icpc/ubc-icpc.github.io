#include <iostream>
#include <string>
#include <vector>
using namespace std;
string s = "ABCDEFGHIJKLMNOPQRSTUVWXYZ_.";

int main() {
    int rot;
    cin >> rot;
    while (rot) {
        string t;
        cin >> t;
        for (int i = t.size() - 1; i >= 0; i--) {
            int a = t[i] - 'A';
            if (t[i] == '.')
                a = 27;
            if (t[i] == '_')
                a = 26;
            
            cout << s[(a + rot) % 28];
        }
        cout << endl;
        
        cin >> rot;
    }
}
