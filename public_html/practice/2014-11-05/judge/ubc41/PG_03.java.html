import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;


public class rot {
	public static void main (String[] args){
		Scanner scan = new Scanner(System.in);
		String delim = "[ ]+";
		String[] linearray;
		String line, abc;
		int n, linelength, newchar;
		List<String> answer_list = new ArrayList<String>();
		while (true){
			String cba = new String();
			String answer = new String();
			line = scan.nextLine();
			linearray = line.split(delim);
			n = Integer.parseInt(linearray[0]);
			
			if (n == 0) {
				break;
			}
			abc = linearray[1];
			linelength = abc.length();
			
			for (int i=linelength-1; i >= 0; i--) {
				cba = cba + abc.charAt(i);
			}
			
			for (int i=0; i<linelength; i++) {
				newchar = (int)cba.charAt(i);
				if (newchar == 46) {
					newchar = 27;
				} else if (newchar == 95) { 
					newchar = 26;
				} else {
					newchar = newchar - 65;
				}
				
				newchar = (newchar + n) % 28;
				
				if(newchar == 27) {
					newchar = 46;
				} else if (newchar == 26) {
					newchar = 95;
				} else {
					newchar = newchar + 65;
				}
				
				answer = answer + (char)newchar;
			}
			
			answer_list.add(answer);
		}
		
		for (int i=0; i< answer_list.size() -1; i++) {
			System.out.print(answer_list.get(i) + "\n");
		}
		System.out.print(answer_list.get(answer_list.size()-1));
		scan.close();
	}
}
