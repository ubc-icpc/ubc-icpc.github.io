#include <iostream>

using namespace std;

char enc(char c, int shift) {
	int n;
	if (c >= 'A' && c <= 'Z') {
		n = (int)(c - 'A');
	} else if (c == '_') {
		n = 26;
	} else {
		n = 27;
	}
	n = (n + shift) % 28;
	if (n < 26) {
		return 'A' + n;
	} else if (n == 26) {
		return '_';
	}
	return '.';
}

int main() {
	int n; cin >> n;
	while (n > 0) {
		string s; cin >> s;
		for (int i = s.length() - 1; i >= 0; i--) {
			cout << enc(s[i], n);
		}
		cout << endl;
		cin >> n;
	}
}
