#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int cmax = 0;
ll psize(int cw) {
	return 8 + (40*(cw-4) + (cmax-5))/(cmax-4);
}

ll width(int t, int P) {
	return (9*t*P+15)/16;
}

typedef pair<string, int> psi; //wow pressure is on
int main() {
	int W, N, c = 0;
	while (cin >> W >> N) {
		if (W+N == 0) break;
		vector<psi> v(N);
		cmax = 0;
		for (int i = 0; i < N; i++) {
			 cin >> v[i].first >> v[i].second;
			 cmax = max(cmax, v[i].second);
		}
		
		ll curr = 0, tot_ht = 0, curr_ht = 0;
		for (int i = 0; i < N; i++) {
			ll word_ht = psize(v[i].second);
			ll word_len = width(v[i].first.size(), word_ht);
			if (curr + word_len > W) {
				tot_ht += curr_ht;
				curr_ht = 0;
				curr = 0;
			}
			curr_ht = max(curr_ht, word_ht);
			curr += word_len;
			curr += 10;
		}
		tot_ht += curr_ht;
		cout << "CLOUD " << ++c << ": " << tot_ht << endl;
	}
}
