#include <iostream>
#include <string>

using namespace std;

int main(){
    int n; 
    string s;
    while(cin>>n){
        string ha = "ABCDEFGHIJKLMNOPQRSTUVWXYZ_.";
        cin >> s;
        if(n == 0) break;
        for(int i = s.size() - 1; i>=0; i--){
            if(s[i] == '_'){
                cout << ha[(n+26)%28];
                continue; 
            }
            if(s[i] == '.'){
                cout << ha[(n+27)%28]; 
                continue;
            }
            cout<<ha[((s[i]-'A') + n)% 28];
        }
        cout<<endl;
    }
    return 0;
}
