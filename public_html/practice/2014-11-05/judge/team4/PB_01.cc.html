#include <iostream>
#include <complex>

using namespace std;

char room[20][20];

int main () {
    int W, L;
    int r, c;
    int H = 0;
    while (cin >> W >> L) {
        if (W == 0) break;
        H++;
        cout << "HOUSE " << H << endl;
        int dr = 0, dc = 0;
        for (int l = 0; l < L; ++l) {
            for (int w = 0; w < W; ++w) {
                cin >> room[l][w];
                if (room[l][w] == '*') {
                    r = l;
                    c = w;
                }
            }
        }
        // Find initial dx dy
        if (r == 0) {
            dr = 1;
        }
        else if (r == L - 1) {
            dr = -1;
        }
        else if (c == 0) {
            dc = 1;
        }
        else {
            dc = -1;
        }
        while (room[r][c] != 'x') {
            if ((dr == 0 && room[r][c] == '/') || (dc == 0 && room[r][c] == '\\')) {
                int tempr = dr;
                dr = -dc;
                dc = tempr;
            }
            else if ((dc == 0 && room[r][c] == '/') || (dr == 0 && room[r][c] == '\\')){
                int tempr = dr;
                dr = dc;
                dc = -tempr;
            }
            r += dr;
            c += dc;
        }
        room[r][c] = '&';
        for (int l = 0; l < L; ++l) {
            for (int w = 0; w < W; ++w) {
                cout << room[l][w];
            }
            cout << endl;
        }
    }
    return 0;
}
