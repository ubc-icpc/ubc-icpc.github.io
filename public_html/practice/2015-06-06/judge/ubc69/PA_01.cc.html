#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	string N;
	cin>>N;
	int digits[10]={0};
	int sum = 0;
	for(int i=0;i<N.size();i++){
		digits[N[i]-'0']++;
		sum+=N[i]-'0';
	}
	if(sum%3!=0||digits[0]==0){
		cout<<-1<<endl;
		return 0;
	}
	for(int i=9;i>=0;i--){
		for(int j=0;j<digits[i];j++)
			cout<<i;
	}
	cout<<endl;
    return 0;
}
