#include<bits/stdtr1c++.h>
using namespace std;

int N;
map<int, int> H;

int main () {
  cin >> N;
  while (N--) {
    int temp; cin >> temp;
    ++H[temp];
  }
  while (H.size() > 3) {
    map<int, int>::iterator begin, end;
    begin = H.begin(); end = prev(H.end());
    int m = min(begin->second, end->second);
    begin->second -= m;
    end->second -= m;
    next(begin)->second += m;
    prev(end)->second += m;
    if (!begin->second) H.erase(begin);
    if (!end->second) H.erase(end); 
  }
  if (H.size() < 3) cout << "Slavko" << endl;
  else if (H.begin()->second > prev(H.end())->second) {
    cout << "Slavko" << endl;
    H.erase(prev(H.end()));
  } else {
    cout << "Mirko" << endl;
    H.erase(H.begin());
  }
  cout << H.begin()->first << " " << prev(H.end())->first << endl;
  return 0;
}
