#include<bits/stdtr1c++.h>
using namespace std;

string S;
int T, C[10];
int main () {
  cin >> S;
  for (int i = 0; i < S.size(); ++i) {
    T += S[i] - '0'; C[S[i] - '0']++; }
  if (!(T % 3) && C[0])
    for (int i = 9; i >= 0; --i)
      while (C[i]--) cout << i;
  else cout << -1 << endl;
  return 0;
}
