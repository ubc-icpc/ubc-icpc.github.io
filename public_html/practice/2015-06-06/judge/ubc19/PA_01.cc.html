#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <algorithm>
#include <cmath>
#include <map>
#include <sstream>
#include <complex>
#include <vector>
#include <set>

using namespace std;

int main()
{ 
  string s; cin >> s;
  bool has_zero = false;
  int sum = 0;
  int nums [s.length()];
  for (int i=0; i<s.length(); i++) {
    string c = s.substr(i, 1);
    if (c == "0") has_zero = true;
    sum += stoi(c);
    nums[i] = stoi(c);
  }
  
  if (sum%3 != 0 || !has_zero) {
    cout << "-1" << endl;
    return 0;
  }
  
  sort(nums, nums+s.length(), std::greater<int>());
  
  for (int i=0; i<s.length(); i++) {
    cout << nums[i];
  }
  cout<< endl;
  return 0;
}
