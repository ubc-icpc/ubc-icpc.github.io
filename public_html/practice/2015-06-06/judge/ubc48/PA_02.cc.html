#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	string n; cin >> n;
	vector < int > el, result;
	for (int i = 0; i < n.size(); ++i){
		el.push_back(n[i] - '0');
	}
	sort(el.begin(), el.end());
	if (el[0] != 0){
		cout << -1 << endl;
		return 0;
	}
	int well = 0;
	for (int i = 0; i < el.size(); ++i){
		well += el[i];
	}
	if (well % 3 != 0){
		cout << -1 << endl;
	}
	else{
		for (int i = 0; i < el.size(); ++i){
			cout << el[el.size() - 1 - i];
		}
		cout << endl;
	}
	return 0;
}
