#include<bits/stdtr1c++.h>
using namespace std;
bool comp(int i, int j){ return (i > j);}
int main(){
	int n; cin >> n;
	vector < int > num(n);
	for (int i = 0; i < n; ++i){
		cin >> num[i];
	}
	sort(num.begin(), num.end(), comp);
	if (num.size() == 1){
		cout << num[0] * 2 << endl;
	}
	else if (num.size() == 2){
		if (num[0] == num[1]) cout << num[0] << endl;
		else cout << num[1] + num[1] + num[1] - num[0] << endl;
	}
	else{
		vector < long long > s(3);
		s[2] = num[0];
		s[1] = num[1];
		s[0] = num[2];
		for (int i = 3; i < n; ++i){
			s[0] += num[i];
			sort(s.begin(), s.end());
		}
		if (s[0] + s[1] > s[2] && s[1] + s[2] > s[0] && s[2] + s[0] > s[1]){
			cout << s[0] + s[1] + s[2] << endl;
		}
		else{
			cout << s[0] + s[1] + s[2] + s[2] - s[0] - s[1] << endl;
		}
	}
	return 0;
}
