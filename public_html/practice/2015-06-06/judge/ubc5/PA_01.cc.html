#include <iostream>
#include <algorithm>
#include <iomanip>
#include <vector>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  string s; cin >> s;
  vector<int> v;
  int t = 0;
  bool z = false;
  for (int i = 0; i < s.size(); i++) {
    int n = s[i]-'0';
    if (!n)
      z = true;
    t += n;
    v.push_back(n);
  }
  if (t % 3 || !z)
    cout << "-1\n";
  else {
    sort(v.begin(),v.end());
    for (int i = v.size()-1; i >= 0; i--)
      cout << v[i];
    cout << "\n";
  }
  return 0;
}
