import java.util.Scanner;
import java.util.List;
import java.util.*;
import java.lang.Double;

public class Triangle {
    public static void main(String[] args) {
      Scanner scanner = new Scanner( System.in );

      // num of test
      String firstString = scanner.nextLine();
      int firstNumber = Integer.parseInt(firstString);

      List<String> check = new ArrayList<String>();
      for(int j=0; j<firstNumber; j++){
        // num of ppl
        String secondString = scanner.nextLine();
        int secondNumber = Integer.parseInt(secondString);
        //System.out.println("the second string is "+secondString);

        String thirdString = scanner.nextLine();
        String[] splited = thirdString.split("\\s+");

        int x1=Integer.parseInt(splited[0]);
        int y1=Integer.parseInt(splited[1]);
        int x2=Integer.parseInt(splited[2]);
        int y2=Integer.parseInt(splited[3]);
        int x3=Integer.parseInt(splited[4]);
        int y3=Integer.parseInt(splited[5]);

        for(int i=0;i<secondNumber; i++){
          String fourthString = scanner.nextLine();
          String[] splitedPpl = fourthString.split("\\s+");
          int x=Integer.parseInt(splitedPpl[0]);
          int y=Integer.parseInt(splitedPpl[1]);
          check.add(checkSafety(x1,y1,x2,y2,x3,y3,x,y));
        }
      }
      for(int i=0; i<check.size(); i++){
        System.out.println(check.get(i));
      }

      //System.out.println("third line is" + thirdString);
    }

    public static String checkSafety(int x1,int y1, int x2, int y2, int x3, int y3, int x, int y){
      Double area  =getArea(x1,y1,x2,y2,x3,y3);

      Double sumArea = getArea(x,y,x2,y2,x3,y3)
      + getArea(x,y,x1,y1,x3,y3)
      +getArea(x,y,x2,y2,x1,y1);
      double eps = 0.0000001;
      //System.out.println("the area for triange is "+area + " the area for the sum of area is "+sumArea);
      if((sumArea-area) < eps){
        return "DANGER";
      }else{
        return "SAFE";
      }
    }

    public static Double getArea(int x1,int y1, int x2, int y2, int x3, int y3){
      return Math.abs(new Double(0.5*((x2-x1)*(y3-y1) - (x3-x1)*(y2-y1))));
    }
}
