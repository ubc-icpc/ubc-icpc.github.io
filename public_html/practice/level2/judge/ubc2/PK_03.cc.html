int query(int x1, int y1, int x2, int y2) {
  cout << "Q " << x1 << " " << y1 << " " << x2 << " " << y2 << endl;
  int ans; cin >> ans;
  return ans;
}

int bs(int a, int n, function<int(int)> f, bool b) {
  int l = 0, r = n, ans = -1;
  while (l <= r) {
    int m = (l+r)/2;
    if (f(m) == a) {
      ans = m;
      if (b) l = m+1;
      else r = m-1;
    } else {
      if (b) r = m-1;
      else l = m+1;
    }
  }
  return ans;
}
