#include <bits/stdc++.h>
const char nl = '\n';
using namespace std;
using ll = long long;
using ld = long double;
using pt = complex<ld>;

ld eval(ld a, ld b, ld d, ld vl, ld vs, ld y) {
  return abs(pt(d, y)) / vl + abs(pt(a, b)-pt(d, y)) / vs;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  int T; cin >> T;
  while (T--) {
    ld a, b, d, vl, vs; cin >> a >> b >> d >> vl >> vs;
    ld l = 0, r = b;
    for (int i = 0; i < 100; i++) {
      ld m1 = (2*l+r)/3, m2 = (l+2*r)/3;
      if (eval(a, b, d, vl, vs, m1) < eval(a, b, d, vl, vs, m2)) r = m2;
      else l = m1;
    }
    cout << fixed << setprecision(10) << eval(a, b, d, vl, vs, l) << nl;
  }
  return 0;
}
