#include <bits/stdc++.h>
using namespace std;
double r, x, y;
int main() {
	cin >> r >> x >> y;
	if (x == 0) {
		cout << min(r - y, y) << '\n';
	}
	else {
		double closestx = sqrt(r * r / (1 + (y * y / (x * x))));
		double closesty = sqrt(r * r - closestx * closestx);
		double closest1 = sqrt((y - closesty) * (y - closesty) + (x - closestx) * (x - closestx));
		double closest2 = sqrt((y - closesty) * (y - closesty) + (x + closestx) * (x + closestx));
		//cout << fixed << setprecision(8) << closestx << ' ' << closesty << ' ' << closest1 << endl; 
		cout << fixed << setprecision(8) << min(y, min(closest1, closest2)) << '\n';

	}

}
