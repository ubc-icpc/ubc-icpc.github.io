pii vectorize(pii A, pii B)
{
    return {A.XX-B.XX, A.YY-B.YY};
}

int area(pii A, pii B)
{
    return abs(A.XX*B.YY-A.YY*B.XX);
}

double dist(pii A, pii B)
{
    pii V=vectorize(A, B);
    return sqrt(V.XX*V.XX+V.YY*V.YY);
}
