int CCW(pii A, pii B, pii C)
{
    int ret=(C.XX-B.XX)*(B.YY-A.YY)-(B.XX-A.XX)*(C.YY-B.YY);
    if (ret>0) return 1;
    if (ret<0) return -1;
    return 0;
}

bool checkInside(pii A, pii B, pii C, pii D)
{
    int ab=CCW(D, A, B), bc=CCW(D, B, C), ca=CCW(D, C, A);
    int ba=CCW(D, B, A), cb=CCW(D, C, B), ac=CCW(D, A, C);
    // cerr<<ab<<" "<<bc<<" "<<ca<<"\n";

    if (ab==0) //D lie on ab
    {
        return (ca!=cb);
    }
    if (bc==0)
    {
        return (ab!=ac);
    }
    if (ca==0)
    {
        return (bc!=ba);
    }

    return (ab!=ac && bc!=ba && ca!=cb);
    
}
