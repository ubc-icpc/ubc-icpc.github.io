import java.util.*;

public class A {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int R = sc.nextInt();
        int x = sc.nextInt();
        int y = sc.nextInt();

        double h = Math.sqrt(x*x + y*y);
        double circDist = R-h;

        System.out.println(Math.min(y, circDist));
    }
}
