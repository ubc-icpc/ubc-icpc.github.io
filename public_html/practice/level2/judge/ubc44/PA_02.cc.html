#include <stdio.h>
#include <limits.h>
#include <string.h>
#include <algorithm>
#include <math.h>
int main()
{
    int i,j;
    double x,y,r;
    double dx,dy;
    scanf("%lf %lf %lf", &r, &x, &y);
    double min = 1000000;
    double res;
    dx = -r;
    for(;dx<=r;dx=dx+0.0001)
    {
        dy = sqrt(r*r - dx*dx);
        res = sqrt((dy-y)*(dy-y) + (dx-x)*(dx-x));

        if(res < min)
            min = res;
    }
    if(y<min)
        min = y;
    printf("%lf", min);
    return 0;
}
