#include <stdio.h>
#include <limits.h>
#include <string.h>
#include <algorithm>
#include <math.h>
double tenarysearch(double l, double r, double a, double b, double d, double vl, double vs)
{
    if(d == a)
        return(sqrt(a*a+b*b)/vl);

    double mid1 = l + (r-l)/3;
    double mid2 = r - (r-l)/3;

    double pt1 = mid1;
    double pt2 = mid2;
    double t1 = sqrt(d*d + pt1*pt1)/vl + sqrt((b-pt1)*(b-pt1) + (a-d)*(a-d))/vs;
    double t2 = sqrt(d*d + pt2*pt2)/vl + sqrt((b-pt2)*(b-pt2) + (a-d)*(a-d))/vs;

    if(abs(mid1-mid2) <= 0.0000001)
        return t2;

    if(t1<=t2)
        return(tenarysearch(l,mid2,a,b,d,vl,vs));
    else if(t1>t2)
        return(tenarysearch(mid1,r,a,b,d,vl,vs));
}

int main()
{
    int T, i = 1;
    double a,b,d,vl,vs;

    scanf("%d", &T);
    while(i<=T)
    {
        scanf("%lf %lf %lf %lf %lf", &a, &b, &d, &vl, &vs);
        double res = tenarysearch(0.0,b,a,b,d,vl,vs);
        printf("%.10f\n", res);
        i++;
    }
    return 0;
}
