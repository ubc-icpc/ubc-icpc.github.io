import java.util.Scanner;

public class HalfCircle {
    public static Scanner sc = new Scanner(System.in);
    public static int x;
    public static int y;
    public static int r;

    public static void main(String[] args) {
        r = sc.nextInt();
        x = sc.nextInt();
        y = sc.nextInt();
        sc.nextLine();
        double shortestC = Math.abs(Math.sqrt(x*x + y*y) - r) ;
        double shortestD = shortestC > ((double) y) ? y : shortestC;
        System.out.printf("%.11f",shortestD);

    }
}
