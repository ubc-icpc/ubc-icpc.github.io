double triarea(int x1, int y1, int x2, int y2, int x3, int y3) {
  return abs((double)(x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2))/(double)2);
}

int main() {
  int t;
  cin >> t;
  for(int i = 0; i < t; i++) {
    int n,x1,y1,x2,y2,x3,y3;
    cin >> n;
    cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3;
    double ar = triarea(x1,y1,x2,y2,x3,y3);
    for(int j = 0; j < n; j++) {
      int xi, yi;
      cin >> xi >> yi;
      double a = triarea(x1, y1, x2, y2, xi, yi);
      double b = triarea(x1, y1, xi, yi, x3, y3);
      double c = triarea(xi, yi, x2, y2, x3, y3);
      if (abs((a+b+c)-ar) == 0) {
        cout << "DANGER" << el;
      }
      else {
        cout << "SAFE" << el;
      }
    }
  }
  return 0; 
}
