//package com.company;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
	// write your code here
        Scanner sc = new Scanner(System.in);
        double r = sc.nextInt();
        double x = sc.nextInt();
        double y = sc.nextInt();
        double d = Math.sqrt(x*x + y*y);
        double ans = r - d;
        ans = Math.min(ans, Math.abs(y));
        String strDouble = String.format("%.11f", ans);
        System.out.println(strDouble);
    }
}
