#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<ll, ll> pii;

ld f(ld a, ld b, ld d, ld vl, ld vs, ld q) {
    ld ans = sqrt(q*q+d*d)/vl + sqrt((b-q)*(b-q)+(a-d)*(a-d))/vs;
    return ans;
}

ld ts(ld a, ld b, ld d, ld vl, ld vs) {
    ld lo = 0, hi = b;
    for (int i=0; i<100; ++i) {
        ld m1 = (lo*2+hi)/3;
        ld m2 = (lo+hi*2)/3;
        ld v1 = f(a, b, d, vl, vs, m1);
        ld v2 = f(a, b, d, vl, vs, m2);
        if (v1 < v2) {
            hi = m2;
        } else {
            lo = m1;
        }
    }
    return f(a, b, d, vl, vs, lo);
}

void solve() {
    ld a, b, d, vl, vs;
    cin >> a >> b >> d >> vl >> vs;
    cout << fixed << setprecision(10) << ts(a, b, d, vl, vs) << "\n";
}

int main() {
    ios_base::sync_with_stdio(0), cin.tie(0);
    int TT;
    cin >> TT;
    while (TT--) {
        solve();
    }
    return 0;
}
