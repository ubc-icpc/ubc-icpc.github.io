int bs_left(int n) {
    int lo = 0, hi = n;
    while (lo < hi) {
        int m = (lo+hi+1)/2;
        cout << "Q " << 0 << " " << 0 << " " << m << " " << n << endl;
        int z;
        cin >> z;
        if (z) {
            hi = m-1;
        } else {
            lo = m;
        }
    }
    return lo;
}
