#include <iostream>
#include <algorithm>
#include <iomanip>
#include <math.h>
using namespace std;
int main()
{
	double r, x, y; cin >> r >> x >> y;
	double pythag = sqrt(x * x + y * y);

	cout << fixed<<setprecision(15)<<min((r-pythag),y)<<"\n";
}
