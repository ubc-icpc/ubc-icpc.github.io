#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using vi = vector<int>;

int main (){
  vi mush(8);
  for (int i = 0; i < 8; ++i)
    cin >> mush[i];

  int res = 0;
  for (int i = 0; i < 8; ++i) {
    int alt = 0;
    for (int j = 0; j < 4; ++j)
      alt += mush[(i + j) % 8];
    res = max(res, alt);
  }
  cout << res << endl;
  return 0;
}
