#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using vi = vector<int>;

int main (){
  int N, K;
  cin >> N >> K;
  vi vert(100001, 0);
  vi hori(100001, 0);
  vi dial(200001, 0); // top left - bottom right
  vi diar(200001, 0); // top right - bottom left
  for (int i = 0; i < N; ++i) {
    int X, Y;
    cin >> X >> Y;
    vert[X]++;
    hori[Y]++;
    int dial_ind = (Y + X) % 200001;
    int diar_ind = ((Y - X) + 200001) % 200001;
    dial[dial_ind]++;
    diar[diar_ind]++;

    if (vert[X] == K || hori[Y] == K
        || dial[dial_ind] == K || diar[diar_ind] == K) {
      cout << i + 1 << endl; 
      break;
    }
    if (i == N - 1)
      cout << -1 << endl;
  }
  return 0;
}
