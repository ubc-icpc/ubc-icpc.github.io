#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	vector < int > m(8);
	vector < int > a;
	for (int i = 0; i < 8; ++i){
		cin >> m[i];
	}
	for (int i = 0; i < 8; ++i){
		if (i < 5) a.push_back(m[i] + m[i + 1] + m[i + 2] + m[i + 3]);
		else {
			int j = i, total = 0;
			for (int k = 0; k < 4; ++k){
				if (k + j == 8){
					j = 0 - k;
				}
				total += m[j + k];
			}
			a.push_back(total);
		}
	}
	sort(a.begin(), a.end());
	cout << a[7] << endl;
	return 0;
}
