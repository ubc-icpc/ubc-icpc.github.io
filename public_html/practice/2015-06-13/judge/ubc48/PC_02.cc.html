#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n, k; cin >> n >> k;
	vector < int > h(100001), v(100001), dh(200000), dl(200000);
	for (int i = 0; i < n; ++i){
		int x, y; cin >> x >> y;
		++h[x];
		++v[y];
		++dl[x + y - 1];
		++dh[100001 - y + x];
		if (h[x] == k || v[y] == k || dl[x + y - 1] == k || dh[100001 - y + x] == k){
			cout << i + 1 << endl;
			return 0;
		}
	}
	cout << -1 << endl;
	return 0;
}
