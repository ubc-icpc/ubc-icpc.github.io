#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int r, s; cin >> r >> s;
	vector < vector < char > > m(r, vector<char>(s));
	vector < vector < int > > p(r, vector<int>(2, -1));
	for (int i = 0; i < r; ++i){
		for (int j = 0; j < s; ++j){
			cin >> m[r - i - 1][j];
			if (m[r - i - 1][j] == 'J'){
				if (p[r - i - 1][0] == -1) p[r - i - 1][0] = j;
				p[r - i - 1][1] = j;
			}
		}
	}
	int dir = -1, cur = 0, op = 0, last = 0;
	for (int i = 0; i < r; ++i){
		if (p[i][0] != -1){
			if (dir == 1){
				op += abs(p[i][1] - cur) + p[i][1] - p[i][0];
				cur = p[i][0];
			}
			else{
				op += abs(p[i][0] - cur) + p[i][1] - p[i][0];
				cur = p[i][1];
			}
			last = i;
		}
		dir *= -1;
	}
	if (p[r - 1][0] != -1) op += r - 1;
	else op += last;
	cout << op << endl;
	return 0;
}
