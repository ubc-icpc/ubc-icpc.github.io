#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	int data[8], m = 0, temp = 0;;
	for (int i = 0; i<8; i++)
		cin >> data[i];
	for (int i = 0; i<8+4; i++) {
		if (i < 4)
			temp += data[i];
		else{
			temp += data[i%8];
			temp -= data[(i-4)%8];
		}
		m = max(m, temp);
	}
	cout << m << endl;
}
