def sum4pieces(mysublist):
    mysum = 0
    for num in mysublist:
        mysum = mysum + num
    return mysum


mylist= [0]*11
for i in range(8):
    mylist[i] = int(input())
    
mylist[8] = mylist[0]
mylist[9] = mylist[1]
mylist[10] = mylist[2]

max = 0
for i in range(11):
    sum = sum4pieces(mylist[i:i+4])
    if sum > max:
        max = sum

print(max)
