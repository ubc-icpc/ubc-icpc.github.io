#include <bits/stdtr1c++.h>
using namespace std;

int N, L, i, X, Y, C[100005], R[100005], D1[200005], D2[200005];

int main() {
	cin >> N >> L;
	for (i = 0; i < N; ++i) {
		cin >> X >> Y;
		if (++C[X] >= L ||
			++R[Y] >= L ||
			++D1[X + Y] >= L ||
			++D2[X - Y + 100001] >= L
			) break;
	}
	cout << ((i == N)? -1 : i + 1) << endl;
	return 0;
}
