#include<iostream>
using namespace std;

int a[15], r;

int main () {
	for (int i = 0; i < 8; ++i)
	    cin >> a[i];
	for (int i = 0; i < 4; ++i)
	    a[i + 8] = a[i];
	for (int i = 0; i < 8; ++i) {
		int s = 0;
	    for (int j = 0; j < 4; ++j) s += a[i + j];
	    r = max(r, s);
	}
	cout << r << endl;
	return 0;
}
