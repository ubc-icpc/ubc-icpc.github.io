#include <iostream>
using namespace std;
unsigned int Y[100000] = { 0 };
unsigned int X[100000] = { 0 };
unsigned int Ris[2000001] = { 0 };
unsigned int Decr[200001] = { 0 };
int main()
{
	unsigned int n, k;
	cin >> n >> k;
	unsigned int answer = -1;
	unsigned int x, y;
	bool go = true;
	for (unsigned int i = 1; i <= n; i++){
		cin >> x >> y;
		if (answer != -1) go=false;
		Y[y]++; 
		if ((Y[y] >= k)&&go) answer = i;
		else{
			X[x]++;
			if ((X[x] >= k)&&go) answer = i;
			else{
				Ris[y - x + 100000]++;
				if ((Ris[y - x + 100000] >= k)&&go) answer = i;
				else{
					Decr[x + y]++;
					if ((Decr[x + y] >= k)&&go) answer = i;
				}
			}
		}
	
	}
	cout << answer << endl;
	return 0;
}
