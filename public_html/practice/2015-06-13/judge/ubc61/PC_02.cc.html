#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n,k; cin >> n >> k;
	
	int ans = 1;
	int vert[100005], horiz[100005], diag[200010], diag2[200010];

	for(int i=0; i<n; i++) {
		int x,y; cin >> x >> y;
		vert[x]++;
		horiz[y]++;
		diag[x-y+50000]++;
		diag2[x+y]++;
		if (vert[x] == k || horiz[y] == k || diag[x-y+50000] == k || diag2[x+y] == k) break;
		ans++;
	}
	if (ans > n) ans = -1;
	cout << ans << endl;
	return 0;
}
