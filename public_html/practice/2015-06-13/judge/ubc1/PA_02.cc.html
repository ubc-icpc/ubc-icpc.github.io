#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int a[8];
    for (int i = 0; i < 8; i++) {
        cin >> a[i];
    }
    
    int ans = 0;
    for (int i = 0; i < 8; i++) {
        int tans = 0;
        for (int j = 0; j < 4; j++) {
            tans += a[(i+j)%8];
        }
        ans = max(ans, tans);
    }
    cout << ans << endl;
	return 0;
}
