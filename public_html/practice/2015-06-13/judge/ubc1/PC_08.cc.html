#include <bits/stdtr1c++.h>

using namespace std;

map<int, int> hor, vert, diag, odiag;
int main() {
    int n, k; cin >> n >> k;
    int ans = 0;
    for (int i = 0; i < n; i++) {
        int x, y; cin >> x >> y;
        hor[x]++;
        vert[y]++;
        diag[y-x]++;
        odiag[y+x]++;
        if (max(hor[x], max(vert[y], max(diag[y-x], odiag[y+x]))) >= k) {
            ans = i+1;
            break;
        }            
    }
    cout << ans << endl;
	return 0;
}
