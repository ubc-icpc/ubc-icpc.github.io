#include<bits/stdtr1c++.h>

using namespace std;

int main() {
  int pieces[8];
  int ans[8];
  memset(ans, 0, sizeof ans);
  for (int i=0; i<8; i++) cin >> pieces[i];
  for (int i=0; i<8; i++) {
    for (int j=0; j<4; j++) {
      ans[i] += pieces[(i+j)%8];
    }
  }
  sort(ans, ans+8);
  cout << ans[7] << '\n';
}
