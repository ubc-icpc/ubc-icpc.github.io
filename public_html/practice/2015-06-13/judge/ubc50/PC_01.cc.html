#include<bits/stdtr1c++.h>

using namespace std;

int main() {
  map<int, int> horizontal, vertical, diagonal, revdiagonal;
  int ans = -1;
  int n, k;
  cin >> n >> k;
  for (int i=0; i<n; i++) {
    int x, y;
    cin >> x >> y;
    horizontal[x]++;
    vertical[y]++;
    diagonal[y-x]++;
    revdiagonal[x+y]++;
    if(max(max(horizontal[x], vertical[y]), max(diagonal[y-x], revdiagonal[x+y])) >= k) {
      ans = i+1;
      break;
    }
  }
  cout << ans << '\n';
}
