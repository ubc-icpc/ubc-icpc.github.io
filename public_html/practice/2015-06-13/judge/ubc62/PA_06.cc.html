#include<bits/stdtr1c++.h>

using namespace std;



int main() {
	int pizza[8];
	for (int i = 0; i < 8; i++){
		cin >> pizza[i];
	}

	int max = 0;
	for (int i = 0; i < 8; i++){
		int shrooms = 0;
		for (int p = 0; p < 4; p++){
			shrooms += pizza[(i+p) % 8];
		}
		if (shrooms > max){
			max = shrooms;
		}
	}
	cout << max << endl;
	return 0;

}
