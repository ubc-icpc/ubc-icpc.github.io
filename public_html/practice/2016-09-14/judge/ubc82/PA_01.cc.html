#include<bits/stdc++.h>

using namespace std;


int P, n, b, s[3][10003];

void precomp(){
	for(int i=1;i<=10000;i++){
		s[0][i]= s[0][i-1]+i;
		s[1][i]= s[1][i-1]+2*i-1;
		s[2][i]= s[2][i-1]+2*i;
	}
}

int main(){
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin >> P;
	precomp();
	for(int i=0;i<P;i++){
		cin >> n >> b;
		cout << n << " " << s[0][b] << " " << s[1][b] << " " << s[2][b] << '\n';
	}
}
