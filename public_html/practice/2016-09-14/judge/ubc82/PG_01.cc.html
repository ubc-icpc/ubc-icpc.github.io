#include<bits/stdc++.h>

using namespace std;


int P, ii, n, m, k, dp[103];


int main(){
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin >> P;
	for(int i=0;i<P;i++){
		cin >> ii>> n >> m >> k;
		memset(dp, 0, sizeof dp);
		int r = m%k;
		for(int a=0;a<=n;a++){
			if (a>=m && a%k==r) continue;
			dp[0] = 1;
		}
		for(int a=1;a<=n;a++){
			for(int j=1;j<=a;j++){
				if (j>=m && j%k==r) continue;
				dp[a] +=dp[a-j];
			}
		}
		cout << ii << " " << dp[n] << '\n';
	}
}
