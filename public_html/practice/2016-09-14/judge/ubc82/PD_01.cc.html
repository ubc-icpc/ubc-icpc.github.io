#include<bits/stdc++.h>

using namespace std;

int P, n, b; 
double prim[10301];
long long ans[10331];

void precomp(){
	for(int i=2;i<=10000;i++){
		prim[i] = i;
	}
	for(int i=2;i<10002;i++){
		if ((int)prim[i]!=i) continue;
		for(int j=i;j<10023;j+=i){
			prim[j]*=(1.0-1.0/(double)i);
		}
	}
	ans[1] = 2;
	for(int i=2;i<=10000;i++){
		//cout <<i << " " << prim[i]<<endl;
		ans[i]=prim[i]+ans[i-1];
	}
}

long long foo(int b){
	return ans[b];
}

int main(){
	ios_base::sync_with_stdio(false);cin.tie(0);
	precomp();
	cin >> P;
	for(int i=0;i<P;i++){
		cin >> n >> b;
		cout << n<< " " << foo(b)<< "\n";
	}
}
