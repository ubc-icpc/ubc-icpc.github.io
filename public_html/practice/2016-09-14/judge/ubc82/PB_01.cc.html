#include<bits/stdc++.h>

using namespace std;


long long P, n, b, s[105][105], ans[105];

#define endl '\n'

void precomp(){
	s[0][0] = 1;
	for(int i=1;i<=51;i++){
		for (int j=0;j<=i;j++){
			s[i][j+1] += s[i-1][j];
			if(i>=2) s[i][j]+=s[i-2][j];
		}
	}
	for(int i=1;i<=51;i++){
		for (int j=0;j<=i;j++){
		//	if (i<7)	cout << i << " " << j << " " << s[i][j]<<endl;
			if (j<=i/3) ans[i]+=s[i][j]*s[i][j];
		}
	}
}

int main(){
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin >> P;
	precomp();
	for(int i=0;i<P;i++){
		cin >> n >> b;
		cout << n << " ";
		if (b%2) cout << 0 << endl;
		else cout << ans[b/2] << endl;
	}
}
