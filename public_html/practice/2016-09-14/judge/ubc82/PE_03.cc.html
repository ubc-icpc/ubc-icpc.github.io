#include<bits/stdc++.h>

using namespace std;


int P, n, a, b;

bool cc[1000];

int main(){
	scanf("%d", &P);
	for(int i=0;i<P;i++){
		scanf("%d", &n);
		scanf("%d/%d", &a, &b);
		long long left=0, depth=0;
		while (a!=b){
			if (a>b){
				cc[depth] = 1;
				depth++;
				a-=b;
			}
			if (a<b){
				cc[depth] = 0;
				depth++;
				b-=a;
			}
		}
		for(int j=depth-1;j>=0;j--){
			left = (left<<1)+cc[j];
		}
		printf("%d %lld\n", n, left+(1<<depth));
	//	cout << left<< " " << (1<<depth)<<endl;
	}
}
