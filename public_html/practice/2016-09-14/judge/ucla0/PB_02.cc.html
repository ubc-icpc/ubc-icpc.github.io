#define ll long long
#define N 200005

ll res[105];


void solve() {
    for (ll s = 2; s <= 100; s += 2) {
        ll sum = 0;
        
        for (ll two = s/4; two >= 0; two--) {
            ll one = (s - 4*two)/2;
            if (one > two)
                break;
            //ll perm = fac(s) / (fac(one)*fac(two));
            
            ll ss = 1;
            ll n = one+two;
            for (ll j = 1; j <= two; j++) {
                ss *= n-j+1;
                ss /= j;
            }
            sum += ss*ss;
        }
        res[s] = sum;
    }
    
    ll p;
    cin >> p;
    for (ll i = 0; i < p; i++) {
        ll k, s;
        cin >> k >> s;
        cout << k << " " << res[s] << endl;
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
}
