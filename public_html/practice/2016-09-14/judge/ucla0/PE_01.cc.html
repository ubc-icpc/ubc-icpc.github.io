#define ll long long

void solve() {
    ll k;
    cin >> k;
    string s;
    cin >> s;
    ll slash = 0;
    for (ll i = 0; i < s.size(); i++) {
        if (s[i] == '/') {
            slash = i;
            break;
        }
    }
    
    ll p = stoll(s.substr(0,slash));
    ll q = stoll(s.substr(slash+1, s.size()-slash+1));
    
    string res;
    while (p != 0 && q != 0) {
        if (p > q) {
            string str(p/q,'r');
            p %= q;
            res += str;
        }
        else {
            string str (q/p,'l');
            q %= p;
            res += str;
        }
    }
    res = res.substr(0, res.size()-1);
    
    ll num = 0;
    for (ll i = 0; i < res.size(); i++) {
        if (res[i] == 'r')
            num += (1<<i);
    }
    num += (1<<res.size());
    
    cout << k << " " << num << endl;
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(0);
    ll p; cin >> p;
    for (ll i = 0; i < p; i++)
        solve();
}
