#define ll long long
#define N 10005
#define MOD 1000000000

ll f[N];

void solve() {
    f[1] = 1;
    for (ll i = 1; i < N; i++) {
        f[i] = i*(i+3)/2;
        for (ll k = 2; k <= i; k++) {
            f[i] -= f[i/k];
        }
    }
    
    ll p;
    cin >> p;
    for (ll i = 0; i < p; i++) {
        ll k, n;
        cin >> k >> n;
        cout << k << " " << f[n] << endl;
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
}
