using namespace std;

#define ll long long
#define N 10005
#define MOD 1000000000

ll f[N];

void solve() {
    ll p;
    cin >> p;
    for (ll z = 0; z < p; z++) {
        ll K, n, m, k;
        cin >> K >> n >> m >> k;
        
        memset(f, 0, sizeof(ll)*N);
        f[0] = 1;
        
        for (ll i = 1; i <= n; i++) {
            for (ll j = 0; j < i; j++) {
                if ((i-j-m)%k == 0)
                    continue;
                f[i] += f[j];
            }
        }
        cout << K << " " << f[n] << endl;
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(0);
    solve();
}
