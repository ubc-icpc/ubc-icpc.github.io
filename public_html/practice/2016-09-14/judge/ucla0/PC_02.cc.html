#define ll long long
#define N 10005
#define MOD 1000000000L

void mult(ll a[2][2], ll b[2][2]) {
    ll tl = (a[0][0]*b[0][0])%MOD + (a[0][1]*b[1][0])%MOD;
    ll tr = (a[0][0]*b[1][0])%MOD + (a[0][1]*b[1][1])%MOD;
    ll bl = (a[1][0]*b[0][0])%MOD + (a[1][1]*b[1][0])%MOD;
    ll br = (a[1][0]*b[1][0])%MOD + (a[1][1]*b[1][1])%MOD;
    b[0][0] = tl;
    b[0][1] = tr;
    b[1][0] = bl;
    b[1][1] = br;
}

void solve() {
    ll k, y;
    cin >> k >> y;
    
    ll mat[2][2] =
        {{1,0},
        {0,1}};
    
    ll one[2][2] =
        {{0,1},
        {1,1}};
    
    y--;
    for (ll i = 50; i >= 0; i--) {
        mult(mat,mat);
        if (y&(1L<<i))
            mult(one, mat);
    }
    cout << k << " " << (mat[0][0]+mat[0][1])%MOD << endl;
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(0);
    ll p;
    cin >> p;
    for (ll i = 0; i < p; i++) {
        solve();
    }
}
