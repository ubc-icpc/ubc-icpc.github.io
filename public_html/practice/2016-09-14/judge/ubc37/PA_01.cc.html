#include <iostream>

using namespace std;

int main() {
	int p;
	cin >> p;

	for (int i = 0; i < p; i++) {
		int k;
		int n;
		cin >> k >> n;

		// total
		int t = (n * (1 + n)) / 2;

		// even
		int e = (n * (2 + n * 2)) / 2;

		// odd
		int o = (n * (1 + (n * 2 - 1))) / 2;

		cout << k << " " << t << " " << o << " " << e << endl;
	}
}
