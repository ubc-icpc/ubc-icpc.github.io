#include <iostream>

using namespace std;
int S1[10000];
int S2[10000];
int S3[10000];

int main() {
	ios::sync_with_stdio();
	S1[0] = S2[0] = 1;
	S3[0] = 2;
	for (int i = 1; i < 10000; i++) {
		S1[i] = S1[i - 1] + i + 1;
		S2[i] = S2[i - 1] + ((i + 1) * 2) - 1;
		S3[i] = S3[i - 1] + ((i + 1) * 2);
	}
	int n;
	cin >> n;
	while (n--) {
		int k, p;
		cin >> k >> p;
		p--;
		cout << k << " " << S1[p] << " " << S2[p] << " " << S3[p] << endl;
	}
	return 0;
}
