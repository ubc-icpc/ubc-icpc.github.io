#include <iostream>
#include <cstring>

using namespace std;

const int N = 105;
long long bin[105][105];

void init() {
	memset(bin, 0, sizeof(bin));
	bin[0][0] = bin[1][0] = bin[1][1] = 1;
	for (int i = 2; i < N; i++) {
		bin[i][0] = 1;
		for (int j = 1; j <= i; j++) {
			bin[i][j] = bin[i - 1][j] + bin[i - 1][j - 1];
		}
	}
	return;
}

int main() {
	init();
	int P; cin >> P;
	while (P--) {
		int k; cin >> k;
		int s; cin >> s;
		int x = s / 2;
		int y = 0;
		if (x % 2) {
			x--; y++;
		}
		x /= 2;
		long long total = 0;
		while (x >= y) {
			long long t = bin[x+y][x];
			total += t*t;
			x--;
			y += 2;
		}
		cout << k << " " << total << endl;
	}
	return 0;
}
