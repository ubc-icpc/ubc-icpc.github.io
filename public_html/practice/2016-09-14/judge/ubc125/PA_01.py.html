import sys

def sum_positive(n):
    return (n*(n+1))/2

def sum_odd(n):
    return n*n

def sum_even(n):
    return sum_positive(n)*2

x = int(sys.stdin.readline().strip())
data = []
for i in range(x):
    datum = list(map(int, sys.stdin.readline().strip().split(' ')))
    K = datum[0]
    N = datum[1]
    data.append((K, N))

for i in data:
    K, N = i
    print K, int(sum_positive(N)), int(sum_odd(N)), int(sum_even(N))
