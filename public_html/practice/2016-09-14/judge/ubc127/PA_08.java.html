import java.util.Scanner;

public class Sum {

    public static void main(String[] args) {
        Scanner sc =  new Scanner(System.in);
        int p = sc.nextInt();

        for(int j=1; j<=p; j++){
            int s1 =0;
            int s2 =0;
            int s3 =0;
            int k = sc.nextInt();
            int n = sc.nextInt();
            for (int i=1;i<=n;i++){
                s1 = s1+i;
                s2 = s2 + (i*2-1);
                s3 = s3 + (i*2);
            }
            System.out.println(k + " " + s1 + " " + s2 + " " + s3);
        }

    }
}
