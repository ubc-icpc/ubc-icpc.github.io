#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;

int T, K;
ll N;

ll perm(ll a, ll b) {
  if (a > b) {
    swap(a, b);
  }
  ll res = 1;
  ll d = a;
  for (ll i = b + 1; i <= b + a; ++i) {
    res *= i;
    while (d > 1 && res % d == 0) {
      res /= d;
      --d;
    }
  }
  return res;
}

ll calc() {
  ll res = 0;
  N /= 2;
  ll A, B;
  A = N / 2;
  B = N % 2;
  for (; A >= B; A -= 1, B += 2) {
    ll p = perm(A, B);
    res += p * p;
  }
  return res;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> T;
  while (T--) {
    cin >> K >> N;
    cout << K << " " << calc() << endl;
  }

  return 0;
}
