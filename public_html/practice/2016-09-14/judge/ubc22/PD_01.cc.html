#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;
int T, K, N, top = 1;
ll dp[10005];

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  dp[1] = 2;

  cin >> T;
  while (T--) {
    cin >> K >> N;
    while (N > top) {
      ++top;
      dp[top] = dp[top - 1];
      for (int i = 1; i < top; ++i) {
        if (__gcd(i, top) == 1) {
          dp[top] += 1;
        }
      }
    }
    cout << K << " " << dp[N] << endl;
  }

  return 0;
}
