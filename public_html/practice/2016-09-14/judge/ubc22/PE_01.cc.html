#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;

int T, K, A, B;
char C;

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> T;
  while (T--) {
    cin >> K >> A >> C >> B;
    ll res = 0;
    int cnt = 0;
    for (; A != 1 || B != 1; ++cnt) {
      if (A > B) {
        res |= (1 << cnt);
        A = A - B;
      } else {
        B = B - A;
      }
    }
    res |= (1 << cnt);
    cout << K << " " << res << endl;
  }

  return 0;
}
