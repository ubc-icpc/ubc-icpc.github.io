import sys

def matmult(a,b):
  zip_b = zip(*b)
  return [[sum(ele_a*ele_b % 10**9 for ele_a, ele_b in zip(row_a, col_b)) 
           for col_b in zip_b] for row_a in a]

def pow_helper(mat, n):
  if n <= 1:
    return mat

  pow_half = pow_helper(mat, n/2)
  mult = matmult(pow_half, pow_half)

  if n % 2:
    return matmult(mat, mult)
  else:
    return mult

def pow(n):
  return pow_helper([[1,1],[1,0]], n)

def fib(n):
  return matmult(pow(n), [[1],[0]])[0][0]

T = int(sys.stdin.readline())
for i in range(T):
  c,n = [int(k) for k in sys.stdin.readline().split()]
  print c, fib(n-1)
