#include <iostream>

using namespace std;
typedef long long ll;
const int N = 10005;

int gcd(int a, int b) {
    while (b > 0) {
        int tmp = a;
        a = b;
        b = tmp % b;
    }
    return a;
}

ll ans[N];
int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    ans[0] = 0;
    ans[1] = 2;
    for (int i = 2; i < N; i++) {
        ans[i] = ans[i-1];
        for (int j = 1; j < i; j++) {
            if (gcd(i, j) == 1) ans[i]++;
        }
    }

    int ncase; cin >> ncase;
    while (ncase--) {
        int k, n; cin >> k >> n;
        cout << k << " " << ans[n] << endl;
    }
}
