#include <iostream>
#include <cstring>

using namespace std;
typedef long long ll;

ll dp[35][35][35];
int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    memset(dp, 0, sizeof dp);
    for (int m = 0; m < 30; m++) {
        for (int k = 0; k < 30; k++) {
            dp[m][k][0] = 1;
            for (int n = 1; n <= 30; n++) {
                for (int z = 1; z <= n; z++) {
                    if (z >= m && ((k == 0 && z == m) || (k > 0 && (z-m)%k == 0))) continue;
                    dp[m][k][n] += dp[m][k][n-z];
                    if (z == n) dp[m][k][n]++;
                }
            }
        }
    }

    int ncase; cin >> ncase;
    while (ncase--) {
        int a, n, m, k; cin >> a >> n >> m >> k;
        //cerr << "dp: "; for (int i = 0; i <= n; i++) cerr << dp[m][k][i] << " "; cerr << endl;
        cout << a << " " << dp[m][k][n]/2 << endl;
    }
}
