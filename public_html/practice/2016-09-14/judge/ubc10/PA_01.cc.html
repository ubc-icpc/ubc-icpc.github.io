#include <iostream>

using namespace std;
typedef long long ll;
const int N = 10005;

ll a[N], b[N], c[N];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    a[0] = b[0] = c[0] = 0;
    for (int i = 1; i < N; i++) {
        a[i] = a[i-1] + i;
        b[i] = b[i-1] + 2*i-1;
        c[i] = c[i-1] + 2*i;
    }

    int ncase; cin >> ncase;
    for (int csnum = 1; csnum <= ncase; csnum++) {
        int k, n; cin >> k >> n;
        cout << k << " " << a[n] << " " << b[n] << " " << c[n] << endl;
    }
}
