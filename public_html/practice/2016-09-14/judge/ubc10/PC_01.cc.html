#include <iostream>

using namespace std;
typedef long long ll;
const ll mod = 1000000000;

void mul(ll mat[2][2], ll vec[2]) {
    ll res[2];
    for (int i = 0; i < 2; i++) {
        res[i] = 0;
        for (int j = 0; j < 2; j++) {
            res[i] += mat[i][j] * vec[j];
            res[i] %= mod;
        }
    }
    for (int i = 0; i < 2; i++) {
        vec[i] = res[i];
    }
}
void sq(ll mat[2][2]) {
    ll res[2][2];
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 2; j++) {
            res[i][j] = 0;
            for (int k = 0; k < 2; k++) {
                res[i][j] += mat[i][k] * mat[k][j];
                res[i][j] %= mod;
            }
        }
    }
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 2; j++) {
            mat[i][j] = res[i][j];
        }
    }
}

int main() {
    int ncase; cin >> ncase;
    for (int csnum = 1; csnum <= ncase; csnum++) {
        int k; ll y; cin >> k >> y;
        if (y <= 2) cout << k << " " << 1 << endl;
        else {
            y -= 2;
            ll mat[2][2] = {{0, 1}, {1, 1}};
            ll vec[2] = {1, 1};
            for (int i = 0; i < 50; i++) {
                if (y & (1LL << i)) {
                    mul(mat, vec);
                }
                sq(mat);
            }
            cout << k << " " << vec[1] << endl;
        }
    }
}
