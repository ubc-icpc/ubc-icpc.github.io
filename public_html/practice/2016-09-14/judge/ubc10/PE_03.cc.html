#include <iostream>
#include <algorithm>

using namespace std;
typedef long long ll;

int main() {
    int ncase; cin >> ncase;
    while (ncase--) {
		char dummy;
        int k; ll p, q; cin >> k >> p >> dummy >> q;
		string s;
		while (p > 1 || q > 1) {
			while (p > q) {
				p -= q;
				s += 'R';
			}
			while (q > p) {
				q -= p;
				s += 'L';
			}
		}
		reverse(s.begin(), s.end());
		ll ans = 0;
		ll idx = 1;
		for (int i = 0; i < s.length(); i++) {
			ans += (1LL << i);
			if (s[i] == 'L') idx = 2*idx;
			else idx = 2*idx+1;
		}
		ans += idx - (1LL << s.length());
		cout << k << " " << ans+1 << endl;
    }
}
