#include <iostream>
#include <string>
#include <sstream>
#include <math.h>
using namespace std;



void solve(int N, int number);
int main()
{
	string line;
	int numberOfTests;
	getline(cin, line);
	stringstream(line) >> numberOfTests;


	for (int i = 0; i < numberOfTests; i++)
	{
		int number, N;
		cin >> number;
		cin >> N;
		solve(N, number);
	}
	
	
	cin.get();
	return 0;
}


void solve(int N, int number)
{
	int S1, S2, S3;
	int N2;
	S1 = N*(N + 1) / 2;
	N2 = N * (2 * N + 1);
	S2 = (N2 - N) / 2;
	S3 = S2 + N;

	cout << number << " " << S1 << " " << S2 << " " << S3 << endl;
	return;
}
