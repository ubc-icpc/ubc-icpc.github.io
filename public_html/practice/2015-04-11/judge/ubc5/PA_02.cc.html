#include <iostream>
#include <map>
#include <vector>
using namespace std;

typedef long long ll;
typedef vector<ll> vll;

const int N = 2;
const char cs[7] = {'B','E','S','I','G','O','M'};

int main() {
  map<char,vll> m;
  for (int i = 0; i < 7; i++)
    m[cs[i]] = vll(N,0);
  int n; cin >> n;
  for (int i = 0; i < n; i++) {
    char c; ll v;
    cin >> c >> v;
    m[c][((v%N)+N)%N]++;
  }
  ll a = 0;
  for (int B = 0; B < N; B++)
    for (int E = 0; E < N; E++)
      for (int S = 0; S < N; S++)
        for (int I = 0; I < N; I++)
          for (int G = 0; G < N; G++)
            for (int O = 0; O < N; O++)
              for (int M = 0; M < N; M++)
                if (((B+2*E+2*S+I)*(G+O+E+S)*(2*O+M)) % N == 0)
                  a += m['B'][B]*m['E'][E]*m['S'][S]*m['I'][I]*m['G'][G]*m['O'][O]*m['M'][M];
  cout << a << "\n";
  return 0;
}
