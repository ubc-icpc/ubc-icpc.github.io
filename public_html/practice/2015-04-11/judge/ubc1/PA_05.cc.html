#include <bits/stdtr1c++.h>

using namespace std;

string bessie = "BESSIE", goes = "GOES", moo = "MOO";
char c[] = {'B', 'E', 'S', 'I', 'G', 'O', 'M'};
map<char, map<int, int> > vals;
int dfs(int i, map<char, int>& m) {
    if (i == 7) {
        int A = 0, B = 0, C = 0;
        for (char ch : bessie) A += m[ch];
        for (char ch : goes) B += m[ch];
        for (char ch : moo) C += m[ch];
        if ((A*B*C)%2 == 0) return 1;
        else return 0;
    }
    
    int ans = 0;
    for (int v = 0; v < 2; v++) {
        m[c[i]] = v;
        ans += vals[c[i]][v]*dfs(i+1, m);
    }
    return ans;
}

int main() {
    int n; cin >> n;
    for (int i = 0; i < n; i++) {
        char ch; int x;
        cin >> ch >> x;
        vals[ch][(x%2+2)%2]++;
    }
    
    map<char, int> m;
    cout << dfs(0, m) << endl;
	return 0;
}
