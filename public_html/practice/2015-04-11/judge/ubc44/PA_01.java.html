import java.util.HashMap;
import java.util.Map;
import java.lang.Integer;
import java.util.Scanner;

public class bge {
	private static Map<String, Integer> even;
	static {
		even = new HashMap<String, Integer>();
		even.put("B", 0);
		even.put("E", 0);
		even.put("S", 0);
		even.put("I", 0);
		even.put("G", 0);
		even.put("O", 0);
		even.put("M", 0);
	}
	private static Map<String, Integer> odd;
	static {
		odd = new HashMap<String, Integer>();
		odd.put("B", 0);
		odd.put("E", 0);
		odd.put("S", 0);
		odd.put("I", 0);
		odd.put("G", 0);
		odd.put("O", 0);
		odd.put("M", 0);
	}
	
	public static void main(String[] args) {
		Scanner s = new Scanner(System.in);
		int N = s.nextInt();
		while (N-->0) {
			String x = s.next();
			int val = s.nextInt();
			if (val%2 == 0)
				even.put(x, (int)even.get(x) +1);
			else
				odd.put(x, (int)odd.get(x) +1);
		}
		final Map<String, Integer> total = new HashMap<String, Integer>(); {
			total.put("B", even.get("B")+ odd.get("B"));
			total.put("E", even.get("E")+ odd.get("E"));
			total.put("S", even.get("S")+ odd.get("S"));
			total.put("I", even.get("I")+ odd.get("I"));
			total.put("G", even.get("G")+ odd.get("G"));
			total.put("O", even.get("O")+ odd.get("O"));
			total.put("M", even.get("M")+ odd.get("M"));
		}
		
		long sum = 0;
		sum += even.get("M")*total.get("B")*total.get("E")*total.get("S")*total.get("I")*total.get("G")*total.get("O");
		sum += odd.get("M")*even.get("B")*total.get("E")*total.get("S")*even.get("I")*total.get("G")*total.get("O");
		sum += odd.get("M")*odd.get("B")*total.get("E")*total.get("S")*odd.get("I")*total.get("G")*total.get("O");
		long GOES_is_even = odd.get("M")*odd.get("B")*even.get("I") + odd.get("M")*even.get("B")*odd.get("I");
		sum += GOES_is_even*odd.get("G")*odd.get("O")*odd.get("E")*odd.get("S");
		sum += GOES_is_even*even.get("G")*even.get("O")*even.get("E")*even.get("S");
		sum += GOES_is_even*even.get("G")*even.get("O")*odd.get("E")*odd.get("S");
		sum += GOES_is_even*even.get("G")*odd.get("O")*even.get("E")*odd.get("S");
		sum += GOES_is_even*even.get("G")*odd.get("O")*odd.get("E")*even.get("S");
		sum += GOES_is_even*odd.get("G")*even.get("O")*even.get("E")*odd.get("S");
		sum += GOES_is_even*odd.get("G")*even.get("O")*odd.get("E")*even.get("S");
		sum += GOES_is_even*odd.get("G")*odd.get("O")*even.get("E")*even.get("S");
		System.out.println(sum);
		s.close();
	}

}
