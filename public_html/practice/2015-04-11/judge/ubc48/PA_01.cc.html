#include<iostream>
#include<vector>
#include<map>
using namespace std;

bool calc(int b, int e, int s, int i, int g, int o, int m) {
	if ((b + e + s + s + i + e) % 2 == 0) {
		return true;
	}
	if ((g + o + e + s) % 2 == 0) {
		return true;
	}
	if ((m + o + o) % 2 == 0) {
		return true;
	}
	return false;
}

int main() {
	int n; cin >> n;
	vector < int > blist;
	vector < int > elist;
	vector < int > slist;
	vector < int > ilist;
	vector < int > glist;
	vector < int > olist;
	vector < int > mlist;
	while (n--) {
		char chr; int val; cin >> chr >> val;
		if (chr == 'B') {
			blist.push_back(val);
		}
		if (chr == 'E') {
			elist.push_back(val);
		}
		if (chr == 'S') {
			slist.push_back(val);
		}
		if (chr == 'I') {
			ilist.push_back(val);
		}
		if (chr == 'G') {
			glist.push_back(val);
		}
		if (chr == 'O') {
			olist.push_back(val);
		}
		if (chr == 'M') {
			mlist.push_back(val);
		}
	}
	int total = 0;
	for (int one = 0; one < blist.size(); ++one) {
		for (int two = 0; two < elist.size(); ++two) {
			for (int three = 0; three < slist.size(); ++three) {
				for (int four = 0; four < ilist.size(); ++four) {
					for (int five = 0; five < glist.size(); ++five) {
						for (int six = 0; six < olist.size(); ++six) {
							for (int seven = 0; seven < mlist.size(); ++seven) {
								if (calc(blist[one], elist[two], slist[three], ilist[four], glist[five], olist[six], mlist[seven])) {
									++total;
								}
							}
						}
					}
				}
			}
		}
	}
	cout << total << endl;
	return 0;
}
