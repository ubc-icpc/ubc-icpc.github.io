#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <math.h>

using namespace std;

int main() {
    long long S;
    while (cin >> S) {
        if (S == 0) {
            break;
        }
        
        // -1 + sqrt(1 + 4*2*S) / 2
        long long est = (-1 + sqrt(1+8*S)) / 2;
        while (((est*est + est) / 2) < S) {
            est++;
        }
        
        if (((est*est + est) / 2) == S) {
            printf("Not bored enough!\n");
        }
        else {
            printf("%lld\n", ((est*est+est)/2) - S);
        }
    }
    return 0;
}
