#include <stdio.h>
#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

int dist(int y1, int x1, int y2, int x2) {
    return (abs(y1-y2) + abs(x1-x2));
}

int main()
{
    int T, M, N;
    int startX, startY;
    char map[505][505];
    string line;
    cin >> T;
    while (T-- > 0) {
        cin >> M >> N;
        for (int y = 0; y < M; y++) {
            cin >> line;
            for (int i = 0; i < line.length(); i++) {
                map[y][i] = line.at(i);
                if (map[y][i] == 'S') {
                    startX = i;
                    startY = y;
                }
            }
        }
        
        int ans = 0;
        for (int y = 0; y < M; y++) {
            for (int x = 0; x < N; x++) {
                if (map[y][x] == 'B') {
                    ans = ans + (2 * dist(y,x,startY, startX));
                }
            }
        }
        
        printf("%d seconds to blocks get!\n", ans);
    }
    
    return 0;
}
