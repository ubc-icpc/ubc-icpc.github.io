#include <iostream>
#include <cmath>
using namespace std;
typedef long long ll;
void solve(ll a) {
  ll n = (ll) sqrt(a * 2);
  for(ll i = n; i <= n + 1; i++) {
    if(i * (i + 1) / 2 - a < i && i * (i + 1) / 2 - a > 0) {
      cout<< i * (i + 1) / 2 - a<<endl;;
      return ;
    }
  }
  cout<<"Not bored enough!"<<endl;
}
int main() {
  ll n;
  while(true) {
    cin>>n;
    if(n == 0)
        break;
    solve(n);
  }
}
