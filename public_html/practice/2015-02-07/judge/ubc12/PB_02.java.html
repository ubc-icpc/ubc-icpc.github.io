import java.util.LinkedList;
import java.util.Scanner;
public class Feb07_2015_B {
	public static void main(String args[]){
		int n, opt, optind;
		Scanner sc = new Scanner(System.in);
		while(true){
			n = sc.nextInt();
			if(n == 1)	break;
			int[] nums = new int[n], maxl = new int[n], minl = new int[n], maxr = new int[n], minr = new int[n];
			nums[0] = sc.nextInt(); maxl[0] = nums[0]; minl[0] = nums[0];
			for(int i = 1; i< n; i++){
				nums[i] = sc.nextInt();
				maxl[i] = Math.max(nums[i], maxl[i-1]);
				minl[i] = Math.min(nums[i], minl[i-1]);
			}
			maxr[n-1] = nums[n-1]; minr[n-1] = nums[n-1];
			for(int i = n-2; i>= 0; i--){
				maxr[i] = Math.max(nums[i], maxr[i+1]);
				minr[i] = Math.min(nums[i], minr[i+1]);
			}
			opt = Math.max(maxl[0] - minl[0], maxr[1] - minr[1]);
			optind = 1;
			//System.out.println(opt + " " + maxl[0] + " " + minl[0] + " " + maxr[1] + " " +minr[1] + " " + nums[0] + " "+ optind);
			for(int i = 1; i<n-1; i++){
				//System.out.println(opt + " " + maxl[i] + " " + minl[i] + " " + maxr[i+1] + " " +minr[i+1] + " " + nums[i]);
				if(opt > Math.max(maxl[i] - minl[i], maxr[i+1] - minr[i+1])){
					opt = Math.max(maxl[i] - minl[i], maxr[i+1] - minr[i+1]);
					optind = i+1;
				}
			}
			System.out.println(optind);
		}
	}
}
