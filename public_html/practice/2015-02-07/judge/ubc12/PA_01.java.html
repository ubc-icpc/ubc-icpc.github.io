import java.util.LinkedList;
import java.util.Scanner;


public class Feb07_2015_A {
	public static void main(String args[]){
		int t, m, n, sum;
		Scanner sc = new Scanner(System.in);
		LinkedList<Pair> bs;
		Pair s = new Pair(-1, -1);
		char c;
		String cs;
		t = sc.nextInt();
		for(int i = 0; i< t; i++){
			m = sc.nextInt();	n = sc.nextInt();
			bs = new LinkedList<Pair>();
			for(int j = 0;j< m ;j++){
				cs = sc.next();
				for(int k = 0; k< n ; k++){
					c = cs.charAt(k);
					if(c == 'B'){
						bs.add(new Pair(j, k));
					}
					else if(c == 'S'){
						s.fst = j; s.snd = k;
					}
				}
			}
			sum = 0;
			for (Pair p : bs){
				sum += Math.abs(p.fst - s.fst);
				sum += Math.abs(p.snd - s.snd);
			}
			System.out.println(sum*2+" seconds to blocks get!");
		}
	}
	private static class Pair {
		int fst;
		int snd;
		public Pair(int fst, int snd){
			this.fst = fst;
			this.snd = snd;
		}
	}

}
