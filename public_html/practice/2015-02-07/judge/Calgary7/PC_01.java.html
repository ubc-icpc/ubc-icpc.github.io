import java.io.*;
import java.util.*;

public class C {

	public static void main(String[] args) {
		Scanner in = new Scanner(new BufferedReader(new InputStreamReader(
				System.in)));

		long s = in.nextLong();
		while (s != 0) {
			long c = s * 2;

			long n = (long) Math.ceil(((1 + Math.sqrt(1 + 4 * c)) / 2));

			long actualN = (n * (n - 1)) / 2;
			if(actualN - s != 0){
			System.out.println(actualN - s);
			}
			else
				System.out.println("Not bored enough!");
			
			s = in.nextLong();

		}
	}

}
