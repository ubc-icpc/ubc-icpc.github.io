#include <bits/stdc++.h>
using namespace std;

long long S;
void sv() {
	long long a=1, b=1000000000;
	while (b-a != 1) {
		long long c = (a+b)/2, s = (c*(c+1))/2;
		if (s == S) {
			cout << "Not bored enough!\n";
			return;
		} else if (s > S) b=c;
		else a=c;
	}
	if ((a*(a+1))/2 == S) {
		cout << "Not bored enough!\n";
		return;
	}
	cout << ((a+2)*(a+1))/2-S << '\n';
}

int main() {
	while (cin >> S && S) sv();
}
