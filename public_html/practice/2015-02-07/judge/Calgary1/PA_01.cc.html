#include <bits/stdc++.h>
using namespace std;

int main() {
	int tc;
	cin >> tc;
	while (tc--) {
		int M,N,ans=0,x,y;
		vector<int> x2,y2;
		cin >> M >>N;
		for (int m = 0; m < M; m++) {
			for (int n = 0; n < N; n++) {
				char c;
				cin >> c;
				if (c == 'S') {x=m;y=n;}
				if (c == 'B') {x2.push_back(m);y2.push_back(n);}
			}
		}
		for (int i = 0; i < x2.size(); i++) ans += abs(x2[i]-x)+abs(y2[i]-y);
		cout << ans*2 << " seconds to blocks get!\n";
	}
}
