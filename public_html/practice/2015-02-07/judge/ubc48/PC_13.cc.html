#include<iostream>
#include<cmath>
using namespace std;
int main() {
	long long n; cin >> n;
	while(n != 0){
		{
			double a = sqrt(1 + 8 * n);
			long long b = a;
			if (double(b) == a)
				cout << "Not bored enough!" << endl;
			else {
				b = ceil((a + 0.00000001 - 1) / 2);
				cout << (b + 1) * b / 2 - n << endl;
			}
		}
		cin >> n;
	}
}
