#include<iostream>
#include<cmath>
#include<tuple>
#include<vector>
using namespace std;

vector< tuple<int, int> >* g_blocks;
tuple<int, int> g_start;

void Solve()
{
    vector< tuple<int, int> >::iterator it = g_blocks->begin();
    vector< tuple<int, int> >::iterator itEnd = g_blocks->end();
    int total = 0;
    //cout << "start m: " << get<0>(g_start) << ", start n: " << get<1>(g_start) << endl;
    for (; it!=itEnd; ++it)
    {
        int diffM = 2*abs(get<0>(*it) - get<0>(g_start));
        int diffN = 2*abs(get<1>(*it) - get<1>(g_start));
        //cout << "m: " << get<0>(*it) << ", n: " << get<1>(*it) << endl;
        //cout << "diffM: " << diffM << ", diffN: " << diffN << endl;
        total += diffM + diffN;
        //cout << "total: " << total << endl;
    }
    cout << total << " seconds to blocks get!\n";
}

int main(int argc, char* argv[])
{
    int t;
    cin >> t;
    while (t--)
    {
        int m, n;
        char cell;
        vector< tuple<int, int> > room;
        g_blocks = &room;
        cin >> m >> n;             
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cin >> cell;
                if (cell == 'B')
                {
                    g_blocks->push_back(make_tuple(i, j));
                }
                else if (cell == '.')
                {
                    continue;
                }
                else if (cell == 'S')
                {
                    g_start = make_tuple(i, j);
                }
            }
        }
        Solve();
    }
    return 0;
}
