#include <cstdio>
#include <algorithm>
using namespace std;

int a[400001], b1[400001], b2[400001];

int main() {
  int n;
  while (scanf("%d", &n) && n != 1) {
    for (int i = 1; i <= n; i++)
      scanf("%d", &a[i]);
    int mn = 2000001, mx = -1;
    for (int i = 1; i < n; i++) {
      mn = min(mn, a[i]);
      mx = max(mx, a[i]);
      b1[i] = mx-mn;
    }
    mn = 2000001, mx = -1;
    for (int i = n; i > 1; i--) {
      mn = min(mn, a[i]);
      mx = max(mx, a[i]);
      b2[i-1] = mx-mn;
    }
    mn = 2000001;
    int mi = -1;
    for (int i = 1; i < n; i++) {
      if (max(b1[i],b2[i]) < mn) {
        mn = max(b1[i],b2[i]);
        mi = i;
      }
    }
    printf("%d\n", mi);
  }
  return 0;
}
