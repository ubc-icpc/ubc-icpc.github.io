#include <iostream>
using namespace std;

typedef long double ld;

const ld NT = 1.23, MT = 3.21;

int main() {
  int n, m;
  while (cin >> n >> m) {
    int nd = 0, md = 0;
    ld t = 0, nt = 0, mt = 0;
    while (true) {
      if (nd >= n*4 || md >= m*27)
        break;
      if (nt < mt) {
        t = nt;
        nt += NT;
        md += n-(nd > 3*n ? nd-3*n : 0);
      }
      else if (mt < nt) {
        t = mt;
        mt += MT;
        nd += m-(md/27);
      }
      else {
        t = nt;
        nt += NT;
        int mmd = md;
        md += n-(nd > 3*n ? nd-3*n : 0);
        mt += MT;
        nd += m-(mmd/27);
      }
    }
    if (nd >= n*4 && md >= m*27)
      cout << "draw\n";
    else if (nd >= n*4)
      cout << "lose\n";
    else
      cout << "win\n";
  }
  return 0;
}
