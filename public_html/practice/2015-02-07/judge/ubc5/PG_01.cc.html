#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

int main() {
  int x;
  while (cin >> x) {
    for (int i = 2; i < sqrt(x); i++) {
      if (!(x%i))
        continue;
      int n = x;
      vector<int> b;
      while (n) {
        b.push_back(n%i);
        n /= i;
      }
      bool p = true;
      for (int j = 0, k = b.size()-1; j < k; j++, k--) {
        if (b[j] != b[k]) {
          p = false;
          break;
        }
      }
      if (p)
        cout << i << "\n";
    }
    vector<int> f;
    for (int i = sqrt(x+1)-1; i; i--) {
      if (!(x%i))
        cout << x/i-1 << "\n";
    }
  }
  return 0;
}
