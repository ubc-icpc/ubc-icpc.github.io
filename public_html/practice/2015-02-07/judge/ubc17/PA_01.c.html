#include <stdio.h>
#include <stdlib.h>

char board[512][512];

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int m, n, i, j, sx, sy;
        scanf("%d %d", &m, &n);
        for (i = 0; i < m; ++i) {
            scanf("%s", board[i]);
            for (j = 0; j < n; ++j) {
                if ('S' == board[i][j]) {
                    sx = j;
                    sy = i;
                }
            }
        }
        int total = 0;
        for (i = 0; i < m; ++i) {
            for (j = 0; j < n; ++j) {
                if ('B' == board[i][j]) {
                    total += (abs(i - sy) + abs(j - sx));
                }
            }
        }
        printf("%d seconds to blocks get!\n", total << 1);
    }
    return 0;
}
