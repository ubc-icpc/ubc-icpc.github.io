#include <stdio.h>
#include <math.h>

#define ull unsigned long long

int main() {
    while (1) {
        ull s;
        scanf("%llu", &s);
        if (!s) break;

        // Find approximate value of n
        ull n = (ull)sqrt(2.0 * s);
        ull act;

again:
        act = (n * (n + 1)) >> 1;

        //printf("n %llu act %llu\n", n, act);

        if (act > s) {
            printf("%llu\n", act - s);
        } else if (act < s) {
            ++n;
            goto again;
        } else {
            printf("Not bored enough!\n");
        }
    }
    return 0;
}
