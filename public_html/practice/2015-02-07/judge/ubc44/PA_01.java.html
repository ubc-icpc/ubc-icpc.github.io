import java.util.*;
import java.lang.Math;

public class blocks {

	public static void main(String[] args) {
		Scanner s = new Scanner(System.in);
		int numCases = s.nextInt();
		for(int i=0;i<numCases;i++) {
			int M = s.nextInt();
			int N = s.nextInt();
			char[][] room = new char[M][N];
			int startR = 0;
			int startC = 0;
			int sum = 0;
			for(int row=0;row<M;row++) {
				String rowstr = s.next();
				for(int col=0;col<N;col++){
					room[row][col] = rowstr.charAt(col);
					if (room[row][col] == 'S') {
						startR = row;
						startC = col;
					}	
				}
			}
			for(int row=0;row<M;row++) {
				for(int col=0;col<N;col++){
					if (room[row][col] == 'B') {
						sum += 2*(Math.abs(row-startR) + Math.abs(col-startC));
					}	
				}
			}
			System.out.println(sum + " seconds to blocks get!");
		}
		s.close();
	}

}
