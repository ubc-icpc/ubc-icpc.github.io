#include <iostream>
#include <cmath>

using namespace std;

int main(){
	long long int S,guess;
	long int m,t;
	while (1){
		cin>>S;
		if(S==0){
			break;
		}

		m=(sqrt(2)*sqrt(S));
		guess=m*(m+1)/2;
//		cout<<"Guess:"<<m<<endl;

		//cout<<"Actual:"<<m<<endl;
		
		t=guess-S;
		if (t==0)
			cout<<"Not bored enough!"<<endl;
		else if (t<0)
			cout<<t+m+1<<endl;
		else
			cout<<t<<endl;
	}

	return 0;
}
