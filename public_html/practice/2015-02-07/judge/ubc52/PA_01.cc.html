#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

typedef pair<int,int> ints ;

int main(){
	int T, M, N, tot;
	ints start;
	vector<ints> blocks;
	char c;

	cin>>T;

	while(T--){
		blocks.clear();
		cin>>M;
		cin>>N;
		for(int m=0; m<M;m++){
			for (int n=0;n<=N;n++){
				cin.get(c);
				switch (c){
					case 'B':
						blocks.push_back(pair<int,int>(m,n));
						break;
					case 'S':
						start=pair<int,int>(m,n);
				}
			}
		}

		//sum everything up
		tot=0;
		for(vector<ints >::iterator i=blocks.begin();i<blocks.end();i++){
			tot+=2*abs((*i).first-start.first);
			tot+=2*abs((*i).second-start.second);
		}

	cout<<tot<<" seconds to blocks get!"<<endl;
	}

	
	return 0;
}
