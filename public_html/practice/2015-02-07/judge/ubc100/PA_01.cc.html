#include<iostream>
#include<vector>
#include<utility>

using namespace std;

int main (void) {
  int T;
  cin >> T;
  while(T--) {
    int M,N;
    cin >> M;
    cin >> N;
    
    int start[2] = {0,0};
    vector<pair<int, int> > blocks;
    char in;
    for ( int i = 0; i<M; i++ ) {
      for( int k = 0; k<N; k++ ) {

	cin >> in;
	if(in == 'B') {
	  pair<int, int> temp(i , k);
	  blocks.push_back(temp);
	} else if( in == 'S' ) {
	  start[0] = i;
	  start[1] = k;
	}

      }
    }
    int res = 0;
    for( int n = 0; n < blocks.size(); n++ ) {
      int dx = blocks[n].first - start[0];
      if(dx < 0) dx = -dx;

      int dy = blocks[n].second - start[1];
      if(dy < 0) dy = -dy;

      res += 2*(dx + dy);
    }
    cout << res << " seconds to blocks get!" << endl;
  }
  return 0;
}
