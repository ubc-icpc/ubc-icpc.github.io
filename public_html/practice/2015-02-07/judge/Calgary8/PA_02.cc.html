#include <iostream>
#include <stdlib.h>

using namespace std;

int main() {
    int T, m, n;
    cin >> T;

    for (int t = 0; t < T; t++) {
        cin >> m >> n;

        char grid[m][n];

        int x, y;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                cin >> grid[i][j];
                if (grid[i][j] == 'S') {
                    x = i;
                    y = j;
                }
            }
        }

        int steps = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 'B') {
                    steps += abs(x - i) * 2;
                    steps += abs(y - j) * 2;
                }
            }
        }

        cout << steps << " seconds to blocks get!" << endl;
    }
    return 0;
}
