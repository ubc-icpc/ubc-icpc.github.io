	public static long bSearch(long comp, long low, long high) {
			long mid= 0;
		while (low < high) {

			long midSum = (mid * (mid + 1)) / 2;
			if (comp > midSum)
				low=mid+1;
				//return bSearch(comp, mid+1, high);
			else if (comp < midSum)
				high = mid;
				//return bSearch(comp, low, mid-1);

				return mid;
		}
		return low;
