#include <iostream>
#include <string.h>

using namespace std;

char grid[501][501];

int startx, starty;

int main() {
    int T; cin >> T;
    while (T--) {
        memset(grid, 0, sizeof(grid));
        int M, N; cin >> M >> N;
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                cin >> grid[i][j];
                if (grid[i][j] == 'S') {
                    startx = i; starty = j;
                }
            }
        }
        int total = 0;
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                if (grid[i][j] == 'B') {
                    total += (i - startx >= 0 ? i - startx : startx - i);
                    total += (j - starty >= 0 ? j - starty : starty - j);
                }
            }
        }
        cout << 2 * total << " seconds to blocks get!" << endl;
    }
}
