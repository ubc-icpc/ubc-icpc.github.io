#include <iostream>
#include <cmath>
#include <sstream>

using namespace std;

int main() {
    long long i; cin >> i;
    while (i != 0) {
        long long n = (long long)ceil((sqrt(i * 8 + 1) - 1) / 2);
        long long result = ((n * (n + 1)) / 2) - i;
        stringstream ss;
        ss << result;
        cout << (result == 0 ? "Not bored enough!" : ss.str()) << endl;
        cin >> i;
    }
}
