#include<iostream>
#include<map>
#include<vector>
#include<list>
#include<string>
#include<cmath>

using namespace std;

int main() {
  int t;
  cin >> t;
  while(t--) {
    int m,n;
    cin >> m >> n;
    vector< vector<char> > v( 500, vector<char>( 500 ) );
    long long ans = 0;
    int sx,sy;
    for(int a = 0; a < m; a++)
      for(int b = 0; b < n; b++) {
	cin >> v[a][b];
	if(v[a][b] == 'S') {
	  sx = a;
	  sy = b;
	}
      }

    for(int i = 0; i < m; i++)
      for(int j = 0; j < n; j++) {
	if(v[i][j] == 'B') {
	  ans += 2*(abs(sx-i))+ 2*abs(sy-j);
	}
      }
    
    cout << ans << " seconds to blocks get!" << endl;
  }
}
