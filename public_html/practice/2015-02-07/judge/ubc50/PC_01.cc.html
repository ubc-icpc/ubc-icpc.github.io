using namespace std;

int main() {
  long long s;
  cin >> s;
  while(s) {
    long long ans = 0;
    long long x = (0.5)*(-1+sqrt(1+8*s));
    long long csum = (x+1)*(x+2)/2;
    if(floor(-1+sqrt(1+8*s)) == (-1+sqrt(1+8*s)))
      ans = 0;
    else
      ans = csum - s;
    
    if(ans > 0)
      cout << ans << endl;
    else
      cout << "Not bored enough!" << endl;
    
    cin >> s;
  }
}
