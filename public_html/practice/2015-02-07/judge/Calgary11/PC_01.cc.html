#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <math.h>

using namespace std;

typedef long long ll;

int main() {
    ll s;
    while ((cin >> s) && s != 0) {
        ll n = ceil((1 + sqrt(1 + 8*s)) / 2);
        ll x = (n * (n - 1)) / 2;
        ll diff = x - s;
        if (diff == 0) {
            cout << "Not bored enough!" << endl;
        } else {
            cout << diff << endl;
        }
    }
    return 0;
}
