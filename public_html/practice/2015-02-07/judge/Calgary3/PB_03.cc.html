#include <iostream>
using namespace std;

int a[400005];
int Left[400005];
int Right[400005];

int main () {
	int n;
	while(cin >> n && n > 1) {
		for(int i = 0; i < n; i++)
			cin >> a[i];
		int maxi = 0;
		int mini = 1000000000;
		for(int i = 0; i < n; i++) {
			maxi = max(maxi, a[i]);
			mini = min(mini, a[i]);
			Left[i] = maxi-mini;
		}
		
		maxi = 0;
		mini = 1000000000;
		for(int i = n-1; i >= 0; i--) {
			maxi = max(maxi, a[i]);
			mini = min(mini, a[i]);
			Right[i] = maxi-mini;
		}
		
		int res = 0;
		mini = 1000000000;
		for(int i = 0; i < n-1; i++)
			if(max(Left[i], Right[i+1]) < mini){
				mini = max(Left[i], Right[i+1]);
				res = i;
			}
		cout << (res+1) << endl;
	}
	return 0;
}
