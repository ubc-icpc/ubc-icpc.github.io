#include <iostream>
#include <cmath>
using namespace std;

long long folan(double x) {
	return sqrt(2.0*x);
}

int main () {
	long long sum;
	while(cin >> sum && sum != 0) {
		long long k = folan(sum);
		if(k*(k+1)/2 == sum)
			cout << "Not bored enough!" << endl;
		else if(k*(k+1)/2 < sum)
			cout << int(((k+2)*(k+1)/2)-sum) << endl;
		else
			cout << int((k*(k+1)/2)-sum) << endl;
	}
	return 0;
}
