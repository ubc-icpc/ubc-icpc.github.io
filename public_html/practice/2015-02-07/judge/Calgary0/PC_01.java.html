import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Scanner;

public class C {

	private void work() {
		Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(
				System.in)));

		while (true) {
			long x = sc.nextLong();
			if (x == 0)
				break;
			long lo = 1;
			long hi = 1L << 30;
			while (lo < hi) {
				long m = (lo + hi) / 2;
				long t = m * (m + 1) / 2;
				if (t >= x)
					hi = m;
				else
					lo = m + 1;
			}
			long diff = lo * (lo + 1) / 2 - x;
			System.out.println(diff == 0 ? "Not bored enough!" : diff);
		}

		sc.close();
	}

	public static void main(String[] args) {
		C ccc = new C();
		ccc.work();
	}
}
