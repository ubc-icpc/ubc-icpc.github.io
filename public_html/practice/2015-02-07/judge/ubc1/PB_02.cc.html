#include <bits/stdtr1c++.h>

using namespace std;

int f(multiset<int>& s) {
    if (s.size() == 0) return 0;
    auto it = s.end();
    return *(--it) - *s.begin();
}

int main() {
    int n;
    while (cin >> n) {
        if (n == 1) break;
        multiset<int> s[2];
        vector<int> v(n);
        for (int i = 0; i < n; i++) {
            cin >> v[i];
            s[1].insert(v[i]);
        }
        
        int ans = f(s[1]), ans_i = 0;
        for (int i = 0; i < n; i++) {
            s[0].insert(v[i]);
            s[1].erase(s[1].find(v[i]));
            if (max(f(s[0]), f(s[1])) < ans) {
                ans = max(f(s[0]), f(s[1]));
                ans_i = i;
            }
        }
        cout << ans_i+1 << endl;
    }
	return 0;
}
