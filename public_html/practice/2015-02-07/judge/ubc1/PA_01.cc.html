#include <bits/stdtr1c++.h>

using namespace std;

typedef pair<int, int> pii;

int main() {
    int t; cin >> t;
    while (t--) {
        int n, m; cin >> n >> m;
        vector<pii> locs;
        int sx, sy;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                char c; cin >> c;
                if (c == 'S') {
                    sx = i;
                    sy = j;
                } else if (c == 'B') {
                    locs.push_back({i,j});
                }
            }
        }
        
        int ans = 0;
        for (auto p : locs) {
            ans += abs(p.first-sx) + abs(p.second-sy);
        }
        cout << 2*ans << " seconds to blocks get!" << endl;
    }
	return 0;
}
