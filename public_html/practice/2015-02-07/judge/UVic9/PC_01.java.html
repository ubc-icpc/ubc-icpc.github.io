import java.util.Scanner;

public class C
{
	public static void main(String[] args)
	{
		//test();
		Scanner in = new Scanner(System.in);
		while(in.hasNextLine())
		{
			long a = in.nextLong();
			if (a==0){System.exit(0);}
			else{answer(a); in.nextLine();}
		}
	}
	
	public static void test()
	{
		//Find our range
		System.out.println(sum_of(9L));
		System.out.println((1L)<<56);
		System.out.println(sum_of(400000000L));
		
		answer(44);
		answer(45);
		answer(46);
	}
	
	public static long sum_of(long n){return (n*(n+1))>>1;}//Macros...
	
	public static void answer(long val)
	{
		long min_n = 0L;
		long max_n = 400000000L;
		while(max_n - min_n > 1)
		{
			long central_n = (max_n + min_n)>>1;
			if (sum_of(central_n) > val)
			{
				max_n = central_n;
			}
			else
			{
				min_n = central_n;
			}
		}
		//System.out.println(min_n);
		long missed_value = sum_of(max_n) - val;
		if (missed_value == max_n)
		{System.out.println("Not bored enough!");}
		else{System.out.println(missed_value);}
	}
}
