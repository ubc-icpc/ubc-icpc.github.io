import java.util.Scanner;

public class Main
{
	static final int BLOCK = (int)'B';
	static final int START = (int)'S';//not "."  -_-
	static final int EMPTY = (int)'.';
	
	public static void main(String[] args)
	{
		Scanner in = new Scanner(System.in);
		int cases = in.nextInt(); in.nextLine();
		for(int i = 0; i < cases; i++)
		{
			do_case(in);
		}
	}
	
	public static void do_case(Scanner in)
	{
		//Build room
		int m = in.nextInt();
		int n = in.nextInt(); in.nextLine();
		int[][] room = new int[m][n];
		int startR = -1;
		int startC = -1;
		for (int i = 0; i < m; i++)
		{
			String line = in.nextLine();
			char[] line_chars = line.toCharArray();
			for(int j = 0; j < n; j++)
			{
				room[i][j] = (int)line_chars[j];
				if (room[i][j]==START)
				{
					startR = i; startC = j;
				}
			}
		}
		//Calc time
		int time = 0;
		for(int i = 0; i < m; i++)
		{
			for(int j = 0; j < n; j++)
			{
				if (room[i][j] == BLOCK)
				{
					int dx = j - startC;
					int dy = i - startR;
					if (dx<0){dx = -dx;}
					if (dy<0){dy = -dy;}
					time += 2*(dx+dy);
				}
			}
		}
		//
		System.out.println(time + " seconds to blocks get!");
	}
}
