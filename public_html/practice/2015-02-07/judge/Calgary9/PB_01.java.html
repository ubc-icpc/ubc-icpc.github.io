import java.util.ArrayList;
import java.util.Scanner;

public class B {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		while (sc.hasNext()) {
			int n = sc.nextInt();
			if (n != 1) {
				ArrayList<Integer> num = new ArrayList<>();
				ArrayList<Integer> minleft = new ArrayList<>();
				ArrayList<Integer> maxleft = new ArrayList<>();
				ArrayList<Integer> minright = new ArrayList<>();
				ArrayList<Integer> maxright = new ArrayList<>();
				int mnleft=10000000;
				int mxleft = 0;
				int mnright = 10000000;
				int mxright = 0;
				for (int i = 0; i < n; i++) {
					int temp = sc.nextInt();
					if(temp>mxleft)
						mxleft = temp;
					if(temp<mnleft)
						mnleft = temp;
					minleft.add(mnleft);
					maxleft.add(mxleft);
					num.add(temp);
				}
				for(int i=n-1 ; i>=0 ; i--){
					int temp = num.get(i);
					if(temp>mxright)
						mxright = temp;
					if(temp<mnright)
						mnright = temp;
					minright.add(mnright);
					maxright.add(mxright);
				}
				int minimum = 10000000;
				int ind = 0;
//				for(int i=0 ; i<n ; i++){
//					System.out.println(maxleft.get(i) - minleft.get(i));
//				}
				for(int i=0 ; i<n-1 ; i++){
					int temp = Math.max((maxleft.get(i)-minleft.get(i)), (maxright.get(n-i-2)-minright.get(n-i-2)));
//					System.out.println(temp + "  HIII");
					if(temp<minimum){
						minimum = temp;
						ind = i;
					}
				}
				System.out.println(ind+1);
			}
			else{
				System.exit(0);
			}
		}
	}

}
