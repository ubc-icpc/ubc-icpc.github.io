import java.util.Scanner;

public class C {
	static long number = 400000000;
	private static Scanner sc;

	public static void main(String[] args) {

		sc = new Scanner(System.in);
		while (sc.hasNext()) {
			long n = sc.nextLong();
			if(n!=0)
				binarysearch(n, 0, number);
			else
				System.exit(0);
		}
	}

	public static void binarysearch(long x, long begin, long end) {
//		System.out.println("begin: "+ begin+ "   end: "+end);
//		System.out.println(Math.pow(2, 56));
		long index = (begin + end) / 2;
		long temp = index * (index + 1) / 2;
		if (x == temp) {
			System.out.println("Not bored enough!");
		} else if (end == begin) {
			System.out.println(temp-x);
		} else if (x > temp) {
			binarysearch(x, (begin + end+1) / 2, end);
		} else if (x < temp) {
			binarysearch(x, begin, (begin + end) / 2);
		}
	}
}
