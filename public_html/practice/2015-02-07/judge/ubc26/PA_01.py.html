import sys

def N(): return sys.stdin.next()
t = int(N())
for i in range(t):
    m, n = map(int, N().split(' '))
    yx = []
    for j in range(m):
        yx.append(list(sys.stdin.next())[0:-1])

    sx, sy = 0, 0
    for y in range(m):
        for x in range(n):
            if yx[y][x] == 'S':
                sx, sy = x, y
                break

    d = 0
    for y in range(m):
        for x in range(n):
            if yx[y][x] == 'B':
                d += abs(y - sy) + abs(x - sx)
    d *= 2
    print "%d seconds to blocks get!" % d
