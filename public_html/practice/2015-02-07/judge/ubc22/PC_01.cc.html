#include<iostream>
#include<cmath>
using namespace std;

int main () {
  while (true) {
    long long N; cin >> N;
    if (N == 0) break;
    double d = sqrt(N * 8 + 1);
    long long l = d;
    if ((double) l == d) {
      cout << "Not bored enough!" << endl;
      continue;
    }
    l = ceil((d + 0.00000001 - 1) / 2);
    cout << (l + 1) * l / 2 - N << endl;
  }
  return 0;
}
