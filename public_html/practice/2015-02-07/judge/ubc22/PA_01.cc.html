#include<iostream>
#include<string>
#include<vector>
#include<cmath>
using namespace std;

int main () {
  int T; cin >> T;
  while (T--) {
    int M, N; cin >> M >> N;
    vector<string> map;
    for (int i = 0; i < M; ++i) {
      string str; cin >> str;
      map.push_back(str);
    }
    int sx, sy;
    for (int i = 0; i < M; ++i)
      for (int j = 0; j < N; ++j)
	if (map[i][j] == 'S') {
	  sx = i; sy = j;
	}
    long long total = 0;
    for (int i = 0; i < M; ++i)
      for (int j = 0; j < N; ++j)
	if (map[i][j] == 'B') {
	  total += 2 * abs(sx - i) + 2 * abs(sy - j);
	}
    cout << total << " seconds to blocks get!" << endl;
  }
  return 0;
}
