#include<iostream>
#include<vector>
#include<limits>
using namespace std;

int main () {
  while (true) {
    int N; cin >> N;
    if (N == 1) break;
    vector<int> v(N);
    for (int i = 0; i < N; ++i) cin >> v[i];
    int MAXl = -1;
    int MINl = numeric_limits<int>::max();
    vector<int> l(N - 1);
    int MAXr = -1;
    int MINr = numeric_limits<int>::max();
    vector<int> r(N - 1);
    int res = numeric_limits<int>::max();
    for (int i = 0, j = N - 2; i < N - 1; ++i, --j) {
      if (MAXl < v[i]) MAXl = v[i];
      if (MINl > v[i]) MINl = v[i];
      if (MAXr < v[j + 1]) MAXr = v[j + 1];
      if (MINr > v[j + 1]) MINr = v[j + 1];
      l[i] = MAXl - MINl;
      r[j] = MAXr - MINr;
    }
    int index = -1;
    for (int i = 0; i < N - 1; ++i) {
      int m = max(l[i], r[i]);
      if (res > m) {
	index = i;
	res = m;
      }
    }
    cout << index + 1 << endl;
  }
  return 0;
}
