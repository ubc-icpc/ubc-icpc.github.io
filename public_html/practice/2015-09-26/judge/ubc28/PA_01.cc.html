#include <bits/stdtr1c++.h>

using namespace std;

map<char, int> val({{'A', 4}, {'K', 3}, {'Q', 2}, {'J', 1}});
int main() {
    int t; cin >> t;
    while (t--) {
        vector<string> hand(13);
        for (int i = 0; i < 13; i++) {
            cin >> hand[i];
        }
        
        int hcp = 0;
        for (string s : hand) {
            hcp += val[s[0]];
        }
        
        bool bal = false;
        map<char, int> suit;
        for (string s : hand) {
            suit[s[1]]++;
        }
        
        if (min(suit['S'], min(suit['C'], min(suit['H'], suit['D']))) >= 2) {
            int c = 0;
            for (auto p : suit) if (p.second == 2) c++;
            bal = c <= 1;
        }
        
        if (15 <= hcp && hcp <= 17 && bal) cout << "1NT" << endl;
        else cout << "Pass" << endl;
    }
	return 0;
}
