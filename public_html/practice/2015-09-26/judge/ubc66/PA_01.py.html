for t in range(int(raw_input())):
    hand = raw_input().split()

    HCs = {'A':4, 'K':3, 'Q':2, 'J':1}

    HCP = 0
    S = 0
    D = 0
    H = 0
    C = 0

    for card in hand:
        suit = card[1]
        val = card[0]
        if val in HCs:
            HCP += HCs[val]

        if suit == 'S':
            S += 1
        elif suit == 'D':
            D += 1
        elif suit == 'H':
            H += 1
        elif suit == 'C':
            C += 1

    dist = sorted([S,D,H,C])

    # print(HCP, dist)

    if dist in ([3,3,3,4], [2,3,4,4], [2,3,3,5]) and HCP >= 15 and HCP <= 17:
        print('1NT')
    else:
        print('Pass')
