#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;

int main () {
	int t; cin >> t;
	for (int i=0; i<t; i++){
		int handvalue = 0;
		bool highcard = false;
		int s=0; int d=0; int c=0; int h=0;
		for (int j=0; j<13; j++){
			string str;
			cin >> str[0] >> str[1];
			if (str[0] == 'A'){
				handvalue += 4;
			} else if (str[0] == 'K'){
				handvalue += 3;
			} else if (str[0] == 'Q'){
				handvalue += 2;
			} else if (str[0] == 'J'){
				handvalue += 1;
			} else {}
			
			if (str[1] == 'S'){
				s++;
			} else if (str[1] == 'D'){
				d++;
			} else if (str[1] == 'C'){
				c++;
			} else if (str[1] == 'H'){
				h++;
			}
		}
		if ((handvalue>=15)&&(handvalue<=17)){
			highcard = true;
		}
		bool balanced = false;
		int A[4];
		A[0] = s; A[1] = d; A[2] = c; A[3] = h;
		sort(A,A+4);
		if (((A[0]==3)&&(A[1]==3)&&(A[2]==3)&&(A[3]==4))||((A[0]==2)&&(A[1]==3)&&(A[2]==4)&&(A[3]==4))||((A[0]==2)&&(A[1]==3)&&(A[2]==3)&&(A[3]==5))){
			balanced = true;
		} 

		if((highcard==true)&&(balanced==true)){
			cout << "1NT" << endl;
		} else {
			cout << "Pass" << endl;
		}
	}
	return 0;
}
