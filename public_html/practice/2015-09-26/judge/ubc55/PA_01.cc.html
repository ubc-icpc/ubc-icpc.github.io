#include <iostream>
#include <algorithm>
#include <vector>
#include <string>

using namespace std;

int get_points(string data) {
	if (data[0] == 'A')
		return 4;
	else if (data[0] == 'K')
		return 3;
	else if (data[0] == 'Q')
		return 2;
	else if (data[0] == 'J')
		return 1;
	else
		return 0;
}

bool get_balanced(vector<string> data) {
	int c = 0, d = 0, h = 0, s = 0;
	for (int i = 0; i<13; i++) {
		if (data[i][1] == 'C')
			c++;
		else if (data[i][1] == 'D')
			d++;
		else if (data[i][1] == 'H')
			h++;
		else if (data[i][1] == 'S')
			s++;
	}
	
	if(c < 2 || d < 2 || h < 2 || s < 2)
		return false;
	int count = 0;
	if (c == 2)
		count++;
	if (d == 2)
		count++;
	if (h == 2)
		count++;
	if (s == 2)
		count++;

	if (count <= 1)
		return true;
	return false;
}

int main() {
	int T; cin >> T;

	while(T--) {
		vector<string> data(13);
		for (int i = 0; i<13; i++)
			cin >> data[i];

		int HCP = 0;
		for (int i = 0; i<13; i++)
			HCP += get_points(data[i]);

		bool balanced = false;
		balanced = get_balanced(data);

		if ((15 <= HCP) && (HCP <= 17) && (balanced))
			cout << "1NT";
		else
			cout << "Pass";
		cout << endl;
	}

	return 0;
}
