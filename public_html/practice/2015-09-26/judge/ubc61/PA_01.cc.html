#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n; cin >> n;
	while(n--) {
		int hcp = 0;
		vector<int> suits(4);
		// c = d = h = s;
		for (int i=0; i<13; i++) {
			string s; cin >> s;
			char c1 = s[0];
			char c2 = s[1];
			if (c1 == 'A') hcp += 4;
			if (c1 == 'K') hcp += 3;
			if (c1 == 'Q') hcp += 2;
			if (c1 == 'J') hcp++;
			if (c2 == 'C') suits[0]++;
			if (c2 == 'D') suits[1]++;
			if (c2 == 'H') suits[2]++;
			if (c2 == 'S') suits[3]++;
		}
		sort(suits.begin(),suits.end());
		bool bal = false;
		if (suits[0] == 2 && suits[1] == 3 && suits[2] == 3 && suits[3] == 5) {
			bal = true;
		}
		if (suits[0] == 2 && suits[1] == 3 && suits[2] == 4 && suits[3] == 4) {
			bal = true;
		}
		if (suits[0] == 3 && suits[1] == 3 && suits[2] == 3 && suits[3] == 4) {
			bal = true;
		}
		if (hcp >= 15 && hcp <= 17 && bal) {
			cout << "1NT" << endl;
		} else {
			cout << "Pass" << endl;
		}
	}	
	return 0;
}
