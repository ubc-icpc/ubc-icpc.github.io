#include <bits/stdc++.h>
using namespace std;

using ui = unsigned int;
using ull = unsigned long long;
using ll = long long;
using vs = vector<string>;
using vi = vector<int>;
using vull = vector<ull>;
using vll = vector<ll>;
using ss = set<string>;
using si = set<int>;
using sll = set<ll>;
using sull = set<ull>;
using uss = unordered_set<string>;
using usi = unordered_set<int>;
using usll = unordered_set<ll>;
using usull = unordered_set<ull>;

int main()
{
	int t; cin >> t;
	while (t--)
	{
		double m, b; cin >> b >> m;

		printf("%.1lf\n", 1 + (1.0 / (1 - (m / 2 + b))) * (m / 3 + b / 2));
	}

	return 0;
}
