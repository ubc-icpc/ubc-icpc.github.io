#include <bits/stdc++.h>
using namespace std;

using ui = unsigned int;
using ull = unsigned long long;
using ll = long long;
using vs = vector<string>;
using vi = vector<int>;
using vull = vector<ull>;
using vll = vector<ll>;
using ss = set<string>;
using si = set<int>;
using sll = set<ll>;
using sull = set<ull>;
using uss = unordered_set<string>;
using usi = unordered_set<int>;
using usll = unordered_set<ll>;
using usull = unordered_set<ull>;

int main()
{
	int t; cin >> t;
	while (t--)
	{
		vector<string> cards(13);
		unordered_map<char, int> m;
		for (int i = 0; i < 13; i++)
		{
			string x; cin >> x;
			cards[i] = x;
			m[x[1]]++;
		}

		vector<int> dist;
		for (auto& i : m)
		{
			dist.push_back(i.second);
		}
		sort(dist.begin(), dist.end());

		bool balanced = false;
		if (dist[0] == 3 && dist[1] == 3 && dist[2] == 3 && dist[3] == 4) balanced = true;
		if (dist[0] == 2 && dist[1] == 3 && dist[2] == 4 && dist[3] == 4) balanced = true;
		if (dist[0] == 2 && dist[1] == 3 && dist[2] == 3 && dist[3] == 5) balanced = true;

		if (balanced)
		{
			int value = 0;
			for (string& i : cards)
			{
				if (i[0] == 'A') value += 4;
				else if (i[0] == 'K') value += 3;
				else if (i[0] == 'Q') value += 2;
				else if (i[0] == 'J') value += 1;
			}
			balanced = (value >= 15 && value <= 17);
		}

		cout << (balanced ? "1NT" : "Pass") << endl;
	}
	return 0;
}
