using mm_t = vector<vector<short>>;
const short sm = numeric_limits<short>::max();

short dp(mm_t& mm, short e, short E)
{
	if (e > E) return sm;
	if (mm[e][E]) return mm[e][E];

	short best = sm;
	if (e > 0)
	{
		best = min(best, (short) (1 + dp(mm, max(e - E / 2, 0), E)));
	}

	int ss = 10;
	for (int i = 1; e + ss <= E - i; i++)
	{
		best = min(best, (short) (1 + dp(mm, e + ss, E - i)));
		ss += 10 + 3 * i;
	}
	return mm[e][E] = best;
}
