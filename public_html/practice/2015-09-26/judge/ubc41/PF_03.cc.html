#include <bits/stdc++.h>
using namespace std;
using ull = unsigned long long;
using ld = long double;

int h, v, l, s;

ld d(double theta)
{
    // d = vi t + 1/2 a t t
    // 5 t^2 - vy t + d = 0
    ld vx = v * cos(theta), vy = v * sin(theta);
    ld b = vy / 2;
    ld t = (b + sqrt(b * b + 5 * h)) / 5;
    return min(l + t * s, t * vx);
}
ld ss(ld lo, ld hi)
{
    while (hi - lo > 1E-8)
    {
        ld hi_ = (hi + hi + lo) / 3, lo_ = (lo + lo + hi) / 3;
        if (d(hi_) > d(lo_)) lo = lo_;
        else hi = hi_;
    }
    return (hi + lo) / 2;
}

int main()
{
    while (cin >> h >> v >> l >> s)
    {
        ld n = d(ss(0, M_PI_2));
        if (n >= l) cout << fixed << setprecision(3) << n << endl;
        else cout << "BAD THROW" << endl;
    }
}
