#include <iostream>
using namespace std;

int s[4];

int main() {
  int t; cin >> t;
  while (t--) {
    int hcp = 0;
    for (int i = 0; i < 4; i++)
      s[i] = 0;
    for (int i = 0; i < 13; i++) {
      string S; cin >> S;
      char v = S[0], u = S[1];
      if (v == 'A')
        hcp += 4;
      else if (v == 'K')
        hcp += 3;
      else if (v == 'Q')
        hcp += 2;
      else if (v == 'J')
        hcp++;
      if (u == 'D')
        s[0]++;
      else if (u == 'C')
        s[1]++;
      else if (u == 'H')
        s[2]++;
      else if (u == 'S')
        s[3]++;
    }
    bool pass = true;
    if (hcp >= 15 && hcp <= 17) {
      int two = 0;
      for (int i = 0; i < 4; i++) {
        if (s[i] < 2) {
          two = -1;
          break;
        }
        if (s[i] == 2)
          two++;
      }
      if (two == 1 || two == 0)
        pass = false;
    }
    cout << (pass ? "Pass" : "1NT") << "\n";
  }
  return 0;
}
