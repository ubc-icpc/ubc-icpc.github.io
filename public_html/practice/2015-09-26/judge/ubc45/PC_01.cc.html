#include <iostream>
#include <iomanip>
#include <random>
#include <cmath>
using namespace std;

typedef long double ld;

ld b, m;

inline ld X(ld a) {
  if (!m)
    return a/b;
  return (-b+sqrt(b*b+2*m*a))/m;
}

int main() {
  int t; cin >> t;
  cout << fixed << setprecision(1);
  default_random_engine gen(random_device{}());
  uniform_real_distribution<ld> dist(0.0L,1.0L);
  while (t--) {
    cin >> b >> m;
    ld d = 0.0;
    ld t = 0.0;
    int n = 0;
    ld h = m/2+b;
    for (int i = 0; i < 1000000; i++) {
      ld a = dist(gen);
      //cout << "gen " << a << endl;
      if (a > h) {
        t += 1+d;
        //cout << "adding " << 1+d << endl;
        d = 0;
        n++;
      }
      else {
        d += X(a);
        //cout << "fail " << x << " " << d << endl;
      }
    }
    //cout << t << " " << n << endl;
    cout << t/n << "\n";
  }
  return 0;
}
