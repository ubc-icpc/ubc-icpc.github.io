#include <iostream>
#include <queue>
#include <algorithm>
using namespace std;

int k[9001][701];
int a[70];

int main() {
  for (int i = 1; i <= 9000; i++)
    for (int j = 1; j <= 700; j++)
      k[i][j] = -1;
  a[0] = 1;
  for (int i = 1; i < 70; i++)
    a[i] = 42424242;
  k[100][1] = 100;
  for (int d = 1; d < 700; d++) {
    for (int m = 100; m <= 9000; m++) {
      if (k[m][d] == -1)
        continue;
      k[m][d+1] = max(k[m][d+1],min(m,k[m][d]+m/2));
      int r = 0, cm = m, cc = k[m][d], e = 10;
      while (cc-e >= 0) {
        cc -= e;
        cm++;
        k[cm][d+1] = max(k[cm][d+1],cc);
        e += 3;
        r++;
        if (a[r] > d) {
          a[r] = d;
          if (r == 69)
            goto VELOCIRAPTOR;
        }
      }
    }
  }
  VELOCIRAPTOR:
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    cout << a[n] << "\n";
  }
  return 0;
}
