#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
#define FIL(a,b) memset((a),(b),sizeof(a))
#define SZ(a) ((int)(a).size())
#define ALL(a) begin(a),end(a)
#define PB push_back
#define FI first
#define SE second
typedef long long LL;
typedef pair<int,int> PT;
typedef complex<double> PX;
typedef vector<int> VI;
typedef vector<VI> VVI;
typedef vector<PT> VPT;
template<typename T> ostream& operator<<(ostream& s, vector<T>& v)
{ s << '{'; FOR(i,0,v.size()) s << (i ? "," : "") << v[i]; return s << '}'; }
template<typename S, typename T> ostream& operator<<(ostream &s, pair<S,T> const& p)
{ return s << '(' << p.first << ',' << p.second << ')'; }

int N, Tmax, D[10000];

bool works(int K) {
	multiset<int> ms;
	int curt = 0;
	FOR(n,0,N) {
		if (SZ(ms) == K) {
			auto it = begin(ms);
			curt = *it;
			ms.erase(it);
		}
		ms.insert(curt + D[n]);
	}
	return int(*ms.rbegin()) <= Tmax;
}

int main() {
	scanf("%d%d", &N, &Tmax);
	FOR(n,0,N) scanf("%d", D+n);
	int ka = 0, kb = N; // range (ka, kb]
	while (kb - ka > 1) {
		int k = (ka + kb) / 2;
		if (works(k)) kb = k; else ka = k;
	}
	printf("%d\n", kb);
}
