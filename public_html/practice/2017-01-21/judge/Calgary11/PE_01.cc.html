#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=int(a);i<int(b);i++)
#define FORB(i,a,b) for(int i=int(b-1);i>=int(a);i--)
#define FIL(a,b) memset((a),(b),sizeof(a))
#define SZ(a) ((int)(a).size())
#define ALL(a) begin(a),end(a)
#define PB push_back
#define FI first
#define SE second
typedef long long LL;
typedef pair<int,int> PT;
typedef complex<double> PX;
typedef vector<int> VI;

template<typename T> ostream& operator<<(ostream& s, vector<T>& v)
{ s << '{'; FOR(i,0,v.size()) s << (i ? "," : "") << v[i]; return s << '}'; }
template<typename S, typename T> ostream& operator<<(ostream &s, pair<S,T> const& p)
{ return s << '(' << p.first << ',' << p.second << ')'; }

int main() {
	int N;
	char gd[10][11];
	scanf("%d", &N);
	FOR(n,0,N) scanf("%s", gd[n]);
	int ans = 0;
	FORB(n,0,N) FORB(m,0,N) {
		if (gd[n][m] == '1') {
			FOR(n2,0,n+1) FOR(m2,0,m+1) gd[n2][m2] = (gd[n2][m2] == '1') ? '0' : '1';
			ans++;
		}
	}
	printf("%d\n", ans);
}
