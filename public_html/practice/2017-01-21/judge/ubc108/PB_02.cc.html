#include <bits/stdc++.h>
using namespace std;

int main() {
   ios::sync_with_stdio(0);
   string code;
   long long N, cmp;
   cin >> code >> N;
   // find max
   long long length = code.size();
   long long temp = 1;
   while (temp*length < N) {
      temp = temp << 1; // find power
   }
   while (temp != 1) {
      temp = temp >> 1;
      cmp = temp*length;
      if (N > cmp) {
         N -= cmp;
         N--;
         if (N == 0)
            N += cmp;
      }
   }
   cout << code[N-1];

   return 0;
}
