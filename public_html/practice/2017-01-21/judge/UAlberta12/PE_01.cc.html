#include <iostream>
using namespace std;

int a[12][12];
int n;

void flip(int mi,int mj)
{
    int i,j;
    for(i=0;i<=mi;i++)for(j=0;j<=mj;j++)a[i][j]=1-a[i][j];
}

int main()
{
    cin>>n;
    int i,j,ct=0;
    char t;
    for(i=0;i<n;i++)for(j=0;j<n;j++)
    {
        cin>>t;
        if(t=='0')a[i][j]=0;
        else a[i][j]=1;
    }
    while(1)
    {
        int mi=-1,mj=-1;
        for(i=0;i<n;i++)for(j=0;j<n;j++)if(a[i][j])mi=i,mj=j;
        if(mi==-1)break;
        flip(mi,mj);
        //for(i=0;i<n;i++){for(j=0;j<n;j++)cout<<a[i][j];cout<<endl;}
        ct++;
    }
    cout<<ct<<endl;
    return 0;
}
