#include <iostream>
#include <string>
using namespace std;

string s;
long long bs;

char f(long long l,long long n)
{
    //cout<<l<<" "<<n<<endl;
    if(n<=bs)return s[n-1];
    if(n<=l/2)return f(l/2,n);
    long long r=(n-1)/bs;
    if((n-1)%bs==0&&(r&(r-1))==0)return f(l/2,n-1);
    //cout<<"?"<<l/2<<" "<<l-n<<" "<<l/2-(l-n)-1<<endl;
    return f(l/2,l/2-(l-n)-1);
}

int main()
{
    long long n;
    cin>>s>>n;
    bs=s.size();
    long long l=bs;
    while(l<n)l*=2;
    cout<<f(l,n)<<endl;
    return 0;
}
