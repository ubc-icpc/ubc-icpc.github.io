#include <iostream>

using namespace std;

string s; long long int n;

int search(long long ind) {
	if(ind < s.size()) return ind;
	long long int pwr = s.size();
	while(2*pwr <= ind) {
		pwr*=2;
	}
	if(pwr == ind)
		return search(ind-1);
	return search(ind - pwr - 1);
}

int main() {
	cin >> s >> n; n--;
	cout << s[search(n)] << endl;
	return 0;
}
