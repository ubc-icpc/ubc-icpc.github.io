#include <bits/stdc++.h>
#define endl '\n'
typedef long long ll;
typedef long double ld;
using namespace std;

string s; ll N;

ll solve(ll n) {
	if (n < s.length())
		return n;
	ll cycles = (ll) floor(n / (double) s.length()) + 1;
	ll i = 1;
	while (i < cycles)
		i <<= 1;
	i >>= 1;
	return solve((n - 1 + (s.length() * i)) % (s.length() * i));
}

int main() {
//	ios::sync_with_stdio(0);
//	cin.tie(0);
	cin >> s;
	cin >> N;
	cout << s[solve(N-1)] << endl;
	return 0;
}
