#include<iostream>

using namespace std;

int main () {
	int T; cin >> T;
	bool t = true;
	for (int i = 1; i <= T; ++i) {
		int p[5];
		for (int j = 0; j < 5; ++j) {
			cin >> p[j];
		}
		do {
			t = false;
			for (int k = 0; k < 5; ++k) {
				if (p[k] < 0) {
					p[k] = -p[k];
					if (k == 0) {
						p[4] -= p[0];
						p[1] -= p[0];
					}
					else if (k == 4) {
						p[3] -= p[4];
						p[0] -= p[4];
					}
					else {
						p[k-1] -= p[k];
						p[k+1] -= p[k];
					}
					t = true;
				}
			}
		} while (t);
		cout << "Pentagon #" << i << ":" << endl;
		for (int j = 0; j < 5; ++j) {
			cout << p[j];
			if (j == 4)
				cout << endl << endl;
			else
				cout << " ";
		}
	}
}
