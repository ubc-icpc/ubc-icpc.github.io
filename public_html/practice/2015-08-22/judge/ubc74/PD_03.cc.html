#include<iostream>
#include<string>
#include<vector>
#include<math.h>

using namespace std;

int main () {
	int T; cin >> T;
	for (int i = 1; i <= T; ++i) {
		string ns;
		cin >> ns;
		int a_after, b_after;
		vector<char> n(ns.begin(), ns.end());
		bool match;
		for (int j = 1; j < int (ns.size()); ++j) {
			match = true;
			string a_before, b_before;
			for (int k = 0; k < j; ++k)
				a_before += n[k];
			for (int k = j; k < int (ns.size()); ++k)
				b_before += n[k];
			a_after = stoi(a_before);
			b_after = stoi(b_before);
			for (int m = 2; m <= int (sqrt(a_after)+1); ++m) {
				if (a_after%m == 0) {
					if (b_after%m == 0) {
						match = false;
						break;
					}
				}
			}
			if (b_after%a_after == 0)
				match = false;
			if (a_after == 1)
				match = true;
			if (match)
				break;
		}
		cout << "Ticket #" << i << ":\n";
		if (match)
			cout << a_after << " " << b_after << endl << endl;
		else
			cout << "Not relative" << endl << endl;
	}
}
