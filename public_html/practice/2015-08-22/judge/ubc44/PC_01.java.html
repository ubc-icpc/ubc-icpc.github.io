import java.util.Scanner;

public class pentagon {
	public static void main(String args[]){
		Scanner s = new Scanner(System.in);
		int N = s.nextInt();
		int k = 0;
		while (k++<N) {
			int pent[] = new int[5];
			for (int i=0;i<pent.length;i++)
				pent[i]=s.nextInt();
			while(!isPositive(pent) ) {
				int minInd=minIndex(pent);
				int val = pent[minInd];
				pent[(minInd+4)%5]+=val;
				pent[(minInd+1)%5]+=val;
				pent[minInd]= -val;	
			}
			System.out.println("Pentagon #"+k+":");
			for(int i=0; i<pent.length;i++)
				System.out.print(pent[i]+" ");
			System.out.println();
			System.out.println();
		}
		s.close();
	}

	public static boolean isPositive (int[] arr) {
		for (int i=0; i<arr.length;i++) {
			if (arr[i]<0) return false;
		}
		return true;
	}

	public static int minIndex(int[] arr) {
		int min = arr[0];
		int ind = 0;
		for(int i=0;i<arr.length;i++) {
			if (arr[i]<min) {
				min=arr[i];
				ind=i;
			}
		}
		return ind;
	}
}
