import java.util.Scanner;

public class sweet {

	public static void main (String args[]) {
		Scanner s = new Scanner(System.in);
		int k = 1;
		int sum = 0;
		int money = s.nextInt();
		while (money!=0) {
			sum += money;
			if (sum>=100)
				System.out.println("Input #"+k+": Totally Sweet!");
			else if (sum>=50)
				System.out.println("Input #"+k+": Sweet!");
			sum %= 50;
			k++;	
			money = s.nextInt();
		}
		s.close();
	}
	
}
