#include <iostream>
#include <algorithm>
using namespace std;

int b[100][100], s[100][100];

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    cout << "Game Board #" << T << ": ";
    int r, c;
    cin >> r >> c;
    for (int i = 0; i < r; i++) {
      for (int j = 0; j < c; j++) {
        s[i][j] = 0;
        if (i+j == 0 || i+j == r+c-2) {
          char C; cin >> C;
          b[i][j] = 0;
        }
        else
          cin >> b[i][j];
      }
    }
    for (int i = 0; i < r; i++) {
      for (int j = 0; j < c; j++) {
        int mx = 0;
        if (j > 0)
          mx = max(mx,s[i][j-1]);
        if (i > 0)
          mx = max(mx,s[i-1][j]);
        s[i][j] = mx+b[i][j];
      }
    }
    cout << s[r-1][c-1] << "\n\n";
  }
  return 0;
}
