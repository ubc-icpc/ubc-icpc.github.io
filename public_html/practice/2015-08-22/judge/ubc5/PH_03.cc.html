#include <iostream>
using namespace std;

int sa[100];
int ss[100];
char sx[100];

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    int n, m;
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
      cin >> sa[i] >> ss[i] >> sx[i];
      sa[i]--; ss[i]--;
    }
    int a = 0;
    for (int i = 0; i < (1<<n); i++) { // 1 truth, 0 contra
      bool v = true;
      for (int j = 0; j < m; j++) {
        bool am = ((1<<sa[j]) & i);
        bool sm = ((1<<ss[j]) & i);
        char cm = sx[j];
        if ((am && sm && cm == 'C') ||
            (am && !sm && cm == 'T') ||
            (!am && sm && cm == 'T') ||
            (!am && !sm && cm == 'C')) {
          v = false;
          break;
        }
      }
      if (v)
        a++;
    }
    cout << "Case #" << T << ": " << a << "\n";
  }
  return 0;
}
