#include <iostream>
using namespace std;

string d[100];

int main() {
  int n; cin >> n;
  for (int i = 0; i < n; i++)
    cin >> d[i];
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    string w; cin >> w;
    cout << "Word #" << T << ": " << w << "\n";
    int nm = 0;
    for (int i = 0; i < n; i++) {
      if (d[i].size() != w.size())
        continue;
      bool match = true;
      for (int j = 0; j < w.size(); j++) {
        if (d[i][j] != w[j] && w[j] != '-') {
          match = false;
          break;
        }
      }
      if (match) {
        nm++;
        cout << d[i] << "\n";
      }
    }
    cout << "Total number of candidate words = " << nm << "\n\n";
  }
  return 0;
}
