#include <iostream>
using namespace std;

typedef long long ll;

ll d[101];
int x[8];

int main() {
  d[0] = 1;
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    for (int i = 0; i < 8; i++)
      cin >> x[i];
    int n; cin >> n;
    for (int i = 1; i <= n; i++)
      d[i] = 0;
    for (int i = 0; i < 8; i++) {
      for (int j = x[i]; j <= n; j++)
        d[j] += d[j-x[i]];
    }
    cout << "Equation #" << T << ": " << d[n] << "\n";
  }
  return 0;
}
