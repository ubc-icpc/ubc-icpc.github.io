#include <iostream>
using namespace std;

int p[5];

int main() {
  int t; cin >> t;
  for (int T = 1; T <= t; T++) {
    cout << "Pentagon #" << T << ":\n";
    for (int i = 0; i < 5; i++)
      cin >> p[i];
    while (true) {
      int mn = 90019001, mi = -1;
      for (int i = 0; i < 5; i++) {
        if (mn > p[i]) {
          mn = p[i];
          mi = i;
        }
      }
      if (mn >= 0) {
        cout << p[0];
        for (int i = 1; i < 5; i++)
          cout << " " << p[i];
        cout << "\n\n";
        break;
      }
      p[(mi+4)%5] += p[mi];
      p[(mi+1)%5] += p[mi];
      p[mi] *= -1;
    }
  }
  return 0;
}
