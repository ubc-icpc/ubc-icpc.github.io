#include<bits/stdtr1c++.h>
using namespace std;

int N, P[5];

int& p(int i) {
    return P[(i + 5) % 5];
}

int main () {
    cin >> N;
    for (int n = 1; n <= N; ++n) {
        for (int i = 0; i < 5; ++i) cin >> p(i);
        while (true) {
            int l = 0;
            for (int i = 1; i < 5; ++i)
                if (p(i) < p(l)) l = i;
            if (p(l) < 0) {
                p(l-1) += p(l);
                p(l+1) += p(l);
                p(l) = abs(p(l));
            } else {
                cout << "Pentagon #"<< n << ":" << endl;
                cout << p(0) << " " << p(1) << " " << p(2) << " " << p(3) << " " << p(4) << endl << endl;
                break;
            }
        }
    }
}
