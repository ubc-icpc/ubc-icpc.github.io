#include<bits/stdtr1c++.h>
using namespace std;

struct p {
	int x, y, id;
} P[55];

int N, C[2][55];

bool comp1 (const p& a, const p& b) {
	if (a.x != b.x) return a.x < b.x;
	return a.y < b.y;
}
bool comp2 (const p& a, const p& b) {
	if (a.y != b.y) return a.y < b.y;
	return a.x < b.x;
}

int main () {
	for (int t = 1;; ++t) {
		cin >> N;
		if (!N) break;
		for (int i = 0; i < N; ++i) {
			cin >> P[i].x >> P[i].y;
			P[i].id = i;
		}
		sort(P, P+N, comp1);
		for (int i = 0; i < N; ++i)
			C[0][P[i].id] = P[i-i%2+!(i%2)].id;
		sort(P, P+N, comp2);
		for (int i = 0; i < N; ++i)
			C[1][P[i].id] = P[i-i%2+!(i%2)].id;
		int at = P[0].id, h = true;
		cout << "Polygon #" << t << ":";
		for (int i = 0; i < N; ++i) {
			cout << " " << at+1;
			at = C[h][at];
			h = !h;
		}
		cout << endl;
	}
	return 0;
}
