#include<bits/stdtr1c++.h>
using namespace std;

int t, j;

int main () {
    for (int i = 1;; ++i) {
        cin >> j;
        if (!j) break;
        t += j;
        if (t >= 100) printf("Input #%d: Totally Sweet!\n", i);
        else if (t >= 50) printf("Input #%d: Sweet!\n", i);
        t %= 50;
    }
    return 0;
}
