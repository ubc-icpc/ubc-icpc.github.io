#include<bits/stdtr1c++.h>
using namespace std;

int N;
vector<string> dict;
string pattern;

bool match(string& pattern, string& word) {
    if (pattern.size() != word.size()) return false;
    for (int i = 0; i < pattern.size(); ++i)
        if (pattern[i] != '-' && pattern[i] != word[i])
            return false;
    return true;
}

int main () {
    cin >> N;
    dict = vector<string>(N);
    for (int i = 0; i < N; ++i) cin >> dict[i];
    cin >> N;
    for (int c, i = 1; i <= N; ++i) {
        cin >> pattern;
        cout << "Word #" << i << ": " << pattern << endl;
        for (int i = c = 0; i < dict.size(); ++i)
            if (match(pattern, dict[i])) {
                cout << dict[i] << endl;
                ++c;
            }
        cout << "Total number of candidate words = " << c << endl << endl;
    }
    return 0;
}
