#include<bits/stdtr1c++.h>
using namespace std;

int N, A, B;
string num;

int toint(const string& s) {
    int i;
    stringstream ss(s);
    ss >> i;
    return i;
}

int main () {
    cin >> N;
    for (int c, n = 1; n <= N; ++n) {
        cin >> num;
        cout << "Ticket #" << n << ":" << endl;
        c = 0;
        for (int i = 1; i < num.size(); ++i) {
            A = toint(num.substr(0, i));
            B = toint(num.substr(i));
            if (A != 0 && B != 0 && __gcd(A, B) == 1) {
                c = true;
                cout << A << " " << B << endl << endl;
                break;
            }
        }
        if (!c) cout << "Not relative" << endl << endl;
    }
    return 0;
}
