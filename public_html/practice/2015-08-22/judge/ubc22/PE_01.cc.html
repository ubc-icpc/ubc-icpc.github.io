#include<bits/stdtr1c++.h>
using namespace std;

int N, R, C, M[105][105];
string s;

int toint(const string& s) {
    int i;
    stringstream ss(s);
    ss >> i;
    return i;
}

int main () {
    cin >> N;
    for (int n = 1; n <= N; ++n) {
        cin >> R >> C;
        for (int r = 0; r < R; ++r)
            for (int c = 0; c < C; ++c) {
                cin >> s;
                M[r][c] = (s == "E" || s == "P") ? 0 : toint(s);
            }
        for (int i = 1; i < C; ++i) M[0][i] += M[0][i-1];
        for (int i = 1; i < R; ++i) M[i][0] += M[i-1][0];
        for (int i = 1; i < R; ++i)
            for (int j = 1; j < C; ++j)
                M[i][j] += max(M[i-1][j], M[i][j-1]);
        cout << "Game Board #" << n << ": " << M[R-1][C-1] << endl << endl;
    }
    return 0;
}
