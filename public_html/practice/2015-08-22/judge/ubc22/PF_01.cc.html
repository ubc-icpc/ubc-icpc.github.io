#include<bits/stdtr1c++.h>
using namespace std;

vector<string> M[15];
int N, L;

void fill(vector<string>& a, vector<string>& b, int x, int y) {
    for (int i = 0; i < b.size(); ++i)
        for (int j = 0; j < b[0].size(); ++j)
            a[i+x][j+y] = b[i][j];
}

void comp (int i) {
    if (M[i].size()) return;
    comp(i-1);
    M[i] = vector<string>(M[i-1].size() * 2, string(M[i-1].size() * 2, ' '));
    fill(M[i], M[i-1], 0, M[i-1].size() / 2);
    fill(M[i], M[i-1], M[i-1].size(), 0);
    fill(M[i], M[i-1], M[i-1].size(), M[i-1].size());
    return;
}

int main () {
    M[1] = vector<string>(2);
    M[1][0] = string(2, 'X');
    M[1][1] = string(2, 'X');
    cin >> N;
    for (int n = 1; n <= N; ++n) {
        cout << "Triangle #" << n << ":" << endl;
        cin >> L;
        comp(L);
        for (int i = 0; i < M[L].size(); ++i)
            cout << M[L][i] << endl;
        cout << endl;
    }
}
