#include<bits/stdtr1c++.h>
using namespace std;

int T, C[8], N;
unsigned long long mem[105][10];
bool vis[105][10];

unsigned long long dp(int n, int at) {
    if (n == 0) return 1;
    if (n < 0 || at >= 8) return 0;
    if (vis[n][at]) return mem[n][at];
    vis[n][at] = true;
    unsigned long long r = 0;
    for (int i = 0; i <= n; i += C[at])
        r += dp(n - i, at + 1);
    mem[n][at] = r;
    return r;
}

int main () {
    cin >> T;
    for (int t = 1; t <= T; ++t) {
        for (int i = 0; i < 8; ++i) cin >> C[i];
        cin >> N;
        memset(vis, 0, sizeof vis);
        cout << "Equation #" << t << ": " << dp(N, 0) << endl;
    }
    return 0;
}
