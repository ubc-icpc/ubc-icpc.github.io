#include <iostream>
#include <algorithm>
#include <stdio.h>

using namespace std;

typedef long long ll;

ll gcd (ll a, ll b) {
  if (a == 0) return b;
  return gcd(b%a, a);
}

int main() {
  int n; cin >> n;
  for (int i=1; i<=n; i++) {
    cout << "Ticket #" << i << ":" << endl;
    ll x; cin >> x;
    bool got = false;
    for (ll i=10000000; i>0; i/=10) {
      ll a = x/i; ll b = x%i;
      if (gcd(a,b) == 1) {
        got = true;
        cout << a << " " << b << endl;
        break;
      }
    }
    if (!got) cout << "Not relative" << endl;
    cout << endl;
  }
  return 0;
}
