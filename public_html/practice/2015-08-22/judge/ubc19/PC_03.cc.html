#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main() {
  int n; cin >> n;
  for (int i=1; i<=n; i++) {
    cout << "Pentagon #" << i << ":" << endl;
    int a [5];
    for (int j=0; j<5; j++) cin>> a[j];
    while (*min_element(a, a+5) < 0) {
      for (int j=0; j<5; j++) {
        if (a[j] == *min_element(a, a+5)) {
          a[j] *= -1;
          a[(5+j+1)%5] -= a[j];
          a[(5+j-1)%5] -= a[j];
          break;
        }
      }
    }
    for (int j=0; j<5; j++)
      cout << a[j] << " ";
    cout << endl << endl;
  }
  return 0;
}
