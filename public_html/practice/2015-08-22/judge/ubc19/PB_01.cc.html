#include <iostream>
#include <array>
#include <string>

using namespace std;

int main() {
  int n; cin >> n;
  array<string,100> dict;
  for (int i=0; i<n; i++) {
    cin >> dict[i];
  }
  int w; cin >> w;
  for (int i=0; i<w; i++) {
    int c = 0;
    string s; cin >> s;
    cout << "Word #" << i+1 << ": " << s << endl;
    for (int j=0; j<n; j++) {
      if (dict[j].size() != s.size())
        continue;
      bool can = true;
      for (int k=0; k<s.size(); k++) {
        if (s[k] != '-' && s[k] != dict[j][k]) {
          can = false;
          continue;
        }
      }
      if (can) {
        c++;
        cout << dict[j] << endl;
      }
    }
    cout << "Total number of candidate words = " << c<< endl << endl;
  }
  return 0;
}
