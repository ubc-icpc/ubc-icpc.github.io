#include <iostream>
#include <algorithm>

using namespace std;

int main() {
  int n; cin >> n;
  for (int i=1; i<=n; i++) {
    int r, c; cin >> r >> c;
    int board [r][c];
    int mg[r][c];
    for (int j=0; j<r; j++) {
      for (int k=0; k<c; k++) {
        board[j][k] = 0; mg[j][k] = 0;
      }
    }
    for (int j=0; j<r; j++) {
      for (int k=0; k<c; k++) {
        if ((j==0 && k==0) || (j==r-1 && k==c-1)) {
          char c; cin >> c;
        } else {
          cin >> board[j][k];
        }
      }
    }
    
    for (int j=1; j<c; j++)
      mg[0][j] = board[0][j] + mg[0][j-1];
    
    for (int j=1; j<r; j++)
      mg[j][0] = board[j][0] + mg[j-1][0];
    
    for (int j=1; j<r; j++) {
      for (int k=1; k<c; k++) {
        mg[j][k] = board[j][k] + max(mg[j-1][k], mg[j][k-1]);
      }
    }
    cout << "Game Board #" << i << ": " << mg[r-1][c-1] << endl << endl;
    
  }
  return 0;
}
