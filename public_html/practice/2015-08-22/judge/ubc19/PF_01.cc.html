#include <iostream>
#include <algorithm>

using namespace std;
char tri[1024][1024];

/* x, and y is second corner of tip */
void fill(int r, int c, int size) {
  if (size == 1) {
    tri[r][c] = 'X';
    tri[r][c-1] = 'X';
    tri[r+1][c] = 'X';
    tri[r+1][c-1] = 'X';
    return;
  }
  fill(r,c,size-1);
  fill(r+(1<<(size-1)), c-(1<<(size-2)), size-1);
  fill(r+(1<<(size-1)), c+(1<<(size-2)), size-1);
}

int main() {
  int n; cin >> n;
  int sz;
  for (int i=1; i<=n; i++) {
    cout << "Triangle #" << i << ":" << endl;
    cin >> sz;
    int dim = 1 << sz;
    for (int j=0; j<dim; j++)
      for (int k=0; k<dim; k++)
        tri[j][k] = ' ';
    fill (0,1<<(sz-1),sz);
    
    for (int j=0; j<dim; j++) {
      for (int k=0; k<dim; k++){
        cout << tri[j][k];
      }
      cout << endl;
    }
    cout << endl;
  
  }
  return 0;
}
