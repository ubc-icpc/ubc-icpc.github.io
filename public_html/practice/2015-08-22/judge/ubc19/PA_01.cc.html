#include <iostream>

using namespace std;

int main() {
  int bank = 0; int c = 1;
  int m; cin>>m;
  while (m != 0) {
    bank += m;
    if (bank/50 == 1) {
      cout << " Input #" << c << ": Sweet!" << endl;
    } else if (bank/50 > 1) {
      cout << " Input #" << c << ": Totally Sweet!" << endl;
    }
    bank = bank%50;
    cin >> m;
    c++;
  }
  return 0;
}
