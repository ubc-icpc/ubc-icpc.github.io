#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int t, all = 0, counter = 1;
	do{
		cin >> t;
		all += t;
		if (all >= 100){
			cout << "Input #" << counter << ": Totally Sweet!" << endl;
		}
		else if (all >= 50){
			cout << "Input #" << counter << ": Sweet!" << endl;
		}
		all %= 50;
		++counter;
	} while (t != 0);
	return 0;
}
