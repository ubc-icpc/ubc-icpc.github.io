#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	for (int gar = 1; gar <= n; ++gar){
		string s; cin >> s;
		vector < pair < int, int > > ans;
		for (int i = 0; i < s.size() - 1; ++i){
			int a = atoi(s.substr(0, i + 1).c_str()), b;
			string temp = s.substr(i + 1).c_str();
			if (temp[0] == '0')	b = atoi(temp.substr(1).c_str());
			else b = atoi(temp.c_str());
			int big, sml;
			if (a >= b){
				big = a; sml = b;
			}
			else{
				big = b; sml = a;
			}
			bool cop = true;
			for (int j = 2; j <= sml; ++j){
				if (big % j == 0 && sml % j == 0){
					cop = false;
					break;
				}
			}
			if (cop){
				ans.push_back(make_pair(a, b));
			}
		}
		sort(ans.begin(), ans.end());
		cout << "Ticket #" << gar << ":" << endl;
		if (ans.size() == 0){
			cout << "Not relative" << endl << endl;
		}
		else{
			cout << ans[0].first << " " << ans[0].second << endl << endl;
		}
		
	}
	return 0;
}
