#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	for (int i = 1; i <= n; ++i){
		vector < int > pent(5);
		for (int j = 0; j < 5; ++j){
			cin >> pent[j];
		}
		
		while (true){
			int low = 0, li = 0;
			for (int j = 0; j < 5; ++j){
				if (pent[j] < low){
					low = pent[j];
					li = j;
				}
			}
			if (low == 0){
				cout << "Pentagon #" << i << ":" << endl;
				for (int j = 0; j < 5; ++j){
					cout << pent[j];
					if (j != 4) cout << " "; else cout << endl;
				}
				break;
			}
			else{
				if (li == 0){
					pent[4] -= (pent[0] * -1);
					pent[1] -= (pent[0] * -1);
					pent[0] *= -1;
				}
				else if (li == 4){
					pent[3] -= (pent[4] * -1);
					pent[0] -= (pent[4] * -1);
					pent[4] *= -1;
				}
				else{
					pent[li - 1] -= (pent[li] * -1);
					pent[li + 1] -= (pent[li] * -1);
					pent[li] *= -1;
				}
			}
		}
		
	}


	return 0;
}
