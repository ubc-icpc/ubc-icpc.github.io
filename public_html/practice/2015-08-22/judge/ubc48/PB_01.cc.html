#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	vector < string > dict(n);
	for (int i = 0; i < n; ++i){
		cin >> dict[i];
	}
	cin >> n;
	for (int k = 0; k < n; ++k){
		string s; 
		cin >> s;
		
		cout << "Word #" << (k + 1) << ": " << s << endl;
		int c = 0;
		for (int i = 0; i < dict.size(); ++i){
			if (dict[i].size() == s.size()){
				bool alltrue = true;
				for (int j = 0; j < s.size(); ++j){
					if (s[j] != '-' && s[j] != dict[i][j]){
						alltrue = false;
					}
				}
				if (alltrue){
					cout << dict[i] << endl;
					++c;
				}
			}
		}
		cout << "Total number of candidate words = " << c << endl << endl;
	}
	return 0;
}
