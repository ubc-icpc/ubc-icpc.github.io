def check(a,b):
    if len(a)!=len(b):
        return False
    answer=True
    for i in range(0,len(a)):
        if a[i]!='-' and a[i]!=b[i]:
                answer=False
    return answer

D=[]
n=int(input())
for i in range(0,n):
    w=raw_input()
    D.append(w)
c=int(input())
count=0
for i in range(1,c+1):
    w=raw_input()
    count=0
    print("Word #"+str(i)+": "+w)
    for word in D:
        if check(w,word):
            print(word)
            count=count+1
    print("Total number of candidate words = "+str(count))
