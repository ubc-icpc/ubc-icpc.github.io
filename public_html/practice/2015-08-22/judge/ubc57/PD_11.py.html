def gcd(a, b):
    while b:
        a, b = b, a%b
    return a
n=int(input())
for c in range(1,n+1):
    s=str(input())
    print("Ticket #"+str(c)+":")
    ans="Not relative"
    for i in range(0,len(s)-1):
        if gcd(int(s[:i+1]),int(s[i+1:]))==1:
            ans=s[:i+1] +' '+s[i+1:]
            break
    print(ans)
    print('')
