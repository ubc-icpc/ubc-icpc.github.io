n=int(input())
for j in range(0,n):
    P=list(map(int,raw_input().split()))
    #print(P)
    while True:
        done=True
        #print(P)
        for i in range(0,5):
            if P[i]<0:
                done=False
        if done:
            break
        x=min(P)
        for i in range(0,5):
            if P[i]==x:
                break
        P[i]=-P[i]
        P[(i-1)%5]=P[(i-1)%5]-P[i]
        P[(i+1)%5]=P[(i+1)%5]-P[i]
    answer=''
    for i in range(0,4):
        answer=answer+str(P[i])+" "
    answer=answer+ str(P[4])
    print("Pentagon #"+str(j+1)+":")
    print(answer)
