//#include "stdafx.h"
#include <cstring>
#include<iostream>
using namespace std;
long long DP[8][105];
int C[8]; //coefficients
long long dp(int a, int add){
	long long ans = 0;
	if (DP[a][add] != -1) return DP[a][add];
	if (add < 0){
		return 0;
	}
	if (a == 0){
		if (add%C[a] == 0) return 1;
		else return 0;
	}
	for (int i = 0; C[a]*i <= add; i++){
		ans += dp(a - 1, add - C[a]*i);
	}
	DP[a][add] = ans;
	return ans;
}

int main()
{
	int t;
	cin >> t;
	for (int cas = 1; cas <= t; cas++){
		memset(DP, -1, sizeof DP);
		for (int i = 0; i < 8; i++) cin >> C[i];
		int n;
		cin >> n;
		cout << "Equation #" << cas <<": "<< dp(7, n) << endl;
	}
	return 0;
}
