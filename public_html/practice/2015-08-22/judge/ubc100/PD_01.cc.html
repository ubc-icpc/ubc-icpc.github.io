#include<iostream>
#include<string>
#define ll long long
using namespace std;
int gcd(int a, int b) {
  while(a!=b) {
    if(a>b) {
      a=a-b;
    } else {
      b=b-a;
    }
  }
  return a;

}
bool cp(ll n) {
  int p=1;
  while(p<n) p*=10;
  p/=10;
  //cout << p << endl;
  while(p>1) {
    int p1=n/p, p2=n%p;
    if(gcd(p1,p2) == 1) {
      cout << p1 << " " << p2 << endl;
      return true;
    }
    p/=10;
  }
  return false;
}

int main(void) {
  
  int t; cin >> t;

  for(int i=1; i<=t; i++) {
    cout << "Ticket #" << i << ":" << endl;
    int n; cin >> n;
    if(!cp(n )) cout << "Not relative" << endl;
  }
  
  return 0;
}
