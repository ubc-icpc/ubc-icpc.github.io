#include<iostream>
#include<math.h>
#include<string.h>

using namespace std;
bool m[3000][3000];
void f(int n, int l, int x1, int x2, int y1, int y2) {
  for(int i=0; i<l/4; i++) {
    for(int j=0; j<l/2; j++) {
      m[x1+i][y1+j] = false;
      m[x2-i][y1+j] = false;
    }
  }
  if(n>1) {
    f(n-1,l/2,x1+l/4,x2-l/4, y1,y1+l/2);
    f(n-1,l/2, x1,x2-l/2, y1+l/2,y2);
    f(n-1,l/2, x1+l/2,x2, y1+l/2,y2);
  }
}

int main(void) {
  int t; cin >> t;

  for(int i=1; i<=t; i++) {
    cout << "Triangle #" << i << ":" << endl;
    int n; cin >> n;
    int l=4*(int)pow(2,n-1); l/=2;
    //memset(m,0,sizeof(m[1][1])*l*l);
    //cout << l << endl;
    for(int j=0; j<l; j++) {
      for(int k=0; k<l; k++ ) {
	m[j][k]=true;
      }
    }

    
    f(n,l,0,l-1,0,l-1);
    for(int j=0; j<l; j++) {
      for(int k=0; k<l; k++ ) {
	if(m[k][j]) cout << "X";
	else cout << " ";
      }
      cout << endl;
    }
  }
  
  return 0;
}
