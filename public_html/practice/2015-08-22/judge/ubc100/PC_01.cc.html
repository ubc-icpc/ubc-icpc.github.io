#include<iostream>

using namespace std;
int p[5];

int main (void) {
  int t; cin >> t;
  for(int i=1; i<=t; i++) {
    cout << "Pentagon #" << i << ":" << endl;
    for(int k=0; k<5; k++) {
      int x; cin >> x; p[k] = x;
    }

    int flag=true;

    while(flag) {
      int m=p[0], ix=0;
      for(int j=1; j<5; j++) {
	if(p[j]<m) {
	  m=p[j]; ix=j;
	}
      }
      if(m>=0) {
	flag=false;
      } else {
	if(ix == 0) p[4] += p[ix];
	else p[ix-1] += p[ix];
	p[(ix+1)%5] += p[ix];
	p[ix] *= -1;
      }
    }
    for(int j=0; j<5; j++) {
      cout << p[j] << " ";
    }
    cout << endl;
  }
  return 0;
}
