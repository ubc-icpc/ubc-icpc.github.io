#include<iostream>
#include<algorithm>

using namespace std;


int main(void) {
  
  int t; cin >> t;

  for(int i=1; i<=t; i++) {
    cout << "Game Board #" << i << ": " ;
    int rows, cols; cin >> rows >> cols;
    int m[rows][cols];
    for(int j=0; j<rows; j++) {
      for(int k=0; k<cols; k++) {
	if( (j==0 && k==0) || (j==rows-1 && k==cols-1) ) {
	  char c; cin >> c;
	  m[j][k] = 0;
	} else {
	  int x; cin >> x;
	  m[j][k] = x;
	}
      }
    }

    for(int r=0; r<rows; r++) {
      for(int c=0; c<cols; c++) {
	if(r==0 && c==0) continue;
	else if(r==0) {
	  m[r][c] += m[r][c-1];
	} else if(c==0) {
	  m[r][c] += m[r-1][c];
	} else {
	  m[r][c] += max(m[r-1][c], m[r][c-1]);
	}
      }
    }
    cout << m[rows-1][cols-1] << endl;
  }

  return 0;
}
