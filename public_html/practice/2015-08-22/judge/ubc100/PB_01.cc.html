#include<iostream>
#include<vector>
#include<string>
using namespace std;

int main (void) {
  
  int t; cin>>t;
  vector<string> dic;
  for(int i=0; i<t; i++) {
	string s;cin >> s; dic.push_back(s);
  }
  
  int n; cin>>n;
  for(int i=0; i<n; i++) {
	string s; cin >> s;
	int r=0;
	cout << "Word #" << i+1 << ": " << s << endl;
	for( int j=0; j<t; j++ ) {
	  if(dic[j].size()==s.size()) {
		bool flag = true;
		for( int k=0; k<s.size(); k++ ) {
		  if(s[k]=='-' || s[k]==dic[j][k])continue;
		  flag = false;
		  break;
		}
		
		if(flag) {
		  r++;
		  cout << dic[j] << endl;
		}
	  }
	}
	cout << "Total number of candidate words = " << r << endl << endl;
  }
  return 0;
}
