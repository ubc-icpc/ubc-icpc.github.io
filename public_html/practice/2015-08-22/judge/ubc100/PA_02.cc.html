#include<iostream>

using namespace std;


int main(void) {
  int total = 0;
  int i=1;
  while(true) {
    int d; cin >> d;
    if(d==0) break;
    total += d;
    if(total >= 50 ) {
      cout << "Input #" << i << ": ";
      int g = 0;//total/50;
      while(total >=50) {
	total -= 50; g++;
      }
      if(g > 1) {
	cout << "Totally Sweet!" << endl;
      } else {
	cout << "Sweet!" << endl;
      }
      //total = total - 50*g;
    }
    i++;
  }

  return 0;
}
