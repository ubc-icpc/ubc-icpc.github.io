#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int main() {
    int s = 0, x, c = 0;
    while (cin >> x) {
        if (!x) break;
        c++;
        s += x;
        if (s >= 100) {
            cout << "Input #" << c << ": Totally Sweet!" << endl;
            s %= 50;
        } else if (s >= 50) {
            cout << "Input #" << c << ": Sweet!" << endl;
            s %= 50;
        }
    }
	return 0;
}
