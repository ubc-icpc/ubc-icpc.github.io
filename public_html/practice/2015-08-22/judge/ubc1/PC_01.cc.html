#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int a[5];
int main() {
    int t; cin >> t;
    for (int c = 1; c <= t; c++) {
        cout << "Pentagon #" << c << ":" << endl;
        for (int i = 0; i < 5; i++) cin >> a[i];
        while (true) {
            int v = *min_element(a, a+5);
            if (v >= 0) break;
            
            int j = 0;
            for (j = 0; j < 5; j++) {
                if (a[j] == v) break;
            }
            a[j] *= -1;
            int l = (j+4)%5, r = (j+1)%5;
            a[l] += v;
            a[r] += v;
        }
        
        for (int i = 0; i < 5; i++) cout << a[i] << " ";
        cout << endl << endl;
    }
	return 0;
}
