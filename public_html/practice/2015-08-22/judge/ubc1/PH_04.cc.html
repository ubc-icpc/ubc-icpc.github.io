#include <bits/stdtr1c++.h>

using namespace std;

struct edge {
    int u, v; char t;
};

int col[20];
int main() {
    int t; cin >> t;
    for (int c = 1; c <= t; c++) {
        cout << "Case #" << c << ": ";
        
        vector<edge> edges;
        int n, m; cin >> n >> m;
        for (int i = 0; i < m; i++) {
            int u, v; char ch;
            cin >> u >> v >> ch;
            edges.push_back({u, v, ch});
        }
        
        int ans = 0;
        for (int i = 0; i < (1<<n); i++) {
            for (int j = 0; j < n; j++) {
                col[j] = (i & (1<<j)) == (1<<j);
            }
            
            bool good = true;
            for (edge e : edges) {
                if (e.t == 'C') {
                    if (col[e.u] == col[e.v]) {
                        good = false;
                        break;
                    }
                } else {
                    if (col[e.u] != col[e.v]) {
                        good = false;
                        break;
                    }
                }
            }
            ans += good;
        }
        cout << ans << endl;
    }
	return 0;
}
