#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

const int lim = 8;
ll memo[10][105], c[10];
ll dp(int i, int C) {
    if (C < 0) return 0;
    if (i == lim) return (C == 0);
    if (memo[i][C] != -1) return memo[i][C];
    
    memo[i][C] = 0;
    for (int j = 0; j <= 100; j++) {
        memo[i][C] += dp(i+1, C-j*c[i]);
    }
    return memo[i][C];
}

int main() {
    int t; cin >> t;
    for (int a = 1; a <= t; a++) {
        memset(memo, -1, sizeof memo);
        for (int i = 0; i < lim; i++) cin >> c[i];
        int N; cin >> N;
        cout << "Equation #" << a << ": " << dp(0, N) << endl;
    }
	return 0;
}
