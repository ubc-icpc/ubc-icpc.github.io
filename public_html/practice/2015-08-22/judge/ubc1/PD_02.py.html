def gcd(a, b):
	if b == 0: return a
	return gcd(b, a%b)

n = int(raw_input())
for c in range(1, n+1):
	print "Ticket #" + str(c) + ":"
	s = raw_input().strip()
	ans = "Not relative"
	for i in range(1,len(s)):
		if gcd(int(s[:i]), int(s[i:])) == 1:
			ans = s[:i] + ' ' + s[i:]
			break
	print ans
	print
