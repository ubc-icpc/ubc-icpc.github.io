#include <bits/stdtr1c++.h>

using namespace std;

vector<string> dict;
int main() {
    int m; cin >> m;
    for (int i = 0; i < m; i++) {
        string s; cin >> s;
        dict.push_back(s);
    }
    
    int n; cin >> n;
    for (int i = 1; i <= n; i++) {
        string s; cin >> s;
        cout << "Word #" << i << ": " << s << endl;
        int cnt = 0;
        for (string w : dict) {
            if (w.size() == s.size()) {
                bool match = true;
                for (int j = 0; j < w.size(); j++) {
                    if (s[j] == '-') continue;
                    if (w[j] != s[j]) {
                        match = false;
                        break;
                    }
                }
                if (match) {
                    cout << w << endl;
                    cnt++;
                }
            }
        }
        
        cout << "Total number of candidate words = " << cnt << endl << endl;;
    }
	return 0;
}
