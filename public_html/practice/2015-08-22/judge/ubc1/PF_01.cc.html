#include <bits/stdtr1c++.h>

using namespace std;

char ans[1<<11][1<<11];
int main() {
    int t; cin >> t;
    for (int ca = 1; ca <= t; ca++) {
        cout << "Triangle #" << ca << ":" << endl;
        int n; cin >> n;
        
        int m = (1<<n);
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < m; j++) {
                ans[i][j] = ' ';
            }
        }
        
        ans[0][m/2] = ans[0][m/2-1] = ans[1][m/2] = ans[1][m/2-1] = 'X';
        for (int iter = 0; iter < n-1; iter++) {
            for (int j = (1<<(iter+1)); j < (1<<(iter+2)); j++) {
                for (int k = m/2-(1<<iter); k < m/2+(1<<iter); k++) {
                    ans[j][k-(1<<iter)] = ans[j-(1<<(iter+1))][k];
                    ans[j][k+(1<<iter)] = ans[j-(1<<(iter+1))][k];
                }
            }
        }
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < m; j++) {
                cout << ans[i][j];
            }
            cout << '\n';
        }
        cout << endl;
    }
	return 0;
}
