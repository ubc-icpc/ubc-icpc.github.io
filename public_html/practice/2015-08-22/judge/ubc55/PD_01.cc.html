int splice(int x)
{
	int length = (int)(log10(x));
	for (int i = 0; i < length; i++)
	{
		int number1, number2;
		number1 = x / pow(10, length - i);
		number2 = x % (int)pow(10, length - i);
		int no = 1;
		for (int j = 2; j < max(number1, number2); j++)
		{
			if (number1%j == 0 && number2%j == 0)
			{
				no = 0;
				break;
			}
		}
		//works
		if (no)
		{
			return length - i;
		}
	}
	return -1;
}
