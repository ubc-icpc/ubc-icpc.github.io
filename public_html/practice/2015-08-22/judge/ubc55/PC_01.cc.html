#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	int trials;
	cin >> trials;
	int count = 0;
	while (count != trials)
	{
		vector<int> arr;
		count++;
		for (int i = 0; i < 5; i++)
		{
			int n;
			cin >> n;
			arr.push_back(n);
		}
		vector<int>::iterator smallest = min_element(arr.begin(), arr.end());
		while ((*smallest) < 0)
		{
			int index = smallest - arr.begin();
			if (index == 0)
			{
				*smallest *= -1;
				arr[4] -= *smallest;
				arr[1] -= *smallest;
			}
			else if (index == 4)
			{
				*smallest *= -1;
				arr[3] -= *smallest;
				arr[0] -= *smallest;
			}
			else
			{
				*smallest *= -1;
				*(smallest + 1) -= *smallest;
				*(smallest - 1) -= *smallest;
			}
			smallest = min_element(arr.begin(), arr.end());
		}
		cout << "Pentagon #" << count << ":" << endl;
		for (int i = 0; i < 5; i++)
			cout << arr[i] << " ";
		cout << endl;
	}
}
