#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

int arr[105][105];
int max_arr[105][105];

int main()
{
	int n;
	cin >> n;
	int count = 0;
	while (count < n)
	{
		count++;
		int x, y;
		cin >> y >> x;
		memset(arr, 0, sizeof(arr));
		memset(max_arr, 0, sizeof(max_arr));
		for (int j = 0; j < y; j++)
		{
			for (int i = 0; i < x; i++)
			{
				if ((i == 0 && j == 0) || (i == x - 1 && j == y - 1))
				{
					char c;
					cin >> c;
				}
				else
				{
					cin >> arr[j][i];
				}
			}
		}
		for (int j = y - 1; j >= 0; j--)
		{
			for (int i = x - 1; i >= 0; i--)
			{
				if (j == y - 1 && i == x - 1)
				{
					max_arr[j][i] = arr[j][i];
				}
				else if (j == y - 1)
				{
					max_arr[j][i] = max_arr[j][i + 1] + arr[j][i];
				}
				else if (i == x - 1)
				{
					max_arr[j][i] = max_arr[j + 1][i] + arr[j][i];
				}
				else
				{
					max_arr[j][i] = max(max_arr[j][i + 1], max_arr[j + 1][i]) + arr[j][i];
				}
			}
		}
		cout << "Game Board #" << count << ": " << max_arr[0][0] << endl << endl;
	}
}
