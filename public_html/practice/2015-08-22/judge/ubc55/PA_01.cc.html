#include <iostream>

using namespace std;
int main()
{
	int count = 0;
	int money = 0;
	while (true)
	{
		count++;
		int m;
		cin >> m;
		if (m)
		{
			money += m;
			if (money >= 100)
			{
				money = money % 50;
				cout << "Input #" << count << ": Totally Sweet!" << endl;
			}
			else if (money >= 50)
			{
				money -= 50;
				cout << "Input #" << count << ": Sweet!" << endl;
			}
		}
		else
			break;
	}
}
