#include <iostream>
#include <string>

using namespace std;

bool compare(string one, string two)
{
	if (one.length() != two.length())
	{
		return 0;
	}
	for (int i = 0; i < one.length(); i++)
	{
		if (one[i] == '-')
			continue;
		if (one[i] != two[i])
			return 0;
	}
	return 1;
}

int main()
{
	int n;
	cin >> n;
	string* dic = new string[n];
	for (int i = 0; i < n; i++)
	{
		cin >> dic[i];
	}
	int w;
	cin >> w;
	string word;
	for (int i = 0; i < w; i++)
	{
		cin >> word;
		cout << "Word #" << i + 1 << ": " << word << endl;
		int count = 0;
		for (int j = 0; j < n; j++)
		{
			if (compare(word, dic[j]))
			{
				cout << dic[j] << endl;
				count++;
			}
		}
		cout << "Total number of candidate words = " << count << endl << endl;
	}
}
