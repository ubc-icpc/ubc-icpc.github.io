#include <bits/stdtr1c++.h>
using namespace std;

bool coprime(int a, int b) {
	while (true) {
		if (!(a %= b)) return b == 1 ;
    	if (!(b %= a)) return a == 1 ;
	}
}

void check(string s) {
	for (int i=1; i<s.size(); i++) {
		string s1, s2;
		s1 = s.substr(0,i);
		s2 = s.substr(i);
		int a = stoi(s1);
		int b = stoi(s2);
		if (coprime(a,b)) {
			cout << a << " " << b << endl;
			return;
		}
	}
	cout << "Not relative" << endl;
	return;
}

int main() {
	int n; cin >> n;
	for (int N=1; N<=n; N++) {
		cout << "Ticket #" << N << ":" << endl;
		string s; cin >> s;
		check(s);
		cout << endl;
	}
}
