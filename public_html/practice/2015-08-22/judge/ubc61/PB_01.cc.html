#include <bits/stdtr1c++.h>
using namespace std;

bool check(string s, string dict) {
	if (dict.size() != s.size()) return false;
	for (int i=0; i<s.size(); i++) {
		if (s[i] == '-') continue;
		if (s[i] != dict[i]) return false;
	}
	return true;
}

int main() {
	int n; cin >> n;
	string dict[n];
	for (int N=0; N<n; N++) {
		string w; cin >> w;
		dict[N] = w;
	}
	int m; cin >> m;
	for (int M=1; M<=m; M++) {
		string w; cin >> w;
		cout << "Word #" << M << ": " << w << endl;
		int numw = 0;
		for (int i=0; i<n; i++) {
			if (check(w,dict[i])) {
				cout << dict[i] << endl;
				numw++;
			}
		}
		cout << "Total number of candidate words = " << numw << endl << endl;
	}
	return 0;
}
