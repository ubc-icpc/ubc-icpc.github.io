#include <bits/stdtr1c++.h>
using namespace std;

void iter(int penta[]) {
	int min = 1000;
	int index = -1;
	for (int i=0; i<5; i++) {
		if (penta[i] < min) {
			min = penta[i];
			index = i;
		}
	}
	int pos = penta[index] *= -1;
	penta[(index+4)%5] -= pos;
	penta[(index+6)%5] -= pos;
	return;
}

bool findneg(int penta[]) {
	for (int i=0; i<5; i++) {
		if (penta[i] < 0) return true;
	}
	return false;
}

int main() {
	int n; cin >> n;
	for (int N=1; N<=n; N++) {
		int penta[5];
		for (int i=0; i<5; i++) {
			cin >> penta[i];
		}
		while (findneg(penta)) {
			iter(penta);
		}
		cout << "Pentagon #" << N << ": " << endl;
		for (int i=0; i<5; i++) {
			cout << penta[i] << " ";
		}
		cout << endl << endl;
	}
}
