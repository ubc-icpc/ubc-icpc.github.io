#include <bits/stdtr1c++.h>
using namespace std;
int main() {
	int n; cin >> n;
	
	vector<string> tri[11];
	
	tri[1] = {"XX","XX"};
	int spaces = 1;
	
	for (int i=2; i<=10; i++) {
		int l = tri[i-1].size();
		
		for (int j=0; j<l; j++) {
			string s(spaces,' ');
			string o = tri[i-1][j];
			tri[i].push_back(s + o + s);
		}
		
		for (int j=0; j<l; j++) {
			string o = tri[i-1][j];
			tri[i].push_back(o+o);
		}
		spaces *=2;
	}

	for (int N=1; N<=n; N++) {
		cout << "Triangle #" << N << ":" << endl;

		int k; cin >> k;
		for (int i=0; i<tri[k].size(); i++) {
			for (int j=0; j<tri[k][i].size(); j++) {
				cout << tri[k][i][j];
			}
			cout << endl;
		}
		cout << endl;
	}
}
