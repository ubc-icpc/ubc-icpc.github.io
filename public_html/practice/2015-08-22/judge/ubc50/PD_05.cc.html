#include<iostream>
#include<cmath>
using namespace std;

int gcd(int a, int b) {
  while (b!=0) {
    int t=b;
    b=a%b;
    a=t;
  }
  return a;
}

int main() {
  int test; cin >> test;
  for (int t=1; t<=test; t++) {
    bool flag=false;
    int n; cin >> n;
    int dig=0, num=n, n1, n2;
    while(num) {dig++; num/=10;}
    for (int i=dig-1; i>=1; i--) {
      int f=n/((int)pow(10,i));
      int s=n%((int)pow(10,i));
      if (f>s &&  gcd(f,s)==1) {n1=f; n2=s; flag=true; break;}
      else if (s>f && gcd(s,f)==1) {n1=f; n2=s; flag=true; break;}
    }

    cout << "Ticket #" << t << ":" << endl;
    if (!flag) cout << "Not relative" << endl << endl;
    else cout << n1 << " " << n2 << endl << endl;
  }
}
