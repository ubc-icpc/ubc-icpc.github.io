#include<iostream>
#include<vector>
using namespace std;

int main() {
  int n; cin >> n;
  for (int t=1; t<=n; t++) {
    vector<int> pt;
    bool flag = false;
    for (int i=0; i<5; i++) { int x; cin >> x; pt.push_back(x); if(x<0) flag=true; }
    while (flag) {
      int mn=0, pos=0;
      for (int i=0; i<pt.size(); i++) if (pt[i]<mn) {mn=pt[i]; pos=i; }
      if (mn>=0) break;
      pt[pos] *=-1;
      pt[(pos+1)%5] +=mn;
      if(pos==0) pt[4]+=mn;
      else pt[pos-1]+=mn;
    }
    cout << "Pentagon #" << t << ": "<< endl;
    for (auto e : pt)
      cout << e << " ";
    cout << endl;
  }
}
