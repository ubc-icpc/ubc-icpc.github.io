#include<iostream>
using namespace std;

int main() {
  int sum=0;
  int cnt=0;
  while(1) {
    int amt; cin >> amt;
    cnt++;
    if (amt==0) break;
    sum+=amt;
    if (sum>=50 && sum<100) {
      cout << "Input #" << cnt << ": " << "Sweet!" << endl;
      sum-=50;
    }
    else if (sum>=100) {
      int num=sum/50;
      sum -= (num*50);
      cout << "Input #" << cnt << ": " << "Totally Sweet!" << endl;
    }
  }
}
