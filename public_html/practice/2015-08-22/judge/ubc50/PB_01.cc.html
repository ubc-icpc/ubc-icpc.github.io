#include<iostream>
#include<regex>
#include<vector>
#include<string>
using namespace std;

int n, m;
vector<string> dict;
int main() {
  cin >> n;
  for (int i=0; i<n; i++) { string s; cin >> s; dict.push_back(s); }
  cin >> m;
  for (int w=1; w<=m; w++) {
    string word; cin >> word;
    vector<string> v;
    for (auto s : dict) {
      if (word.size() != s.size()) continue;
      bool match=true;
      for (int pos=0; pos<s.size(); pos++) {
	if(word[pos]=='-') continue;
	if (word[pos]!=s[pos]) {match=false; break;}
      }
      if (match) v.push_back(s);
    }
    cout << "Word #" << w << ": " << word << endl;
    for (auto e : v)
      cout << e << endl;
    cout << "Total number of candidate words = " << v.size() << endl << endl;
  }
}
