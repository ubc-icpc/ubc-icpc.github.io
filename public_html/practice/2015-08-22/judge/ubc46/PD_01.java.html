import java.util.Scanner;

/**
 * Created by Peter on 2015-08-23.
 */
public class Lottery {
    public static void main(String[] args) {

        Scanner s = new Scanner(System.in);

        int noOfCases = Integer.parseInt(s.nextLine());

        for(int i = 0; i < noOfCases; i++){
            boolean foundCommonFactor = false;
            String line = s.nextLine();

            System.out.println("Ticket #" + (i+1) + ":");
            for(int j = 1; j < line.length(); j++){
                int a = Integer.parseInt(line.substring(0,j));
                int b = Integer.parseInt(line.substring(j));
                if(!HaveCommonFactor(a,b)){
                    System.out.println("" + a + " " + b);
                    foundCommonFactor = true;
                    break;
                }

            }
            if(!foundCommonFactor){
                System.out.println("Not relative");
            }
            System.out.println();
        }
        //System.out.println(HaveCommonFactor(4, 7108));
        //System.out.println(HaveCommonFactor(1, 697203));
        //System.out.println(HaveCommonFactor(2, 2));
    }

    public static boolean HaveCommonFactor(int a, int b){
        for(int i = 2; i <= a; i++){
            if(a%i == 0 && b%i == 0){
                return true;
            }
        }
        return false;
    }
}
