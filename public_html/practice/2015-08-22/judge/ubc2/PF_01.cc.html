#include <bits/stdc++.h>

using namespace std;

char everything[1024][1024];

int main() {
  memset(everything, ' ', sizeof(everything[0][0])*1024*1024);
  everything[1022][0] = 'X';
  everything[1022][1] = 'X';
  everything[1023][0] = 'X';
  everything[1023][1] = 'X';
  for (int i = 0; i < 9; ++i) {
    int shift = 1 << i;
    int width = 2*shift;
    for (int j = 1024 - width; j < 1024; ++j) {
      for (int k = 0; k < width; ++k) {
        everything[j][k + width] = everything[j][k]; // Right
        everything[j - width][k + shift] = everything[j][k]; // Top
      }
    }
  }
  int T; cin >> T;
  for (int t = 1; t <= T; ++t) {
    cout << "Triangle #" << t << ":" << endl;
    int k; cin >> k;
    int width = 1 << k;
    for (int j = 1024 - width; j < 1024; ++j) {
      for (int k = 0; k < width; ++k) {
        cout << everything[j][k];
      }
      cout << endl;
    }
    cout << endl;
  }
  return 0;
}
