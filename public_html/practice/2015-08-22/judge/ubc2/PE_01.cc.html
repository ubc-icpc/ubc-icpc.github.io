#include <bits/stdc++.h>

using namespace std;

int bd_val[100][100];
int bd[100][100];

int main() {
  int N; cin >> N;
  for (int n = 1; n <= N; ++n) {
    int R; int C; cin >> R >> C;
    for (int r = 0; r < R; ++r) {
      for (int c = 0; c < C; ++c) {
        bd_val[r][c] = 0;
        if (r == 0 && c == 0 || r == R-1 && c == C-1) {
          string temp; cin >> temp;
        }
        else {
          cin >> bd_val[r][c];
        }
      }
    }
    for (int r = 0; r < R; ++r) {
      for (int c = 0; c < C; ++c) {
        bd[r][c] = 0;
        int up = 0, left = 0;
        if (r > 0) up = bd[r-1][c];
        if (c > 0) left = bd[r][c-1];
        bd[r][c] = bd_val[r][c] + max(up, left);
      }
    }
    cout << "Game Board #" << n << ": " << bd[R-1][C-1] << endl;
  }
  return 0;
}
