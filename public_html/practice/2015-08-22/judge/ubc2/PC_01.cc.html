#include <bits/stdc++.h>

using namespace std;

bool all_positive(int* pentagon) {
  for (int i = 0; i < 5; ++i) {
    if (*(pentagon + i) < 0) return false;
  }
  return true;
}

int get_mindex(int* pentagon) {
  int minval = *pentagon;
  int mindex = 0;
  for (int i = 1; i < 5; ++i) {
    if (*(pentagon + i) < minval) {
      minval = *(pentagon + i);
      mindex = i;
    }
  }
  return mindex;
}

int main() {
  int N; cin >> N;
  int pentagon[5];
  for (int n = 1; n <= N; ++n) {
    cout << "Pentagon #" << n << ":" << endl;
    for (int i = 0; i < 5; ++i) {
      cin >> pentagon[i];
    }
    while(!all_positive(pentagon)) {
      int mindex = get_mindex(pentagon);
      pentagon[mindex] *= -1;
      pentagon[(mindex + 4) % 5] -= pentagon[mindex];
      pentagon[(mindex + 1) % 5] -= pentagon[mindex];
    }
    for (int i = 0; i < 4; ++i) {
      cout << pentagon[i] << " ";
    }
    cout << pentagon[4] << endl;
  } 
  return 0;
}
