#include <bits/stdc++.h>

using namespace std;

int gcd(int a, int b) {
  if (b == 0) return a;
  return gcd(b, a % b);
}

int main() {
  int N; cin >> N;
  for (int n = 1; n <= N; ++n) {
    cout << "Ticket #" << n << ":" << endl;
    string ticket; cin >> ticket;
    bool found = false;
    for (int i = 1; i < ticket.size(); ++i) {
      string numstr1 = ticket.substr(0, i);
      string numstr2 = ticket.substr(i);
      int num1 = stoi(numstr1);
      int num2 = stoi(numstr2);
      if (gcd(num1, num2) == 1) {
        cout << num1 << " " << num2 << endl;
        found = true;
        break;
      }
    }
    if (!found) {
      cout << "Not relative" << endl;
    }
  }
  return 0;
}
