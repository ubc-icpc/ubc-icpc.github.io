#include <bits/stdc++.h>

using namespace std;

int ms[100][2];
bool mb[100];

bool tauto(int i, int n) {
  return ((i >> (n-1)) & 1) == 1;
}

bool consistent(int i, int m) {
  bool t1 = tauto(i, ms[m][0]), t2 = tauto(i, ms[m][1]);
  if (t1) return t2 == mb[m];
  else return t2 != mb[m];
}

bool all_consistent(int i, int M) {
  for (int m = 0; m < M; ++m) {
    if (!consistent(i, m)) return false;
  }
  return true;
}

int main() {
  int T; cin >> T;
  for (int t = 1; t <= T; ++t) {
    int N, M; cin >> N >> M;
    for (int m = 0; m < M; ++m) {
      char x; cin >> ms[m][0] >> ms[m][1] >> x;
      mb[m] = (x == 'T');
    }
    int total = 0;
    for (int i = 0; i < 1 << N; ++i) {
      if (all_consistent(i, M)) total++;
    }
    cout << "Case #" << t << ": " << total << endl;
  }
  return 0;
}
