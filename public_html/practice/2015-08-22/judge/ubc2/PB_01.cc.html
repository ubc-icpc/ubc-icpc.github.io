#include <bits/stdc++.h>

using namespace std;

bool isMatching(const string& pattern, const string& word) {
  if (pattern.size() != word.size()) return false;
  for (int i = 0; i < pattern.size(); ++i) {
    if (pattern[i] != word[i] && pattern[i] != '-') return false;
  }
  return true;
}

int main() {
  int N; cin >> N;
  vector<string> dict;
  for (int i = 0; i < N; ++i) {
    string word; cin >> word;
    dict.push_back(word);
  }
  int W; cin >> W;
  for (int i = 1; i <= W; ++i) {
    string pattern; cin >> pattern;
    int total = 0;
    cout << "Word #" << i << ": " << pattern << endl;
    for (int j = 0; j < dict.size(); ++j) {
      if (isMatching(pattern, dict[j])) {
        cout << dict[j] << endl;
        total++;
      }
    }
    cout << "Total number of candidate words = " << total << endl;
  }
  return 0;
}
