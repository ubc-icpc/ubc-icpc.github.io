#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

ll dp[101][9];
int cs[8];

ll dpv(int a, int b) {
	if (a < 0 || b <= 0) return 0;
	if (a == 0) return 1;
	return dp[a][b];
}

int main() {
	memset(dp, 0, sizeof(dp[0][0])*101*9);
	int T; cin >> T;
	for (int t = 1; t <= T; ++t) {
		for (int i = 0; i < 8; ++i) {
			cin >> cs[i];
		}
		int N; cin >> N;
		for (int n = 1; n <= N; ++n) {
			for (int i = 1; i <= 8; ++i) {
        dp[n][i] = dpv(n, i-1) + dpv(n - cs[i-1], i);
      }
		}
		cout << "Equation #" << t << ": " << dp[N][8] << endl;
	}
	return 0;
}
