#include <iostream>
#include <cstdlib>
#include <string>

using namespace std;

bool iscoprime(int a, int b) {

	int gcd;
	
	while (a > 0 && b > 0) {
		
		int m = max(a, b);
		
		if (m == a)
			a = a - b;
		else
			b = b - a;
	
	}
	
	gcd = max(a, b);
	
	if (gcd == 1) {
		return true;
	
	} else
		return false;

}

int main() {

	int n; cin >> n;
	
	for (int i = 0; i < n; i++) {
		string num; cin >> num;
		
		cout << "Ticket #" << i + 1 << ":\n";
		bool isrel = false;
		for (int j = 1; j < num.length(); j++) {
			string left = num.substr(0, j);
			string right = num.substr(j, num.length() - j);
			
			if (iscoprime(atoi(left.c_str()), atoi(right.c_str()))) {
				cout << left << " " << right << '\n';
				isrel = true;
				break;
			}		
		}
		
		if (!isrel) 
			cout << "Not relative\n";
		
	
	}


}
