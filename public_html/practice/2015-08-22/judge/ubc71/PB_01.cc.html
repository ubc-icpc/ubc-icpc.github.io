#include <iostream>
#include <regex>
#include <string>
#include <algorithm>

using namespace std;

int main(int argc, char** argv) {

	int n; cin >> n;
	string dict[n];
	
	for (int i = 0; i < n; i++) {
		string word;
		cin >> word;
		dict[i] = word;
	}
	
	int m; cin >> m;
	
	for (int i = 1; i <= m; i++) {
		string word;
		cin >> word;
		
		cout << "Word #" << i << ": " << word << '\n';
		
		replace(word.begin(), word.end(), '-', '.');
		
		//cout << "REPLACED: " << word << endl;
		
		regex reg(word);
		
		int tot = 0;
		for (int j = 0; j < n; j++) {
			
			if (regex_match(dict[j], reg)) {
				cout << dict[j] << '\n';
				tot += 1;
			} //else {
			//	cout << word << '\n';
			//}
		
		}
		
		cout << "Total number of candidate words = " << tot << endl;
	
	}	 

	return 0;
}
