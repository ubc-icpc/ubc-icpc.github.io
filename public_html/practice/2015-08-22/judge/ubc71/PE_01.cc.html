#include <iostream>
#include <string>

using namespace std;


int main() {
	
	int t; cin >> t;
	
	for (int test = 1; test <= t; test++) {
		int r; cin >> r;
		int c; cin >> c;
		
		int board[r][c];
		
		for (int row = 0; row < r; row++) {
			for (int col = 0; col < c; col++) {
				
				string lol; cin >> lol;
				
				if (lol.compare("P") == 0 || lol.compare("E") == 0) {
					//cout << lol << endl;
					board[row][col] = 0;
				} else {
					board[row][col] = atoi(lol.c_str());
				
				}
			}
		}
		
		//board[0][0] = 0;
		//board[r-1][c-1] = 0;
		
/*
for (int row = 0; row < r; row++) {
for (int col = 0; col < c; col++) {

cout << board[row][col] << " ";

}
cout << endl;
}
*/		
		int sumboard[r][c];
		
		sumboard[0][0] = board[0][0];
		
		for (int col = 1; col < c; col++) {
			sumboard[0][col] = board[0][col] + sumboard[0][col-1];
		}
		
		for (int row = 1; row < r; row++) {
			sumboard[row][0] = board[row][0] + sumboard[row-1][0];
		
		}
		
		for (int row = 1; row < r; row++) {
			for (int col = 1; col < c; col++) {
				sumboard[row][col] = max(sumboard[row-1][col], sumboard[row][col-1]) + board[row][col];				
			}
		}
/*	
for (int row = 0; row < r; row++) {
for (int col = 0; col < c; col++) {

cout << sumboard[row][col] << " ";

}
cout << endl;
}*/

		cout << "Game Board #" << test << ": " << sumboard[r-1][c-1] << "\n\n";
		
//cout << r << " " << c << " " << sumboard[r-1][c-1] << endl;
	
	}

	return 0;

}
