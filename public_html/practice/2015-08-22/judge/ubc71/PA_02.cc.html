#include <iostream>

using namespace std;

int main(int argc, char** argv) {
	
	int totalmoney = 0;
	int money;
	int input = 1;
	
	cin >> money;
	
	while (money != 0) {
	
		totalmoney += money;
	
		int games = totalmoney / 50;
		int remainder = totalmoney % 50;
	
		if (games > 1) {
			cout << "Input #"<< input << ": " << "Totally Sweet!\n";
			totalmoney = remainder;
		}
		else if (games > 0) {
			cout << "Input #" << input << ": " << "Sweet!\n";
			totalmoney = remainder;
		}
	
		cin >> money;
		input += 1;
	
	}

	return 0;
}
