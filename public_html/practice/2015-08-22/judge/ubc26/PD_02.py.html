from sys import stdin

def gcd(p, q):
    while q > 0:
        p, q = q, p%q
    return p

m = int(stdin.next())
for i in range(m):
    n = stdin.next()
    print("Ticket #%d:" % i)
    for j in range(1, len(n)-2):
        p, q = int(n[:j]), int(n[j:])
        if gcd(p, q) == 1:
            print p, q
            break
    else:
        print "Not relative"
    print ""
