from sys import stdout
from pprint import pprint
t1 = [[1,1],[1,1]]

def expand(xss):
    u = len(xss)
    v = u / 2
    yss = [[0 for j in xrange(2*u)] for i in xrange(2*u)]

    for i, j in ((0, v), (u, 0), (u, u)):
        for ii in xrange(0, u):
            for jj in xrange(0, u):
                yss[i + ii][j + jj] = xss[ii][jj]

    return yss

def show(xss):
    for xs in xss:
        for x in xs:
            stdout.write('X' if x else ' ')
        print ""

m = int(raw_input())

for p in xrange(m):
    a = int(raw_input())
    t = t1
    for x in xrange(a-1):
        t = expand(t)

    print "Triangle #%d:" % (p+1)
    show(t)
    print ""
