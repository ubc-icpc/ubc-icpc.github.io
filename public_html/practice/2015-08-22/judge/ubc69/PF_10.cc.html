#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <algorithm>
#define INF 1000000

using namespace std;
char triangle[1024][1024];
int powers[11];

void fill(int y, int x, int n){
	if(n==1){
		triangle[y][x]='X';
		triangle[y-1][x]='X';
		triangle[y][x+1]='X';
		triangle[y-1][x+1]='X';
		return;
	}
	fill(y,x,n-1);
	fill(y,x+powers[n-1],n-1);
	fill(y-powers[n-1],x+powers[n-2],n-1);
}

int main() {
	int n;
	cin>>n;
	powers[0]=1;
	for(int i=1;i<11;i++){
		powers[i]=powers[i-1]*2;
	}
	for(int i=1;i<=n;i++){
		int m;
		cin>>m;
		for(int j=0;j<1024;j++)
			for(int k=0;k<1024;k++)
				triangle[j][k]=' ';
		fill(powers[m]-1,0,m);
		cout<<"Triangle #"<<i<<':'<<endl;
		for(int j=0;j<powers[m];j++){
			for(int k=0;k<powers[m];k++)
				cout<<triangle[j][k];
			cout<<endl;
		}
	}
    return 0;
}
