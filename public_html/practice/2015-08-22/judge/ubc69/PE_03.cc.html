int maxi(int a, int b){
	if(a>b) return a;
	return b;
}

int toint(string s){
	int n = 0;
	for(int i=0;i<s.size();i++)
		n = 10*n+s[i]-'0';
	return n;
}

int main() {
	int n,r,c;
	cin>>n;
	for(int i=1;i<=n;i++){
		r = c = 333;
		cin>>r>>c;
		int board[r+1][c+1];
		
		for(int j=0;j<=r;j++)
			board[j][0]=0;
		for(int j=0;j<=c;j++)
			board[0][j]=0;
			
		for(int j=1;j<=r;j++)
			for(int k=1;k<=c;k++){
				string z; int p;
				cin>>z;
				if((j==1&&k==1)||(j==r&&k==c)) p=0;
				else p=toint(z);
				board[j][k] = maxi(board[j-1][k],board[j][k-1])+p;
			}
			
		cout<<"Game Board #"<<i<<": "<<board[r][c]<<endl<<endl;
	}
    return 0;
}
