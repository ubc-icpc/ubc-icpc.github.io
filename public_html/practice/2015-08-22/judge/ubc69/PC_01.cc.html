using namespace std;

int min(int p[]){
	int n=0;
	for(int i=0;i<5;i++)
		if(p[i]<p[n]) n=i;
	return n;
}

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		int p[5];
		for(int j=0;j<5;j++)
			cin>>p[j];
		cout<<"Pentagon #"<<i<<':'<<endl;
		
		while(p[0]<0||p[1]<0||p[2]<0||p[3]<0||p[4]<0){
			int low = min(p);
			if(p[low]>0) break;
			p[low] = -p[low];
			p[(5+low-1)%5]-=p[low];
			p[(low+1)%5]-=p[low];
		}
		cout<<p[0]<<' '<<p[1]<<' '<<p[2]<<' '<<p[3]<<' '<<p[4]<<endl;
	}
    return 0;
}
