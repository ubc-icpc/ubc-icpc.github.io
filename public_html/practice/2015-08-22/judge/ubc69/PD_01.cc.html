using namespace std;

int a,b;

bool rel(int a, int b){
	for(int i=2;i<=a;i++){
		if(a%i==0 && b%i==0)
			return false;
	}
	return true;
}

void find(string t){
	a=b=-1;
	for(int i=1;i<t.size();i++){
		int c=0,d=0;
		for(int j=0;j<i;j++){
			c*=10;
			c+=t[j]-'0';
		}
		for(int j=i;j<t.size();j++){
			d*=10;
			d+=t[j]-'0';
		}
		if(rel(c,d)){
			a=c;
			b=d;
			return;
		}
	}
}

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		cout<<"Ticket #"<<i<<':'<<endl;
		string t;
		cin>>t;
		find(t);
		if(a==-1)
			cout<<"Not relative"<<endl;
		else
			cout<<a<<' '<<b<<endl;
		cout<<endl;
	}
    return 0;
}
