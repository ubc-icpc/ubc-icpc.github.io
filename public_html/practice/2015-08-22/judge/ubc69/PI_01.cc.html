#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <algorithm>
#define INF 1000000

using namespace std;

int c[8],n;
long long dp[8][101];

long long solve(int a, int sum){
	if(sum==n) return 1;
	if(sum>n||a>=8) return 0;
	if(dp[a][sum]>=0) return dp[a][sum];
	dp[a][sum] = solve(a,sum+c[a]) +solve(a+1,sum);
	return dp[a][sum];
}

int main() {
	int e;
	cin>>e;
	for(int i=1;i<=e;i++){
		for(int j=0;j<8;j++)
			for(int k=0;k<101;k++)
				dp[j][k]=-1;
		for(int j=0;j<8;j++)
			cin>>c[j];
		cin>>n;
		cout << "Equation #" << i <<": "<< solve(0,0) << endl;
	}
    return 0;
}
