using namespace std;

bool match(string a, string b){
	if(a.size()!=b.size()) return false;
	
	for(int i=0;i<a.size();i++){
		if(a[i]!=b[i]&&b[i]!='-')
			return false;
	}
	return true;
}

int main() {
	int n, m;
	cin>>n;
	string dict[n];
	for(int i=0;i<n;i++)
		cin>>dict[i];
	cin>>m;
	for(int i=1;i<=m;i++){
		string word;
		int count=0;
		cin>>word;
		cout<<"Word #"<<i<<": "<<word<<endl;
		for(int j=0;j<n;j++){
			if(match(dict[j],word)){
				cout<<dict[j]<<endl;
				count++;
			}
		}
		cout<<"Total number of candidate words = "<<count<<endl;
	}
    return 0;
}
