#include <bits/stdc++.h>
using namespace std;
int main(){
	bool found = false; 
	string alphabet = "abcdefghijklmnopqrstuvwxyz";
	char a;
	int n; cin >> n; string qq; getline(cin,qq);
	while(n--){
	string missing = " ";
	string line;
	getline(cin, line);
	for (char letter: alphabet){
		found = false;
		for (char a:line){
			if (letter == tolower(a)){
				found = true;
				break;
			}
		}
		if (found == false){
		missing.push_back(letter);
		}
		
	}
	
	if (missing == " "){ 
		cout << "pangram"<< endl;
	}
	
	else{
		cout << "missing" << missing << endl;
		 
	
	}
	}
}
