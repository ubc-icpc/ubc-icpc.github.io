ull binomial(ull n, ull k)
{
	ull den = 1, num = 1;
	for (int i = 1; i <= k; i = i + 1) den = den * i;
	for (int j = n - k + 1; j <= n; j = j + 1) num = num * j;
	return num / den;
}

int main()
{
	int t; cin >> t;
	while (t--)
	{
		int r, s, x, y, w;
		cin >> r >> s >> x >> y >> w;

		ld prob = 0;
		ld sing = 1.0L - ((ld) (r - 1) / s);
		for (int i = x; i <= y; i++)
		{
			// cout << binomial(y, i) * pow(sing, i) * pow(1 - sing, y - i) << endl;
			prob += binomial(y, i) * pow(sing, i) * pow(1 - sing, y - i);
		}
		// cout << prob << endl;

		if (prob * w > 1.0L)
		{
			cout << "yes" << endl;
		}
		else
		{
			cout << "no" << endl;
		}
	}
	return 0;
}
