#include <bits/stdc++.h>
using namespace std;

using uc = unsigned char;
using ui = unsigned int;
using ull = unsigned long long;
using ll = long long;
using vs = vector<string>;
using vi = vector<int>;
using vui = vector<ui>;
using vuc = vector<uc>;
using vull = vector<ull>;
using vll = vector<ll>;
using ss = set<string>;
using suc = set<uc>;
using si = set<int>;
using sui = set<ui>;
using sll = set<ll>;
using sull = set<ull>;
using uss = unordered_set<string>;
using usui = unordered_set<ui>;
using usuc = unordered_set<uc>;
using usi = unordered_set<int>;
using usll = unordered_set<ll>;
using usull = unordered_set<ull>;

ll a[450], dp[450];
ll solve(int n)
{
	if (n == 1) return a[0];
	if (n == 2) return a[1];
	if (dp[n] != -1) return dp[n];
	dp[n] = min(solve(n-1) + a[0] + a[n-1], solve(n-2) + a[0] + 2 * a[1] + a[n-1]);
	return dp[n];
}

int main()
{
	int n; cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
	}

	sort(a, a+n);
	memset(dp, -1, sizeof dp);
	cout << solve(n) << endl;
	return 0;
}
