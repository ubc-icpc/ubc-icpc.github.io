#include <bits/stdc++.h>
using namespace std;

using uc = unsigned char;
using ui = unsigned int;
using ull = unsigned long long;
using ll = long long;
using vs = vector<string>;
using vi = vector<int>;
using vui = vector<ui>;
using vuc = vector<uc>;
using vull = vector<ull>;
using vll = vector<ll>;
using ss = set<string>;
using suc = set<uc>;
using si = set<int>;
using sui = set<ui>;
using sll = set<ll>;
using sull = set<ull>;
using uss = unordered_set<string>;
using usui = unordered_set<ui>;
using usuc = unordered_set<uc>;
using usi = unordered_set<int>;
using usll = unordered_set<ll>;
using usull = unordered_set<ull>;

int main()
{
	int t; cin >> t;
	string s;
	getline(cin, s);

	while (t--)
	{
		getline(cin, s);
		vector<char> alphabet(26);
		iota(alphabet.begin(), alphabet.end(), 'a');
		set<char> oo(alphabet.begin(), alphabet.end());

		for (char i : s)
		{
			oo.erase(tolower(i));
		}

		if (oo.size() == 0)
		{
			cout << "pangram" << endl;
		}
		else
		{
			cout << "missing ";
			for (char i : oo)
			{
				cout << i;
			}
			cout << endl;
		}
	}

	return 0;
}
