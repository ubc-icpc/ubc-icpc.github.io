#include <bits/stdc++.h>
using namespace std;

using uc = unsigned char;
using ui = unsigned int;
using ull = unsigned long long;
using ll = long long;
using vs = vector<string>;
using vi = vector<int>;
using vui = vector<ui>;
using vuc = vector<uc>;
using vull = vector<ull>;
using vll = vector<ll>;
using ss = set<string>;
using suc = set<uc>;
using si = set<int>;
using sui = set<ui>;
using sll = set<ll>;
using sull = set<ull>;
using uss = unordered_set<string>;
using usui = unordered_set<ui>;
using usuc = unordered_set<uc>;
using usi = unordered_set<int>;
using usll = unordered_set<ll>;
using usull = unordered_set<ull>;

int main()
{
	int t; cin >> t;
	string s;
	getline(cin, s);

	while (t--)
	{
		getline(cin, s);

		int side = 100;
		for (int i = 1; i <= 100; i++)
		{
			if (i * i >= s.size())
			{
				side = i;
				break;
			}
		}

		vector<vector<char>> mat(side);
		for (auto& i : mat) i.resize(side);

		int ind = 0;
		for (int rr = 0; rr < side; rr++)
		{
			for (int cc = 0; cc < side; cc++)
			{
				if (ind < s.size())
				{
					mat[rr][cc] = s[ind++];
				}
				else
				{
					mat[rr][cc] = '*';
				}
			}
		}

		for (int cc = 0; cc < side; cc++)
		{
			for (int rr = side - 1; rr >= 0; rr--)
			{
				if (mat[rr][cc] != '*')
				{
					cout << mat[rr][cc];
				}
			}
		}
		cout << endl;
	}
	return 0;
}
