ll operate(ll a, char op, ll b)
{
	switch (op)
	{
		case '+': return a + b;
		case '-': return a - b;
		case '*': return a * b;
		case '/': return b == 0 ? -1 : a / b;
		default: return 0;
	}
}

int main()
{
	int t; cin >> t;
	while (t--)
	{
		string a, op, b, eq, res;
		cin >> a >> op >> b >> eq >> res;

		char highest = max(max(*max_element(a.begin(), a.end()), *max_element(b.begin(), b.end())), *max_element(res.begin(), res.end()));
		int start;
		if (highest == '1')
		{
			start = 1;
		}
		else if (highest <= '9')
		{
			start = highest - '0' + 1;
		}
		else
		{
			start = highest - 'a' + 11;
		}

		bool ok = false;
		for (int i = start; i <= 36; i++)
		{
			if (op[0] == '/')
			{
				//cout << i << endl;
				//cout << a << " " << b << " " << res << endl;
			}
			
			try
			{
				if (i == 1)
				{
					if (a.find('0') == -1 && b.find('0') == -1 && res.find('0') == -1)
					if (operate(a.size(), op[0], b.size()) == res.size())
					{
						if (op[0] != '/' || (res.size() * b.size() == a.size()))
						{
							ok = true;
							cout << '1';
						}
					}
				}
				else
				{
					ll a_, b_, r_;
					a_ = stoll(a, 0, i);
					b_ = stoll(b, 0, i);
					r_ = stoll(res, 0, i);
					if (operate(a_, op[0], b_) == r_)
					{
						if (op[0] != '/' || r_ * b_ == a_)
						{
							ok = true;

							if (i == 36)
							{
								cout << '0';
							}
							else if (i <= 9)
							{
								cout << (char) (i + '0');
							}
							else
							{
								cout << (char) (i - 10 + 'a');
							}
						}
					}
				}
			}
			catch (...)
			{
			}
		}
		
		if (ok)
		{
			cout << endl;
		}
		else
		{
			cout << "invalid" << endl;
		}
	}
	return 0;
}
