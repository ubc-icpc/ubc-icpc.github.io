#include <bits/stdtr1c++.h>
using namespace std;
#define endl '\n'

typedef long long ll;
typedef long double ld;

ll bin(int n, int k) {
	ll den = 1, num = 1;
    for (int i = 1; i <= k; i = i + 1) den = den * i;
    for (int j = n - k + 1; j <= n; j = j + 1) num = num * j;
    return num / den;
}

int main() {
	int n; cin >> n;
	while(n--) {
		int r,s,x,y,w;
		cin >> r >> s >> x >> y >> w;
		ld prob = 0;
		ld win = 1.0L - ((ld) (r-1.0L)/s);
		for (int i=x; i<=y; i++) {
			prob += bin(y,i)*pow(win,i)*pow(1-win,y-i);
		}
		if (prob*w > 1.0L) cout << "yes";
		else cout << "no";
		cout << endl;
	}
	return 0;
}
