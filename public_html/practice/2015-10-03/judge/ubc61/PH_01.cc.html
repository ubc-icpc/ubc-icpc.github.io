#include <bits/stdtr1c++.h>
using namespace std;
#define endl '\n'


int main() {
	int n; cin >> n;
	while(n--) {
		string s; cin >> s;
		int l = s.size();
		int m;
		if ((double) sqrt(l) != (int) sqrt(l)) m = sqrt(l)+1;
		else m = sqrt(l);
		m*=m;

		int k = sqrt(m);
		char arr[k][k];
		int x=0;
		for (int i=0; i<k; i++) {
			for (int j=0; j<k; j++) {
				if (x<s.size()) arr[i][j] = s[x];
				else arr[i][j] = '*';
				x++;
			}
		}

		for (int i=0; i<k; i++) {
			for (int j=k-1; j>=0; j--) {
				if (arr[j][i] != '*') cout << arr[j][i];
			}
		}
		cout << endl;
	}
	return 0;
}
