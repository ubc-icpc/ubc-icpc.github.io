#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;
ll a[450], dp[450];
ll solve(int n) {
	if (n == 1) return a[0];
	if (n == 2) return a[1];
	if (dp[n] != -1) return dp[n];
	dp[n] = min(solve(n-1) + a[0] + a[n-1], solve(n-2) + a[0] + 2*a[1] + a[n-1]);
	return dp[n];
}

int main() {
	int n;
	while (cin >> n) {
		for (int i = 0; i < n; i++) {
			cin >> a[i];
		}
		
		sort(a, a+n);
		memset(dp, -1, sizeof dp);
		cout << solve(n) << endl;
	}
	return 0;
}
