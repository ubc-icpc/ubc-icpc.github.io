#include <bits/stdtr1c++.h>

using namespace std;

char a[105][105];
int main() {
    int t; cin >> t;
    while (t--) {
        string s; cin >> s;
        if (s.back() == '\r' || s.back() == '\n') s.resize(s.size()-1);
        int l = sqrt(s.length());
        if (l*l < s.length()) l++;
        s.resize(l*l);
        
        for (int i = 0; i < l; i++) {
            for (int j = 0; j < l; j++) {
                a[i][j] = s[i*l+j];
            }
        }
        
        for (int j = 0; j < l; j++) {
            for (int i = l-1; i >= 0; i--) {
                if (!isalpha(a[i][j])) continue;
                cout << a[i][j];
            }
        }
        cout << endl;
    }
	return 0;
}
