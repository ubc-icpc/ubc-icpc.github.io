from math import *
from decimal import *

getcontext().prec = 20
t = int(raw_input())

for i in range(t):
    r, s, x, y, w = [int(x) for x in raw_input().split()]
    
    # expected probability of winning > 0.5 = take bet
    # prob of winning 1 roll = (S-R+1)/S
    # prob of winning at least x out of y rolls
    # sum binomial yCx * p^i * (1-p)^(y-i) for i = x to y
    p = Decimal(s-r+1)/s
    pwin = 0
    for i in range(x, y+1):
        binom = (factorial(y)/factorial(i))/factorial(y-i)
        pwin += binom * pow(p, i) * pow(1-p, y-i)
    
    if pwin*w > 1:
        print 'yes'
    else:
        print 'no'
