#include <bits/stdc++.h>

using namespace std;

int mintime(int end, vector<int>& times, unordered_map<int,int>& soln) {
  if (end == 1) return times[1];
  if (end == 2) return times[0] + times[1] + times[2];
  if (soln.find(end) != soln.end()) return soln[end];
  else {
    soln[end] = min(times[0]+times[end]+mintime(end-1, times, soln),
      times[0]+2*times[1]+times[end]+mintime(end-2, times, soln));
    return soln[end];
  }
}

int main() {
  int N; cin >> N;
  vector<int> v;
  while (N--) {
    int num; cin >> num;
    v.push_back(num);
  }
  sort(v.begin(), v.end());
  unordered_map<int, int> soln;
  cout << mintime(v.size()-1, v, soln) << endl;
  return 0;
}
