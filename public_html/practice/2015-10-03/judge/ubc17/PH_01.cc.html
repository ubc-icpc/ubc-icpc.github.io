#include <bits/stdc++.h>
using namespace std;

char grid[100][100];

int main() {
    int t; cin >> t;
    while (t--) {
        string s;
        cin >> s;

        int len = s.length();

        int gridsize = (int)pow(len, 0.5);
        if (gridsize * gridsize < len) ++gridsize;

        for (int i = 0; i < gridsize; ++i) {
            for (int j = 0; j < gridsize; ++j) {
                int index = i * gridsize + j;
                if (index >= len) {
                    grid[i][j] = '*';
                } else {
                    grid[i][j] = s[index];
                }
            }
        }

        for (int i = 0; i < gridsize; ++i) {
            for (int j = gridsize - 1; j >= 0; --j) {
                if (grid[j][i] != '*') {
                    printf("%c", grid[j][i]);
                }
            }
        }
        printf("\n");
    }
}
