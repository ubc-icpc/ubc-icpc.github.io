#include <bits/stdc++.h>
using namespace std;

#define ld long double

int nCr(int n, int r) {
    int result = 1;
    if (r > n / 2) r = n - r;
    for (int i = 1; i <= r; ++i) {
        result *= (n + 1 - i);
        result /= i;
    }
    return result;
}

int main() {
    int t; cin >> t;
    while (t--) {
        int r, s, x, y, w;
        cin >> r >> s >> x >> y >> w;

        ld odds = 1.0 / w;

        ld sum = 0.0;
        for (int i = x; i <= y; ++i) {
            //printf("here %d %f %f\n", nCr(y, i), pow((r - 1.0) / s, y - i), pow((s - r + 1.0) / s, i));
            sum += nCr(y, i) * 1.0 * pow((r - 1.0) / s, y - i) * pow((s - r + 1.0) / s, i);
        }

        //printf("sum %Lf odds %Lf\n", sum, odds);
        cout << ((sum > odds) ? "yes" : "no") << endl;
    }
}
