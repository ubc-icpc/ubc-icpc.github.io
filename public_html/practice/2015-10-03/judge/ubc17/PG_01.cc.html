#include <bits/stdc++.h>
using namespace std;

#define vll vector<long long>

int main() {
    int n; cin >> n;
    vll vals;

    for (int i = 0; i < n; ++i) {
        long long x;
        cin >> x;
        vals.push_back(x);
    }

    /*for (int i = 0; i < vals.size(); ++i) {
        cout << vals[i] << endl;
    }*/

    sort(vals.begin(), vals.end());

    long long result = 0;

    while (n > 0) {
        if (n == 2) {
            result += vals[1];
            break;
        } else if (n == 3) {
            result += vals[0] + vals[1] + vals[2];
            break;
        } else {
            long long a = 2 * vals[1];
            long long b = vals[0] + vals[n-2];
            if (a < b) {
                result += vals[0] + 2 * vals[1] + vals[n-1];
                n -= 2;
            } else {
                result += vals[0] + vals[n-1];
                n -= 1;
            }
        }
    }

    cout << result << endl;


}
