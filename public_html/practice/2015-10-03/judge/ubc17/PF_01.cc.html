#include <bits/stdc++.h>
using namespace std;

bool c[256];

#define ASCII_A 65
#define ASCII_a 97

int main() {
    int t;
    string s;
    cin >> t;
    getline(cin, s);
    while (t--) {
        getline(cin, s);
        memset(c, 0, sizeof(c));

        for (int i = 0; i < s.length(); ++i) {
            char ch = s[i];
            if (ch >= ASCII_a) {
                c[ch - ASCII_a] = true;
            } else if (ch >= ASCII_A) {
                c[ch - ASCII_A] = true;
            }
        }

        bool pangram = true;

        for (int i = 0; i < 26; ++i) {
            if (!c[i]) {
                pangram = false;
                break;
            }
        }

        if (pangram) {
            printf("pangram\n");
        } else {
            printf("missing ");
            for (int i = 0; i < 26; ++i) {
                if (!c[i]) {
                    printf("%c", i + ASCII_a);
                }
            }
            printf("\n");
        }
    }
}
