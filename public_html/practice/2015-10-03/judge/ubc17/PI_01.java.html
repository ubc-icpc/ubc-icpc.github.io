import java.util.*;

public class I {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);

        int t = s.nextInt();

        s.nextLine();

        while (t-- != 0) {
            String curr = s.nextLine();

            if (curr.startsWith("Simon says ")) {
                System.out.println(curr.substring(10));
            }
        }
    }
}
