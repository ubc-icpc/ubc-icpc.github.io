using namespace std;

map<string, vector<string> > m;
set<string> vis;
vector<string> path;
bool dfs(string start, string end) {
  //return false;
  if(start.compare(end)==0) return true;
  vis.insert(start);
  for(unsigned i=0;i<m[start].size();i++) {
	if(vis.find(m[start][i])==vis.end() ) {
	  if(dfs(m[start][i],end)) {
		path.push_back(m[start][i]);
		return true;
	  }
	}
  }
  return false;
}
