#include<bits/stdtr1c++.h>
using namespace std;

int N, M, P, X, Y, C, A[100005], V[100005];

int main () {
  cin >> N >> M >> P;
  while (N--) {
    cin >> X >> Y;
    if (!A[Y]) A[Y] = X;
  }
  while (true) {
    if (A[P] == 0) break;
    if (V[P]) { C = -1; break; }
    V[P] = true;
    P = A[P];
    ++C;
  }
  cout << C << endl;
  return 0;
}
