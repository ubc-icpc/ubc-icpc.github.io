#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n, m, p; cin >> n >> m >> p;
	vector < int > cj(m + 1, 0);
	for (int i = 0; i < n; ++i){
		int ga, gb; cin >> ga >> gb;
		if (cj[gb] == 0) cj[gb] = ga;
	}
	int num = 0;
	bool flag = true;
	for (int i = 0; i < 100001; ++i){
		if (cj[p] != 0) {
			p = cj[p];
			++num;
		}
		else{
			flag = false;
			break;
		}
	}
	if (flag){
		cout << "-1" << endl;
	}
	else{
		cout << num << endl;
	}
	return 0;
}
