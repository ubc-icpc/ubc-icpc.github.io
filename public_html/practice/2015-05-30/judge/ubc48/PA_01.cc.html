#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	vector < string > dict(8);
	dict[0] = "1QAZ";
	dict[1] = "2WSX";
	dict[2] = "3EDC";
	dict[3] = "4RFV5TGB";
	dict[4] = "6YHN7UJM";
	dict[5] = "8IK,";
	dict[6] = "9OL.";
	dict[7] = "0P;/-['=]";
	string gar; cin >> gar;
	int ans[8];
	memset(ans, 0, sizeof ans);
	for (int k = 0; k < gar.size(); ++k){
		for (int i = 0; i < 8; ++i){
			for (int j = 0; j < dict[i].size(); ++j){
				if (gar[k] == dict[i][j]) ans[i]++;
			}
		}
	}
	for (int i = 0; i < 8; ++i){
		cout << ans[i] << endl;
	}
	return 0;
}
