#include<iostream>
#include<string.h>

using namespace std;

int coff=100000;

int main (void) {
  int P[100001];//pensioner - fav channel
  int N[100001];//channel - youngest pensioner
  

  int n,m,c; cin >> n >> m >> c;

  //for( int i=0; i<m+5 && i<100001; i++ ) N[i] = -1;
  memset(N,-1,sizeof(int)*100001);
  //  for( int k=0; k<n+1; k++ ) cout << P[k] << endl;
  //  for( int k=0; k<m+1; k++ ) cout << N[k] << endl;
  for( int i=1; i<=n; i++ ) {
    int a,b; cin >> a >> b;
    P[i] = a;
    if(N[b] < 0 || N[b]>i) N[b] = i;
  }
  /*
  for( int i=1; i<=n; i++ ) {
    cout<< P[i] << endl;
  }
  cout<< endl;
  for( int i=1; i<=m; i++ ) {
    cout << N[i] << endl;
  }
  */
  int count = 0;
  bool ok = true;
  while(N[c] >= 0 ) {
    //cout << "nc " << N[c] << " " << "pnc " << P[N[c]] << endl;
    c = P[N[c]];
    count ++ ;
    if(count>coff) {
      ok = false;
      break;
    }
  }
  
  if(ok)cout << count << endl;
  else cout << -1 << endl;
  return 0;
}
