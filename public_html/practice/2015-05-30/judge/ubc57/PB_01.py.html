a=list(map(int,raw_input().split()))
N=a[0]
M=a[1]
P=a[2]
Visited=[False]*M
H=[-1]*M #H[i] is the youngest who hates channel i
L=[-1]*N #L[i] is th channel i loves
for i in range (0,N):
    a=list(map(int,raw_input().split()))
    L[i]=a[0]
    if H[a[1]-1]==-1:
        H[a[1]-1]=i
count=0
while True:
    if H[P-1]==-1:
        print(count)
        break
    P=L[H[P-1]]
    count=count+1
    if Visited[P-1]:
        print(-1)
        break
    Visited[P-1]=True
