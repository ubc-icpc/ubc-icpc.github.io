import java.util.Scanner;

public class channelDOM {

	public static void main(String[] args) {
		Scanner s = new Scanner(System.in);
		int N = s.nextInt();
		int M = s.nextInt();
		int P = s.nextInt();
		int pointsTo[] = new int[M]; //initialized 0, indexed 0...M-1?; CONTENT of array is 1...M
		int visited[] = new int[M];
		while(N-- > 0) { //process seniors
			int favorite = s.nextInt();
			int hate = s.nextInt();
			if (pointsTo[hate-1] == 0)
				pointsTo[hate-1] = favorite;
		}
		int length = 0;
		int currChannel = P;
		while (true) {
			if (pointsTo[currChannel-1] == 0)
				break;
			if (visited[currChannel-1] == 1) {
				length = -1;
				break;
			}
			visited[currChannel-1] = 1;
			currChannel = pointsTo[currChannel-1];
			length++;
		}
		System.out.println(length);
		s.close();
	}

}
