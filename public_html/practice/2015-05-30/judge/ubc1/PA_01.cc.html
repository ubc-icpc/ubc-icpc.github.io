#include <bits/stdtr1c++.h>

using namespace std;

string a[] = {"1QAZ", "2WSX", "3EDC", "4RFV5TGB", "6YHN7UJM", "8IK,", "9OL.", "0P;/'[-=]"};
int main() {
    map<char, int> m;
    for (int i = 0; i < 8; i++) {
        for (char c : a[i]) {
            m[c] = i;
        }
    }
    
    map<int, int> ans;
    string s; cin >> s;
    for (char c : s) {
        ans[m[c]]++;
    }
    
    for (int i = 0; i < 8; i++) {
        cout << ans[i] << endl;
    }
	return 0;
}
