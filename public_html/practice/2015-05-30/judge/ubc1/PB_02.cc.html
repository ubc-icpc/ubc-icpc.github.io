#include <bits/stdtr1c++.h>

using namespace std;

int g[2][100005];
int main() {
    ios::sync_with_stdio(0);
    int n, m, p;
    cin >> n >> m >> p;
    
    memset(g, 0x3f, sizeof g);
    for (int i = 0; i < n; i++) {
        int a, b; cin >> a >> b;
        g[0][i] = a;
        g[1][b] = min(g[1][b], i);
    }
    
    int c = p, ans = 0;
    while (g[1][c] != 0x3f3f3f3f) {
        ans++;
        c = g[0][g[1][c]];
        if (ans > 1e6) {
            ans = -1;
            break;
        }
    }
    cout << ans << endl;
	return 0;
}
