#include<bits/stdtr1c++.h>

using namespace std;

int n, m, p;
vector<int> liked;
vector<int> hated;
bool seen[100001];

int main() {
  memset(seen, false, sizeof seen);
  cin >> n >> m >> p;
  for (int i=0; i<n; i++) {
    int l, h; cin >> l >> h;
    liked.push_back(l); hated.push_back(h);
  }
  bool flag = true;
  int ans = 0;
  int ch = p;
  while (flag) {
    auto it = find(hated.begin(), hated.end(), ch);
    int offset = it - hated.begin();
    if (seen[offset]) {
      ans = -1;
      flag = false;
    }
    else if (it != hated.end()) {
      seen[offset] = true;
      ch = liked[offset];
      ans++;
    }
    else
      flag = false;
  }
  cout << ans << endl;
}
