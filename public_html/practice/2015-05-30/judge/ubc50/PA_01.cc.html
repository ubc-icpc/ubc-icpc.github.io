#include<bits/stdtr1c++.h>

using namespace std;

// 0 means left pinky, 7 means right pinky
map<string, int> m;
int counts[8];

int main() {
  memset(counts,0,sizeof counts);
  m["~"] = 0;  m["`"] = 0;  m["1"] = 0;  m["Q"] = 0;  m["A"] = 0;  m["Z"] = 0;
  m["2"] = 1;  m["W"] = 1;  m["S"] = 1;  m["X"] = 1;
  m["3"] = 2;  m["E"] = 2;  m["D"] = 2;  m["C"] = 2;
  m["4"] = 3;  m["R"] = 3;  m["F"] = 3;  m["V"] = 3;  m["5"] = 3; m["T"]=3; m["G"]=3; m["B"]=3;
  m["6"] = 4;  m["Y"] = 4;  m["H"] = 4;  m["N"] = 4;  m["7"] = 4;  m["U"] = 4;  m["J"] = 4;  m["M"] = 4;
  m["8"] = 5; m["I"] = 5; m["K"] = 5; m[","] = 5;
  m["9"] = 6;  m["O"] = 6;  m["L"] = 6;  m["."] = 6;
  m["0"] = 7; m["-"] = 7; m["="] = 7; m["P"] = 7; m["["] = 7; m["]"] = 7; m[";"] = 7; m["'"] = 7; m["/"] = 7;

  string s; cin >> s;
								    
  for (int i=0; i<s.length(); i++) {
    int res = m[s.substr(i,1)];
    counts[res]++;
  }
  for (int j=0;j<8;j++)
    cout << counts[j] << endl;
}
