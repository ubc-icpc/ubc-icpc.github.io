#include<iostream>

using namespace std;

int main (void) {
  int T; cin >> T;
  while ( T-- ) {
    int n; cin >> n;
    int x[n];
    for( int i=0; i<n; i++ ) {
      int t; cin >> t;
      x[i] = t;
    }
    if(n <= 4) {
      cout << "YES" << endl;
      continue;
    }
    int a,b,c,d;
    d = ( 6*x[0]);
    c = ( 2*x[3] - 9*x[2] + 18*x[1] - 11*x[0]);
    b = (-3*x[3] + 12*x[2] - 15*x[1] + 6*x[0]);
    a = (x[3] - 3*x[2] + 3*x[1] - x[0]);
   

    //cout << a << b << c << d << endl;
    bool ok = true;
    int i;
    for (i=4; i<n; i++){
      if( (a*(i)*(i)*(i) + b*(i)*(i) + c*(i) + d)/6 != x[i]) {
	ok = false;
	break;
      }
    }
    if(ok) cout << "YES" << endl;
    else cout << "NO" << endl;
  }
  return 0;
}
