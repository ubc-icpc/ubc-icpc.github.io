from math import floor, ceil
import sys
L = sys.stdin.next
for _ in range(int(L())):
    x = int(L())
    y = x
    k = 1
    while y % 10 == 0:
        y /= 10
        k *= 10
    hi, lo = (int(floor(1.05*x)), int(ceil(x*0.95)))

    def diff_and_within(s):
        return s != x and lo <= s <= hi

    r = x % (10*k)
    d5 = r - (5*k)

    # How to lower absurdity:
    c5 = x - d5           # make last digit 5
    c0 = x - r            # make last digit 0 by decreasing
    c10 = x + (10*k - r)  # make last digit 0 by increasing
    # Make sure the 3 candidates are within range

    if diff_and_within(c5) or diff_and_within(c0) or diff_and_within(c10):
        print "absurd"
    else:
        print "not absurd"
