#include <iostream>
using namespace std;

int s[11][11];

int backtrack(int idx, int sum, long occ) {
  if (idx == 11) {
    //for (int i = 0; i < 11; i++) printf(" %d", occ[i]);
    //printf(": %d\n", sum);
    return sum;
  }

  int best = -1;
  for (int i = 0; i < 11; i++) {
    long msk = 1 << i;
    if (s[idx][i] != 0 && 0 == (occ & msk)) {
      int t = backtrack(idx + 1, sum + s[idx][i], occ | msk);
      if (t > best && t != -1) best = t;
    }
  }
  return best;
}

void cas() {
  for (int i = 0; i < 11; i++) {
    for (int j = 0; j < 11; j++) {
      cin >> s[i][j];
    }
  }

  cout << backtrack(0, 0, 0) << endl;
}

int main() {
  int t; cin >> t;
  for (int i = 0; i < t; i++) cas();
}
