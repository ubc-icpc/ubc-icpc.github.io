#include <iostream>
using namespace std;



int main() {
  int t; cin >> t;
  while (t--) {
    int n, d; cin >> n;
    int* a = new int[n];
    for (int i = 0; i < n; i++) cin >> a[i];

    if (n <= 4) {
      cout << "YES" << endl;
      goto L;
    }

    d = -a[0] + 3*a[1] - 3*a[2] + a[3];
    for (int i = 1; i+3<n; i++) {
      int d1 = -a[i+0] + 3*a[i+1] - 3*a[i+2] + a[i+3];
      if (d1 != d) {
        cout << "NO" << endl;
        goto L;
      }
      d = d1;
    }
    cout << "YES" << endl;

L:;
  delete[] a;
  }
}
