cases=int(raw_input())
for i in range(0,cases):
    Dic={}
    for i in range(0,16):
        a=raw_input().split()
        if a[1] not in Dic:
            Dic[a[1]]=0
        if a[0] not in Dic:
            Dic[a[0]]=0
        if int(a[3])>int(a[2]):
            Dic[a[1]]=Dic[a[1]]+1
        else:
            Dic[a[0]]=Dic[a[0]]+1
    for x in Dic.keys():
        if Dic[x]==4:
            print(x)
            break
