#include <iostream>
#include <sstream>
#include <cstdlib>
#include <cmath>
using namespace std;

#define s2l(x) atoll((x).c_str())

typedef long long ll;

int main() {
  int t; cin >> t;
  while (t--) {
    ll p; cin >> p;
    stringstream ss;
    ss << p;
    string s = ss.str();
    int ind = s.length()-1;
    while (s[ind] == '0')
      ind--;
    if (ind == 0 && s[ind] == '5') {
      cout << "not absurd\n";
      continue;
    }
    ll lo, hi;
    if (s[ind] == '5') {
      ll d = s2l(s.substr(ind));
      lo = p-d;
      hi = p+d;
    }
    else if (s[ind] < '5') {
      string s2 = s;
      s2[ind] = '0';
      lo = s2l(s2);
      s2 = s;
      s2[ind] = '5';
      hi = s2l(s2);
    }
    else {
      string s2 = s;
      s2[ind] = '5';
      lo = s2l(s2);
      string s3 = s.substr(ind);
      s3[0] = '9'+1-(s3[0]-'0');
      hi = p+s2l(s3);
    }
    ll loa = ceil(0.95L*p), hia = floor(1.05L*p);
    if (loa <= lo || hi <= hia)
      cout << "absurd\n";
    else
      cout << "not absurd\n";
  }
  return 0;
}
