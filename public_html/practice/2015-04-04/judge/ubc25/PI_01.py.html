from sys import stdin

c = int(stdin.readline().strip())

for _ in range(c):
    winners = {}
    for _ in range(16):
        game = stdin.readline().strip().split()
        if game[2] > game[3]:
            if game[0] in winners:
                winners[game[0]] += 1
            else:
                winners[game[0]] = 1
        else:
            if game[1] in winners:
                winners[game[1]] += 1
            else:
                winners[game[1]] = 1
    v = list(winners.values())
    k = list(winners.keys())
    print(k[v.index(max(v))])
