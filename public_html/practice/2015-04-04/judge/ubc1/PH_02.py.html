t = int(raw_input())

for c in range(t):
	inp = [int(x) for x in raw_input().split()]
	n = inp[0]
	
	# a + bi + ci^2 + di^3
	# adjacent diffs
	# b + c(2i+1) + d(3i^2 + 3i + 1)
	# 2c + 3d(2i+1)  + 3d
	# 6d
	# should be at most cubic iff adjacent diffs all 0 after
	# 4 iterations
	if n <= 4:
		print 'YES'
	else:
		for k in range(4):
			for i in range(len(inp)-1, 1+k, -1):
				inp[i] -= inp[i-1]
		yes = True
		for i in range(5, len(inp)):
			if inp[i] != 0:
				yes = False
				break
		print 'YES' if yes else 'NO'
