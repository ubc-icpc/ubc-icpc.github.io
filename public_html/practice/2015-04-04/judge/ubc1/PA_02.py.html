t = int(raw_input())

for c in range(t):
	sx = raw_input()
	tx = sx.rstrip('0')
	length = len(sx) - len(tx)
	lx, rx = (95*int(sx)+99)/100, 105*int(sx)/100

	cand0 = 1000000000000
	if tx[-1] != '5':
		cand0 = (int(tx)-int(tx[-1])+5)*10**length
	
	cand1 = (int(tx)-int(tx[-1]))*10**length
	cand2 = (int(tx)+10-int(tx[-1]))*10**length
	
	if lx <= cand0 <= rx or lx <= cand1 <= rx or lx <= cand2 <= rx:
		print 'absurd'
	else:
		print 'not absurd'
