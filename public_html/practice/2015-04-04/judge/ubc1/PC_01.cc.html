#include <bits/stdtr1c++.h>

using namespace std;

int pass[2][100005], dribble[2][100005], memo[2][100005];
int l[2], s[2], n;

int dp(int i, int j) {
    if (j == n-1) return s[i];
    if (memo[i][j] != -1) return memo[i][j];
    return memo[i][j] = min(dp(i^1, j+1) + pass[i][j], dp(i, j+1) + dribble[i][j]);
}

int main() {
    ios::sync_with_stdio(0);
    int t; cin >> t;
    while (t--) {
        cin >> n >> l[0] >> l[1] >> s[0] >> s[1];
        
        for (int k = 0; k < 2; k++) {
            for (int i = 0; i < n-1; i++) {
                cin >> pass[k][i];
            }
            for (int i = 0; i < n-1; i++) {
                cin >> dribble[k][i];
            }
        }
        
        memset(memo, -1, sizeof memo);
        cout << min(l[0] + dp(0, 0), l[1] + dp(1, 0)) << endl;
    }
	return 0;
}
