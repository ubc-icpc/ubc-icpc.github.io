import java.util.HashSet;
import java.util.Scanner;
import java.util.Set;

/**
 * Created by Peter on 2015-04-05.
 */
public class SoccerBets {
    private static HashSet<String> allTeams;
    private static String[][] games;
    public static void main(String[] args) {
      /*  String [] input = {"ITA URU 2 0", "ITA IRE 1 0", "ITA ARG 3 4", "YUG ARG 2 3", "GER CZE 1 0",
        "ENG GER 3 4", "ITA ENG 2 1", "CAM COL 2 1", "ENG CAM 3 2", "ENG BEL 1 0",
        "GER ARG 1 0","CZE CRC 4 1", "NET GER 1 2", "BRZ ARG 0 1", "SPA YUG 1 2", "ROM IRE 4 5"};*/

        Scanner s = new Scanner(System.in);
        int noOfTests = Integer.parseInt(s.nextLine());
        for(int k = 0; k< noOfTests; k++) {
            allTeams = new HashSet<String>();
            games = new String[16][];
            for (int i = 0; i < 16; i++) {  //Get all teams into the set
                String[] line = /*input[i].split("\\s+");*/s.nextLine().split("\\s+");
                games[i] = line;
                allTeams.add(line[0]);
                allTeams.add(line[1]);
            }

            for (int j = 0; j < 16; j++) {
                int score1 = Integer.parseInt(games[j][2]);
                int score2 = Integer.parseInt(games[j][3]);
                String team1 = games[j][0];
                String team2 = games[j][1];

                if (score1 < score2) {
                    allTeams.remove(team1);
                } else {
                    allTeams.remove(team2);

                }
            }

            String winner = "";
            for (String str : allTeams) {
                if (str.length() > winner.length()) {
                    winner = str;
                }
            }

            System.out.println(winner);
        }
    }

}
