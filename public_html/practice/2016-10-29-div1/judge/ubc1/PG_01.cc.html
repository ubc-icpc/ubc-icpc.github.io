#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;

pii a[100005];
int main() {
	ios::sync_with_stdio(0);
	int n; cin >> n;
	ll r, w, h; //project onto vectors (1,r), (-1,r)
	cin >> r >> w >> h;
	for (int i = 0; i < n; i++) {
		ll x, y; cin >> x >> y;
		x -= w; y += r * w;
		a[i] = pii(r * x + y, -r * x + y);
	}
	
	sort(a, a+n);
	
	multiset<ll> s;
	for (int i = 0; i < n; i++) {
		ll v = a[i].second;
		auto it = s.upper_bound(v);
		if (it != s.end()) s.erase(it);
		s.insert(v);
	}
	cout << s.size() << endl;
	return 0;
}
