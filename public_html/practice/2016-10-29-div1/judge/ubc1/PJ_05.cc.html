#include <bits/stdtr1c++.h>

using namespace std;

typedef double ld;

int Et, Es, Ef;
ld E[10000005];
int main() {
	cin >> Et >> Es >> Ef;
	// weight W s.t. R-W <= E[fail] and W-L <= E[pass]
	// E[curr] = min number of guesses ill need to determine the entire weight range with energy curr
	
	// E[curr] = E[fail] + E[pass] + 1
	
	E[1] = 1;
	for (int i = 2; i <= Et; i++) {
		if (i < min(Es, Ef)) E[i] = E[i-1];
		else {
			E[i] = 1;
			if (i >= Es) E[i] += E[i-Es];
			if (i >= Ef) E[i] += E[i-Ef];
		}
	}
	cout << setprecision(6) << fixed << min(200.0 / E[Et], 225.0 / (E[Et] + 1)) << endl;
	return 0;
}
