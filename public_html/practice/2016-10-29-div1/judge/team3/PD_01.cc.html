#include <bits/stdc++.h>

using namespace std;

int main() {
  int a, b, c, d, e, f; cin >> a >> b >> c >> d >> e >> f;
  vector<int> v1 = {a,b,c};
  vector<int> v2 = {d,e,f};
  sort(v1.begin(), v1.end());
  sort(v2.begin(), v2.end());
  if (v1 == v2 && (v1[0]*v1[0] + v1[1]*v1[1] == v1[2]*v1[2])) {
    cout << "YES" << endl;
  } else {
    cout << "NO" << endl;
  }
  return 0;
}
