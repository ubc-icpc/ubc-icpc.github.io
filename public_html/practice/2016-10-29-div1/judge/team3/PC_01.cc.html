#include <bits/stdc++.h>

using namespace std;

typedef vector<int> vi;
typedef pair<vi, string> pvis;

int main() {
  int N; cin >> N;
  vector<pvis> names;
  string l; getline(cin, l);
  int msize = 0;
  while (N--) {
    getline(cin, l);
    stringstream ss(l);
    string s; ss >> s;
    vi order;
    string name = s.substr(0, s.size()-1);
    while (ss >> s) {
      if (s == "upper") {
        order.push_back(0);
      } else if (s == "middle") {
        order.push_back(1);
      } else if (s == "lower") {
        order.push_back(2);
      }
    }
    reverse(order.begin(), order.end());
    int ordersize = order.size();
    msize = max(msize, ordersize);
    names.push_back({order, name});
  }
  for (pvis & p: names) {
    int toadd = msize - p.first.size();
    while (toadd--) {
      p.first.push_back(1);
    }
  }
  sort(names.begin(), names.end());
  for (pvis & p: names) {
    cout << p.second << endl;
  }
  return 0;
}
