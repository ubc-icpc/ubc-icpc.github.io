#include <bits/stdtr1c++.h>
using namespace std;

const int N = 100005;
int a[N];
int main() {
    int n; scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    sort(a, a+n);
    int ans = a[0] + a[n-1];
    for (int i = 1; i*2 < n; i++) {
        ans = min(ans, a[i] + a[n-1-i]);
    }
    printf("%d\n", ans);
    return 0;
}
