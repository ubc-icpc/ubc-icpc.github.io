#include <bits/stdc++.h>

using namespace std;

int dp[10000001];
int main() {
    int e, es, ef; cin >> e >> es >> ef;
    int nleaf = -1;
    for (int i = 1; i <= e; i++) {
        dp[i] = (i-es > 0 ? dp[i-es] : 1) + (i-ef > 0 ? dp[i-ef] : 1);
        if (225.0 / dp[i] < 1e-7) {
            nleaf = dp[i];
            break;
        }
    }
    if (nleaf == -1) nleaf = dp[e];
    cout << setprecision(6) << fixed << min(abs(225.0/nleaf), abs(200.0/(nleaf-1))) << endl;
}
