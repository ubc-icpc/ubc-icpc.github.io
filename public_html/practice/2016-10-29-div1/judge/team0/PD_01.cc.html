#include <bits/stdtr1c++.h>

using namespace std;
typedef long long ll;

int main() {
    ll a[3], b[3];
    cin >> a[0] >> a[1] >> a[2] >> b[0] >> b[1] >> b[2];
    sort(a, a+3); sort(b, b+3);
    if (a[0] == b[0] && a[1] == b[1] && a[2] == b[2] && a[0]*a[0] + a[1]*a[1] == a[2]*a[2]) {
        cout << "YES" << endl;
    } else {
        cout << "NO" << endl;
    }
    return 0;
}
