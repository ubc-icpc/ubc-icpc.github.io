#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef long double ld;

ll n,r,w,h;
ll x[100002],y[100004];

pair<ll,ll> pi[100004];

int m[100009];

int LIS(){
    //find LIS of y[n] in log(n) time;
    int L = 0;
    for(int i=0;i<n;i++){
        int lo = 1, hi = L;
        while (lo <= hi){
            int mid = (hi+lo+1)/2;
            if (x[m[mid]]<=x[i])
                lo = mid+1;
            else 
                hi = mid-1;
            //cout << lo << " " << y[m[mid]] << endl;
        }
       //cout << lo << " ";
        m[lo] = i;
        if (lo>L) L = lo;
    }
    //cout << endl;
    return L;
}

int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);
    cin >> n >> r >> w >> h;
    for(int i=0;i<n;i++){
        cin >>x[i] >> y[i];
    }
    w*=r;
    for(int i=0;i<n;i++){
        x[i]*=r;
    }
    // 1/sqrt(2) 1/sqrt(2)
    // -1/sqrt(2) 1/sqrt(2)
    for(int i=0;i<n;i++){
        pi[i] = {-x[i]+y[i],x[i]+y[i]};
    }
    sort(pi,pi+n);
    for(int i=0;i<n;i++){
        x[i] = pi[i].second;
        //cout << pi[i].first << " "<< pi[i].second << endl;
        //cout << x[i] << " ";
    }
    //cout << endl;
    cout << LIS() << endl;
}
