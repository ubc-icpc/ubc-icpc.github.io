#include<bits/stdc++.h>

using namespace std;

long long dp[10000008];

const long long INF = 225* 1e9;

int e,s,f;

int main(){
	cin >> e>>s>>f;
	if (s>f) swap(s,f);
	for(int i=1;i<=s;i++){
		dp[i] = 2;
	}
	dp[0] =0;
	for(int i=s+1;i<=e;i++){
		dp[i] += dp[i-s];
		if (dp[i] > INF) continue;
		if (i>f) dp[i]+=dp[i-f];
		else dp[i]+=1;
	}
	//for(int i=0;i<=e;i++) cout << dp[i] <<" ";
	double ans1 = (double)225 / (double)(dp[e]);
	double ans2 = (double)200 / (double)(dp[e]-1);
	//cout << dp[e] << endl;
	cout << setprecision(6)<<fixed << min(ans1, ans2) <<endl;
}
