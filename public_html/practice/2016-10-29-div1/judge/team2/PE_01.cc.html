#include <bits/stdc++.h>

using namespace std;

int main() {
	int n; cin >> n; 
	vector<int> nums;
	for(int i = 0; i < n; i++) {
		int curr; cin >> curr; nums.push_back(curr);
	}

	sort(nums.begin(), nums.end());
	int ans = 10000000;
	for(int i = 0, j = nums.size() - 1; i < j; i++, j--) {
		if(nums[i] + nums[j] < ans) {
			ans = nums[i] + nums[j];
		}
	}

	cout << ans << endl;
}
