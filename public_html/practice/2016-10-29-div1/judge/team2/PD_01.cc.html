#include <bits/stdc++.h>

using namespace std;

int main() {
	vector<int> tri1, tri2;

	for(int i = 0; i < 3; i++){
		int n; cin >> n; tri1.push_back(n);
	}

	for(int i = 0; i < 3; i++){
		int n; cin >> n; tri2.push_back(n);
	}

	sort(tri1.begin(), tri1.end());
	sort(tri2.begin(), tri2.end());

	if(tri1[0] == tri2[0] && tri1[1] == tri2[1] && tri1[2] == tri2[2]) {
		if(tri1[0]*tri1[0] + tri1[1]*tri1[1] == tri1[2]*tri1[2]) {
			cout << "YES" << endl;
		} else {
			cout << "NO" << endl;
		}
	} else {
		cout << "NO" << endl;
	}	
}
