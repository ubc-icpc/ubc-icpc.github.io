n=int(raw_input())
L=[]
for i in range(n):
    s=raw_input().split()
    name=s[0]
    name=name[:len(name)-1]
    val=''
    for i in reversed(range(1,len(s)-1)):
        if s[i]=='lower':
            val+='c'
        if s[i]=='middle':
            val+='b'
        if s[i]=='upper':
            val+='a'
    L.append([val,name])
l=0
for x in L:
    l=max(l,len(x[0]))
for i in range(n):
    if (len(L[i][0])<l):
        L[i][0]+='b'*(l-len(L[i][0]))
L.sort()
for x in L:
    print(x[1])
