#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll A[1000005];
int n;
bool good(ll x){
    multiset<ll> s;
    for (int i=1;i<=n;i++) s.insert(A[i]);
    while (s.size()){
        int m=*s.begin(); s.erase(s.find(m)); 
        auto it=s.lower_bound(x-m); 
        if (it==s.end()) return false; 
        s.erase(it); 
    }
    return true;
}
int main(){
    ios::sync_with_stdio(0);
    ll l=1,r=2e6+10;
    cin>>n; 
    for (int i=1;i<=n;i++) cin>>A[i];
    ll ans=0;
    while (true){
        if (l+1==r){
            if (good(r)) ans=r;
            else ans=l;
            break;
        }
        if (l==r){
            ans=l; break;
        }
        ll m=(l+r)/2;
        if (good(m)) l=m;
        else r=m-1;
    }
    cout<<ans<<endl;
    return 0;
}
