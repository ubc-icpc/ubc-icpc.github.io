#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll A[3],B[3];
int main(){
    ios::sync_with_stdio(0);
    cin>>A[0]>>A[1]>>A[2];
    cin>>B[0]>>B[1]>>B[2];
    sort(A,A+3); sort(B,B+3);
    bool good = false;
    if (((A[0]==B[0])&&(A[1]==B[1]))&&(A[2]==B[2])){
        if ((A[0]*A[0]+A[1]*A[1])==A[2]*A[2]) good=true;
    }
    if (good) cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    return 0;
}
