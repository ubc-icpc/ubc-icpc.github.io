#include <iostream>
#include <algorithm>
#include <string>
#include <sstream>
#include<vector>
#include <utility>

using namespace std;

int main() {
	int n; cin >> n;
	vector<pair<string, string> > ppl;
	while(n--) {
		string nn,c ; cin >> nn;
		getline(cin, c);
		stringstream ss(c);
		string s;
		string v;
		while(ss>>s) {
			if(s=="upper")
				v="0"+v;
			if(s=="middle")
				v="1"+v;
			if(s=="lower")
				v="2"+v;
		}
		while(v.size() < 50) {
			v+="1";
		}
		ppl.push_back(make_pair(v, nn));
	} 
	sort(ppl.begin(), ppl.end());
	for(int i = 0; i < ppl.size(); i++) {
		cout << ppl[i].second.substr(0, ppl[i].second.size()-1) << endl;	
	}
}
