#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

int main() {
  int n; cin >> n;
  vector<ll> v(n);
  for (int i = 0; i < n; i++)
    cin >> v[i];
  sort(v.begin(), v.end());
  ll ans = numeric_limits<ll>::max();
  for (int i = 0; i < n/2; i++) {
    ans = min(ans, v[i] + v[n-1-i]);
  }
  cout << ans << endl;
}
