#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;


int main(void) {
  vector<int> v1, v2;
  for(int i=0; i<3; i++) {
    int x; cin >> x;
    v1.push_back(x);
  }
  for(int i=0; i<3; i++) {
    int x; cin >> x;
    v2.push_back(x);
  }
  sort(v1.begin(), v1.end());
  sort(v2.begin(), v2.end());
  if(v1[0] != v2[0] || v1[1] != v2[1] || v1[2] != v2[2]) {
    cout << "NO" << endl;
    return 0;
  }
  if(v1[0] * v1[0] + v1[1] * v1[1] != v1[2] * v1[2]) {
    cout << "NO" << endl;
    return 0;
  }
  cout << "YES" << endl;
  return 0;
}
