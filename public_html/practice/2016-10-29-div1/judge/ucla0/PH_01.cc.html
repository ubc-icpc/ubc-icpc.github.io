#define ll long long
#define N 500005
#define INF (ll)(1e18+3)
#define EPS (1e-7)
#define PI (3.14159265358979323846)
#define ld long double
#define MOD (int)(1e9+7)

pair<string, string> p[N];

void solve() {
    ll n, s;
    cin >> n >> s;
    for (ll i = 0; i < n; i++) {
        ll x, y;
        cin >> x >> y;
        string name;
        cin >> name;
        string dist;
        for (ll j = 0; j < 32; j++) {
            if (y*2 > s) {
                y = 2*y - s;
                if (x*2 > s) {
                    dist += '2';
                    x = 2*x - s;
                }
                else {
                    dist += '1';
                    x = 2*x;
                }
  
            }
            else {
                y = 2*y;
                if (x*2 > s) {
                    dist += '3';
                    x = 2*x - s;
                    ll t = s-x;
                    x = s-y;
                    y = t;
                }
                else {
                    dist += '0';
                    x = 2*x;
                    swap(x, y);
                }
            }
        }
        p[i] = {dist, name};
    }
    sort(&p[0], &p[n]);
    for (ll i = 0; i < n; i++) {
        cout << p[i].second << endl;
    }
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    solve();
}
