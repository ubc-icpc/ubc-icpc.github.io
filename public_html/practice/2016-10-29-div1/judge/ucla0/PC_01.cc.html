#define ll long long
#define N 500005
#define INF (ll)(1e18+3)
#define EPS (1e-7)
#define PI (3.14159265358979323846)
#define ld long double
#define MOD (int)(1e9+7)


pair<string, string> a[N];

bool comp(pair<string,string> p1, pair<string, string> p2) {
    if (p1.second == p2.second)
        return p1.first < p2.first;
    return p2.second < p1.second;
}

void solve() {
    ll n;
    cin >> n;
    ll m = 0;
    for (ll i = 0; i < n; i++) {
        string s;
        cin >> s;
        a[i].first = s.substr(0, s.size()-1);
        cin >> s;
        while (s != "class") {
            if (s == "upper")
                a[i].second = '2' + a[i].second;
            else if (s == "lower")
                a[i].second = '0' + a[i].second;
            else
                a[i].second = '1' + a[i].second;
            cin >> s;
        }
        m = max(m, (ll)(a[i].second.size()));
    }
    
    for (ll i = 0; i < n; i++) {
        string pad(m-a[i].second.size(), '1');
        a[i].second += pad;
    }
    
    sort(&a[0], &a[n], comp);
    
    for (ll i = 0; i < n; i++) {
        cout << a[i].first << endl;
    }
    
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    solve();
}
