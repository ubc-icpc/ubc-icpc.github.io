ll fac(ll n) {
    if (n <= 1)
        return 1;
    return fac(n-1)*n;
}

void solve() {
    ll a[3], b[3];
    cin >> a[0] >> a[1] >> a[2];
    cin >> b[0] >> b[1] >> b[2];
    
    sort(&a[0], &a[3]);
    sort(&b[0], &b[3]);
    
    for (ll i = 0; i < 3; i++) {
        if (a[i] != b[i]) {
            cout << "NO" << endl;
            return;
        }
    }
    
    if (a[2]*a[2] != a[0]*a[0]+a[1]*a[1])
        cout << "NO" << endl;
    else
        cout << "YES" << endl;
    
}
