#include<bits/stdtr1c++.h>
using namespace std;

bool me(pair<string, string>a, pair<string, string>b){
	if (a.second == b.second) return a.first < b.first;
	else return a.second < b.second;
}

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);

	int n; cin >> n;
	while (n--){
		long double a, b, c, s, t; cin >> a >> b >> c >> s >> t;
		if (b * b - 4 * a * c < 0){
			cout << "No" << endl;
			continue;
		}
		long double sq = sqrt(b * b - 4 * a * c);
		long double rootp = (-1 * b + sq) / (2 * a), rootn = (-1 * b - sq) / (2 * a);
		if ((rootp >= s && rootp <= t) || (rootn >= s && rootn <= t)) cout << "Yes" << endl;
		else cout << "No" << endl;
	}
	return 0;
}
