#include<bits/stdtr1c++.h>
using namespace std;

bool me(pair<string, string>a, pair<string, string>b){
	if (a.second == b.second) return a.first < b.first;
	else return a.second < b.second;
}

int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);

	int n; cin >> n;
	vector < pair < string, string > > v(n);
	unordered_map < string, unordered_map < string, pair < pair < string, string >, pair < string, string > > > > um;
	for (int i = 0; i < n; ++i){
		cin >> v[i].first >> v[i].second;
	}
	sort(v.begin(), v.end(), me);
	for (int i = 0; i < n; i += 3){
		um[v[i].first][v[i].second].first = make_pair(v[i + 1].first, v[i + 1].second);
		um[v[i].first][v[i].second].second = make_pair(v[i + 2].first, v[i + 2].second);

		um[v[i + 1].first][v[i + 1].second].first = make_pair(v[i].first, v[i].second);
		um[v[i + 1].first][v[i + 1].second].second = make_pair(v[i + 2].first, v[i + 2].second);

		um[v[i + 2].first][v[i + 2].second].first = make_pair(v[i].first, v[i].second);
		um[v[i + 2].first][v[i + 2].second].second = make_pair(v[i + 1].first, v[i + 1].second);
	}
	int q; cin >> q;
	while (q--){
		string f, l; cin >> f >> l;
		cout << um[f][l].first.first << " " << um[f][l].first.second << endl << um[f][l].second.first << " " << um[f][l].second.second << endl;
	}
	return 0;
}
