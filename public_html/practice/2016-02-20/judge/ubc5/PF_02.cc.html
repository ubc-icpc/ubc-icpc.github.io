#include <bits/stdc++.h>

using namespace std;

typedef long double ld;

const ld eps = 1e-8;
bool win(ld x, ld a, ld b) {
	return a-eps <= x && x <= b+eps;
}

int main() {
	int t; cin >> t;
	while (t--) {
		ld a, b, c, s, t;
		cin >> a >> b >> c >> s >> t;
		
		ld d = b*b - 4 * a * c;
		
		if (d < 0) {
			cout << "No" << endl;
		} else {
			ld x0 = (-b + sqrt(d))/(2*a);
			ld x1 = (-b - sqrt(d))/(2*a);
			cerr << x0 << " " << x1 << endl;
			cout << (win(x0, s, t) || win(x1, s, t) ? "Yes" : "No") << endl;
		}
	}
	
	return 0;
}
