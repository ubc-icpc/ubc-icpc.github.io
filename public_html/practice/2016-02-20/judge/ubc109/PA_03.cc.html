//Given two positive integers A and B, where B ≤ A and B divides A, which represent the length of ONE SIDE of a equilateral triangle, 
//how many equilateral triangles of size B you need to completely cover the triangle of size A.



#include <iostream>
using namespace std;

long long calculateareas(long long a, long long b) {
	return (a * a) / (b*b);
}

int main(void) {
	int tc;
	cin >> tc;
	cin.ignore(100, '\n');
	for (int i = 0; i < tc; i++) {
		long long a;
		long long b;
		cin >> a;
		cin.ignore(100, ' ');
		cin >> b;
		cin.ignore(100, '\n');
		long long res = calculateareas(a, b);
		cout << res << endl;
	}

}
