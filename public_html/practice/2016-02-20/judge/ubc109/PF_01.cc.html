// We can combine two matching adjacent numbers to form larger numbers. 
// We can combine a 1 adjacent to a 2 into a single 3. 
// We can combine two adjacent 3s into a single 6, we can combine two adjacent 6s into a single 12, and we can in general combine two identical adjacent numbers higher than 3 into a single number that will be their sum. The 1s and 2s are special in that a 1 can only be matched with a 2, and a 2 can only be matched with a 1.
// The goal of the game is to form the largest integer possible. The game ends when we can no longer combine two adjacent numbers. Consider we e.g. start with the array {12, 24, 6, 1, 2, 3, 12, 3, 3, 6, 1, 2, 2, 1, 24} then the largest number we can form is 48.


#include<iostream>
#include<string>
#include<vector>
#include<sstream>
#include<math.h>

double calculateDistcriminant(double a, double b, double c) {
	double ans = (b*b) - (4 * a * c);
	return ans;


}
void calculateRoots(double a, double b, double ans, double s, double t) {
	double fr = (-b + sqrt(ans)) / (2 * a);
	double sr = (-b - sqrt(ans)) / (2 * a);
	if ((fr >= s) && (fr <= t)) {
		std::cout << "Yes" << std::endl;
		return;
	}
	if ((sr >= s) && (sr <= t)) {
		std::cout << "Yes" << std::endl;
		return;
	}
	std::cout << "No" << std::endl;
	return;
}


int main(void) {
	int tcs;
	std::cin >> tcs;
	std::cin.ignore(100, '\n');
	for (int i = 0; i < tcs; i++) {
		double a, b, c, s, t;
		std::cin >> a;
		std::cin >> b;
		std::cin >> c;
		std::cin >> s;
		std::cin >> t;
		std::cin.ignore(100, '\n');
		double ans = calculateDistcriminant(a,b,c);
		if (ans < 0) {
			std::cout << "No" << std::endl;
			continue;
		}
		calculateRoots(a, b, ans, s, t);

	}


}
