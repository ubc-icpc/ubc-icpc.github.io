#include <iostream>
#include <algorithm>
#include <utility>
#include <string>
#include <vector>

using namespace std;

typedef pair<string, string> pss;

int main()
{
	int n;

	vector<pss> v;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		string first, last;
		cin >> first >> last;
		v.push_back(pss(last, first));
	}
	sort(v.begin(), v.end());
	int n2;
	cin >> n2;
	while (n2--)
	{
		string first, last;
		cin >> first >> last;
		auto it = lower_bound(v.begin(), v.end(), pss(last,first));
		int j = it - v.begin();
		if (v[j].first == last && v[j].second == first)
		{
			if (j % 3 == 0)
			{
				cout << v[j + 1].second << " " << v[j + 1].first << endl;
				cout << v[j + 2].second << " " << v[j + 2].first << endl;
			}
			else if ((j + 2) % 3 == 0)
			{
				cout << v[j - 1].second << " " << v[j - 1].first << endl;
				cout << v[j + 1].second << " " << v[j + 1].first << endl;
			}
			else
			{
				cout << v[j - 2].second << " " << v[j - 2].first << endl;
				cout << v[j - 1].second << " " << v[j - 1].first << endl;
			}
		}
	}
	return 0;
}
