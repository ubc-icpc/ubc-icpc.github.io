#include <iostream>
#include <cmath>

using namespace std;

typedef long long ll;
typedef long double ld;
int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		ld a, b, c, s, t;
		cin >> a >> b >> c >> s >> t;
		ld q = b*b - 4 * a * c;
		if (q < 0)
		{
			cout << "No" << endl;
			continue;
		}
		q = sqrt(q);
		ld r1, r2;
		r1 = (-b + q) / (2 * a);
		r2 = (-b - q) / (2 * a);
		if ((s <= r1 && r1 <= t) || (s <= r2 && r2 <= t))
		{
			cout << "Yes" << endl;
		}
		else
		{
			cout << "No" << endl;
		}
	}
		return 0;
}
