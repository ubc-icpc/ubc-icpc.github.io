#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;
typedef pair<string,string> pss;

#define endl '\n'

bool cmp(pss a, pss b) {
	if (a.second != b.second) {
		return a.second < b.second;
	} else {
		return a.first < b.first;
	}
}

pss team1[100005];
pss team2[100005];

int main() {
	ios::sync_with_stdio(0);
	int n; cin >> n;
	string f, l;
	vector<pss> names;
	while(n--) {
		cin >> f >> l;
		names.push_back({f,l});
	}
	sort(names.begin(),names.end(),cmp);
	map<pss,int> m1;
	for (int i=0; i<names.size(); i++) {
		m1[names[i]] = i;
	}

	for (int i=0; i<names.size(); i++) {
		bool first = true;
		for (int j=i-(i%3); j<=i-(i%3)+2; j++) {
			if (j == i) continue;
			if (first) {
				team1[m1[names[i]]] = names[j];
				first = false;
			} else {
				team2[m1[names[i]]] = names[j];
			}
		}
	}
	int q; cin >> q;
	while(q--) {
		cin >> f >> l;
		pss fl = {f,l};
		cout << team1[m1[fl]].first << " " << team1[m1[fl]].second << endl;
		cout << team2[m1[fl]].first << " " << team2[m1[fl]].second << endl;
	}
	return 0;
}
