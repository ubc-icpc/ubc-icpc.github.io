#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

int main() {
	ios::sync_with_stdio(0);
	int n; cin >> n;
	while (n--) {
		ld a,b,c,s,t;
		cin >> a >> b >> c >> s >> t;
		ld det = (b*b)-(4*a*c);
		if (det < 0) cout << "No" << endl; 
		else {
			ld x1 = (b*(-1.0)+sqrt(det))/(2.0*a);
			ld x2 = (b*(-1.0)-sqrt(det))/(2.0*a);
			if (s <= x1 && x1 <= t) cout << "Yes" << endl;
			else if (s <= x2 && x2 <= t) cout << "Yes" << endl;
			else cout << "No" << endl;
		}
	}
	return 0;
}
