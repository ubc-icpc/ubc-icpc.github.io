#include<bits/stdtr1c++.h>
using namespace std;

int N, M;
vector<pair<string, string> > A;
pair<string, string> B;

int main () {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> N;
    A.resize(N);
    for (int i = 0; i < N; i++)
        cin >> A[i].second >> A[i].first;
    
    sort(A.begin(), A.end());
    
    cin >> M;
    while (M--) {
        cin >> B.second >> B.first;
        vector<pair<string, string> >::iterator at = lower_bound(A.begin(), A.end(), B);
        int i = at - A.begin();
        int k = i / 3;
        k *= 3;
        for (int j = 0; j < 3; ++j) {
            if (j + k != i) {
                cout << A[j + k].second << " " << A[j + k].first << endl;
            }
        }
    }
    
    return 0;
}
