#include<bits/stdtr1c++.h>
using namespace std;

long long T, A, B, C, D, E;

int main () {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> T;
    while (T--) {
        cin >> A >> B >> C >> D >> E;
        long long d = B * B - 4LL * A * C;
        bool yes = false;
        if (d >= 0) {
            long double xa = (-1.0l * B + sqrt(1.0l * d)) / (2.0l * A);
            long double xb = (-1.0l * B - sqrt(1.0l * d)) / (2.0l * A);
            
            long double ll = 1.0l * D;
            long double rl = 1.0l * E;
            
            if (ll <= xa && xa <= rl) yes = true;
            if (ll <= xb && xb <= rl) yes = true;
        }
        
        cout << (yes ? "Yes" : "No") << endl;
    }
    
    return 0;
}
