#include<bits/stdtr1c++.h>
using namespace std;
typedef long long ll;
typedef pair<string, string> pss;

int n, q;
vector<pss> v;

bool cmp(const pss &a, const pss &b) {
    if (a.second == b.second)
        return a.first < b.first;
    return a.second < b.second;
}

int main() {
    ios_base::sync_with_stdio(0);
    cin >> n;
    for (int i=0; i<n; i++) {
        string f, l; cin >> f >> l;
        v.push_back({f,l});
    }
    sort(v.begin(), v.end(), cmp);
    cin >> q;
    while (q--) {
        string f, l;
        cin >> f >> l;
        auto it = lower_bound(v.begin(), v.end(), make_pair(f,l), cmp);
        int pos = it - v.begin();
        if (pos%3==0) {
            cout << v[pos+1].first << " " << v[pos+1].second << '\n';
            cout << v[pos+2].first << " " << v[pos+2].second << '\n';
        }
        else if (pos%3==1) {
            cout << v[pos-1].first << " " << v[pos-1].second << '\n';
            cout << v[pos+1].first << " " << v[pos+1].second << '\n';
        }
        else {
            cout << v[pos-2].first << " " << v[pos-2].second << '\n';
            cout << v[pos-1].first << " " << v[pos-1].second << '\n';
        }
    }
}
