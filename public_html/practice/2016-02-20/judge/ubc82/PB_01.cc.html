string fr, la;
vector<pair<string,string>> par;
int N, Q;

void prtr(pair<string,string> name, int be, int en){
	int mid = (be+en)/2;
	if (mid == be) {
		int tm = (be/3)*3;
		forall(i,tm,tm+3){
			if (i!=be)
				cout << par[i].second << " " << par[i].first<< endl;
		}
		return;
	}
	else if (par[mid] > name) prtr(name, be, mid);
	else prtr(name,mid, en);
}

int main(){
	s(N);
	forall(i,0,N){
		cin >> fr >> la;
		par.pb(mp(la,fr));
	}
	sort(par.begin(),par.end());
	//forall(i,0,N)	cout << par[i].first << " " << par[i].second << endl;
	s(Q);
	forall(i,0,Q){
		cin >> fr >> la;
		prtr(mp(la,fr),0,N);
	}
}
