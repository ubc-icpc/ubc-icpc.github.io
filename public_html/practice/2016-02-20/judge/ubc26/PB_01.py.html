import itertools as it
lookup = {}

n = int(raw_input())
names = [ tuple(raw_input().split(' ')) for i in xrange(n) ]
names.sort(key=lambda (a,b):(b,a))
x = iter(names)

for j in xrange(n / 3):
    a, b, c = next(x), next(x), next(x)
    lookup[a] = b, c
    lookup[b] = a, c
    lookup[c] = a, b

s = int(raw_input())
for j in xrange(s):
    first, last = raw_input().split(' ')
    for p, q in lookup[(first, last)]:
        print "{} {}".format(p, q)
