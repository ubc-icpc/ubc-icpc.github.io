from math import sqrt
e = 1e-8

for i in xrange(int(raw_input())):
    a, b, c, s, t = map(float, raw_input().split())
    delta = b*b - 4.0*a*c

    try:
        sd = sqrt(delta)
        xm = (-b - sd) / (2.0*a)
        xp = (-b + sd) / (2.0*a)

        if (s - e <= xm <= t + e) or (s - e <= xp <= t + e):
            print "Yes"
        else:
            print "No"

    except ValueError:
        print "No"
