#include <bits/stdc++.h>

using namespace std;

int main() {
  double a, b, c, d, m, t;
  while (cin >> a >> b >> c >> d >> m >> t) {
    double high = (t/m - d) / c;
    double low = 0;
    while (high - low > 10e-7) {
      double v = (high + low)/2.0;
      double x = a*v*v*v + b*v*v + c*v + d - t/m;
      if (x < 0) {
        low = v;
      } else {
        high = v;
      }
    }
    low -= 0.005;
    cout << fixed << setprecision(2) << low << endl;
  }
}
