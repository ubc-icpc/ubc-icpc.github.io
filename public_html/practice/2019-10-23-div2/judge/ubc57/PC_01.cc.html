#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define INF = 2000000000
#define sz(a) int((a).size())
#define pb push_back
#define mp make_pair
#define all(c) (c).begin(),(c).end()
#define X first
#define Y second

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  int T, p, w, q;
  double x, y, z;
  string str, str1, str2;
  map<int, string> planet;
  map<string, int> key;
  map<string, pair<double, pair<double, double> > > coords;
  cin >> T;
  for (int a = 0; a < T; a++) {
    cout << "Case " << a+1 << ":" << "\n";
    coords.clear();
    planet.clear();
    key.clear();
    cin >> p;
    for (int i = 0; i < p; i++) {
      cin >> str >> x >> y >> z;
      planet.insert(mp(i, str));
      key.insert(mp(str, i));
      coords.insert(mp(str, mp(x, mp(y, z))));
    }
    double distance[p][p];
    for (int i = 0; i < p; i++) {
      distance[i][i] = 0;
      for (int j = i+1; j < p; j++) {
	distance[i][j] = sqrt(pow(coords[planet[i]].X-coords[planet[j]].X, 2)+pow(coords[planet[i]].Y.X-coords[planet[j]].Y.X, 2)+pow(coords[planet[i]].Y.Y-coords[planet[j]].Y.Y, 2));
	distance[j][i] = distance[i][j];
      }
    }
    cin >> w;
    for (int i = 0; i < w; i++) {
      cin >> str1 >> str2;
      distance[key[str1]][key[str2]] = 0;
    }
    for (int k = 0; k < p; k++) {
      for (int i = 0; i < p; i++) {
	for (int j = 0; j < p; j++) {
	  distance[i][j] = min(distance[i][j], distance[i][k] + distance[k][j]);
	}
      }
    }
    cin >> q;
    for (int i = 0; i < q; i++) {
      cin >> str1 >> str2;
      cout << "The distance from " << str1 << " to " << str2 << " is ";
      cout << fixed << setprecision(0) << distance[key[str1]][key[str2]] << " parsecs." << "\n";
    }
  }
}
