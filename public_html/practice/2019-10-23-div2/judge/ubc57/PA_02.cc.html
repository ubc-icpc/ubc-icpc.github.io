bool ok(double a, double b, double c, double d, double m, double t, double v) {
  if (a*v*v*v+b*v*v+c*v+d <= t/m) return true;
  else return false;
}

  cin.tie(NULL);
  double a, b, c, d, m, t;
  while (cin >> a) {
    cin >> b >> c >> d >> m >> t;
    double ans = 0;
    for (double x = pow(2.0, 30.0)*0.01; x >= 0.01; x /= 2) {
      while (ok(a, b, c, d, m, t, ans+x)) ans += x;
    }
