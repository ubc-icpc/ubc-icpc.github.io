import sys
import math
import functools
from collections import deque
sin = sys.stdin


def trunc(f):
    st = str(f)
    stt = st.split(".")
    if len(stt[1]) < 2:
        for i in range((2 - len(stt[1]))):
            stt[1] += str(0)
    return stt[0] + "." + stt[1][0:2]


abcdmt = sin.readline().split()
while abcdmt != []:
    a = float(abcdmt[0])
    b = float(abcdmt[1])
    c = float(abcdmt[2])
    d = float(abcdmt[3])
    m = float(abcdmt[4])
    t = float(abcdmt[5])
    low = 0.0
    high = 100000000000.0
    eps = .000001
    while True:
        mid = (low + high)/2
        fuel = (a*(mid**3)+b*(mid**2)+c*mid+d)*m
        if abs(t-fuel) < eps:
            print(trunc(mid))
            break
        elif fuel > t:
            high = mid
        elif fuel < t:
            low = mid
    abcdmt = sin.readline().split()
