#include <iostream>
#include <bitset>
#include <cstring>

using namespace std;
typedef long long ll;

int main() {
	int n, m;
	cin >> n >> m;

	ll lol[n][32];
	memset(lol, 0, sizeof lol);

	int k; cin >> k;
	for (int i = 0; i < k; i++) {
		int a, b; cin >> a >> b; a--; b--;
		lol[a][b >> 6] |= (1LL << (b & 63));
	}

	for (int i = 0; i < n; i++) {
		for (int j = i+1; j < n; j++) {
			int cnt = 0;
			for (int s = 0; s < 32; s++) {
				ll ccc = (lol[i][s] & lol[j][s]);
				if (ccc != 0) {
					cnt++;
					if (ccc != (ccc & (-ccc))) {
						cnt++;
					}
					if (cnt >= 2) {
						cout << "NO" << endl;
						return 0;
					}
				}
			}
		}
	}
	cout << "YES" << endl;
}
