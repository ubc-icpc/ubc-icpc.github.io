#include <iostream>
#include <set>

using namespace std;

int arr[100005];
int main() {
	int n; cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> arr[i];
	}

	multiset<int> available;
	available.insert(arr[0]);
	available.insert(arr[0]);
	for (int i = 1; i < n; i++) {
		multiset<int>::iterator it = available.lower_bound(arr[i]);
		if (it == available.begin()) {
			cout << "NO" << endl;
			return 0;
		}
		--it;
		available.erase(it);
		available.insert(arr[i]);
		available.insert(arr[i]);
	}
	cout << "YES" << endl;
}
