#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<ll, ld> pp;

int main() {
	int n; cin >> n;
	vector<pp> ranks;
	for (int i = 0; i < n; i++) {
		ll a; ld b;
		cin >> a >> b;
		ranks.push_back(pp(a, b));
	}
	sort(ranks.begin(), ranks.end());
	reverse(ranks.begin(), ranks.end());
	ll score; cin >> score;
	ll next = 1;
	for (int i = 0; i < ranks.size(); i++) {
		ll after = next + 2 * (ranks[i].second - next);
		if (ranks[i].first == score) {
			cout << next << endl;
			cout << after << endl;
			return 0;
		}
		next = after+1;
	}
}
