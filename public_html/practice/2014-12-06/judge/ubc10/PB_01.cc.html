#include <iostream>
using namespace std;

int main() {
	int k; cin >> k;
	char blah[k];
	string ssss[k];
	for (int i = 0; i < k; i++) {
		cin >> blah[i] >> ssss[i];
	}
	string lol; cin >> lol;
	int n = lol.length();
	int ind = 0;
	while (ind < n) {
		for (int i = 0; i < k; i++) {
			if (ind + ssss[i].length() > n) continue;
			if (lol.substr(ind, ssss[i].length()) == ssss[i]) {
				cout << blah[i];
				ind += ssss[i].length();
				break;
			}
		}
	}
	cout << endl;
}
