#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
typedef long long ll;
struct C {
	string name;
	ll score;
};
struct cmp {
	bool operator() (const C &c1, const C &c2) const {
		return (c1.score > c2.score || (c1.score == c2.score &&
				c1.name < c2.name));
	}
};

int main() {
	int n; cin >> n;
	vector<C> computers;
	for (int i = 0; i < n; i++) {
		C c; ll r, s, d;
		cin >> c.name >> r >> s >> d;
		c.score = 2*r + 3*s + d;
		computers.push_back(c);
	}
	sort(computers.begin(), computers.end(), cmp());
	if (n >= 1) {
		cout << computers[0].name << endl;
		if (n >= 2) {
			cout << computers[1].name << endl;
		}
	}
}
