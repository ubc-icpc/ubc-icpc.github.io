import java.util.*;

public class S1 {

	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);

		int n = input.nextInt();
		String name[] = new String[n];
		long score[] = new long[n];

		input.nextLine();

		for (int i = 0; i < n; i++) {
			String x = input.nextLine();
			name[i] = x.split(" ")[0];
			score[i] = (2 * Integer.parseInt(x.split(" ")[1]))
					+ (3 * Integer.parseInt(x.split(" ")[2]))
					+ Integer.parseInt(x.split(" ")[3]);
		}
		// Find Highest
		long max = -1;
		int index = 0;
		for (int i = 0; i < n; i++) {
			if (score[i] > max) {
				max = score[i];
				index = i;
			}
		}
		// Find second highest
		long max2 = -1;
		int index2 = -1;
		if (n > 1) {
			for (int i = 0; i < n; i++) {
				if (score[i] > max2 && i != index) {
					max2 = score[i];
					index2 = i;
				}
			}
		}
		// Find lex....
		if (n > 1) {
			int smaller = -1;
			if (max == max2) {
				String a = name[index];
				String b = name[index2];
				smaller = a.compareTo(b);
				if (smaller < 0) {
					System.out.println(name[index]);
					System.out.println(name[index2]);
				} else if (smaller > 0) {
					System.out.println(name[index2]);
					System.out.println(name[index]);
				}
			}else{
				System.out.println(name[index]);
				System.out.println(name[index2]);
			}

		}else if (n == 1){
			System.out.println(name[index]);
		}
	}

}
