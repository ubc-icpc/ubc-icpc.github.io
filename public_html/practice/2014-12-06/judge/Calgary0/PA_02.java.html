import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.Scanner;

public class A {

	class Comp implements Comparable<Comp> {
		String name;
		int r, s, d, score;

		Comp(String name, int r, int s, int d) {
			this.name = name;
			this.r = r;
			this.s = s;
			this.d = d;
			score = 2 * r + 3 * s + d;
		}

		@Override
		public int compareTo(Comp o) {
			if (score == o.score)
				return name.compareTo(o.name);
			return o.score - score;
		}
	}

	private void work() {
		Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(
				System.in)));

		int n = sc.nextInt();
		Comp[] all = new Comp[n];
		for (int i = 0; i < n; i++) {
			String name = sc.next();
			int r = sc.nextInt();
			int s = sc.nextInt();
			int d = sc.nextInt();
			all[i] = new Comp(name, r, s, d);
		}
		Arrays.sort(all);
		if (n > 0)
			System.out.println(all[0].name);
		if (n > 1)
			System.out.println(all[1].name);

		sc.close();
	}

	public static void main(String[] args) {
		new A().work();
	}

}
