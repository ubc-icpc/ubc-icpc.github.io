#include <iostream>
#include <vector>
using namespace std;
int main(){
  int lines;
  cin >> lines;
  vector <string> table_l;
  vector <string> table_v;
  string temp, code, answer;
  for(int a = 0; a < lines; a++){
    cin >> temp;
    table_l.push_back(temp);
    cin >> temp;
    table_v.push_back(temp);
  }
  cin >> code;
  int indexcounter = 0;
  for(;indexcounter < code.length(); indexcounter ++){
  indexincrease:
    for(int a = 0; a < table_v.size(); a++){
      if(table_v[a] == code.substr(indexcounter, table_v[a].length())){
	answer += table_l[a];
	indexcounter += table_v[a].length();
	goto indexincrease;
      }
    }
  }
  cout << answer << endl;
  return 0;
}
