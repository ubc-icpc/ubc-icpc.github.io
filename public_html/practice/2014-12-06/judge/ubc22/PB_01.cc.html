#include<iostream>
#include<vector>
#include<map>
using namespace std;

int main() {
  map<string, char> m;
  int N;
  cin >> N;
  while (N--) {
    string str, code;
    cin >> str >> code;
    for (int i = 1; i < code.length(); i++)
      m[code.substr(0,i)] = 0;
    m[code] = str[0];
  }
  string str;
  cin >> str;
  int index = 0;
  for (int i = 1; i <= str.length(); i++) {
    char c = m[str.substr(index, i - index)];
    if (c != 0) {
      cout << c;
      index = i;
    }
  }
  cout << endl;
  return 0;
}
