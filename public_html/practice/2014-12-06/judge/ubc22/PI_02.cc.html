#include<iostream>
#include<vector>
#include<set>
using namespace std;

int main() {
  int N;
  cin >> N;
  bool y = true;
  multiset<int> s;
  s.insert(0);
  while (N--) {
    int temp;
    cin >> temp;
    multiset<int>::iterator it = s.lower_bound(temp);
    if (it == s.begin()) {
      y = false;
      break;
    }
    --it;
    s.erase(it);
    s.insert(temp);
    s.insert(temp);
  }
  if (y) cout << "YES" << endl;
  else cout << "NO" << endl;
  return 0;
}
