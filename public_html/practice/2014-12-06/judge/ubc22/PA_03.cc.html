#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

struct computer {
  string name;
  int value;
  bool operator<(const computer &b) const {
    if (value != b.value) return value > b.value;
    return name < b.name;
  }
};

int main() {
  int N;
  cin >> N;
  vector<computer> list;
  while (N--) {
    computer temp;
    cin >> temp.name;
    int R,S,D;
    cin >> R >> S >> D;
    temp.value = 2 * R + 3 * S + D;
    list.push_back(temp);
  }
  sort(list.begin(), list.end());
  if (list.size() >= 1) cout << list[0].name << endl;
  if (list.size() > 1) cout << list[1].name << endl;
  return 0;
}
