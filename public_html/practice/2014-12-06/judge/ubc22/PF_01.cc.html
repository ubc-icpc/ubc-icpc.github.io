#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

struct scores {
  double score;
  int rank;
  bool operator<(const scores &b) const {
    return score > b.score;
  }
};

int main () {
  int N;
  cin >> N;
  vector<scores> s(N);
  while (N--) {
    double temp;
    cin >> s[N].score >> temp;
    s[N].rank = (int)(temp * 2);
  }
  sort(s.begin(), s.end());
  double sc;
  cin >> sc;
  int c = 1;
  for (int i = 0; i < s.size(); i++) {
    if (sc == s[i].score) {
      cout << c << endl << s[i].rank - c << endl;
      break;
    }
    c = s[i].rank - c + 1;
  }
  return 0;
}
