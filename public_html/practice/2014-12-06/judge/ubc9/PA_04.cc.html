#include<iostream>
#include<vector>
#include<string>

int main()
{
	struct comp
	{
		std::string name;
		int R;
		int S;
		int D;
	};

	comp max1;
	comp max2;
	int maxVal1 = 0;
	int maxVal2 = 0;

	int n;
	int R;
	int S;
	int D;
	std::string name;

	std::cin >> n;
	std::vector<comp> v(n);

	for (int i = 0; i < n; i++)
	{
		std::cin >> name;
		std::cin >> R;
		std::cin >> S;
		std::cin >> D;

		v[i].name = name;
		v[i].D = D;
		v[i].R = R;
		v[i].S = S;
	}

	for (int i = 0; i < n; i++)
	{
		int temp = v[i].D + 2 * v[i].R + 3 * v[i].S;
		if (temp > maxVal1)
		{
			maxVal2 = maxVal1;
			max2 = max1;
			maxVal1 = temp;
			max1 = v[i];
		}
		else if (temp > maxVal2)
		{
			maxVal2 = temp;
			max2 = v[i];
		}
	}

	if (max1.name != max2.name || max1.D != max2.D || max1.R != max2.R || max1.S != max2.S)
	{
		if (maxVal1 != maxVal2)
		{
			std::cout << max1.name << std::endl;
			std::cout << max2.name;
		}
		else
		{
			if (std::string(max1.name) < std::string(max2.name))
			{
				std::cout << max1.name << std::endl;
				std::cout << max2.name;
			}
			else
			{
				std::cout << max2.name << std::endl;
				std::cout << max1.name;
			}
		}

	}
	else
	{
		std::cout << max1.name;
	}
}
