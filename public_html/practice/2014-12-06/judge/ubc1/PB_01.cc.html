#include <bits/stdtr1c++.h>

using namespace std;

map<string, string> m;
int main() {
	int k; cin >> k;
	for (int i = 0; i < k; i++) {
		string s, t; cin >> s >> t;
		m[t] = s;
	}
	
	string code; cin >> code;
	int i = 0;
	while (i < code.size()) {
		int k = 1;
		while (!m.count(code.substr(i, k))) k++;
		cout << m[code.substr(i,k)];
		i += k;
	}
	cout << endl;
	return 0;
}
