#include <bits/stdtr1c++.h>

using namespace std;

typedef pair<long long, double> pid;

int main() {
	int n; cin >> n;
	vector<pid> v(n);
	for (int i = 0; i < n; i++) {
		cin >> v[i].first >> v[i].second;
	}
	sort(v.rbegin(), v.rend());
	long long s; cin >> s;
	
	long long last = 1;
	for (auto p : v) {
		long long nxt = last + 2*(p.second - last) + 1e-6;
		if (p.first == s) {
			cout << last << endl;
			cout << nxt << endl;
			break;
		}
		last = nxt+1;
	}
	return 0;
}
