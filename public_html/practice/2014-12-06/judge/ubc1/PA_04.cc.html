#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, string> pls;

bool comp(const pls& a, const pls& b) {
	if (a.first == b.first) return a.second < b.second;
	return a.first > b.first;
}

int main() {
	int n; cin >> n;
	vector<pls> plsss(n);
	for (int i = 0; i < n; i++) {
		cin >> plsss[i].second;
		int x, y, z; cin >> x >> y >> z;
		plsss[i].first = 2*x + 3*y + z;
	}
	sort(plsss.begin(), plsss.end(), comp);
	
	if (!plsss.empty()) cout << plsss[0].second << endl;
	if (plsss.size() >= 2) cout << plsss[1].second << endl;
	return 0;
}
