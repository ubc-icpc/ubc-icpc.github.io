#include <bits/stdtr1c++.h>

using namespace std;

int main() {
	int n; cin >> n;
	
	bool yes = true;
	multiset<int> q;
	q.insert(0);
	for (int i = 0; i < n; i++) {
		int x; cin >> x;
		auto it = q.lower_bound(x);
		if (it == q.begin()) {
			yes = false;
			break;
		} else {
			q.erase(--it);
			q.insert(x); q.insert(x);
		}
	}
	cout << (yes ? "YES" : "NO") << endl;
	return 0;
}
