#include <bits/stdc++.h>
#include <string.h>

using namespace std;

struct code {
    char c;
    string encoding; 
};


int main() {
    
    int K;
    cin >> K;
    string encoding;
    char c;
    vector<code> codes;
    for(int i = 0; i < K; i++){
        cin >> c >> encoding;
        code co;
        co.encoding  = encoding;
        co.c = c;
        codes.push_back(co);
    }
    
    char binary[1000];
    cin >> binary;
    int index = 0;
    while(index < strlen(binary)){
        for(int i = 0; i < codes.size(); i++){
            if(strncmp(binary + index, codes[i].encoding.c_str(), codes[i].encoding.size()) == 0){
                cout << codes[i].c;
                index += codes[i].encoding.size();
                break;
            }
        }
    }
    cout << endl;
    return 0;

}
