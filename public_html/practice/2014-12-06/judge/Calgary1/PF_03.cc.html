#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
vector<pair<ll, ll > > scores;

int main() {
    int N;
    cin >> N;
    ll s;
    double r;
    for(int i = 0; i < N; i++){
        cin >> s >> r;
        
        scores.push_back(make_pair(-s,ll(2*r)));
    }
    ll S;
    cin >> S;
    sort(scores.begin(), scores.end());
    ll left = 1;
    for(int i = 0; i < N; i++){
        // calculate range for this one
        if(-scores[i].first == S){
            cout << left << endl;
            cout << scores[i].second  - left << endl;
            break;
        }
        left = scores[i].second - left + 1;
    }
    return 0;
}
