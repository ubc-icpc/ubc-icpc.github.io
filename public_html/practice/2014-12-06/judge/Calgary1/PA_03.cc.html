#include <bits/stdc++.h>
#include <string.h>

using namespace std;

struct computer {
    long long R;
    long long S;
    long long D;
    string name;
};

inline bool operator< (const computer& lhs, const computer& rhs){
    long long numl = 2*lhs.R + 3*lhs.S + lhs.D;
    long long numr = 2*rhs.R + 3*rhs.S + rhs.D;
    if(numl == numr){
        return strcmp(lhs.name.c_str(), rhs.name.c_str()) < 0;
    }
    return numl > numr;
}

int main() {
    
    int N;
    cin >> N;
    string name;
    long long R,S,D;
    vector<computer> computers;
    for(int i = 0; i < N; i++){
        cin >> name >> R >> S >> D;
        computer c;
        c.R = R;
        c.S = S;
        c.D = D;
        c.name = name;
        computers.push_back(c);
    }
    
    sort(computers.begin(), computers.end());
    if(N > 0){
        cout << computers[0].name << endl;
    }
    if(N > 1){
        cout << computers[1].name << endl;
    }
    
    return 0;

}
