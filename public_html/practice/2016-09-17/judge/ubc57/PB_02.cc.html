#include <bits/stdc++.h>
using namespace std;
typedef pair<int,int> pii;
typedef long long ll;

int x[1005],y[1005];
int main(){
    ios::sync_with_stdio(0);
    int n; int ca=0;
    while (cin>>n){
        ca++; 
        if (n==0) break;
        for (int i=1;i<=n;i++) cin>>x[i]>>y[i];
        sort(x+1,x+n+1); sort(y+1,y+n+1);
        int minx=1e9+5,solx=0;
        for (int i=1;i<=n;i++){
            int now=0; 
            for (int j=1;j<=n;j++) now+=abs(x[i]-x[j]); 
            if (now<minx){
                minx=now,solx=x[i];
            }
        }
        int miny=1e9+5,soly=0;
        for (int i=1;i<=n;i++){
            int now=0;
            for (int j=1;j<=n;j++) now+=abs(y[i]-y[j]);
            if (now<miny){
                miny=now,soly=y[i];
            }
        }
        cout<<"Case "<<ca<<": "<<'('<<solx<<','<<soly<<") "<<minx+miny<<endl;
    }
    return 0;
}
