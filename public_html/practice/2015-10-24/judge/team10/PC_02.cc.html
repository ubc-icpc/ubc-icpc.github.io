#include <cstdio>
#include <cctype>

using namespace std;

const char name[] = "PER";

int main() {
    int current = 0;
    char letter;
    int counter = 0;
    while ((letter = getchar()) != EOF) {
        if(isspace(letter)) continue;
        if (letter != name[current]) counter++;
        current = (current + 1) % 3;
    }

    printf("%d\n", counter);
}
