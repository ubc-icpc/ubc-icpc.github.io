#include <cstdio>
#include <queue>

using namespace std;

int max(int a, int b) {
    if (a > b) return a;
    else return b;
}

int main() {
    int requests, server_count;
    scanf("%d %d", &requests, &server_count);

    queue<int> queue;
    int max_count = 0;
    for (int i = 0; i < requests; i++) {
        // Add new process arriving at a certain time.
        int new_process;
        scanf("%d", &new_process);
        //printf("NEW: %d\n", new_process);

        queue.push(new_process + 1000);
        while(queue.front() <= new_process) queue.pop();
        //printf("%d: %d -> %d\n", queue.size(), queue.front(), queue.back());

        max_count = max(max_count, queue.size());
    }

    if (max_count % server_count == 0) printf("%d\n", max_count / server_count);
    else printf("%d\n", (max_count + server_count) / server_count);
}
