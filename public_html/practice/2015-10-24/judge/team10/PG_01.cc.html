#include <cstdio>

using namespace std;

int gobx[150000];
int goby[150000];
bool gobspray[150000];

int main() {
    long g;
    scanf("%ld", &g);
    for (int i = 0; i < g; i++) scanf("%d %d", &gobx[i], &goby[i]);
    for (int i = 0; i < g; i++) gobspray[i] = false;

    long m;
    scanf("%ld", &m);
    for (int i = 0; i < m; i++) {
        int x,y,r;
        scanf("%d %d %d", &x, &y, &r);

        for (int j = 0; j < g; j++) {
            if ((x - gobx[j])*(x - gobx[j]) + (y - goby[j])*(y - goby[j]) <= r*r) gobspray[j] = true;
        }
    }

    int count = 0;
    for (int i = 0; i < g; i++) {
        if (gobspray[i] == false) count++;
    }
    printf("%d\n", count);
}
