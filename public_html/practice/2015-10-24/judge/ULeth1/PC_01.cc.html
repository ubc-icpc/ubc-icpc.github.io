#include<iostream>
#include<iostream>
using namespace std;

int main()
{
   string s;
   getline(cin, s);
   int tally=0;
   for(int i=0; i<(int)s.size(); i++)
   {
      if(i%3==0)
      {
	 if(s[i]!='P')
	    tally++;
      }
      else if(i%3==1)
      {
	 if(s[i]!='E')
	    tally++;
      }
      else
      {
	 if(s[i]!='R')
	    tally++;
      }
   }
   cout << tally << endl;


   return 0;
}
