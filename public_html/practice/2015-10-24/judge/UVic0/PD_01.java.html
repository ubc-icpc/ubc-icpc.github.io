import java.io.File;
import java.io.FileNotFoundException;
import java.util.PriorityQueue;
import java.util.Scanner;

public class D 
{
	public static void main(String[]args) throws FileNotFoundException
	{
		Scanner in = new Scanner(System.in);
		//Scanner in = new Scanner(new File("E:\\Development\\CompProgWS\\compprog\\src\\compprog\\input.txt"));
		
		
		Scanner line1 = new Scanner(in.nextLine());
		int n = line1.nextInt();
		int k = line1.nextInt();
		
		PriorityQueue<Integer> pq = new PriorityQueue<Integer>();
		
		int count = 0;
		int maxCount=0;
		while(in.hasNext())
		{
			int line = Integer.parseInt(in.nextLine());
			
			while(!pq.isEmpty())
			{
				int test = pq.peek();
				if(test <= line)
				{
					count--;
					pq.poll();
				}
				else
				{
					break;
				}
			}
			
			pq.add(line+1000);
			count++;
			if(count>maxCount) maxCount=count;
			
		}	
		System.out.println((int)Math.ceil(((double)maxCount)/k));
		in.close();
	}
	
}
