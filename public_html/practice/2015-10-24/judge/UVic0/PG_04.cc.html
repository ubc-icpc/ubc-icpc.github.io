#include <iostream>
#include <algorithm>

#define FIELD_WIDTH 10001

int goblins[FIELD_WIDTH][FIELD_WIDTH];

int main()
{
	int g, m;
	std::cin >> g;
	int x, y;
	for (int i = 0; i < g; ++i)
	{
		std::cin >> x >> y;
		++goblins[x][y];
	}
	std::cin >> m;
	int r;
	int alive = g;
	int xd, yd;
	for (int i = 0; i < m; ++i)
	{
		std::cin >> x >> y >> r;
		for (int sx = std::max(0, x - r); sx <= std::min(x + r, FIELD_WIDTH - 1); ++sx)
		{
			for (int sy = std::max(0, y - r); sy <= std::min(y + r, FIELD_WIDTH - 1); ++sy)
			{
				xd = sx - x;
				yd = sy - y;
				if (goblins[sx][sy] > 0 && xd*xd + yd*yd <= r*r)
				{
					alive -= goblins[sx][sy];
					goblins[sx][sy] = 0;
				}
			}
		}
	}
	std::cout << alive;
	return 0;
}
