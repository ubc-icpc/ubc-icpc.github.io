import java.io.File;
import java.io.FileNotFoundException;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Scanner;

public class B 
{
	public static void main(String [] args) throws FileNotFoundException
	{
		Scanner in = new Scanner(System.in);
		//Scanner in = new Scanner(new File("E:\\Development\\CompProgWS\\compprog\\src\\compprog\\input.txt"));
		
		int n = Integer.parseInt(in.nextLine());
		ArrayList<String> ans = allComb(n);
		
		
		for(String a:ans)
		{
			for(int i=0; i<a.length(); ++i)
			{
				if(i!=0)System.out.print(" ");
				System.out.print(a.charAt(i));
			}
			System.out.println();
		}

	}
	public static ArrayList<String> allComb(int n)
	{
		ArrayList<String> ret = new ArrayList<String>();
		
		if(n == 1)
		{
			ret.add("1");
			return ret;
		}
		if(n==2)
		{
			ret.add("12");
			ret.add("21");
			return ret;
		}
		if(n==3)
		{
			ret.add("123");
			ret.add("213");
			ret.add("231");
			ret.add("321");
			ret.add("312");
			ret.add("132");
			return ret;
		}
		
		ArrayList<String> Nm1 = allComb(n-1);
		
		Boolean rev = false;
		while(!Nm1.isEmpty())
		{
			String current = Nm1.remove(0);
			
			if(!rev)
			{
				for(int i = current.length(); i>=0; --i)
				{
					if(i==0)
					{
						ret.add(n+current);
					}
					else if(i==current.length())
					{
						ret.add(current+n);
					}
					else
					{
						ret.add(current.substring(0,i)+ n + current.substring(i));
					}
				}
			}
			else
			{
				for(int i = 0; i<=current.length(); ++i)
				{
					if(i==0)
					{
						ret.add(n+current);
					}
					else if(i==current.length())
					{
						ret.add(current+n);
					}
					else
					{
						ret.add(current.substring(0,i)+ n + current.substring(i));
					}
				}
			}
			rev = !rev;
		}
		return ret;
	}
}
