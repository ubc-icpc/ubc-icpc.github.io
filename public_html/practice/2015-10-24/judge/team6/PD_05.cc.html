#include <iostream>

using namespace std;

int main(){
	int n, k;
	cin >> n >> k;
	int t[100000];
	for (int i = 0; i < n; i++){
		cin >> t[i];
	}
	int count = 0;

	int max = 0;
	for (int i = 0; i < n; i++){
		if ((max + i)<n&&t[max + i] >= t[i] + 1000)
			continue;
		else{
			while ((max + i + count + 1) < n&&t[max + i + count + 1] < t[i] + 1000){
				count++;
			}
			max += count;
			count = 0;
		}
	}
	cout << 1+ (max)/ k << endl;
	return 0;
}
