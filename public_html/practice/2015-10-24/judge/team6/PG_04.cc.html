#include <iostream>
#include <complex>
#include <vector>
#include <utility>

using namespace std;

typedef  pair < int, int > pii ;
typedef complex<double> cd;

int main()
{
	double lengths[20005];
	vector<pii> goblins;
	vector<pii>sprinklers;
	int g;
	cin >> g;
	for (int i = 0; i < g; i++)
	{
		int x, y;
		cin >> x >> y;
		goblins.push_back(pii(x, y));
	}
	int s;
	cin >> s;
	for (int i = 0; i < s; i++)
	{
		int x, y;
		cin >> x >> y;
		sprinklers.push_back(pii(x, y));
		cin >> lengths[i];
	}
	int count = 0;
	for (int i = 0; i < g; i++)
	{
		bool alive = true;
		for (int j = 0; j < s; j++)
		{
			int dx = sprinklers[j].first - goblins[i].first;
			int dy = sprinklers[j].second - goblins[i].second;
			if (dx*dx + dy*dy <= lengths[j]*lengths[j])
			{
				alive = false;
				break;
			}
		}
		if (alive)
			count++;
	}
	cout << count << endl;
	return 0;
}
