#include <iostream>
#include <string>
using namespace std;
int main()
{
	string s;
	cin >> s;
	int count = 0;
	for (int i = 0; i < s.length(); i++)
	{
		int remainder = i % 3;
		if (remainder == 0 && s[i] != 'P')
			count++;
		if (remainder == 1 && s[i] != 'E')
			count++;
		if (remainder == 2 && s[i] != 'R')
			count++;
	}
	cout << count << endl;
	return 0;
}
