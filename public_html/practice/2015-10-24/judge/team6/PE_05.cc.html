#include <iostream>
#include <utility>
#include <queue>
#include <vector>
#include <functional>
#include <algorithm>
#include <set>

using namespace std;

typedef pair<int, int> pii;

int main()
{
	vector<pii> v;
	int n, k;
	int shows = 0;
	cin >> n >> k;
	for (int i = 0; i < n; i++)
	{
		pii temp;
		cin >> temp.first >> temp.second;
		v.push_back(temp);
	}
	multiset<int> q;
	for (int i = 0; i < k; i++)
	{
		q.insert(0);
	}
	sort(v.begin(), v.end());
	for (pii show : v)
	{
		int temp = *q.begin();
		int temp2 = *q.rbegin();
		if (temp <= show.first)
		{
			int num = *q.begin();
			q.erase(q.find(num));
			q.insert(show.second);
			shows++;
		}
		//within show.second > temp
		else if (show.second < temp2)
		{
			q.erase(q.find(*(q.rbegin())));
			q.insert(show.second);
		}
	}
	cout << shows << endl;
}
