#include<iostream>

using namespace std;
const int MAX = 100010;
int X[MAX];

int main(){
   int n, k;
   cin >> n >> k;
   for(int i = 0; i < n; i++){
      cin >> X[i];
   }
   if(n == 1){
      cout << 1 << endl;
      return 0;
   }
   int j = 1;
   int max = 1;
   int curr = 1;
   for(int i = 0; i < n; i++){
      while(j < n and X[i] + 999 >= X[j]){
	 curr++;
	 j++;
      }
      if(curr > max)
	 max = curr;
      curr--;
   }
   cout << (max+k-1)/k << endl;
   return 0;
}
