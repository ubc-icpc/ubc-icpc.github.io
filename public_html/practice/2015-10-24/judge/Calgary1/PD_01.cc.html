#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
typedef long long LL;
typedef pair<int,int> PT;

int main() {
	vector<int> reqs;
	int N, K, ans = 0;
	scanf("%d%d", &N, &K);
	FOR(n,0,N) {
		int t;
		scanf("%d", &t);
		int i = reqs.end() - upper_bound(reqs.begin(), reqs.end(), t-1000);
		if (LL(ans)*K <= i) ans++;
		reqs.push_back(t);
	}
	printf("%d\n", ans);
}
