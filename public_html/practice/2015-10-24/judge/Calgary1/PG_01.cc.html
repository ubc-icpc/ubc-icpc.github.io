#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
typedef long long LL;
typedef pair<int,int> PT;

int main() {
	int G, M, ans=0;
	map<int, multiset<int>> pts;
	scanf("%d", &G);
	FOR(g,0,G) {
		int x, y;
		scanf("%d%d", &x, &y);
		pts[y].insert(x);
	}
	scanf("%d", &M);
	FOR(m,0,M) {
		int X, Y, R;
		scanf("%d%d%d", &X, &Y, &R);
		FOR(y,-R,R+1) {
			int xd = sqrt(R*R - y*y);
			multiset<int>& st = pts[Y+y];
			auto it = st.lower_bound(X-xd);
			while (it != st.end() && *it <= X+xd) {
				st.erase(it++);
				ans++;
			}
		}
	}
	printf("%d\n", G-ans);
}
