#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
typedef long long LL;
typedef pair<int,int> PT;

int main() {
	set<PT> reqs;
	vector<PT> pts;
	int N, K, ans=0;
	scanf("%d%d", &N, &K);
	pts.resize(N);
	FOR(n,0,N) scanf("%d%d", &pts[n].second, &pts[n].first);
	sort(pts.begin(), pts.end());
	FOR(k,0,K) reqs.emplace(0, k);
	for (PT pt : pts) {
		auto it = reqs.upper_bound(PT(pt.second, 1000000000));
		if (it != reqs.begin()) {
			//printf("Added %d -> %d\n", pt.second, pt.first);
			--it;
			int k = it->second;
			reqs.erase(it);
			reqs.emplace(pt.first, k);
			ans++;
		}
	}
	printf("%d\n", ans);
}
