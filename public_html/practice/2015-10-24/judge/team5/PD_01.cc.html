#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main() {
  int n,k; cin >> n >> k;
  vector<int> v;
  for (int i=0; i<n; i++) {
    int x; cin >> x; v.push_back(x);
  }
  int res=-1;
  for (int i=0; i<n; i++) {
    int l=v[i];
    auto ridx=upper_bound(v.begin(), v.end(),v[i]+999);
    int num=ridx-(v.begin()+i)-1;
    res=max(res,1+(num/k));
  }
  cout << res << endl;
}
