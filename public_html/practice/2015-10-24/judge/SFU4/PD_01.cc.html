#include <iostream>
#include <stdio.h>
#include <string>
#include <math.h>

using namespace std;

int times[100010];

int main()
{
	int n, k;
	int t;
	scanf("%d%d", &n, &k);
	for(int i=0; i<n; i++)
	{
		scanf("%d", &t);
		times[i] = t;
	}
	int p, q;
	p = q = 0;
	int cnt = 0;
	int maxOv = 0;
	while(p<n)
	{
		while(p<n && times[p] < times[q]+1000)
		{
			p++;
			cnt++;
		}
		if(cnt>maxOv)
			maxOv = cnt;
		q++;
		cnt--;
	}
	int res = (int)ceil((float)maxOv/k);
	printf("%d\n", res);
	return 0;
}
