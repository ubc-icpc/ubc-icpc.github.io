#include <cstdio>

using namespace std;

int main() {
    int count = 0;
    int answer = 0;
    char c;
    while (1) {
        scanf("%c", &c);
        if (c == '\n') break;
        if (count == 0) {
            if (c != 'P' && c !='p')
                answer++;
        } else if (count == 1) {
            if (c != 'E' && c !='e')
                answer++;
        } else {
            if (c != 'R' && c !='r')
                answer++;
        }
        count++;
        if (count == 3) count = 0;
    }
    printf("%d", answer);
}
