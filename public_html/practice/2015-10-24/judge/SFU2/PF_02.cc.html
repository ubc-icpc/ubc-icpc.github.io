#include <stdio.h>

int main(){
	int f;
	scanf("%d",&f);
	int flag=1;
	while (f--){
		int t,n;
		scanf("%d%d",&t,&n);
		int dp[110][10010];
		for (int i=0;i<=t;i++)
			dp[0][i]=1;
		for (int i=1;i<=n;i++){
			int s1,s2,s;
			scanf("%d%d",&s1,&s2);
			s=s2-s1;
			for (int j=0;j<=t;j++){
				dp[i][j]=0;
				if (j+s<=t)dp[i][j]|=dp[i-1][j+s];
				if (j-s>=0)dp[i][j]|=dp[i-1][j-s];
			}
		}
		int i;
		for (i=0;i<=t;i++)
			if (dp[n][i])break;
		if (i==t+1)flag=0;
	}
	if (flag==0)printf("impossible\n");
	else printf("possible\n");
}
