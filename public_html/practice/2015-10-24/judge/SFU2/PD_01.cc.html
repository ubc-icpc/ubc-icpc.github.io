#include <stdio.h>
#include <deque>
using namespace std;
int main() {
	int n,k;
	scanf("%d %d",&n,&k);
	deque<int> q;
	int maxLen=0;
	for(int i=0;i<n;i++) {
		int t;
		scanf("%d",&t);
		q.push_back(t);
		while(q.size()>0 && q.front()<=(t-1000)) q.pop_front();
		if(q.size()>maxLen) maxLen=q.size();
	}
	printf("%d\n",(maxLen+k-1)/k);
}
