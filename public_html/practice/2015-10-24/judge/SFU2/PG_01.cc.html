#include <stdio.h>
#include <set>
#include <math.h>
using namespace std;
int main() {
	int g,m;
	scanf("%d",&g);
	multiset<int> lines[10001];
	for(int i=0;i<g;i++) {
		int x,y;
		scanf("%d %d",&x,&y);
		lines[x].insert(y);
	}
	int count=0;
	//for(int i=0;i<10001;i++) count+=(int)lines[i].size();
	//printf("%d\n",count);

	scanf("%d",&m);
	for(int i=0;i<m;i++) {
		int X,Y,r;
		scanf("%d %d %d",&X,&Y,&r);
		int x1=X-r,x2=X+r;
		if(x1<0) x1=0;
		if(x2>10000)x2=10000;
		for(int x=x1;x<=x2;x++) {
			int w=X-x;
			int h=(int)floor(sqrt(r*r-w*w));
			int y1=Y-h;
			int y2=Y+h;
			multiset<int>& s=lines[x];
			s.erase(s.lower_bound(y1),s.upper_bound(y2));
			//printf("erase x=%d, y: [%d,%d)\n",x,y1,y2);
		}
	}
	count=0;
	for(int i=0;i<10001;i++) count+=(int)lines[i].size();
	printf("%d\n",count);
}
