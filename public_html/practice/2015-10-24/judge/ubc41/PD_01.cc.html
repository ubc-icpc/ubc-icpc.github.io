#include <iostream>
#include <vector>

using namespace std;

int func(vector<int> val) {
  // basically find longest sequence within 1000 ms
  int first = 0;
  int second = 0;
  int max_dist = 0;
  while(first < val.size()) {
    while(second+1 < val.size() && val[second+1] - val[first] < 1000) {
      second++;
    }
    if(second-first > max_dist)
      max_dist = second - first;

    first++;
  }
  return max_dist;
}

int main() {
  int n;
  int k;
  cin >> n >> k;
  int tmp;
  vector<int> val;
  while(n--) {
    cin >> tmp;
    val.push_back(tmp);
  }
  int most = func(val);
  int srvr = most / k + 1;
  cout << srvr << endl;
  return 0;
}
