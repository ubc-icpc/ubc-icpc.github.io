import java.util.Scanner;

public class MainClass {
	
	private static int changeCipher(String str) {
		
		if (str == null) return -1;
		int days = 0;
		
		char[] strArr = str.toCharArray();
		
		for (int i = 0; i < strArr.length; i++) {
			if (i % 3 == 0 && strArr[i] != 'P') {
				strArr[i] = 'P';
				days++;
				continue;
			}
			if (i % 3 == 1 && strArr[i] != 'E') {
				strArr[i] = 'E';
				days++;
				continue;
			}
			if (i % 3 == 2 && strArr[i] != 'R') {
				strArr[i] = 'R';
				days++;
				continue;
			}
		}
		return days;		
		
	}
	
	public static void main (String[] args) {	
		Scanner sc = new Scanner(System.in);		
		String input = sc.nextLine();
		System.out.println(changeCipher(input));
	}

}
