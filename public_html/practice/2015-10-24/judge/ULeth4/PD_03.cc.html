#include <iostream>
#include <cmath>
using namespace std;

int request[100100];
int compreq[100100];
int main(){
   int n;
   double k;
   cin >> n >> k;

   for(int i = 0; i < n; i++){
      int temp;
      cin >> temp;
      request[i] = temp;
      compreq[i] = 1;
   }
   int maxcomp = 1;
   
   
   for (int i = 0; i < n; i++){
      int j = i+1;
      
      if (!(j >= n))
     	 while(request[i]+999>=request[j]){
	 
	    compreq[i];
	    compreq[j]++;
	    j++;
	    if (j >= n)
	       break;
      }
      if (compreq[i]> maxcomp){
	 maxcomp = compreq[i];
      }

   }

  
   cout << ceil( maxcomp/k) <<  endl;

}
