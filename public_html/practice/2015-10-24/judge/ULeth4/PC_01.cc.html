#include <iostream>
#include <string>

using namespace std;

int main ()
{
   int ans = 0;
   string s;
   string per = "PER";
   
   cin >> s;

   for (int i = 0; i < s.length (); i++)
   {
      if (s[i] == per[i % 3])
	 continue;
      else
	 ans++;
   }

   cout << ans << endl;

   return 0;
}
