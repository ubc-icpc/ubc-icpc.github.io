#include <cstdio>
#include <cstdlib>
#include <deque>
#include <iostream>
#include <algorithm>

using namespace std;
struct pc
{
	int t;
	bool s;
	int ind;
};
deque<pc> p;

bool cmp(pc f, pc s)
{
	//cout<<f.ind<<','<<s.ind<<".  "<<f.t<<' '<<f.s<<':'<<s.t<<' '<<s.s<<endl;
	if (f.t < s.t)
	{
		return true;
	}
	if (f.t == s.t && f.s == s.s) {
		return false;
	}
	if (f.t == s.t && f.s == false)
	{
		return true;
	}


	return false;
}

int main()
{
    
	int n, k;
	int j = 0;
	p.clear();
	scanf("%d %d", &n, &k);
	for (int i = 0; i < n; ++ i)
	{
		int t;
		scanf("%d", &t);
		p.push_back({t, true, j++});
		p.push_back({t+1000, false, j++});
	}
	sort(p.begin(), p.end(), cmp);
	int sv = 0;
	int inp = 0;
	for (auto &i: p)
	{
		if (i.s)
		{
			++ inp;
			if (inp > k*sv)
			{
				++sv;
			}
		}
		else
		{
			-- inp;
		}
	}
	printf("%d\n", sv);
}
