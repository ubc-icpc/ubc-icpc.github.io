#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdio>

using namespace std;
vector<int> a;

int f[20000][200];
int main() {
	int ca;
	scanf("%d", &ca);
	while (ca--) {
		int t;
		int n;
		scanf("%d%d", &t, &n);
		a.clear();
		memset(f, 0, sizeof(f));
		for (int i = 0; i < n; i++) {
			int x, y;
			scanf("%d%d", &x, &y);
			a.push_back(y - x);
			
		}
		bool flag = false;
		for (int i = 1; i <= t; i++) {
			f[i][0] = 1; 
		}
		int cur = 0;
		for (int i = 0; i < n; i++) {
			flag = false;
			for (int j = 0; j <= t; j++) {
				if (f[j][i]) {
					int l = j - a[i];
					int r = j + a[i];
					if (l >= 0) {
						f[l][i+1] = 1;
						flag = true;
					}
					if (r <= t) {
						f[r][i+1] = 1;
						flag = true;
					}
				}
			}
			if (!flag) {
				printf("impossible\n");
				return 0;
			}
		}
	}
	printf("possible\n");
}
