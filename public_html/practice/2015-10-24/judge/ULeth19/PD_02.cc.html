#include <iostream>
#include <deque>
#include <utility>
using namespace std;

int main()
{
  int nReq, pSec;
  cin >> nReq >> pSec;
  int time=0;
  deque<int> waitlist;
  int a;
  cin >> a;
  int max=1;
  waitlist.push_back(a);
  for (int i=1;i<nReq;i++)
    {
      cin >> a;
      while (waitlist.front()+1000<=a&&waitlist.size()!=0)
	waitlist.pop_front();
      waitlist.push_back(a);
      if (waitlist.size()>max)
	max=waitlist.size();
    }
  int num=0;
  if (max%pSec!=0)
    num++;
  num+=max/pSec;
  cout << num << endl;
}
