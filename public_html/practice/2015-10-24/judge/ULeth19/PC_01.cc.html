#include <iostream>

using namespace std;

int main()
{
  string pswd;
  cin >> pswd;
  int days=0;
  for (int i=0;i<pswd.size(); i++)
    {
      if (pswd.at(i)!='P')
	days++;
      i++;
      if (pswd.at(i)!='E')
	days++;
      i++;
      if (pswd.at(i)!='R')
	days++;
    }
  cout << days << endl;
}
