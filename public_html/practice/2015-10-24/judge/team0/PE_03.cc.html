#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;

pii a[100005];
int main() {
    int n, k; cin >> n >> k;
    for (int i = 0; i < n; i++) {
        cin >> a[i].first >> a[i].second;
    }
    sort(a, a+n);
    int ans = 0;
    multiset<int> q;
    for (int i = 0; i < n; i++) {
        q.insert(a[i].second);
        while (*q.begin() <= a[i].first) {
            q.erase(q.begin());
            ans++;
        }
        if (q.size() > k) q.erase(--q.end());
    }
    ans += q.size();
    cout << ans << endl;
	return 0;
}
