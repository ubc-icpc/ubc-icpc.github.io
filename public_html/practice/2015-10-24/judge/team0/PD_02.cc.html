#include <iostream>

using namespace std;

const int sz = 1e5 + 2000;
int a[sz];
int main() {
	int n, k; cin >> n >> k;
	for (int i = 0; i < n; i++) {
		int t; cin >> t;
		a[t] += 1;
		a[t+1000] -= 1;
	}
	
	int ans = 0;
	for (int i = 0; i < sz; i++) {
		if (i > 0) a[i] += a[i-1];
		ans = max(ans, (a[i]+k-1)/k);
	}
	cout << ans << endl;
	return 0;
}
