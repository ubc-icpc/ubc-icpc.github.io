#include <bits/stdc++.h>

using namespace std;

int main() {
	int n; cin >> n;
	vector<vector<int>> v({{1}});
	for (int i = 2; i <= n; i++) {
		int dir = 1;
		vector<vector<int>> w;
		for (const auto& p : v) {
			if (dir == 0) {
			    for (int k = 0; k < p.size(); k++) {
			        w.push_back(vector<int>());
			        for (int j = 0; j < p.size(); j++) {
			            if (k == j) w.back().push_back(i);
			            w.back().push_back(p[j]);
			        }
			    }
			    w.push_back(p);
			    w.back().push_back(i);
			} else if (dir == 1) {
			    w.push_back(p);
			    w.back().push_back(i);
			    for (int k = p.size()-1; k >= 0; k--) {
			        w.push_back(vector<int>());
			        for (int j = 0; j < p.size(); j++) {
			            if (k == j) w.back().push_back(i);
			            w.back().push_back(p[j]);
			        }
			    }
			}
			dir ^= 1; 
		}
		v = w;
	}
	
	cerr << v.size() << endl;
	for (const auto& p : v) {
	    for (int w : p) {
	        cout << w << " ";
	    }
	    cout << '\n';
	}
	return 0;
}
