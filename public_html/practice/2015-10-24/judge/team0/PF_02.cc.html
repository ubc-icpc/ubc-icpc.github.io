#include <iostream>
#include <cstring>
using namespace std;

bool p[10001][101];

int main() {
  int f;
  while (cin >> f) {
    bool pos = true;
    while (f--) {
      int t, n; cin >> t >> n;
      memset(p, 0, sizeof p);
      for (int i = 0; i <= t; i++)
        p[i][0] = true;
      bool fpos = true;
      for (int i = 1; i <= n; i++) {
        bool rpos = false;
        int ti, tj; cin >> ti >> tj;
        int tv = tj-ti;
        for (int j = 0; j <= t; j++) {
          if (!p[j][i-1])
            continue;
          if (0 <= j-tv) {
            p[j-tv][i] = true;
            rpos = true;
          }
          if (j+tv <= t) {
            p[j+tv][i] = true;
            rpos = true;
          }
        }
        if (!rpos)
          fpos = false;
      }
      if (!fpos)
        pos = false;
    }
    if (!pos)
      cout << "im";
    cout << "possible\n";
  }
  return 0;
}
