#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
   int n, k;

   int A[100000];

   cin >> n >> k;
      for(int i=0; i < n; i++)
      {
	 cin >> A[i];
      }

      int diff = 0;
      int last = 0;
      int servercount = 1;
      int elements;
      for(int i=1; i < n; i++)
      {
	 diff += A[i] - A[i-1];
	 while(diff > 999)
	 {
	    diff -= A[last+1] - A[last];
	    last++;
	 }
	 elements = i+1 -last;
	 if(elements > k)
	 {
	    if(elements%k == 0)
	    {
	       if(k==1)
		  servercount = max(elements/k, servercount);
	    }
	    else
		servercount = max(elements/k+1, servercount);
	 }	
      }
      cout << servercount << endl;
      return 0; 
}
