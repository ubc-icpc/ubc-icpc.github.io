#include <iostream>
using namespace std;

#define FOR(i,a,b) for (int i = a; i < b; i++)

int main() {
    const char *p = "PER";
    string s;
    int result = 0;

    cin >> s;

    for (int i = 0; i < s.size(); i++)
        if (s[i] != p[i%3]) 
            result++;

    cout << result << endl;

    return 0;
}
