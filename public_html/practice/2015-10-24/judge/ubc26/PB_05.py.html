def f(n):
    if n == 1:
        yield [1]
    elif n == 2:
        yield [1, 2]
        yield [2, 1]
    else:
        r = [(n-1, -1, -1), (n,)]
        for i, z in enumerate(f(n - 1)):
            for x in xrange(*r[i % 2]):
                yield z[:x] + [n] + z[x:]

for w in f(int(raw_input())):
    print " ".join(map(str, w))
