#include <iostream>
#include <queue>
#include <algorithm>

using namespace std;

int main()
{
  int n, k;
  cin >> n >> k;

  int req = 0;
  queue<int> q;
  for (int i = 0; i < n; i++) {
    int t;
    cin >> t;
    while (!q.empty() && q.front() <= t-1000) {
      q.pop();
    }
    q.push(t);
    req = max(req, (int)q.size());
  }

  cout << (req + k-1)/k << endl;
  return 0;
}
