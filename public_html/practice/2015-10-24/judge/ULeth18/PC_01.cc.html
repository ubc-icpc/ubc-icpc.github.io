#include <iostream>
#include <string>

using namespace std;

int main()
{
  string s;
  cin >> s;

  int count = 0;
  for (int i = 0; i < s.length(); i++) {
    count += (i % 3 == 0 && s[i] != 'P');
    count += (i % 3 == 1 && s[i] != 'E');
    count += (i % 3 == 2 && s[i] != 'R');
  }
  cout << count << endl;
  return 0;
}
