#include <iostream>
#include <array>
#include <vector>
#include <unordered_map>
#include <stdio.h>
#include <utility>
#include <cmath>

typedef long long ll;
using namespace std;

string s;



int main() {
  int g; cin >> g;
  vector<int> xlist[10001];
  vector<bool> gobs_removed[10001];
  for (int i=0; i<g; i++) {
    int x, y; cin >> x >> y;
    xlist[x].push_back(y);
    gobs_removed[x].push_back(false);
  }
  
  int s; cin >> s;
  int removed = 0;
  for (int i=0; i<s; i++) {
    int x, y, rad; cin >> x >> y >> rad;
    
    int high = min(10000, x+rad+1);
    int low = max(0, x-rad-1);
    
    for (int j=low; j<=high; j++) {
      for (int k=0; k<xlist[j].size(); k++) {
        int dist = (j-x)*(j-x) + (xlist[j][k] - y)*(xlist[j][k] - y);
        if (dist <= rad*rad) {
          if(!gobs_removed[j][k]) {
            removed++;
            gobs_removed[j][k] = true;
          }
          
        }
      }
      
    }
    
    
//    for (int j=0; j<gobs.size(); j++) {
//      int dist = (gobs[j].first - x) * (gobs[j].first - x);
//      dist +=(gobs[j].second - y) * (gobs[j].second - y);
//      if (dist <= rad*rad) {
//        if (!gobs_removed[j]) {
//          removed++;
//          gobs_removed[j] = true;
//        }
//      }
//    }
  }
  
  cout << g - removed;
  
}
