#include<bits/stdtr1c++.h>
using namespace std;

int garden[10240][10240];

inline bool within(int x, int y, int r) {
    return x*x + y*y <= r*r;
}

int main () {
    int g;
    cin >> g;
    
    memset(garden, 0, sizeof(garden));
    
    for (int i = 0; i < g; ++i) {
        int x, y;
        cin >> x >> y;
        ++garden[y][x];
    }
    
    int m;
    cin >> m;
    
    while (m--) {
        int x, y, r;
        cin >> x >> y >> r;
        
        for (int i = max(0, y - r); i <= min(10000, y + r); ++i) {
            for (int j = max(0, x - r); j <= min(10000, x + r); ++j) {
                if (within(i-y, j-x, r)) {
                    g -= garden[i][j];
                    garden[i][j] = 0;
                }
            }   
        }
    }
    
    cout << g << endl;
    
    return 0;
}
