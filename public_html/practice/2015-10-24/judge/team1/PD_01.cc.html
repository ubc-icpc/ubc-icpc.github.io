#include<bits/stdtr1c++.h>
using namespace std;

#define vi vector<int>
#define pqi priority_queue<int, vi, greater<int> >

int main () {
    int n, k, t;
    cin >> n >> k;
    
    pqi pq;

    pq.push(0);
    
    while (n--) {
        cin >> t;
        int curr = pq.top();
        if (curr <= t) {
            pq.pop();
        }
        pq.push(t + 1000);
    }
    
    int threads = pq.size();
    
    cout << (int)((threads + k - 1) / k) << endl;
    return 0;
}
