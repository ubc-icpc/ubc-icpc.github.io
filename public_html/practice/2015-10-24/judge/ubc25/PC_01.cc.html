#include <iostream>
#include <string>

using namespace std;

int main() {
    string input; cin >> input;
    char current = 'P';
    int count = 0;
    for (int i = 0; i < input.size(); i++) {
        if (input[i] != current) {
            count++;
        }
        if (current == 'P') {
            current = 'E';
        } else if (current == 'E') {
            current = 'R';
        } else if (current == 'R') {
            current = 'P';
        }
    }
    cout << count << endl;
}
