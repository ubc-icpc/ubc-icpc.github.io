def permutations(length):
    if length == 1:
        return [["1"]]
    perms = permutations(length-1)
    answer = []
    for i, perm in enumerate(perms):
        if i % 2 == 0:
            for j in range(length - 1, -1, -1):
                answer.append(perm[:j] + [str(length)] + perm[j:])
        else:
            for j in range(length):
                answer.append(perm[:j] + [str(length)] + perm[j:])
    return answer

for perm in permutations(int(input())):
    print(" ".join(perm))
