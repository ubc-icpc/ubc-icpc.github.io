#include <iostream>
#include <string>
 
int main()
{
  std::string str;
  std::getline (std::cin, str);
  std::cin >> str;
  std::string str_pers = "PER";
  int n_dif = 0;
  for (int n=0; n<str.length(); n++)
    {
    if ( str[n] == str_pers[n%3] )
      continue;
    else
      n_dif++;
    }
  std::cout << n_dif << std::endl;
  return 0;
}
