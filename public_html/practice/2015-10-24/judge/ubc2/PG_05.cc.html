#include <bits/stdc++.h>

using namespace std;

typedef pair<int,int> pii;

struct sprinkler { int x; int y; int r; };

int goblins[10001][10001];

int main() {
  memset(goblins, 0, sizeof(goblins));
  int G; cin >> G;
  vector<sprinkler> ss;
  for (int g = 0; g < G; ++g) {
    int x, y; cin >> x >> y;
    goblins[x][y]++;
  }
  int count = G;
  int M; cin >> M;
  for (int m = 0; m < M; ++m) {
    int x, y, r; cin >> x >> y >> r;
    for (int i = x-r; i <= x+r; ++i) {
      if (i < 0 || i > 10000) continue;
      for (int j = y-r; j <= y+r; ++j) {
        if (j < 0 || j > 10000) continue;
        int dx = i-x, dy = j-y;
        if (dx*dx + dy*dy <= r*r && goblins[i][j] > 0) {
          count -= goblins[i][j];
          goblins[i][j] = 0;
        }
      }
    }
  }
  cout << count << endl;
  return 0;
}
