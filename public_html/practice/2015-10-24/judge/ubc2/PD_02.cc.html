#include <bits/stdc++.h>

using namespace std;

int main() {
  int N, K; cin >> N >> K;
  queue<int> q;
  int servers = 1;
  for (int i = 0; i < N; ++i) {
    int t; cin >> t;
    if (!q.empty() && q.front() + 1000 <= t) q.pop();
    q.push(t);
    if (q.size() > servers*K) servers++;
  }
  cout << servers << endl;
  return 0;
}
