#include <bits/stdc++.h>
using namespace std;

int main(){
	string l; cin >> l;
	int ans = 0;
	for (int i = 0; i < l.size(); i++){
		if (i % 3 == 0){
			if (l[i] != 'p'&&l[i] != 'P')ans++;
		}else if (i % 3 == 1){
			if (l[i] != 'e'&&l[i] != 'E')ans++;
		}
		else{
			if (l[i] != 'r'&&l[i] != 'R')ans++;
		}
	}
	cout << ans << '\n';
}
