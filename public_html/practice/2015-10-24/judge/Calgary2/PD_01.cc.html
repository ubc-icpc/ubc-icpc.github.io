#include <bits/stdc++.h>
using namespace std;

int main(){
	int n, k; cin >> n >> k;
	vector<int> plus(200000, 0),minus(200000,0);
	int x;
	while (n--){
		cin >> x; plus[x]++; minus[x + 1000]++;
	}
	int m = 0,current=0;
	for (int i = 0; i < 150000; i++){
		current += plus[i] - minus[i];
		m = max(m, current);
	}
	cout << ((m + k - 1) / k);
}
