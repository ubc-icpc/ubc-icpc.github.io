import java.io.*;
import java.util.*;

public class B {
	
	public static void main(String args[]) throws IOException {
		
		BufferedReader kinput = new BufferedReader(new InputStreamReader(System.in));
		
		int num = Integer.parseInt(kinput.readLine());
		
		ArrayList<String> people = new ArrayList<String>();
		
		for(int a = 0; a < num; a++) {
			people.add(kinput.readLine());
		}
		
		Collections.sort(people);
		
		ArrayList<String> finish = new ArrayList<String>();
		
		for(int a = 0; a < num - 1; a++) {
			finish.add(kinput.readLine());
		}
		
		Collections.sort(finish);
		
		String not = "";
		
		for(int a = 0; a < finish.size(); a++) {
			
			if(!(people.get(a).equals(finish.get(a)))) {
				not = people.get(a);
				break;
			}
			
			if(a == finish.size()-1) {
				not = people.get(finish.size());
			}
			
		}
		
		System.out.println(not);
		
	}
	
}
