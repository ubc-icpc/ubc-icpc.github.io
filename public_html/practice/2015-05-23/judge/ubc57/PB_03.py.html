n=int(raw_input())
Dic={}
L=[True]*n
People=[]
for i in range(0,n):
    a=raw_input()
    if a not in Dic:
        Dic[a]=[]
    Dic[a].append(i)
    People.append(a)
for i in range(0,n-1):
    a=raw_input()
    L[Dic[a][0]]=False
    Dic[a].pop(0)
for i in range(0,n):
    if L[i]:
        print(People[i])
        break
