L=list(map(int,raw_input().split()))
Dic={}
for i in range(1,10):
    a=L[i-1]
    Dic[a]=i #to get the result of a, press i
word=raw_input()
M={'a': '2', 'c': '222', 'b': '22', 'e': '33', 'd': '3', 'g': '4', 'f': '333', 'i': '444', 'h': '44', 'k': '55', 'j': '5', 'm': '6', 'l': '555', 'o': '666', 'n': '66', 'q': '77', 'p': '7', 's': '7777', 'r': '777', 'u': '88', 't': '8', 'w': '9', 'v': '888', 'y': '999', 'x': '99', 'z': '9999'}
answer=""
final=""
for i in range(0,len(word)):
   a=word[i]
   answer=answer+M[a]
   if i<len(word)-1:
       b=word[i+1]
       if M[a][0]==M[b][0]:
           answer=answer+"#"
for i in range(0,len(answer)):
    if answer[i]=="#":
        final=final+"#"
    else:
        final=final+str(Dic[int(answer[i])])
print(final)
