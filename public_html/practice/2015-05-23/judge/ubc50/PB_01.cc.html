#include<bits/stdtr1c++.h>

using namespace std;

int n;
map<string, int>  contestants;

string solve() {
  for (auto it=contestants.begin(); it!=contestants.end(); ++it) {
    if (it->second != 0)
      return it->first;
  }
}

int main() {
  cin >> n;
  for (int i=0; i<n; i++) {
    string name; cin >> name;
    contestants[name]++;
  }
  for (int i=0; i<n-1; i++){
    string name; cin >> name;
    contestants[name]--;
  }

  cout << solve() << endl;
  return 0;
}
