#include <bits/stdtr1c++.h>

using namespace std;

map<int, string> key({ {1, "   "}, {2, "abc"}, {3, "def"}, 
                       {4, "ghi"}, {5, "jkl"}, {6, "mno"},
                       {7, "pqrs"}, {8, "tuv"}, {9, "wxyz"} });
                       
int main() {
    map<int, int> m;
    for (int i = 1; i <= 9; i++) {
        int x; cin >> x;
        m[x] = i;
    }
    
    map<char, string> trans;
    for (char c = 'a'; c <= 'z'; c++) {
        for (int k = 1; k <= 9; k++) {
            if (find(key[k].begin(), key[k].end(), c) != key[k].end()) {
                string s;
                for (int i = 0; i < key[k].size(); i++) {
                    s.push_back('0' + m[k]);
                    if (key[k][i] == c) {
                        break;
                    }
                }
                trans[c] = s;
            }
        }
    }
    
    string t; cin >> t;
    for (int i = 0; i < t.size(); i++) {
        cout << trans[t[i]];
        if (i+1 < t.size() && trans[t[i]][0] == trans[t[i+1]][0]) {
            cout << '#';
        }
    }
    cout << endl;
	return 0;
}
