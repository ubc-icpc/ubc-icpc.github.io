#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int main() {
    ios::sync_with_stdio(0);
    
    int n; cin >> n;
    multiset<string> s;
    for (int i = 0; i < n; i++) {
        string t; cin >> t;
        s.insert(t);
    }
    
    for (int i = 0; i < n-1; i++) {
        string t; cin >> t;
        s.erase(s.find(t));
    }
    
    cout << *s.begin() << endl;
	return 0;
}
