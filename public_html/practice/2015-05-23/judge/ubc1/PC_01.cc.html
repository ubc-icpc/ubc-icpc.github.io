#include <bits/stdtr1c++.h>

using namespace std;

int n, k, a[5005], b[5005];
int main() {
    cin >> n >> k;
    map<int, int> m;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        b[i] = a[i];
    }
    sort(b, b+n);
    
    for (int i = 0; i < n; i++) {
        m[b[i]] = i;
    }
    
    for (int i = 0; i < n; i++) {
        a[i] = m[a[i]]/k;
    }
    
    multiset<int> s;
    for (int i = 0; i < n; i++) {
        auto it = s.lower_bound(a[i]+1);
        if (it != s.end()) {
            s.erase(it);
        }
        s.insert(a[i]);
    }
    cout << n-s.size() << endl;
	return 0;
}
