#include<bits/stdtr1c++.h>
using namespace std;

int N, K, V[100005];
pair<int, int> W[100005];
multiset<int> M;

int main () {
  cin >> N >> K;
  for (int i = 0; i < N; ++i) { cin >> W[i].first; W[i].second = i; }
  sort(W, W + N);
  for (int i = 0; i < N; ++i)
    V[W[i].second] = i / K;
  for (int i = 0; i < N; ++i) {
    multiset<int>::iterator it = M.upper_bound(V[i]);
    if (it == M.end()) M.insert(it, V[i]);
    else {
      it = M.erase(it);
      M.insert(V[i]);
    }
  }
  cout << N - M.size() << endl;
  return 0;
}
