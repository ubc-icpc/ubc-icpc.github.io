#include<bits/stdtr1c++.h>
using namespace std;

string S, C = "22233344455566677778889999",
          N = "12312312312312312341231234";
int K[10];

int main () {
  for (int i = 1; i <= 9; ++i) {
    int t; cin >> t;
    K[t] = i;
  }
  cin >> S;
  for (int i = 0; i < S.size(); ++i) {
    if (i > 0 && C[S[i] - 'a'] == C[S[i - 1] - 'a']) cout << '#';
    for (int j = '0'; j < N[S[i] - 'a']; ++j)
      cout << K[C[S[i] - 'a'] - '0'];
  }
  cout << endl;
  return 0;
}
