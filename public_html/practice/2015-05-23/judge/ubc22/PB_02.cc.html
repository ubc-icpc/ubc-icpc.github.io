#include<bits/stdtr1c++.h>
using namespace std;

int N;
string L[100005], M[100005];

int main () {
  cin >> N;
  for (int i = 0; i < N; ++i) cin >> L[i];
  for (int i = 0; i < N - 1; ++i) cin >> M[i];
  sort(L, L + N);
  sort(M, M + N - 1);
  int i;
  for (i = 0; i < N; ++i)
    if (L[i] != M[i]) break;
  cout << L[i] << endl;
  return 0;
}
