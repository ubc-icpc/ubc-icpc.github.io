#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int t; cin >> t;
	while(t--) {
		int d; cin >> d;
		int curr = 0;
		int last = 0;
		bool ans = true;
		cout << "Denominations: ";
		for (int i=0; i<d; i++) {
			cin >> curr;
			cout << curr << " ";
			if (curr < 2*last) ans = false;
			last = curr;
		}
		cout << endl;
		if (ans) {
			cout << "Good ";
		} else {
			cout << "Bad ";
		}
		cout << "coin denominations!" << endl;
	}
	return 0;
}
