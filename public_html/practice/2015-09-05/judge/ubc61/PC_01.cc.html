#include <bits/stdtr1c++.h>
using namespace std;

const double PI = 3.14159265359;

int main() {
	int n; cin >> n;
	for (int N=1; N<=n; N++) {
		double a,b; cin >> a >> b;
		double c = sqrt(a*a + b*b);
		double tarea = a*b/2;
		double r = (a + b - c)/2;
		double b2 = (b + c - a)/2;
		double c2 = sqrt(b2*b2 + r*r);
		double r2 = (c2*r-r*r)/(c2+r);
		double ratio = (r2*r2)/(r*r);
		double carea = (r*r*PI)/(1-ratio);
		cout << fixed << setprecision(4) << "Case #" << N << ": " << carea/tarea << endl << endl;
	}
	return 0;
}
