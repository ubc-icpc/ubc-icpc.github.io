#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n; cin >> n;

	vector<int> primes;
	vector<int> sieve (11000,0);

	for (int i=2; i<sieve.size(); i++) {
		if (!sieve[i]) {
			int p = i;
			primes.push_back(p);
			for (int j=p; j<sieve.size(); j+=p) {
				sieve[j] = 1;
			}
		}
	}

	while(n--) {
		int v; cin >> v;
		cout << "Input value: " << v << endl;
		if (find(primes.begin(),primes.end(),v) != primes.end()) {
			cout << "Would you believe it; it is a prime!" << endl;
		} else {
			int nextindx = lower_bound(primes.begin(),primes.end(),v)-primes.begin();
			int prev = primes[nextindx-1];
			int next = primes[nextindx];
			int diff = min(abs(v-next),abs(v-prev));
			cout << "Missed it by that much (" << diff << ")!" << endl;
		}
		cout << endl;
	}
	return 0;
}
