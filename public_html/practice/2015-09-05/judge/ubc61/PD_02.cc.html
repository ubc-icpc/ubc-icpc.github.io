#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n; cin >> n;
	for (int N=1; N<=n; N++) {
		int r,c; cin >> r >> c;
		int arr[r][c];
		for (int i=0; i<r; i++) {
			for (int j=0; j<c; j++) {
				cin >> arr[i][j];
			}
		}

		for (int i=0; i<r; i++) {
			for (int j=0; j<c-1; j++) {
				arr[i][j+1] -= arr[i][j];
				arr[i][j] = 0;
			}
			if (i!=r-1) {
				arr[i+1][c-1] -= arr[i][c-1];
				arr[i][c-1] = 0;
			}
		}

		cout << "Case #" << N << ": ";
		if (arr[r-1][c-1])
			cout << "NO";
		else
			cout << "YES";
		cout << endl << endl;
	}
	return 0;
}
