#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main() {
  int n; cin >> n;
  while (n--) {
    vector<int> v,t;
    int d; cin >> d;
    for (int i=0; i<d; i++) {
      int x; cin >> x; v.push_back(x); t.push_back(x);
    }
    sort(v.begin(), v.end());
    bool flag=false;
    for (int i=1; i<d; i++) {
      if (v[i]<2*v[i-1]) {
	flag=true; break;
      }
    }
    cout << "Denominations:";
    for (auto e : t) cout << " " << e;
    cout << endl;
    if (flag) cout << "Bad coin denominations!" << endl;
    else cout << "Good coin denominations!" << endl;
  }
}
