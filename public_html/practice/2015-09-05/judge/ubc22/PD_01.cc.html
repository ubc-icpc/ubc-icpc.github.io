#include<bits/stdtr1c++.h>
using namespace std;

int T, R, C, M, S[2];

int main () {
    cin >> T;
    for (int t = 1; t <= T; ++t) {
        S[0] = S[1] = 0;
        cin >> R >> C;
        for (int i = 0; i < R; ++i)
            for (int j = 0; j < C; ++j) {
                cin >> M;
                S[(i+j)%2] += M;
            }
        cout << "Case #" << t << ": " << (S[0]==S[1]?"YES":"NO") << "\n\n";
    }
    return 0;
}
