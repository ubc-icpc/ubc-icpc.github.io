#include<bits/stdtr1c++.h>
using namespace std;

int T, N, M[15];

int main () {
    cin >> T;
    while (T--) {
        cin >> N;
        for (int i = 0; i < N; ++i) cin >> M[i];
        cout << "Denominations:";
        for (int i = 0; i < N; ++i) cout << " " << M[i];
        cout << "\n";
        bool b;
        for (int i = b = 1; i < N; ++i)
            b = b && (M[i]/M[i-1] >= 2);
        cout << (b?"Good":"Bad") << " coin denominations!\n\n";
    }
}
