#include<bits/stdtr1c++.h>
using namespace std;

int N, M;
bool S[10500];
vector<int> P;

int main () {
    for (int i = 2; i < 10500; ++i)
        if (!S[i])
            for (int j = i+i; j < 10500; j+=i)
                S[j] = true;
    for (int i = 2; i < 10500; ++i)
        if (!S[i]) P.push_back(i);
    cin >> N;
    while (N--) {
        cin >> M;
        vector<int>::iterator it = upper_bound(P.begin(), P.end(), M)-1;
        cout << "Input value: " << M << "\n";
        if (*it == M) cout << "Would you believe it; it is a prime!\n\n";
        else cout << "Missed it by that much (" << min(M-*it, *(it+1)-M) << ")!\n\n";
    }
    return 0;
}
