#include <iostream>	
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <cmath>
#include <math.h>

using namespace std;

int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		cout<<"Case #"<<i<<": ";
		int r,c;
		cin>>r>>c;
		int sum=0;
		for(int j=0;j<r;j++)
			for(int k=0;k<c;k++){
				int a;
				cin>>a;
				if((j+k)%2) sum+=a;
				else sum-=a;
			}
		if(sum==0) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
