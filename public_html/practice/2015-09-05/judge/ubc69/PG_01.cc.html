#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <cmath>
#include <math.h>

using namespace std;

int main() {
	int n;
	cin>>n;
	bool primes[20001];
	for(int i=0;i<20001;i++)
		primes[i]=true;
	primes[1]=false;
	for(int i=2;i<=150;i++){
		if(!primes[i]) continue;
		for(int j=2;j*i<20001;j++){
			primes[j*i]=false;
		}
	}
	for(int i=1;i<=n;i++){
		int v;
		cin>>v;
		cout<<"Input value: "<<v<<endl;
		if(primes[v])
			cout<<"Would you believe it; it is a prime!"<<endl;
		else{
			int j=1;
			while(j<10001){
				if(primes[v+j]||(v-j>=0&&primes[v-j]))
					break;
				j++;
			}
			cout<<"Missed it by that much ("<<j<<")!"<<endl;
		}
	}
	return 0;
}
