#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>

using namespace std;

int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		int d;
		cin>>d;
		int den[d];
		cout<<"Denominations:";
		bool good=true;
		for(int j=0;j<d;j++){
			cin>>den[j];
			cout<<" "<<den[j];
			if(j>0&&den[j]<den[j-1]*2) good=false;
		}
		cout<<endl;
		if(good)
			cout<<"Good coin denominations!"<<endl;
		else
			cout<<"Bad coin denominations!"<<endl;
		
	}
	return 0;
}
