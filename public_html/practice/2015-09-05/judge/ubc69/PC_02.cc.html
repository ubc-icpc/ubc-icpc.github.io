#include <iostream>
#include <stdio.h>
#include <string.h>
#include <fstream>
#include <iomanip>
#include <cmath>
#include <math.h>

using namespace std;

int main() {
	int n;
	cin>>n;
	cout <<std::fixed<<std::setprecision(4);
	for(int i=1;i<=n;i++){
		int L1,L2;
		cin>>L1>>L2; //L2 is longer
		double H=sqrt(L1*L1+L2*L2);
		double R1=L1*L2/(L1+L2+H);
		double B=(L2+H-L1)/2;
		double H1=sqrt(B*B+R1*R1);
		double K=((H1-R1)/(H1+R1))*((H1-R1)/(H1+R1));
		double area=R1*R1*M_PI/(1-K);
		cout<<"Case #"<<i<<": "<<area/(L1*L2/2)<<endl;
	}
	return 0;
}
