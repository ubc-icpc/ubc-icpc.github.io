#include <bits/stdc++.h>

using namespace std;

int main(){
  int n; cin >>n;
  while(n--){
    int d; cin >> d;
    int denom[d];
    for(int i = 0; i < d; i++){
      cin >> denom[i];
    }

    cout<< "Denominations: ";
    for(int i = 0; i < d; i++){
      cout << denom[i] << " ";
    }
    cout << endl;
    
    int p = denom[0];
    bool possible = true;
    for(int i = 1; i < d; i++){
      if(2 *p > denom[i]){
	cout << "Bad coin denominations!" << endl;
	possible = false;
	break;
      }
      p = denom[i];
    }
    if(possible)
      cout << "Good coin denominations!" << endl;
  }
}
