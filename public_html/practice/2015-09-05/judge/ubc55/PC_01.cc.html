#define _USE_MATH_DEFINES
#include <iostream>
#include <cmath>

using namespace std;

typedef long double ld;

int main()
{
	int n;
	cin >> n;
	int count = 0;
	while (n--)
	{
		count++;
		ld little, big;
		cin >> little >> big;
		ld area_triangle = little * big / 2;
		ld theta1 = atan(little / big);
		ld theta2 = M_PI / 2 - theta1;
		ld bisect = theta1 / 2;
		ld radius2 = little * sin(theta2) / (1+sqrt(2));
		ld radius = (little + big - sqrt(little*little + big*big))/2;
		ld x_ratio = 1 / sin(bisect);
		ld circle_ratio = (x_ratio - 1) / (x_ratio + 1);
		ld area_circle = M_PI * radius * radius;
		//for (int i = 0; i < 10000; i++)
		//{
		//	radius *= circle_ratio;
		//	area_circle += M_PI*radius*radius;
		//}
		area_circle *= (1 / (1 - (circle_ratio* circle_ratio)));
		cout.precision(4);
		cout << "Case #" << count << ": " << area_circle / area_triangle << endl << endl;
	}
	return 0;
}
