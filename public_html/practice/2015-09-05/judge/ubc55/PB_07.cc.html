#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	while (n--)
	{
		int d;
		bool lol = true;
		cin >> d;
		int num = 0;
		cout << "Denominations:";
		for (int i = 0; i < d; i++)
		{
			int compare;
			cin >> compare;
			cout << " " << compare;
			if (num * 2 > compare)
			{
				lol = false;
			}
			num = compare;
		}
		if (lol)
		{
			cout << endl << "Good coin denominations!" << endl << endl;
		}
		else
		{
			cout << endl << "Bad coin denominations!" << endl << endl;
		}
	}
	return 0;
}
