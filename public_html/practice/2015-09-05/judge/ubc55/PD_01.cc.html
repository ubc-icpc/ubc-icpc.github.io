#include <iostream>
#include <cstring>

using namespace std;

int arr[36][36];
int main()
{
	int n; cin >> n;
	int count = 0;
	while (n--)
	{
		count++;
		cout << "Case #" << count << ": ";
		int y, x;
		cin >> y >> x;
		for (int j = 0; j < y; j++)
		{
			for (int i = 0; i < x; i++)
			{
				int num; cin >> num;
				arr[j][i] = num;
			}
		}
		//sum rows
		for (int j = 0; j < y; j++)
		{
			for (int i = 0; i < x - 1; i++)
			{
				if (arr[j][i] > 0)
				{
					do
					{
						arr[j][i + 1]--;
					} while (--arr[j][i]);
				}
				else if (arr[j][i] < 0)
				{
					do
					{
						arr[j][i + 1]++;
					} while (++arr[j][i]);
				}
			}
		}
		//sum columns
		for (int j = 0; j < y - 1; j++)
		{
			if (arr[j][x-1] > 0)
			{
				do
				{
					arr[j+1][x-1]--;
				} while (--arr[j][x-1]);
			}
			else if (arr[j][x-1] < 0)
			{
				do
				{
					arr[j+1][x-1]++;
				} while (++arr[j][x-1]);
			}
		}
		if (arr[y - 1][x - 1] == 0)
		{
			cout << "YES" << endl << endl;
		}
		else
		{
			cout << "NO" << endl << endl;
		}
	}
}
