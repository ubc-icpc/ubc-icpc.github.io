#include <iostream>
#include <cstring>

using namespace std;

bool primes[10105];
int main()
{
	memset(primes, true, sizeof(primes));
	primes[0] = false;
	primes[1] = false;
	for (int i = 2; i < 6000; i++)
	{
		for (int j = 2; j < 6000; j++)
		{
			if (i*j > 10100)
				break;
			primes[i*j] = false;
		}
	}
	int n; cin >> n;
	while (n--)
	{
		int cur; cin >> cur;
		cout << "Input value: " << cur << endl;
		if (primes[cur])
		{
			cout << "Would you believe it; it is a prime!" << endl << endl;
		}
		else
		{
			cout << "Missed it by that much (";
			for (int dif = 1; dif < 10000; dif++)
			{
				if (cur - dif > 0)
				{
					if (primes[cur - dif])
					{
						cout << dif << ")!" << endl << endl;
						break;
					}
				}
				if (primes[cur + dif])
				{
					cout << dif << ")!" << endl << endl;
					break;
				}
			}
		}
	}
}
