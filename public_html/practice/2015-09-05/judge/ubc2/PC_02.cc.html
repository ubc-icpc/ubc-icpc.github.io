#include <bits/stdc++.h>

using namespace std;

typedef long double ld;

long double PI = 3.1415926535L;

int main() {
  int N; cin >> N;
  for (int n = 1; n <= N; ++n) {
    ld L1, L2; cin >> L1 >> L2;
    ld s = sqrtl(2/(L1*L2));
    ld a = s*min(L1, L2), b = s*max(L1, L2);
    ld r1 = (a + b - sqrtl(a*a + b*b)) / 2;

    a = r1, b = b - r1;
    ld h0 = sqrtl(a*a + b*b), h1 = h0 + r1, h2 = h0 - r1;
    ld r2 = r1*h2/h1;
    ld ratio = (r2*r2)/(r1*r1);
    ld area = PI*r1*r1/(1-ratio);

    cout << "Case #" << n << ": " << setprecision(4) << fixed << area << endl << endl;
  }
  return 0;
}
