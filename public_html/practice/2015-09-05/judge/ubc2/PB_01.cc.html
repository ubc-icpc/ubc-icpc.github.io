#include <bits/stdc++.h>

using namespace std;

int main() {
  int N; cin >> N;
  while (N--) {
    int D; cin >> D;
    bool bad = false;
    cout << "Denominations:";
    int prev = 0;
    while (D--) {
      int curr; cin >> curr;
      cout << " " << curr;
      if (curr < 2*prev) {
        bad = true;
      }
      prev = curr;
    }
    cout << endl;
    if (bad) {
      cout << "Bad coin denominations!" << endl << endl;
    }
    else {
      cout << "Good coin denominations!" << endl << endl;
    }
  }
  return 0;
}
