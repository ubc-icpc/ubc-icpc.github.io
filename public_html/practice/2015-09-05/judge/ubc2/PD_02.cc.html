#include <bits/stdc++.h>

using namespace std;

int main() {
  int N; cin >> N;
  for (int n = 1; n <= N; ++n) {
    int R, C; cin >> R >> C;
    int curr = 0;
    for (int r = 0; r < R; ++r) {
      vector<int> row;
      for (int c = 0; c < C; ++c) {
        int x; cin >> x;
        row.push_back(x);
      }
      if (r % 2 == 0) {
        for (int i = 0; i < row.size(); ++i) {
          curr = row[i] - curr;
        }
      }
      else {
        for (int i = row.size() - 1; i >= 0; --i) {
          curr = row[i] - curr;
        }
      }
    }
    cout << "Case #" << n << ": ";
    if (curr == 0) cout << "YES" << endl << endl;
    else cout << "NO" << endl << endl;
  }
	return 0;
}
