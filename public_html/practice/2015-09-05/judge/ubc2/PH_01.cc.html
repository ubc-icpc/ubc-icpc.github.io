#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> pii;

int dist[20][20];
bool visited[20][20];

int dr[8] = {-2, -2, 2, 2, -1, 1, -1, 1};
int dc[8] = {-1, 1, -1, 1, -2, -2, 2, 2};

int main() {
  int T; cin >> T;
  for (int t = 1; t <= T; ++t) {
    memset(dist, 0, sizeof(dist[0][0])*20*20);
    memset(visited, false, sizeof(visited[0][0])*20*20);
    int N, R1, C1, R2, C2; cin >> N >> R1 >> C1 >> R2 >> C2;
    R1--; C1--; R2--; C2--;
    queue<pii> q;
    q.push({R1, C1});
    visited[R1][C1] = true;
    while (!q.empty()) {
      pii origin = q.front(); q.pop();
      int ro = origin.first, co = origin.second;
      if (ro == R2 && co == C2) break;
      for (int i=0; i < 8; ++i) {
        int rd = ro + dr[i], cd = co + dc[i];
        if (rd >= 0 && cd >=0 && rd < N && cd < N && !visited[rd][cd]) {
          visited[rd][cd] = true;
          dist[rd][cd] = dist[ro][co] + 1;
          q.push({rd, cd});
        }
      }
    }
    cout << "Case #" << t << ": " << dist[R2][C2] << endl << endl;
  }
  return 0;
}
