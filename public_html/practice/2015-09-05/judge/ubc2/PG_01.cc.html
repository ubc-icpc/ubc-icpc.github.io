#include <bits/stdc++.h>

using namespace std;

bool sieve[11001];

int main() {
  memset(sieve, true, sizeof(sieve));
  for (int i = 2; i*i <= 11000; ++i) {
    if (sieve[i]) {
      for (int j = i*i; j <= 11000; j += i) {
        sieve[j] = false;
      }
    }
  }
  vector<int> primes;
  for (int i = 2; i <= 11000; ++i) {
    if (sieve[i]) {
      primes.push_back(i);
    }
  }
  int N; cin >> N;
  while (N--) {
    int v; cin >> v;
    cout << "Input value: " << v << endl;
    int prev = 0;
    for (int i = 0; i < primes.size(); ++i) {
      if (primes[i] == v) {
        cout << "Would you believe it; it is a prime!" << endl;
        break;
      }
      if (prev < v && v < primes[i]) {
        int mindiff = min(v - prev, primes[i] - v);
        cout << "Missed it by that much (" << mindiff << ")!" << endl;
        break;
      }
      prev = primes[i];
    }
    cout << endl;
  }
  return 0;
}
