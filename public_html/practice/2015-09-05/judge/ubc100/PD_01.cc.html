#include<iostream>

using namespace std;
int m[35][35];

int main (void) {
  int t; cin >> t;
  for(int i=0; i<t; i++){
	int r,c; cin >>r>>c;
	for(int j=0; j<r; j++) {
	  for(int k=0; k<c; k++) {
		int x; cin>>x;
		m[j][k] = x;
	  }
	}
	for(int j=0; j<r; j++) {
	  for(int k=0; k<c; k++) {
		if(k<c-1) {
		  m[j][k+1] += 0-m[j][k];m[j][k]=0;
		} else if(j<r-1){
		  m[j+1][k] += 0-m[j][k];m[j][k]=0;
		}
		
	  }
	}
	cout << "Case #" << i+1 << ": ";
	if(m[r-1][c-1] == 0) cout << "YES" << endl << endl;
	else cout << "NO" << endl << endl;
  }
  return 0;
}
