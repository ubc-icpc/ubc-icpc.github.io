#include<iostream>
#include<algorithm>

using namespace std;

int main(void) {
  int T; cin >> T;

  while(T--) {
	int n; cin >> n;
	int arr[n];
	for(int i=0; i<n; i++) {
	  int x; cin >> x;
	  arr[i]=x;
	}
	sort(arr, arr+n);
	bool flag = true;
	for(int i=1; i<n; i++) {
	  if(arr[i] < 2*arr[i-1]){
		flag = false; break;
	  }
	}
	cout << "Denominations: ";
	for(int i=0; i<n; i++ ) cout << arr[i] << " ";
	cout << endl;
	if(flag) cout << "Good ";
	else cout << "Bad ";
	cout << "coin denominations!" << endl << endl;
  }
  return 0;
}
