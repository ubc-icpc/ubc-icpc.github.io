n = int(raw_input())
for i in range(n):
    line = raw_input()
    xs = map(int, line.split(' '))[1:]
    print "Denominations: {}".format(' '.join(map(str, xs)))
    for j in range(len(xs) - 1):
        try:
            a, b = xs[j], xs[j+1]
            if b < 2*a:
                break
        except IndexError:
            break
    else:
        print "Good coin denominations!"
        continue

    print "Bad coin denominations!"
