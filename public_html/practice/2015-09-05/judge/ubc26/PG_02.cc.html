bool prime(int x) {
    if (x == 2) return true;
    if (x % 2 == 0) return false;
    int xx = int(ceil(sqrt(x))) + 1;
    for (int i = 3; i < xx; i += 2) {
        if (x % i == 0)
            return false;
    }
    return true;
}
