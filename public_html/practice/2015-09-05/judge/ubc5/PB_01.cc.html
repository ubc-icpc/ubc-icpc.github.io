#include <iostream>
using namespace std;

int d[10];

int main() {
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    cout << "Denominations:";
    for (int i = 0; i < n; i++) {
      cin >> d[i];
      cout << " " << d[i];
    }
    cout << "\n";
    for (int i = 1; i < n; i++) {
      if (d[i] < 2*d[i-1]) {
        cout << "Bad";
        goto VELOCIRAPTOR;
      }
    }
    cout << "Good";
    VELOCIRAPTOR:
    cout << " coin denominations!\n\n";
  }
  return 0;
}
