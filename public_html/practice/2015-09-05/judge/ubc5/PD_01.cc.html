#include <iostream>
using namespace std;

int m[30][30];

int main() {
  int n; cin >> n;
  for (int N = 1; N <= n; N++) {
    cout << "Case #" << N << ": ";
    int r, c; cin >> r >> c;
    int t = 0;
    for (int i = 0; i < r; i++)
      for (int j = 0; j < c; j++)
        cin >> m[i][j];
    for (int i = 0; i < r; i++) {
      if (i % 2) { // rtl
        for (int j = c-1; j; j--) {
          m[i][j-1] -= m[i][j];
          m[i][j] = 0;
        }
        if (i < r-1) {
          m[i+1][0] -= m[i][0];
          m[i][0] = 0;
        }
      }
      else { // ltr
        for (int j = 0; j < c-1; j++) {
          m[i][j+1] -= m[i][j];
          m[i][j] = 0;
        }
        if (i < r-1) {
          m[i+1][c-1] -= m[i][c-1];
          m[i][c-1] = 0;
        }
      }
    }
    for (int i = 0; i < r; i++) {
      for (int j = 0; j < c; j++) {
        if (m[i][j]) {
          cout << "NO";
          goto VELOCIRAPTOR;
        }
      }
    }
    cout << "YES";
    VELOCIRAPTOR:
    cout << "\n\n";
  }
  return 0;
}
