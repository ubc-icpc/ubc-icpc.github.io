#include <iostream>
using namespace std;

bool np[13370];

int main() {
  for (int i = 4; i < 13370; i += 2)
    np[i] = true;
  for (int i = 3; i < 13370; i += 2) {
    if (!np[i]) {
      for (int j = i*i; j < 13370; j += i)
        np[j] = true;
    }
  }
  int n; cin >> n;
  while (n--) {
    int x; cin >> x;
    cout << "Input value: " << x << "\n";
    if (!np[x])
      cout << "Would you believe it; it is a prime!\n\n";
    else {
      int y = x, z = x;
      while (np[y]) y--;
      while (np[z]) z++;
      cout << "Missed it by that much (" << (z-x > x-y ? x-y : z-x) << ")!\n\n";
    }
  }
  return 0;
}
