import java.util.Scanner;
import java.util.ArrayList;

public class Test {

	public static void main(String[] args) {
		
		Scanner in = new Scanner(System.in);
		int testNum = in.nextInt();
		int[][] testCases = new int[testNum][];
		

		for (int i = 0; i < testNum; i++) {
			int elementSize = in.nextInt();
			int[] testCaseElement = new int[elementSize];
			for (int j = 0; j < elementSize; j++) {
				testCaseElement[j] = in.nextInt();
			}
			testCases[i] = testCaseElement;
		}
		
		
		for (int[] arr : testCases) {
			solve(arr);
		}
		return;
	}
	
	public static void solve(int[] cases) {
		System.out.print("Denominations: ");
		for (int i = 0; i < cases.length; i++) {
			System.out.print(cases[i] + " ");
		}
		System.out.println();
//		if (cases.length == 1) {
//			System.out.println(cases[0]);
//			System.out.println("Good coin denominations!");
//			return;
//		}
		int elem = cases[0];
		int next;
		for (int i = 0; i < cases.length - 1; i++) {
			next = cases[i+1];
			if (next < 2*cases[i]) {
				System.out.println("Bad coin denominations!\n");
				return;
			}
			elem = next;
		}
		System.out.println("Good coin denominations!\n");
	}

}
