#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	for (int tc = 1; tc <= n; ++tc){
		int d; cin >> d;
		vector < int > c(d);
		cout << "Denominations: ";
		for (int i = 0; i < d; ++i){
			cin >> c[i];
			if (i == d - 1){
				cout << c[i] << endl;
			}
			else{
				cout << c[i] << " ";
			}
		}
		bool prop = true;
		for (int i = 1; i < c.size(); ++i)
			if (!(c[i] >= 2 * c[i - 1])) prop = false;
		
		if (prop) cout << "Good coin denominations!" << endl;
		else cout << "Bad coin denominations!" << endl;
	}
	return 0;
}
