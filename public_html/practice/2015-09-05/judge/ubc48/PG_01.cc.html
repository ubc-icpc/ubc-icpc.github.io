#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	vector < int > s(10202);
	for (int i = 2; i <= 101; ++i)
		for (int j = 2; i * j <= 10201; ++j)
			s[i * j] = 1;

	while (n--){
		int v; cin >> v;
		cout << "Input value: " << v << endl;;
		if (s[v] == 0){
			cout << "Would you believe it; it is a prime!" << endl << endl;
			continue;
		}
		for (int i = 1; i < 10201; ++i){
			if (v + i < 10202)
				if (s[v + i] == 0){
					cout << "Missed it by that much (" << i << ")!" << endl << endl;
					break;
				}
			if (v - i >= 0)
				if (s[v - i] == 0){
					cout << "Missed it by that much (" << i << ")!" << endl << endl;
					break;
				}
		}
	}
}
