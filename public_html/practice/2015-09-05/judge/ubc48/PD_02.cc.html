#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int n; cin >> n;
	for (int tc = 1; tc <= n; ++tc){
		int r, c; cin >> r >> c;
		int ori = 1, first = 0, second = 0;
		for (int i = 0; i < r; ++i){
			for (int j = 0; j < c; ++j){
				int temp; cin >> temp;
				if (ori % 2 == 1) {
					if (j % 2 == 0) first += temp;
					else second += temp;
				}
				if (ori % 2 == 0) {
					if (j % 2 == 1) first += temp;
					else second += temp;
				}
			}
			if (ori == 1) ori = 0;
			else ori = 1;
		}
		if (first == second) cout << "Case #" << tc << ": YES" << endl << endl;
		else cout << "Case #" << tc << ": NO" << endl << endl;
	}
}
