#include <iostream>
#include <cmath>
#include <utility>
#include <vector>
#include <math.h>

using namespace std;

double rad(double x, double y) {
  return 0.5*(x + y - sqrt(x*x + y*y));
}

int main() {
  int t; cin >> t;
  for (int w=0; w<t; w++) {
    double a, b; cin >> a >> b;
    double bisector = sqrt((rad(a,b))*(rad(a,b)) + (max(a,b) - rad(a,b))*(max(a,b)-rad(a,b)));
    double little_rad = (rad(a,b)*bisector - rad(a,b)*rad(a,b))/(rad(a,b) + bisector);
    
    double theta = acos(rad(a,b)/bisector);
    double area_pent = (rad(a,b) + little_rad)*sqrt(rad(a, b)*little_rad);
    double area_pent_circles = 0.5*rad(a,b)*rad(a,b)*theta + 0.5*little_rad*little_rad*(M_PI-theta);
    
    double sec1 = 0.5*rad(a, b)*rad(a,b)*(1.5*M_PI - 2*theta);
    double sec2 = 0.25*M_PI*rad(a, b)*rad(a,b);
    double sec3 = (rad(a,b)*(max(a,b)-rad(a,b)))*area_pent_circles/area_pent;
    
    printf("Case #%i: %.4f\n\n", w+1, 2*(sec1 + sec2 + sec3)/(a*b));
  }
}
