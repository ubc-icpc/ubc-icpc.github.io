#include <iostream>
#include <cmath>

using namespace std;

int main() {
  int t; cin >> t;
  
  for (int w=0; w<t; w++) {
    int sum = 0;
    int r, c; cin >> r >> c;
    for (int i=0; i<r; i++) {
      for (int j=0; j<c; j++) {
        int q; cin >> q;
        if ((i+j)%2 == 0) sum += q;
        else sum -= q;
      }
    }
    cout << "Case #" << w+1 << ": ";
    if (sum==0) {
      cout << "YES" << endl << endl;
    } else {
      cout << "NO" << endl << endl;
    }
  }
}
