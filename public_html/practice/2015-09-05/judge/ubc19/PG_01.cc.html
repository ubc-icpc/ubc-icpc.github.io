#include <iostream>
#include <cmath>

using namespace std;

bool is_prime(int n) {
  if (n <= 1) return false;
  for (int i=2; i<sqrt(n)+1; i++) {
    if (i!=n && n%i==0) {
      return false;
    }
  }
  return true;
}

int main() {
  int t; cin >> t;
  
  for (int i=0; i<t; i++) {
    int n; cin >> n;
    cout << "Input value: " << n << endl;
    if (is_prime(n)) {
      cout << "Would you believe it; it is a prime!" << endl;
    } else {
      int j = 1;
      while (!is_prime(n-j) && !is_prime(n+j)) {
        j++;
      }
      cout << "Missed it by that much (" << j << ")! " << endl;
    }
    cout << endl;
  }
}
