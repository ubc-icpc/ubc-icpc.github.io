#include <iostream>

using namespace std;

int main() {
  int t; cin >> t;
  
  for (int i=0; i<t; i++) {
    int n; cin >> n; bool ok = true;
    int prev; cin >> prev;
    cout << "Denominations: " << prev << " ";
    for (int j=1; j<n; j++) {
      int cur; cin >> cur;
      cout << cur << " ";
      if (cur < 2*prev) {
        ok = false;
      }
      prev = cur;
    }
    cout << endl;
    if (!ok) {
      cout << "Bad coin denominations!" << endl;
    } else {
      cout << "Good coin denominations!" << endl;
    }
    cout << endl;
  }
}
