#include <iostream>
#include <vector>
#include <iterator>
#include <algorithm>
using namespace std;
using vi = vector<int>;

int main() {
  int T;
  for (cin >> T; T > 0; --T) {
    int d; cin >> d;
    vi coins(d);
    copy_n(istream_iterator<int>(cin), d, coins.begin());
    sort(coins.begin(), coins.end());
    bool good = true;
    for (int i = 0; i < d - 1; ++i)
      if (coins[i + 1] < 2 * coins[i])
        good = false;

    cout << "Denominations:";
    for (auto coin : coins)
      cout << ' ' << coin;
    cout << endl << (good? "Good " : "Bad ") << "coin denominations!" << endl;
    if (T > 1)
      cout << endl;
  }
  return 0;
}
