#include <iostream>
#include <vector>
#include <iterator>
#include <algorithm>
using namespace std;
using vi = vector<int>;

int main() {
  vi sieve(20000, 0);
  vi primes;
  for (int i = 2; i < sieve.size(); ++i) {
    if (sieve[i] == 0) {
      primes.push_back(i);
      for (int j = 2; i * j < sieve.size(); ++j)
        sieve[i * j] = 1;
    }
  }
  int T;
  for (cin >> T; T > 0; --T) {
    int n; cin >> n;
    cout << "Input value: " << n << endl;
    if (sieve[n] == 0) {
      cout << "Would you believe it; it is a prime!" << endl;
    } else {
      auto lb = lower_bound(primes.begin(), primes.end(), n);
      int diff = *lb - n;
      if (lb - primes.begin() > 0)
        diff = min(diff, n - *(lb - 1));
      cout << "Missed it by that much (" << diff << ")!" << endl;
    }
    if (T > 1)
      cout << endl;
  }
  return 0;
}
