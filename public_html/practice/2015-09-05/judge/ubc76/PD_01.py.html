test = int(raw_input())
for i in range(test):
    row,col = [int(x) for x in raw_input().split()]
    m = []
    for j in range(row):
        m.append([int(x) for x in raw_input().split()])
    for r in range(row-1):
        for c in range(col):
            if (m[r][c]!= 0):
                diff = 0-m[r][c]
                m[r+1][c] = m[r+1][c]+diff
                diff = 0
                m[r][c] = 0    
    # The last row
    last = m[row-1]
    diff = 0
    for c in range(col-1):
        if (last[c]!= 0):
            diff = 0-last[c]
            last[c+1] = last[c+1]+diff
            diff = 0
            
    if (last[col-1] == 0):
        print 'Case #'+str(i+1)+': YES'
    else:
        print 'Case #'+str(i+1)+': NO'
