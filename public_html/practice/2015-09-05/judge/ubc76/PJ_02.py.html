ans = 1

def dfs(node, level):
	global ans
	if d[node] > -1:
		ans *= level - d[node]
		ans %= 1007
		return
	d[node] = level
	for i in e[node]:
		if (node, i) not in vis:
			vis[(node, i)] = 1
			vis[(i, node)] = 1
			dfs(i, level + 1)

test = int(raw_input())
for re in range(test):
	V, E = map(int, raw_input().split())
	e = [[] for i in range(V)]
	d = [-1 for i in range(V)]
	for i in range(E):
		x, y = map(int, raw_input().split())
		x -= 1
		y -= 1
		e[x].append(y)
		e[y].append(x)
	ans = 1
	vis = {}
	dfs(0, 0)
	print 'Case #'+str(re+1)+': ', ans
	print
