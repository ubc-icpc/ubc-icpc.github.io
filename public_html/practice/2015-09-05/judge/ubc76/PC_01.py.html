import math

t = int(raw_input())

for re in range(t):
	a, b = map(float, raw_input().split())
	R = (a * b) / (math.sqrt(a*a+b*b) + a + b)
	theta = math.atan(a/b)
	r = (1 - math.sin(theta / 2)) * R / (1 + math.sin(theta / 2))

	beta = (math.pi / 2 - theta / 2)

	l = (R + r) * math.cos(theta / 2)

	AreaR = math.pi*R*R*beta/(2*math.pi)
	Arear = math.pi*r*r*(math.pi-beta)/(2*math.pi)

	ratio1 =  (AreaR + Arear) / (l * (R + r) / 2)

	A1 = R * (b - R) * ratio1 + math.pi*R*R*(1 - 2*beta/(2*math.pi))

	print "Case #"+str(re+1)+":", "{:.4f}".format(A1 / (a * b / 2))
	print
