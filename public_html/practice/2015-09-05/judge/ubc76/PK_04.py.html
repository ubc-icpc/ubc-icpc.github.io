import sys

data = map(int, sys.stdin.read().split())
cnt = 0

test = data[cnt]
cnt += 1

for re in range(test):
	V = data[cnt]
	cnt += 1
	E = data[cnt]
	cnt += 1
	e = [[] for i in range(V)]
	d = [-1 for i in range(V)]
	vis = [0 for i in range(E)]
	for i in range(E):
		x = data[cnt]
		cnt += 1
		y = data[cnt]
		cnt += 1
		x -= 1
		y -= 1
		e[x].append((y, i))
		e[y].append((x, i))
	
	ans = 1
	stack = []
	stack.append((0, 0, -1))
	while len(stack) > 0:
		node, level, p = stack.pop()
		if p > -1 and vis[p] == 1: continue
		if p > -1: vis[p] = 1
		#print node, level, ans
		if d[node] > -1:
			ans *= level - d[node]
			ans %= 1007
			continue
		d[node] = level
		for i, ind in e[node]:
			if vis[ind] == 0:
				stack.append((i, level + 1, ind))

	print 'Case #'+str(re+1)+':', ans
	print
