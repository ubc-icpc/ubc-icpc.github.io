test = int(raw_input())
for i in range(test):
    coins1 = [str(x) for x in raw_input().split()]
    output1 = ' '.join(coins1[1:])
    num = int(coins1[0])
    coins = [ int(x) for x in coins1[1:]]
    pre = coins[0]
    flag = True
    for i in range(1,num):
        cur = coins[i]
        flag = flag and ((cur/pre) >=2)
        pre = cur
    if (flag):
        print 'Denominations: '+output1
        print 'Good coin denominations!'
    else:
        print 'Denominations: '+output1
        print 'Bad coin denominations!'
