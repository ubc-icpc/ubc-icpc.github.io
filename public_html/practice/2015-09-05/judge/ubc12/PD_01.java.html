import java.util.Scanner;


public class D {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int r, c, w, b, n = sc.nextInt();
		for(int i = 0; i< n;i++){
			r = sc.nextInt();
			c = sc.nextInt();
			b = w = 0;
			for (int j = 0;j< r; j++) {
				for (int k = 0; k< c ;k++){
					if (j % 2 == k%2) {
						b += sc.nextInt();
					} else {
						w += sc.nextInt();
					}
				}
			}
			System.out.println("Case #"+(i+1)+": "+ (b == w ? "YES" : "NO"));
			System.out.println();
		}
	}
}
