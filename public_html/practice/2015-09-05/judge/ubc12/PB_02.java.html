import java.util.Scanner;


public class B {
	public static void main(String[] args) {
		int n, d, c, pc;
		boolean good = true;
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		for(int i = 0; i<n; i++) {
			System.out.print("Denominations:");
			d = sc.nextInt();
			c = 0;
			good = true;
			for(int j = 0;j< d;j++){
				pc = c;
				c = sc.nextInt();
				good &= (c >= 2*pc);
				System.out.print(" " + c);
			}
			System.out.println();
			System.out.println((good ? "Good " : "Bad " ) + "coin denominations!");
			System.out.println();
		}
	}
}
