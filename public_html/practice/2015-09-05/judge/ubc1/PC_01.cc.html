#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

// R1 = (a+b-sqrt(a^2 + b^2))/2

//(R1-R2)^2 + x1^2 = (R1+R2)^2
//4R1R2 = x1^2 --> x1 = 2Sqrt(R1R2)
//x1 + x2 + x3 + ... = a - r
//x1 * 1/(1-p) = a-r

int main() {
    int t, ca = 0; cin >> t;
    while (t--) {
        cout << "Case #" << ++ca << ": ";
        ld b, a; cin >> b >> a;
        
        ld r0 = 0.5 * (a + b - sqrt(a*a + b*b));
        ld l1 = sqrt((a-r0)*(a-r0) + r0*r0) - r0;
        ld l0 = sqrt((a-r0)*(a-r0) + r0*r0) + r0;
        ld ratio = pow(l0/l1, -2);
        
        ld pi = acos(-1.0L);
        cout << setprecision(4) << fixed << pi * r0 * r0 / (1-ratio) / (a*b/2) << endl;
        cout << endl;
    }
	return 0;
}
