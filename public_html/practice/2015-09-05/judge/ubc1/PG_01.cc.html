#include <bits/stdtr1c++.h>

using namespace std;

const int maxn = 20005;
bool sieve[20005];
int main() {
    sieve[0] = sieve[1] = true;
    for (int i = 0; i < maxn; i++) {
        if (sieve[i]) continue;
        for (long long j = (1LL*i)*i; j < maxn; j+=i) {
            sieve[j] = true;
        }
    }
    
    int t; cin >> t;
    while (t--) {
        int x; cin >> x;
        cout << "Input value: " << x << endl;
        if (!sieve[x]) cout << "Would you believe it; it is a prime!" << endl;
        else {
            int l = x-1, r = x+1;
            while (l >= 0 && sieve[l]) l--;
            while (sieve[r]) r++;
            
            if (l == -1) l = -10000;
            int res = min(x-l, r-x);
            cout << "Missed it by that much (" << res << ")!" << endl;
        }
        cout << endl;
    }
	return 0;
}
