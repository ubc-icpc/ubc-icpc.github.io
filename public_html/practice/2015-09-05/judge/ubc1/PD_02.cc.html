#include <bits/stdtr1c++.h>

using namespace std;

//1
int a[35][35];
int main() {
    int t, ca = 0; cin >> t;
    while (t--) {
        int n, m; cin >> n >> m;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                cin >> a[i][j];
            }
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m-1; j++) {
                a[i][j+1] -= a[i][j];
                a[i][j] = 0;
            }
            if (i != n-1) {
                a[i+1][m-1] -= a[i][m-1];
                a[i][m-1] = 0;
            }
        }
        
        cout << "Case #" << ++ca << ": ";
        if (a[n-1][m-1] == 0) cout << "YES" << endl;
        else cout << "NO" << endl;
        cout << endl;
    }
	return 0;
}
