#include <bits/stdtr1c++.h>

using namespace std;

int a[15];
int main() {
    int t; cin >> t;
    while (t--) {
        int d; cin >> d;
        for (int i = 0; i < d; i++) cin >> a[i];
        
        bool good = true;
        for (int i = 1; i < d; i++) {
            if (a[i] < 2*a[i-1]) good = false;
        }
        cout << "Denominations: ";
        for (int i = 0; i < d; i++) {
            cout << a[i] << " ";
        }
        cout << endl;
        
        cout << (good ? "Good" : "Bad") << " coin denominations!" << endl << endl;
    }
	return 0;
}
