#include <bits/stdc++.h>

using namespace std;

int main(){
  int t; cin >> t;
  int tc = 1;
  while (t--){
    int r, c; cin >> r >> c;
    int even, odd;
    even = 0;
    odd = 0;
    for (int i = 0; i < r ; i++){
      for (int j = 0; j < c; j++){
	int val; cin >> val;
	if((i+j) %2)
	  even += val;
	else
	  odd += val;
      }
    }
    if(odd -even)
      cout << "Case #"<< tc<< ":" << " NO"<< endl<<endl;
    else
      cout << "Case #"<< tc<< ":" << " YES"<< endl<<endl;
    tc++;
  }

}
