#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
typedef pair<int,int> PT;
typedef vector<pair<int, int>> VP;
typedef long long LL;
const int MAX_P = 70000;

vector<int> primes;
int sp[MAX_P];

void sieve() {
	for (LL i = 2; i < MAX_P; i++) {
		if (sp[i]) continue;
		sp[i] = i;
		primes.push_back(i);
		for (LL j=i*i;j<MAX_P;j+=i)
			if(!sp[j]) sp[j] = i;
	}
}

int main() {
	sieve();
	int tc; cin >> tc;
	while (tc--) {
		int x; cin >> x;
		cout << "Input value: " << x << endl;
		if (sp[x] == x) {
			cout << "Would you believe it; it is a prime!\n\n";
		} else {
			auto it = lower_bound(begin(primes), end(primes), x);
			int ans = *it - x;
			if (it != begin(primes)) {
				--it;
				ans = min(ans, x - *it);
			}
			cout << "Missed it by that much (" << ans << ")!\n\n";
		}
	}
}
