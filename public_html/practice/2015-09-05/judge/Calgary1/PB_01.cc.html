int main() {
	int tc;
	cin >> tc;
	while (tc--) {
		int D, x[10];
		cin >> D;
		FOR(d,0,D) cin >> x[d];
		cout << "Denominations:";
		FOR(d,0,D) cout << ' ' << x[d];
		cout << endl;
		sort(x,x+D);
		bool good = true;
		FOR(d,1,D) if (x[d] < 2*x[d-1]) good=false;
		cout << (good?"Good coin denominations!\n\n":"Bad coin denominations!\n\n");
	}
}
