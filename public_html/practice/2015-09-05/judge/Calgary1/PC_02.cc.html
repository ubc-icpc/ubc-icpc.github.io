#include <bits/stdc++.h>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<int(b);i++)
typedef long long LL;
typedef pair<int,int> PT;

double getR(double& a, double& b, double& c) {
	double angleA = acos((c*c+b*b-a*a)/2.0/c/b);
	double s = 0.5*(a+b+c);
	double A = sqrt(s*(s-a)*(s-b)*(s-c));
	double r = A/s;
	double L = r/sin(0.5*angleA)-r;
	double new_a = 2.0*L*tan(0.5*angleA);
	double new_bc = new_a*sin(0.5*(M_PI-angleA))/sin(angleA);
	a = new_a;
	b = c = new_bc;
	return r;
}

int main() {
	int TC;
	cin >> TC;
	FOR (tc,1,TC+1) {
		double ans = 0.0, a, b;
		cin >> a >> b;
		double oarea=0.5*a*b, c = sqrt(a*a+b*b);
		for (;;) {
			double r = getR(a, b, c);
			double dans = M_PI*r*r/oarea;
			ans += dans;
			if (dans < 1e-10) break;
		}
		cout << "Case #" << tc << ": " << fixed << setprecision(4) << ans << "\n\n";
	}
}
