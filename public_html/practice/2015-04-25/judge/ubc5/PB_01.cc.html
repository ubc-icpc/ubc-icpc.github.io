#include <iostream>
using namespace std;

int t[4];

int main() {
  int w, n;
  cin >> w >> n;
  int tw = 0, tn = 0;
  for (int i = 0; i < n; i++) {
    int c; cin >> c;
    tw -= t[0];
    t[0] = t[1];
    t[1] = t[2];
    t[2] = t[3];
    t[3] = c;
    tw += c;
    if (tw <= w)
      tn++;
    else
      break;
  }
  cout << tn << "\n";
  return 0;
}
