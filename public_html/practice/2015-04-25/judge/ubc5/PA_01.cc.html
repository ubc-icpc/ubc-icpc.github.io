#include <iostream>
#include <set>
using namespace std;

inline bool d(int y) {
  set<int> s;
  int n = 0;
  while (y) {
    s.insert(y%10);
    y /= 10;
    n++;
  }
  return s.size() == n;
}

int main() {
  int y;
  while (cin >> y) {
    y++;
    while (!d(y))
      y++;
    cout << y << "\n";
  }
  return 0;
}
