#include <iostream>
#include <vector>
#include <iomanip>
#include <queue>
#include <cstring>
using namespace std;

vector<int> adj[1000001];
bool v[1000001];

bool bfs(int s, int e) {
  memset(v,0,sizeof v);
  queue<int> q;
  q.push(s);
  while (q.size()) {
    int f = q.front();
    q.pop();
    if (f == e)
      return true;
    for (int i = 0; i < adj[f].size(); i++) {
      if (!v[adj[f][i]]) {
        v[adj[f][i]] = true;
        q.push(adj[f][i]);
      }
    }
  }
  return false;
}

int main() {
  ios::sync_with_stdio(0);
  int n, m;
  cin >> n >> m;
  for (int i = 0; i < m; i++) {
    int x, y;
    cin >> x >> y;
    adj[x].push_back(y);
  }
  int p, q;
  cin >> p >> q;
  if (bfs(p,q))
    cout << "yes\n";
  else if (bfs(q,p))
    cout << "no\n";
  else
    cout << "unknown\n";
  return 0;
}
