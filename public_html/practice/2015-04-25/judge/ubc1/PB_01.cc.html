#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int W, n; cin >> W >> n;
    queue<int> q;
    int s = 0;
    
    int ans = 0;
    while (n--) {
        int x; cin >> x;
        if (s+x > W) break;
        q.push(x);
        s += x;
        if (q.size() == 4) {
            s -= q.front();
            q.pop();
        }
        ans++;
    }
    cout << ans << endl;
	return 0;
}
