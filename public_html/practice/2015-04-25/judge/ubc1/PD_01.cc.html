#include <bits/stdtr1c++.h>

using namespace std;

vector<int> g[1000005];
int deg[1000005], seen[1000005];
int main() {
    ios::sync_with_stdio(0);
    int n, m; cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int x, y; cin >> x >> y;
        g[x].push_back(y);
        deg[y]++;
    }
    
    int res = -1;
    int loc[2]; cin >> loc[0] >> loc[1];
    for (int j = 0; j < 2; j++) {
        memset(seen, 0, sizeof seen);
        queue<int> q({loc[j]});
        seen[loc[j]] = 1;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : g[u]) {
                if (!seen[v]) {
                    q.push(v);
                    seen[v] = 1;
                }
            }
        }
        if (seen[loc[j^1]]) res = j;
    }
    
    if (res == -1) {
        cout << "unknown" << endl;
    } else {
        cout << (res ? "no" : "yes") << endl;
    }
    
	return 0;
}
