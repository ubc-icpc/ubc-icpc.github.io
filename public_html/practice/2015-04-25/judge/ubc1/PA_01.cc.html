#include <bits/stdtr1c++.h>

using namespace std;

bool good(int n) {
    set<int> d;
    int l = 0;
    while (n > 0) {
        l++;
        d.insert(n%10);
        n /= 10;
    }
    return d.size() == l;
}
int main() {
    int n; cin >> n;
    while (!good(++n)); 
    cout << n << endl;
	return 0;
}
