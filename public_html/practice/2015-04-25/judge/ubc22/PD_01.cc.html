#include<bits/stdtr1c++.h>
using namespace std;

int N, M, A, B;
vector<int> taller[1000005];
bool T, S, V[1000005];

int main () {
  cin >> N >> M;
  while (M--) {
    cin >> A >> B;
    taller[A].push_back(B);
  }
  cin >> A >> B;
  queue<int> q;
  q.push(A);
  while (!q.empty()) {
    int top = q.front(); q.pop();
    if (V[top]) continue; V[top] = true;
    if (top == B) T = true;
    for (int i : taller[top]) q.push(i);
  }
  memset(V, 0, sizeof V);
  q = queue<int>();
  q.push(B);
  while (!q.empty()) {
    int top = q.front(); q.pop();
    if (V[top]) continue; V[top] = true;
    if (top == A) S = true;
    for (int i : taller[top]) q.push(i);
  }
  if (T) cout << "yes" << endl;
  else if (S) cout << "no" << endl;
  else cout << "unknown" << endl;
  return 0;
}
