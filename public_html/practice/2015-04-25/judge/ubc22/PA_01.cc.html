#include<bits/stdtr1c++.h>
using namespace std;

int N, C[10];

bool U(int i) {
  memset(C, 0, sizeof C);
  while (i) { ++C[i % 10]; i /= 10; }
  bool r = false;
  for (int i = 0; i < 10; ++i) r = r || C[i] > 1;
  return r;
}

int main () {
  cin >> N;
  while(U(++N));
  cout << N << endl;
  return 0;
}
