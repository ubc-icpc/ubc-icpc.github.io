#include<bits/stdtr1c++.h>
using namespace std;

int N, C;
vector<int> primes;
bool P[3000];

int main () {
  cin >> N;
  for (int i = 2; i < 3000; ++i)
    if (!P[i]) {
      primes.push_back(i);
      for (int j = i + i; j < 3000; j += i) P[j] = true;
    }
  // lol
  while (N != 1) {
    bool b = true;
    for (int p : primes) {
      if (p * p > N) break;
      if (N % p == 0) {
        C += p - 1;
        N = N / p * (p - 1);
        b = false;
        break;
      }
    }
    if (b) C += --N;
  }
  cout << C << endl;
  return 0;
}
