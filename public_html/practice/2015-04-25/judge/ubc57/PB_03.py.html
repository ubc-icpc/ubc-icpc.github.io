w=int(input())
n=int(input())
L=[]
for i in range(0,n):
    x=int(input())
    L.append(x)
answer=n
if n<=4:
    total=0
    for i in range(0,n):
        x=L[i]
        total=total+x
        if total>w:
            answer=i
            break
if n>4:
    total=L[0]+L[1]+L[2]+L[3]
    if total>w:
        total=0
        for i in range(0,4):
            x=L[i]
            total=total+x
            if total>w:
                answer=i
                break
    else:
        for i in range(0,n-4):
            total=total-L[i]+L[i+4]
            if total>w:
                answer=i+4
                break
        
print(answer)
