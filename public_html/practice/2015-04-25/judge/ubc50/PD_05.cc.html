#include<bits/stdtr1c++.h>

using namespace std;

int p, q;
// m[x] contains people shorter than x
map<int, vector<int> >m;
int seen[1000001];

bool solve (int p, int q) {
    memset(seen, 0, sizeof seen);
  queue<int> sthng;
  sthng.push(p);
  while(!sthng.empty()) {
    int top = sthng.front();
    sthng.pop();
    if (find(m[top].begin(), m[top].end(), q)!=m[top].end())
      return true;
    for (auto e : m[top])
      if (!seen[e]) {
	seen[e] = 1;
	sthng.push(e);
      }
  }
  return false;
}

int main() {
  ios::sync_with_stdio(0);
  int n, ma; cin >> n >> ma;
  for (int i=0; i<ma; i++) {
    int x, y; cin >> x >> y;
    m[x].push_back(y);
  }
  cin >> p >> q;
  if (solve(p,q))
    cout << "yes\n";
  else if (solve(q,p))
    cout << "no\n";
  else
    cout << "unknown\n";
  return 0;
}
