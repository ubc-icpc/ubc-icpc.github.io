#include<iostream>

using namespace std;

int main() {
  int ans = 0;
  int max_weight; cin >> max_weight;
  int n; cin >> n;
  int arr[n];
  for (int i=0; i<n; i++)
    cin >> arr[i];
  bool flag = false;
  int temp = 0;
  int i = 0;
  while (i<n && !flag) {
    if(i-4>=0)
      temp-=arr[i-4];
    temp+=arr[i++];
    if (temp<=max_weight)
      ans++;
    else
      flag = true;
  }
  cout << ans << endl;
  return 0;
}
