#include<iostream>
#include<map>

using namespace std;

bool distinct(int y) {
  map<int, int> m;
  while(y>0) {
    m[y%10]++;
    if (m[y%10]>1)
      return false;
    y/=10;
  }
  return true;
}

int solve (int y) {
  bool flag = false;
  while (!flag) {
    y++;
    if(distinct(y))
      flag=true;
  }
  return y;
}

int main() {
  int y; cin >> y;
  int ans = 0;
  ans = solve(y);
  cout << ans << endl;
  return 0;
}
