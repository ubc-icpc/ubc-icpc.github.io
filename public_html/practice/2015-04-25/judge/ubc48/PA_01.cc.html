#include<bits/stdtr1c++.h>
using namespace std;
int main() {
	int n; cin >> n;
	while (true) {
		++n;
		stringstream ss;
		ss << n;
		string temp; ss >> temp;
		int a[10]{ 0 };
		bool flag = true;
		for (int i = 0; i < temp.size(); ++i) {
			a[temp[i] - '0']++;
			if (a[temp[i] - '0'] > 1) {
				flag = false;
				break;
			}
		}
		if (flag) {
			cout << n << endl;
			break;
		}
	}
	return 0;
}
