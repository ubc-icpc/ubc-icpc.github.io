#include<bits/stdtr1c++.h>
using namespace std;
int main() {
	int w, n; cin >> w >> n;
	deque < int > rail;
	int rw = 0;
	int car = 0;
	for (int i = 0; i < n; ++i) {
		int gar; cin >> gar;
		if ((rw + gar) <= w) {
			rail.push_back(gar);
			rw += gar;
			++car;
		}
		else {
			cout << car << endl;
			return 0;
		}
		if (rail.size() == 4) {
			rw -= rail[0];
			rail.pop_front();
		}
	}
	cout << car << endl;
	return 0;
}
