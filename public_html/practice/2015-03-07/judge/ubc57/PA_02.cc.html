// Triangulate This.cpp : Defines the entry point for the console application.
//


#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	long long Answers[1000];
	for (int i=1;i<=n;i++){
		long long a,b;
		cin>>a>>b;
		Answers[i-1]=(a/b)*(a/b);
	}
	for (int j=0;j<n;j++){
		cout<<Answers[j]<<endl;
	}
	return 0;
}
