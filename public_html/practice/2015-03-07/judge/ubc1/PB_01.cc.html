#include <bits/stdtr1c++.h>

using namespace std;

typedef pair<string, string> pss;
vector<pss> names[100000];
int main() {
    ios::sync_with_stdio(0);
    int n; cin >> n;
    vector<pss> nm(n);
    for (int i = 0; i < n; i++) {
        cin >> nm[i].second >> nm[i].first;
    }
    sort(nm.begin(), nm.end());
    
    map<pss, int> mp;
    for (int i = 0; i < n/3; i++) {
        names[i].resize(3);
        for (int j = 0; j < 3; j++) {
            names[i][j] = nm[3*i+j];
            mp[names[i][j]] = i;
        }
        sort(names[i].begin(), names[i].end());
    }
    
    int q; cin >> q;
    while (q--) {
        pss p;
        cin >> p.second >> p.first;
        int idx = mp[p];
        for (int j = 0; j < 3; j++) {
            if (names[idx][j] == p) continue;
            cout << names[idx][j].second << " " << names[idx][j].first << '\n';
        }
    }
	return 0;
}
