#include<bits/stdc++.h>
#define pb   push_back
using namespace std;
typedef long long ll;
typedef long double ld;

ll cnt[4444];
ll cnt2[4444];

int main(){
  ios_base::sync_with_stdio(false);cin.tie(0);
  int a,n;
  cin >> n;
  for(int i=0;i<n;i++){
    cin>>a;
    cnt[a]++;
  }
  for(int i=1;i<2001;i++){
    cnt2[i+i] += (cnt[i])/2;
    //if (cnt[i]) cout << cnt[i]*(cnt[i]-1)/2 <<endl;
    for(int j=i+1;j<2001;j++){
      cnt2[i+j]+=min(cnt[i],cnt[j]);
    }
  }
  ll mx = 0;
  for(int i=0;i<4001;i++){
    //if (cnt2[i]) cout <<"HAS " << cnt2[i] << " " << i <<endl;
    mx = max(mx,cnt2[i]);
  }
  ll ans = 0;
  for(int i=0;i<4001;i++){
    ans+=(mx==cnt2[i]);
  }
  cout<<mx<< " " <<ans;
}
