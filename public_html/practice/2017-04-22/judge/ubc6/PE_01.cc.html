#include<bits/stdc++.h>
#define pb   push_back
using namespace std;
typedef long long ll;
typedef long double ld;

int m[333];

int main(){
  ios_base::sync_with_stdio(false);cin.tie(0);
  int N;
  cin >> N;
  for(int i=0;i<N;i++){
    cin>>m[i];
  }
  sort(m,m+N);
  int cen = (N+1)/2-1;
  for(int i=0;i<=N/2+1;i++){
    if (cen+i<N) cout<<m[cen+i]<<" ";
    if(i!=0 && cen-i>=0) cout<<m[cen-i]<<" ";
  }
}
