#include<bits/stdc++.h>
#define pb   push_back
using namespace std;
typedef long long ll;
typedef long double ld;

int n;
int sw[111111];

int main(){
  ios_base::sync_with_stdio(false);cin.tie(0);
  cin >> n;
  for(int i=0;i<n;i++){
    cin >> sw[i];
    if(i!=0) sw[i]+=sw[i-1];
  }
  ll fir = -1;
  ll c = 0, cs=0;
  for(int i=0;i<n;i++){
    cin >> c;
    cs+=c;
    if (sw[i]==cs) fir = i;
  }
  cout << fir+1 <<endl;
}
