#define _USE_MATH_DEFINES
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
	ios::sync_with_stdio(0); cin.tie(0);

	int num, temp;
	cin >> num;
	int height[2001], ans[4001]; 
	memset(height, 0, sizeof(height));
	memset(ans, 0, sizeof(ans));
	for (int i = 0; i < num; i++) {
		cin >> temp;
		height[temp]++;
	}
	for (int i = 1; i <= 2000; i++) {
		for (int j = 1; j < i; j++)
			ans[i+j] += min(height[i], height[j]);
		ans[i+i] += height[i]/2;
	}
	int diffheight, maxlen;
	diffheight = maxlen = 0;
	for (int i = 1; i <= 4000; i++) {
		if (ans[i] > maxlen) {
			maxlen = ans[i];
			diffheight = 0;
		}
		if (ans[i] == maxlen) {
			diffheight++;
		}
	}
	cout << maxlen << " " << diffheight << endl;

	return 0;
}
