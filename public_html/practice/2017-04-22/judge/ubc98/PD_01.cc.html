#include <iostream>
#include <queue>
using namespace std;

int main()
{
	long N;
	cin >> N;
	queue<long> swifts;
	for (long i = 0; i < N; ++i) {
		int r;
		cin >> r;
		swifts.push(r);
	}
	long K = 0;
	long sum = 0;
	for (long i = 0; i < N; ++i) {
		long r;
		cin >> r;
		sum += r - swifts.front();
		swifts.pop();
		if (sum == 0) {
			K = i + 1;
		}
	}
	cout << K << endl;
}
