#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int N;
	cin >> N;
	vector<int> tides;
	for (int i = 0; i < N; ++i) {
		int k;
		cin >> k;
		tides.push_back(k);
	}
	sort(tides.begin(), tides.end());
	for (int i = 0; i < N; ++i) {
		int k = i/2;
		bool high = i % 2 == 1;
		if (high) {
			int a = 0;
			if (N % 2 == 1) {
				a = 1;
			}
			cout << tides.at(N/2+k+a) << " ";
		}
		else {
			cout << tides.at((N-1)/2 - k) << " ";;
		}
	}
}
