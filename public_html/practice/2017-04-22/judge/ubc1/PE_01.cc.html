#include <bits/stdtr1c++.h>
using namespace std;

int main(){

  ios_base::sync_with_stdio(false);
  cin.tie(0);

  int n;
  int v[105];
  int ans[105];

  cin >> n;
  for(int i = 1;i <= n;i++) cin >> v[i];

  sort(v+1, v+n+1);
  
  int half = (n+1)/2;
  reverse(v+1, v+half+1);

  int li = 1, lj = half+1;
  
  for(int i = 1;i <= n;i++){
    if(i%2) ans[i] = v[li++];
    else ans[i] = v[lj++];
  }

  for(int i = 1;i <= n;i++) cout << ans[i] << (i==n?"\n":" ");

  return 0;
}
