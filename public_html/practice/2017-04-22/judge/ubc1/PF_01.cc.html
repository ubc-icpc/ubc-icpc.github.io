#include <bits/stdtr1c++.h>
using namespace std;

//-------------------
#define MAXN 2020

int n;
int l[2*MAXN];
int bestw[2*MAXN];
//-------------------

int main(){

  ios_base::sync_with_stdio(false);
  cin.tie(0);

  cin >> n;
  for(int i = 1;i <= n;i++){
    int x;
    cin >> x;
    l[x]++;
  }

  for(int h = 1;h <= 4000;h++){
    for(int i = 1;2*i <= h;i++){
      assert(h-i >= 0);
      assert(h-i < 2*MAXN);
      assert(i >= 0);
      assert(i < 2*MAXN);
      if(2*i == h) bestw[h] += (l[i]/2);
      else bestw[h] += min(l[i], l[h-i]);
    }
  }

  int answ = -1, anst = 0;
  for(int i = 1;i <= 4000;i++){
    if(bestw[i] > answ){
      answ = bestw[i];
      anst = 0;
    }
    if(bestw[i] == answ) anst++;
  }

  cout << answ << " " << anst << endl;

  return 0;
}
