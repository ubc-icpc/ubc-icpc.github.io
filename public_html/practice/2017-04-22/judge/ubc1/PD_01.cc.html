#include <bits/stdtr1c++.h>
using namespace std;

#define MAXN 100100

int n;
int a[MAXN];
int b[MAXN];

int main(){

  ios_base::sync_with_stdio(false);
  cin.tie(0);

  cin >> n;

  for(int i = 1;i <= n;i++){
    cin >> a[i];
    a[i] += a[i-1];
  }

  for(int i = 1;i <= n;i++){
    cin >> b[i];
    b[i] += b[i-1];
  }

  int ans = 0;
  for(int i = 1;i <= n;i++) if(a[i] == b[i]) ans = i;

  cout << ans << endl;

  return 0;
}
