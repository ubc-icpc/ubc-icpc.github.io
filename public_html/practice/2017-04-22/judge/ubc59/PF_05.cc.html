#include <bits/stdc++.h>
#define endl '\n'
using namespace std;
typedef long long ll;
typedef long double ld;
typedef complex<ld> pt;
const int MOD = 1e9 + 7;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	vector<int> count(2001);
	vector<int> ans(4001);
	int N; cin >> N;
	vector<int> in(N);
	for (int i = 0; i < N; i++) {
		cin >> in[i];
		count[in[i]]++;
	}
	for (int i = 0; i < 2001; i++) {
		for (int j = i+1; j < 2001; j++) {
			ans[i + j] += min(count[i], count[j]);
		}
	}
	for (int i = 0; i < 2001; i++) {
		ans[2*i] += count[i] / 2;
	}
	int best = 0; set<int> bestSet;
	for (int i = 0; i < 4001; i++) {
		if (ans[i] > best) {
			bestSet.clear();
			best = ans[i];
			bestSet.insert(i);
		} else if (ans[i] == best)
			bestSet.insert(i);
	}
	cout << best << " " << bestSet.size() << endl;
	return 0;
}
