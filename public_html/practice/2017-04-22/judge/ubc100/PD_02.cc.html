#include<iostream>
#include<vector>

using namespace std;

int main(void) {
    int n; cin >> n;
    vector<int> a,b;
    for(int i=0; i<n; i++) {
        int x; cin >> x;
        a.push_back(x);
    }
    for(int i=0; i<n; i++) {
        int x; cin >> x;
        b.push_back(x);
    }
    int ans = 0, sa=0, sb=0;
    for(int i=0; i<n; i++) {
       sa += a[i];
       sb += b[i];
       if(sa == sb) ans = i+1; 
    }
    cout << ans << endl;;
    return 0;
}
