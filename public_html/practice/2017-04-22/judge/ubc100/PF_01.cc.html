#include<iostream>
#include<vector>
#include<cstring>
#include<algorithm>

using namespace std;

int arr[2001];

int main(void) {
    memset(arr, 0, sizeof arr);
    int n; cin >> n;
    for(int i=0; i<n; i++) {
        int x; cin >> x;
        arr[x]++;
    }
    int ans = 0;
    int sz = 0;
    for(int i=1; i<4001; i++) {//i==h of board
        int len = 0;
        //first wood is j. second is i-j
        for(int j=1; j<=(i-1)/2; j++) {
            if(i-j>=1 && i-j<=2000 && j>=1 && j<=2000) {
                len += min(arr[j], arr[i-j]);
            }
            
        }
        if(i%2 == 0) {
            len += arr[i/2]/2;
        }

        if(len>ans) {
            ans = len;
            sz = 1;
        } else if(len == ans) {
            sz++;
        }
    }
    cout << ans << " " << sz << endl;
    return 0;
}
