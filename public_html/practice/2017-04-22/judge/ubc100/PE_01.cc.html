#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main(void) {
    int n; cin >> n;
    vector<int> vec;
    for(int i=0; i<n; i++) {
        int x; cin >> x; vec.push_back(x);
    }
    sort(vec.begin(), vec.end());
    int a=(n+1)/2-1, b=((n+1)/2);
    while(a>=0 && b<n) {
        cout << vec[a] << " " << vec[b] << " ";
        a--;
        b++;
    }while(a>=0 ) {
        cout << vec[a] << " "; 
        a--;
    }
    cout << endl;
}
