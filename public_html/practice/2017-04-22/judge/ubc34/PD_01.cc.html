#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

#define debug printf
/* #define debug(...) */
typedef long long ll;

int A[2][100000];

int main() {
    int N; scanf("%d", &N);
    for (int i = 0; i < N; i++)
        scanf("%d", &A[0][i]);
    for (int i = 0; i < N; i++)
        scanf("%d", &A[1][i]);

    int ans = 0;
    int cur[2] = {0};
    for (int i = 0; i < N; i++) {
        cur[0] += A[0][i];
        cur[1] += A[1][i];
        if (cur[0] == cur[1]) ans = i+1;
    }
    printf("%d\n", ans);
    return 0;
}
