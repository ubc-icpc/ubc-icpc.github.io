#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

#define debug printf
/* #define debug(...) */
typedef long long ll;

int A[100];

int main() {
    int N; scanf("%d", &N);
    for (int i = 0; i < N; i++) {
        scanf("%d", &A[i]);
    }
    sort(A, A+N);
    int total = 0;
    for (int j = (N-1)/2, i = (N+1)/2; j >= 0; j--, i++) {
        printf("%d", A[j]);
        if (++total != N) printf(" ");
        if (i < N) {
            printf("%d", A[i]);
            if (++total != N) printf(" ");
        }
    }
    printf("\n");
    return 0;
}
