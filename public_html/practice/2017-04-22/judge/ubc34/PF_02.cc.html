#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
using namespace std;

#define debug printf
/* #define debug(...) */
typedef long long ll;

int H[2001];

int main() {
    int N; scanf("%d", &N);
    for (int i = 0; i < N; i++) {
        int a; scanf("%d", &a);
        H[a] += 1;
    }
    int best_len = 0;
    set<int> best_heights;
    for (int h = 2; h <= 4000; h++) {
        int len = 0;
        for (int i = 1; i < h-i; i++)
            if (h-i <= 2000) len += min(H[i], H[h-i]);
        if (h % 2 == 0)
            len += H[h/2] / 2;

        if (len > best_len)
            best_len = len, best_heights = {h};
        else if (len == best_len)
            best_heights.insert(h);
    }
    printf("%d %lu\n", best_len, best_heights.size());
    return 0;
}
