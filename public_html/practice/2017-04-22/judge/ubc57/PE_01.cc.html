#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll A[105];
int main(){
    ios::sync_with_stdio(0);
    int n; cin>>n;
    for (int i=0;i<n;i++) cin>>A[i];
    sort(A,A+n);
    int m=(n-1)/2;
    cout<<A[m];
    for (int i=1;i<=n;i++){
        if ((0<=(m+i))&&((m+i)<n)) cout<<' '<<A[m+i];
        if ((0<=(m-i))&&((m-i)<n)) cout<<' '<<A[m-i];
    } cout<<endl;
    return 0;
}
