#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll freq[2005];
int main(){
    ios::sync_with_stdio(0);
    int n; cin>>n; 
    for (int i=1;i<=n;i++){
        int x; cin>>x;
        freq[x]++;
    }
    multiset<int> ans;
    for (int l=2;l<=4000;l++){
        ll cnt=0;
        for (int i=1;i<=2000;i++){
            int j=l-i;
            if ((j<=0)||(j>2000)) continue;
            if (i==j) cnt+=int(freq[i]/2);
            if (i<j) cnt+=min(freq[i],freq[j]);
        }
        //if (cnt) cout<<l<<endl;
        ans.insert(cnt);
    }
    ll res=*(--ans.end());
    cout<<res<<' '<<ans.count(res)<<endl;
    return 0;
}
