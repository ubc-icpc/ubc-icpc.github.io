#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll A[100005],B[100005];
int main(){
    int n; cin>>n;
    for (int i=1;i<=n;i++) cin>>A[i];
    for (int i=1;i<=n;i++) cin>>B[i];
    ll aa=0,bb=0;
	int ans=0;
    for (int i=1;i<=n;i++){
        aa+=A[i],bb+=B[i];
        if (aa==bb){
            ans=i;
        }
    }
    cout<<ans<<endl; return 0;
}
