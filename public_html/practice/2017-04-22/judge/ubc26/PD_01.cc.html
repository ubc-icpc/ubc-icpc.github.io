#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, e;
    cin >> n;
    vector<int> a;
    vector<int> b;
    for (int i = 0; i < n; i++) {
        cin >> e;
        a.push_back(e);
    }
    for (int i = 0; i < n; i++) {
        cin >> e;
        b.push_back(e);
    }

    int r = 0;
    int suma = 0;
    int sumb = 0;
    for (int i = 0; i < n; i++) {
        suma += a[i];
        sumb += b[i];
        if (suma == sumb) {
            r = max(r, i + 1);
        }
    }
    cout << r;
    
    return 0;
}
