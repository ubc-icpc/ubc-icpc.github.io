#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, e;
    cin >> n;
    vector<int> v;
    vector<int> w;
    vector<int> u;
    for (int i = 0; i < n; i++) {
        cin >> e;
        v.push_back(e);
    }
    sort(v.begin(), v.end());
    for (int i = n / 2; i < n; i++) {
        cout << v[n - i - 1] << ' ';
        if (n & 1) {
            if (i + 1 < n) {
                cout << v[i + 1] << ' ';
            }
        } else {
            cout << v[i] << ' ';
        }
    }
    return 0;
}
