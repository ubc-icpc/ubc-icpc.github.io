#include <bits/stdc++.h>

#define MOD 1000000007

using namespace std;
typedef long long int lli;

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0);
	int N, v;
	cin >> N;
	vector<int> vals(N);
	for(int i=0;i<N;++i)
	{
		cin >> v;
		vals[i] = v;
	}
	sort(vals.begin(), vals.end());
	int maxLow = (N+1)/2-1;
	int minHigh = maxLow+1;
	int low = maxLow;
	int high = minHigh;
	for(int i=0;i<N;++i)
	{
		if(i%2 == 0)
		{
			cout << vals[low--] << " ";
		}
		else
		{
			cout << vals[high++] << " ";
		}
	}
	cout << endl;
	
	return 0;
}
