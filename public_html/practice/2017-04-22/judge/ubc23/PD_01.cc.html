#include <bits/stdc++.h>

#define MOD 1000000007

using namespace std;
typedef long long int lli;

lli A[100000], B[100000], As[100000], Bs[100000];

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0);
	int D;
	cin >> D;
	lli sum = 0;
	for(int i=0;i<D;++i)
	{
		cin >> A[i];
		sum += A[i];
		As[i] = sum;
	}
	sum = 0;
	for(int i=0;i<D;++i)
	{
		cin >> B[i];
		sum += B[i];
		Bs[i] = sum;
	}
	bool found = false;
	for(int i=D-1;i>=0;--i)
	{
		if(As[i] == Bs[i])
		{
			found = true;
			cout << i+1 << endl;
			break;
		}
	}
	if(found == false)
		cout << 0 << endl;
	
	
	return 0;
}
