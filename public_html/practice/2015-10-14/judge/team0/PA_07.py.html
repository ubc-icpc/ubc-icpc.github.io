import math
import sys

for line in sys.stdin:
	if len(line.strip()) == 0:
		break
		
	stuff = line.split()
	a = float(stuff[0])
	b = float(stuff[1])
	c = float(stuff[2])
	d = float(stuff[3])
	m = float(stuff[4])
	t = float(stuff[5])


	epsilon = 0.0000001
	v = 1e15

	while True:
		v1 = v - ((a*v**3 + b*v**2 + c*v + d)*m - t)/((3*a*v**2 + 2*b*v + c)*m)
		if abs(v1 - v) < epsilon:
			break
		v = v1

	v = 3.1415

	while True:
		v2 = v - ((a*v**3 + b*v**2 + c*v + d)*m - t)/((3*a*v**2 + 2*b*v + c)*m)
		if abs(v1 - v) < epsilon:
			break
		v = v2

	print '%.2f'%(int(max(v1, v2)*100)/100.0)
