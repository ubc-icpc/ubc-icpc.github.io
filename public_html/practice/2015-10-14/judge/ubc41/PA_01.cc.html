#include <bits/stdc++.h>
using namespace std;
using ull = unsigned long long;
using ld = long double;

double a, b, c, d, m, t;

ld r(double v)
{
	return m * (a * v * v * v + b * v * v + c * v + d) - t;
}
ld ss(ld lo, ld hi)
{
    while (hi - lo > 1E-8)
    {
        ld mid = (hi + lo) / 2;
        if (r(mid) > 0) hi = mid;
        else lo = mid;
    }
    return (hi + lo) / 2;
}

int main()
{
    while (cin >> a >> b >> c >> d >> m >> t)
    {
        ld n = ss(0, 1000000);
		printf("%.2f\n", ((int) (100 * n)) / 100.0);
    }
}
