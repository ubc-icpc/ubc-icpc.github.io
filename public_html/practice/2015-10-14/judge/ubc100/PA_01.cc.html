#include<iostream>
#include<limits>
#include<cmath>
#include<iomanip>

#define ld long double


using namespace std;
ld a,b,c,d,m,t;

ld bs(ld lo, ld hi, ld mid) {
  ld temp=a*pow(mid,3)+b*pow(mid,2)+c*mid+d;
  if( abs(temp-t/m) < 0.0000001) return mid;
  if( temp < t/m )
    return bs(mid,hi,(hi+mid)/2);
  return bs(lo,mid,(mid+lo)/2);

}

int main (void) {
  while(cin>>a>>b>>c>>d>>m>>t) {
  ld v1=0,v2=numeric_limits<ld>::max(),v=v1+v2/2;
  //cout << v1 <<endl<<v2<<endl<<v<<endl;
  cout << fixed << setprecision(2) << ((int)(bs( v1, v2, v)*100))/double(100) << endl;
  }
}
