#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N=200005;
bool visited[N]; 
int gone[N];
vector<int> G[N];
int main(){
    ios::sync_with_stdio(0);
    int n,m,home,s; cin>>n>>m>>home>>s; 
    while (m--){
        int a,b;
        cin>>a>>b; G[a].push_back(b); G[b].push_back(a);
    }
    queue<int> q; q.push(s); visited[s]=true;
    while (!q.empty()){
        int now=q.front(); q.pop();
        //cout<<now<<endl;
        for (int nex:G[now]) if (!visited[nex]) {
            gone[nex]++;
            if ((gone[nex]*2)>=G[nex].size()){
                q.push(nex); visited[nex]=true;
            }
        }
    }
    if (visited[home]) cout<<"leave"<<endl;
    else cout<<"stay"<<endl;
    return 0;
}
