#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    ll n; cin>>n; 
    ll prime=-1;
    set<ll> s;
    while (n>1){
        bool isPrime=true; 
        for (ll i=2;i*i<=n;i++) if (n%i==0){
            s.insert(i); n/=i; isPrime=false; break;
        }
        if (isPrime){
            s.insert(n); break;
        }
    }
    if (s.size()==1) cout<<"yes"<<endl;
    else cout<<"no"<<endl;
    return 0;
}
