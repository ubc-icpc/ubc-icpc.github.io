#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    string s; cin>>s;
    string ans=""; 
    int i=0;
    while (i<s.size()){
        bool done=false;
        if ((i+2)<s.size()){
            set<char> ss; 
            ss.insert(s[i]),ss.insert(s[i+1]),ss.insert(s[i+2]);
            if (ss.size()==3) {
                ans+='C', i+=3; done=true;
            }
        }
        if (!done){
            if (s[i]=='R') ans+='S';
            if (s[i]=='B') ans+='K';
            if (s[i]=='L') ans+='H';
            i++;
        }
    }
    cout<<ans<<endl;
    return 0;
}
