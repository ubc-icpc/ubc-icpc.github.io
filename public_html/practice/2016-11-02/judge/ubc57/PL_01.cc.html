#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll A[20005];
int main(){
    ios::sync_with_stdio(0);
    int n; cin>>n;
    for (int i=1;i<=n;i++) cin>>A[i];
    sort(A+1,A+n+1);
    bool good=false;
    for (int i=1;i<=n;i++) {
        for (int j=i+1;j<n;j++) if (A[i]+A[j]>A[j+1]){
            good=true; break;
        }
        if (good) break;
    }
    if (good) cout<<"possible"<<endl;
    else cout<<"impossible"<<endl;
    return 0;
}
