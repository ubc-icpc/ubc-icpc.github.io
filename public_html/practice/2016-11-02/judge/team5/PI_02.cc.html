#include<bits/stdtr1c++.h>

using namespace std;
vector<int> primes;
bool check[(int)sqrt(1e9) + 1];
void sieve() {
  for(int i = 2; i <= sqrt(1e9) + 1; i++) {
    if(!check[i]) {
      primes.push_back(i);
      for(int j = i*i; j <=sqrt(1e9) + 1; j+=i) check[j] = true;
    }

  }

}
int main() {
  int q; cin >> q;
  bool isa = false;
  sieve();
  if(q == 1) {
      cout << "no" << endl;
      return 0;
  }
  for(auto p : primes) {
    if(q%p == 0) {
      while(q%p == 0) q = q/p;
      isa = true;
      break;
    }

  }
  if(q != 1) {
    if(!isa) cout << "yes" <<endl;
    else cout << "no" << endl;
  }
  else cout << "yes" << endl;

}
