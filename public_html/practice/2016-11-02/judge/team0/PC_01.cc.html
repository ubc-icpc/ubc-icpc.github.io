#include<bits/stdtr1c++.h>
using namespace std;

int C, P, X, L;
vector<vector<int> > A;
vector<int> cnt;
vector<bool> out;

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> C >> P >> X >> L;
  --X; --L;
  A.resize(C);
  cnt.resize(C);
  out.resize(C);

  while (P--) {
    int a, b; cin >> a >> b; --a; --b;
    A[a].push_back(b);
    A[b].push_back(a);
  }

  stack<int> q;
  q.push(L);
  while (!q.empty()) {
    int top = q.top();
    q.pop();
    if (out[top]) continue;
    out[top] = true;
    for (int i : A[top]) {
      if (++cnt[i] >= (A[i].size()+1) / 2)
        q.push(i);
    }
  }

  cout << (out[X] ? "leave" : "stay") << endl;

  return 0;
}
