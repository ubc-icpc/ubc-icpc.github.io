#include<bits/stdtr1c++.h>
using namespace std;

int V;

bool solve() {
  if (V < 2) return false;
  int p = 0;
  for (int i = 2; i <= sqrt(V) + 1; ++i)
    if (V % i == 0) {
      p = i;
      break;
    }
  if (p == 0) return true;
  while (V % p == 0)
    V /= p;
  return (V == 1);
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> V;
  cout << (solve() ? "yes" : "no") << endl;

  return 0;
}
