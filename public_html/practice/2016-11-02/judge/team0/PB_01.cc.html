#include<bits/stdtr1c++.h>
using namespace std;

string S;

bool combo(int i) {
  int r, b, l;
  r = b = l = 0;
  int end = i + 3;
  if (end > (int)S.size()) return false;
  for (; i < end; ++i) {
    if (S[i] == 'R') r++;
    else if (S[i] == 'B') b++;
    else if (S[i] == 'L') l++;
  }
  return (r == 1 && b == 1 && l == 1);
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> S;
  for (int i = 0; i < (int)S.size(); ++i) {
    if (combo(i)) {
      i += 2;
      cout << 'C';
    } else if (S[i] == 'R')
      cout << 'S';
    else if (S[i] == 'B')
      cout << 'K';
    else if (S[i] == 'L')
      cout << 'H';
  }
  cout << endl;

  return 0;
}
