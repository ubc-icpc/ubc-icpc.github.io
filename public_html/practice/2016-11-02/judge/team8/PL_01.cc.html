#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
	ll L;
	cin >> L;

	std::vector<ll> a(L, 0);
	for (ll i = 0; i < L; i++) {
		cin >> a[i]; 
	}

	std::sort(a.begin(), a.end());

	for (ll i = a.size()-1; i >= 2; i--) {
		ll c1 = a[i];
		ll c2 = a[i-1];
		ll c3 = a[i-2];

		if (c1 + c3 > c2 && c2 + c3 > c1 && c1 + c2 > c3) {
			cout << "possible" << endl;
			return 0;
		}
	}

	cout << "impossible" << endl;
	return 0;
}
