import sys
import math
input = sys.stdin

moves = list(input.readline().strip())

mapping = {
	"R": "S",
	"B": "K",
	"L": "H"
}

s = 0
i = min(3, len(moves))
out = ""
while True:
	seg = moves[s:i]
	
	m = {'R': False, 'B': False, 'L': False}
	for c in seg:
		m[c] = True

	if m['R'] and m['L'] and m['B']:
		out += "C"
		s = min(s + 3, len(moves))
		i = min(i + 3, len(moves))
	else:
		out += mapping[seg[0]]
		s = min(s + 1, len(moves))
		i = min(i + 1, len(moves))

	if s == i:
		break

print(out)
