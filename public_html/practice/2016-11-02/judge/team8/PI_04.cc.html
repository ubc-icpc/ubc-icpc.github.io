#include <bits/stdc++.h>
using namespace std;

typedef unsigned long ul;

bool prime_check(ul N) {
	if (N < 2) return false;
	if (N == 2) return true;
	if (N % 2 == 0) return false;
	ul rootN = floor(sqrt(N));
	for (int i = 3; i <= rootN; i += 2) {
		if (N % i == 0) return false;
	}

	return true;
}

int main() {
	ul N;
	cin >> N;
	if (prime_check(N)) {
		cout << "yes" << endl;
		return 0;
	}
	for (int i = 2; i < floor(sqrt(N)) + 1; i++) {
		if (prime_check(i)) {
			for (int p = 2; ; p++) {
				if (pow(i, p) > N) break;
				if (pow(i, p) == N) {
					cout << "yes" << endl;
					return 0;
				}
			}
		}
	}
	cout << "no" << endl;
}
