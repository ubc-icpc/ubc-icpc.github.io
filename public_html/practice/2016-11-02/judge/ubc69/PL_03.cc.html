#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main() {
	int n; cin >> n;
	long long int len[n];
	for(int i = 0; i < n; i++) {
		cin >> len[i];
	}
	sort(len, len+n);
	for(int i = 0; i < n - 2; i++) {
		for(int j = i + 1; j < n - 1; j++) {
			if(len[i] + len[j] > len[j+1]) {
				//cout << len[i] << ' ' << len[j] << endl;
				cout << "possible" << endl;
				return 0;
			}
		}
	}
	cout << "impossible" << endl;
	return 0;
}
