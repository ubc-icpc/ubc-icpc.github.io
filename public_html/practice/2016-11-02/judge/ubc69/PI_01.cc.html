#include <iostream>

using namespace std;

int divisor(int n) {
	for(int i = 2; i*i <= n; i++) {
		if(n%i == 0) {
			return i;
		}
	}
	return -1;
}

bool pp(int q, int d) {
	while(q > 1) {
		//cout << q << endl;
		if(q % d != 0) {
			return false;		
		}
		q /= d;
	}
	return true;
}

int main() {
	int q; cin >> q;
	if(q == 1) {
		cout << "no" << endl;
		return 0;	
	}
	int d = divisor(q);
	if(d == -1) {
		cout << "yes" << endl;	
	}
	else {
		if(pp(q, d))
			cout << "yes" << endl;
		else
			cout << "no" << endl;
	}
	return 0;
}
