#include <iostream>
#include <string>

using namespace std;

int main() {
	string s; cin >> s;
	s+="XX";
	for(int i = 2; i < s.size(); i++) {
		if(s[i] != 'X' && s[i] != s[i-1] && s[i] != s[i-2] && s[i-2] != s[i-1]) {
			cout << 'C';
			i += 2;
		}
		else {
			if(s[i-2] == 'R') {
				cout << 'S';
			}
			else if(s[i-2] == 'B') {
				cout << 'K';
			}
			else if(s[i-2] == 'L') {
				cout << 'H';
			}
		}
	}
	cout << endl;
	//if()
	return 0;
}
