#include <vector>
#include <set>
#include <iostream>
#include <cmath>

using namespace std;
set<int> good;
vector<bool> primes(1e5, 1);
int q;

void pre() {
  good.insert(2);
  int cur, curPrime;
  for (int i = 2; i <= sqrt(q); i++) {
    if (primes[i]) {
      for (int j = i*i; j <= sqrt(q); j+= i) {
        primes[j] = 0;
      }
      cur = i; curPrime = i;
      while (cur <= q) {
        good.insert(cur);
        cur *= curPrime;
      }
    }
  }
}

bool isPrime() {
  if ( q < 2) return false;
  for (int i = 2; i <= sqrt(q); i++) {
    if (q % i == 0) return false;
  }
  return true;
}
      

int main() {
  cin >> q;
  pre();
  //for(int e : good) cout << e << endl;
  if(isPrime() || good.count(q)) {
    cout << "yes" << endl;
  } else {
    cout << "no" << endl;
  }
}
