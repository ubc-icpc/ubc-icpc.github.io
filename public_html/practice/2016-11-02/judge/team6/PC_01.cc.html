#include <iostream>
#include <vector>
#include <queue>
#include <set>
using namespace std;

int main() {
	long c,p,x,l;cin>>c>>p>>x>>l;
	vector<set<long>> graph(c + 1, set<long>()); vector<long> m(c+1,0);
	while(p--){
		long a,b;cin>>a>>b;
		graph[a].insert(b);
		graph[b].insert(a);
		m[a]++;
		m[b]++;
	}

	queue<long> q;
	q.push(l);
	bool stay = true;
	while(!q.empty()) {
		long a = q.front();
		if (a == x) {
			stay = false;
			break;
		}
		q.pop();
		for (long b : graph[a]) {
			graph[b].erase(a);
			if (graph[b].size() <= m[b]/2)
				q.push(b);
		}
	}

	if (stay)
		cout << "stay" << endl;
	else 
		cout << "leave" << endl;
}
