#include <bits/stdc++.h>
#define endl '\n'
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int N; cin >> N;
	vector<long long> sticks;
	for (int i = 0; i < N; i++) {
		long long a; cin >> a;
		sticks.push_back(a);
	}
	sort(sticks.begin(), sticks.end());
	for (int i = 0; i < N-2; i++) {
		for (int j = i+1; j < N-1; j++) {
			long long sum = sticks[i] + sticks[j];
//			cout << sticks[j+1] << " ";
			if (sum > sticks[j+1]) {
				cout << "possible";
				return 0;
			}
		}
	}
	cout << "impossible";
	return 0;
}
