#include <bits/stdc++.h>
#include <string>
#include <cmath>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	int t; cin >> t;
	bool possible = false;
	vector<long long> A; long long in;
	for (int i=0; i<t; i++) {
		cin >> in;
		A.push_back(in);
	}
	sort(A.begin(), A.end());
	for (int i=0; i<t-2; i++) {
		if (A[i] + A[i+1] > A[i+2]) {
			possible = true;
			break;
		}
	}
	if (possible) cout << "possible" << endl;
	else cout << "impossible" << endl;

	return 0;
}
