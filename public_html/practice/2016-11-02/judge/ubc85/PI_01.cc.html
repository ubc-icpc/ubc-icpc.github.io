#include <bits/stdc++.h>
#include <string>
#include <cmath>

using namespace std;

int main() {
	int q; cin >> q;
	if (q == 1) {
		cout << "no" << endl;
		return 0;
	}
	int p = q; 

	for (int i=2; i<=floor(sqrt(q)); i++) {
		if (q % i == 0) {
			p = i;
			break;
		}
	}
	while (q % p == 0) {
		q /= p;
	}
	if (q == 1) cout << "yes" << endl;
	else cout << "no" << endl; 
	

	return 0;
}
