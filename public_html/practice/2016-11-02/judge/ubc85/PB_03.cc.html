#include <bits/stdc++.h>
#include <string>

using namespace std;

int main() {
	string s; cin >> s;
	string ret = "";
	int len = s.length();
	for (int i=0; i<len; i++) {
		if (i < len - 2 && s[i] != s[i+1] && s[i] != s[i+2] && s[i+1] != s[i+2]) {
				ret += "C";
				i += 2;
		}
		else if (s[i] == 'R') ret += "S";
		else if (s[i] == 'B') ret += "K";
		else if (s[i] == 'L') ret += "H";
	}
			
	cout << ret << endl;
	return 0;
}
