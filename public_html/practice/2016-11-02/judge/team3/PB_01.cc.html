#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;
typedef vector<int> vi;

#define endl '\n'

int main() {
	ios::sync_with_stdio(0);
	string a; cin >> a;
	int i=0;
	while (i<a.size()) {
		if (a.size() - i > 2) {
			if ( (a[i] == 'R' || a[i+1] == 'R' || a[i+2] == 'R') &&
			(a[i] == 'B' || a[i+1] == 'B' || a[i+2] == 'B') &&
			(a[i] == 'L' || a[i+1] == 'L' || a[i+2] == 'L') ) {
				i += 3;
				cout << "C";
				continue;
			}
		}
		if (a[i] == 'R') cout << "S";
		if (a[i] == 'B') cout << "K";
		if (a[i] == 'L') cout << "H";
		i++;
	}
	cout << endl;
	return 0;
}
