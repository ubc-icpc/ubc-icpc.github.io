#include <bits/stdc++.h>
using namespace std;
using ll = long long;
const ll int_max = numeric_limits<ll>::max();

int main()
{
	ll n; cin >> n;
	vector<ll> e(n);
	for (ll i = 0; i < n; i++)
	{
		cin >> e[i];
	}

	sort(e.begin(), e.end());
	for (ll i = 1; i < n; i++)
	{
		ll diff = e[i] - e[i-1];
		ll sum = e[i] + e[i-1];

		ll ei = e[i];
		ll eim1 = e[i-1];

		e[i] = int_max;
		e[i-1] = int_max;
		
		if (*upper_bound(e.begin(), e.end(), diff) < sum)
		{
			// cout << "a = " << ei << " b = " << eim1 << " c = " << *lower_bound(e.begin(), e.end(), diff) << endl;
			cout << "possible" << endl;
			return 0;
		}

		e[i] = ei;
		e[i-1] = eim1;
	}
	cout << "impossible" << endl;
}
