#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

bool isprime(ll x) {
    if (x<=1) return false;
    if (x<=3) return true;
    if (!(x%2) || !(x%3)) return false;
    ll limit = (ll)(sqrt((double)x)+1e-7);
    for (ll i=5; i<=limit; i+=6) {
        if (!(x%i) || !(x%(i+2))) return false;
    }
    return true;
}

int main() {
  int i; cin >> i;
  if (isprime(i)) {
    cout << "yes" << endl;
    return 0;
  }
  int j = 2;
  if (i % j == 0) {
    i /= j;
    while (i > 1) {
      if (i % j != 0) {
        cout << "no" << endl;
        return 0;
      }
      i /= j;
    }
    cout << "yes" << endl;
    return 0;
  }
  j++;
  while (j <= 31623) {
    if (i % j == 0) {
      i /= j;
      while (i > 1) {
        if (i % j != 0) {
          cout << "no" << endl;
          return 0;
        }
        i /= j;
      }
      cout << "yes" << endl;
      return 0;
    }
    j += 2;
  }
  cout << "no" << endl;
  return 0;
}
