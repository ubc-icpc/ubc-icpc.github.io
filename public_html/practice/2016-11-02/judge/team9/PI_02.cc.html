#include <iostream>
#include <algorithm> 
#include <utility>
#include <math.h>
#include <cmath>
#include <climits>
#include <vector>
#include <set>
#include<string>
#include <tuple>
#include <iostream>
#include <string>


using namespace std;

int main() {
	ios::sync_with_stdio(0);

	long long q;
	cin >> q;

	if (q == 1) {
		cout << "no" << endl;
		return 0;
	}

	set<long long> s;
	
	while (q > 1) {
		bool isPrime = true;
		for (int i = 2; i <= sqrt(q); i++) {
			if (q%i == 0) {
				s.insert(i);
				q /= i; // update q
				isPrime = false;
				break;
			}
		}
		if (isPrime) {
			s.insert(q);
			break;
		}
	}

	if (s.size() == 1) {
		cout << "yes" << endl;
	}
	else {
		cout << "no" << endl;
	}

	return 0;
}
