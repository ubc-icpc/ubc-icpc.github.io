#include <iostream>
#include <iomanip>

using namespace std;

double f (int, int, double);

int main()
{
   int cases;
   cin >> cases;
   for (int a=0;a<cases;a++)
   {
      double p, win;
      cin >> p;

      win=f(0,0,p);
      cout << fixed << setprecision(5) << win << endl; 
   }
}

double f (int i, int j, double p)
{
   if ((i<4&&j<4)&&!(i==3&&j==3))
   {
      return (p*f(i+1,j,p)+(1-p)*f(i,j+1,p));
   }
   if (i==4&&j<3)
      return 1;
   if (i<3&&j==4)
      return 0;
   return (p*p)/(1-(2*p)*(1-p));
}
