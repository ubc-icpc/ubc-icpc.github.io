#include <iostream>
using namespace std;

void sort (int[], int); //smallest to largest

int main()
{
   int tests;
   cin >> tests;

   for (int i=0;i<tests;i++)
   {
      int numBars, bars[100000];
      cin >> numBars;

      for (int j=0;j<numBars;j++)
	 cin >> bars[j];
      sort(bars, numBars);

      int a=1;
      for (int j=0;j<numBars-1;j++)
      {
	 if (bars[j]<bars[j+1])
	    cout << bars[j+1]-bars[j] << " " << a << endl;
	 a++;
      }
      cout << endl;
   }
   return 0;
}

void sort(int A[], int n)
{
   for (int i=0;i<n-1;i++)
      for (int j=i+1;j<n;j++)
	 if (A[j]<A[i])
	 {
	    int temp = A[j];
	    A[j]=A[i];
	    A[i]=temp;
	 }
}
