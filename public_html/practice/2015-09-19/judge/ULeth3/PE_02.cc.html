#include <iostream>
#include <iomanip>

using namespace std;

double f (int, int, const double, const int, const int, bool);

int main()
{
   int cases;
   cin >> cases;

   for (int a=0; a<cases; a++)
   {
      string dummy;
      getline(cin, dummy);
      getline(cin, dummy);

      double s;
      int up, down;
      cin >> s >> up >> down;

      double win=(20/(s+20.0))*f(1,0,s,up,down,true)+
	 (1-(20/(s+20.0)))*f(0,1,s,up,down,false);
      cout << fixed << setprecision(5) << win << endl;
   }
}

double f (int a, int b, const double s, const int up, const int down, bool win)
   //a: for, b: against)
{
   if (a==4)
      return 1;
   if (b==4)
      return 0;
   if (win)
      return (20+up)/(s+20.0+up)*f(a+1,b,s,up,down,true)+
	 (1-(20+up)/(s+20.0+up))*f(a,b+1,s,up,down,false);
 
   return (20-down)/(s+20.0-down)*f(a+1,b,s,up,down,true)+
      (1-(20-down)/(s+20.0-down))*f(a,b+1,s,up,down,false);
}
