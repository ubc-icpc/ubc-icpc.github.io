#include <iostream>
using namespace std;

void sort (int [],int);

int main()
{
   while (true)
   {
      int n, A[30];
      cin >> n;
      if (n==0)
	 break;
      for (int i=0;i<n;i++)
	 cin >> A[i];
      sort (A,n);
      int finish=0;
      for (int i=0;i<n;i++)
      {
	 int next=(i+1)*30;
	 if (next<finish)
	    next=finish;
	 finish=next+A[i];
      }
      cout << finish/60 << ":";
      if (finish%60<10)
	 cout << "0";
      cout << finish%60 << endl;
   }
   return 0;
}

void sort(int A[], int n)
{
   for (int i=0;i<n-1;i++)
      for (int j=i+1;j<n;j++)
	 if (A[j]>A[i])
	 {
	    int temp=A[i];
	    A[i]=A[j];
	    A[j]=temp;
	 }
}
