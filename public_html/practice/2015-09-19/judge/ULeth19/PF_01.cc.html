#include <iostream>
#include <algorithm>

using namespace std;

void solve()
{
  int n;
  cin >> n;
  int len[100000];
  for (int i = 0; i < n; i++) {
    cin >> len[i];
  }
  sort(len, len+n);

  for (int i = 0, j; i < n; i = j) {
    for (j = i+1; j < n && len[i] == len[j]; j++)
      ;
    if (j < n) {
      cout << len[j] - len[i] << ' ' << j << endl;
    }
  }
  
  
  cout << endl;
}

int main()
{
  int t;
  cin >> t;
  while (t--)
    solve();

  return 0;
}
