#include <iostream>
#include <algorithm>
#include <iomanip>

using namespace std;

bool solve()
{
  int n;
  cin >> n;
  if (!n) return false;

  int A[30];
  for (int i = 0; i < n; i++) {
    cin >> A[i];
  }
  sort(A, A+n);
  reverse(A, A+n);

  int curr = 0;
  for (int i = 0; i < n; i++) {
    int done_wash = (i+1)*30;
    curr = max(done_wash, curr) + A[i];
  }

  cout << curr/60 << ":" << setw(2) << setfill('0') << curr%60 << endl;
  

  return true;
}

int main()
{
  while (solve())
    ;

  return 0;
}
