#include <iostream>
#include <iomanip>
#include <set>
#include <string>
using namespace std;

typedef long double ld;

int main() {
  cout << fixed << setprecision(5);
  int n; cin >> n;
  set<int> wb;
  for (int i = 0; i < (1<<7); i++) {
    int am = 0;
    int cw = 0, cl = 0;
    for (int j = 0; j < 7; j++) {
      if ((1<<j) & i) {
        cw++;
        am |= (1<<j);
      }
      else
        cl++;
      if (cw == 4) {
        wb.insert(am);
        break;
      }
      if (cl == 4)
        break;
    }
  }
  while (n--) {
    string S;
    getline(cin, S);
    getline(cin, S);
    ld s, w, l;
    cin >> s >> w >> l;
    ld tp = 0;
    for (set<int>::iterator it = wb.begin(); it != wb.end(); it++) {
      int i = *it;
      int cw = 0, cl = 0;
      ld p = 1;
      ld cs = 20;
      for (int j = 0; j < 7; j++) {
        if ((1<<j) & i) {
          p *= cs/(cs+s);
          cs = 20+w;
          cw++;
        }
        else {
          p *= s/(cs+s);
          cs = 20-l;
          cl++;
        }
        if (cw == 4)
          break;
      }
      tp += p;
    }
    cout << tp << "\n";
  }
  return 0;
}
