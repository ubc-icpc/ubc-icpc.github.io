#include <iostream>
#include <vector>
using namespace std;

int main() {
  string s;
  while (cin >> s) {
    int ind = 0;
    while (s[ind] != '.')
      ind++;
    ind++;
    vector<int> d;
    while (ind < s.size()) {
      d.push_back(s[ind]-'0');
      ind++;
    }
    while (true) {
      if (d.back() && d.back() != 5) {
        cout << "Some precision lost.\n";
        break;
      }
      if (!d.size()) {
        cout << "Exact.\n";
        break;
      }
      int c = 0;
      for (int i = d.size()-1; i >= 0; i--) {
        int x = d[i]*2+c;
        d[i] = x%10;
        c = x/10;
      }
      while (d.size() && d.back() == 0)
        d.pop_back();
    }
  }
  return 0;
}
