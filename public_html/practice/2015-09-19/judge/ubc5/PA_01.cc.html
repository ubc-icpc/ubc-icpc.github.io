#include <iostream>
#include <algorithm>
using namespace std;

int l[30];

int main() {
  int n;
  while (cin >> n && n) {
    for (int i = 0; i < n; i++)
      cin >> l[i];
    sort(l,l+n);
    int t = 0;
    for (int i = n-1; i >= 0; i--)
      t = max(30*(n-i),t)+l[i];
    int h = t/60, m = t%60;
    cout << h << ":";
    if (!m)
      cout << "00";
    else if (m < 10)
      cout << "0" << m;
    else
      cout << m;
    cout << "\n";
  }
  return 0;
}
