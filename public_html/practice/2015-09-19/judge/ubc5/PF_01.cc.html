#include <iostream>
#include <iomanip>
#include <algorithm>
using namespace std;

int b[100000];

int main() {
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    for (int i = 0; i < n; i++)
      cin >> b[i];
    sort(b,b+n);
    int ind = 0;
    while (ind < n-1) {
      int ind2 = ind;
      while (ind2 < n-1 && b[ind2] == b[ind])
        ind2++;
      if (b[ind2]-b[ind])
        cout << b[ind2]-b[ind] << " " << ind2 << "\n";
      ind = ind2;
    }
    cout << "\n";
  }
  return 0;
}
