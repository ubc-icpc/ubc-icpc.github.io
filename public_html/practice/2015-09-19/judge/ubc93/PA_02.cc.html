#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main() {
	int numLoads;
	cin >> numLoads;

	while (numLoads != 0) {
		vector<int> dryTimes;
		for (int i = 0; i < numLoads; i++) {
			int dryTime;
			cin >> dryTime;
			dryTimes.push_back(dryTime);
		}

		sort(dryTimes.begin(), dryTimes.end(), [](int x, int y) { return y < x; });

		int totalWash = 30;
		int totalDry = totalWash;
		for (int i = 0; i < numLoads; i++) {
			if (totalWash > totalDry) {
				totalDry = totalWash;
			}

			totalDry += dryTimes.at(i);
			totalWash += 30;
		}

		int hours = totalDry / 60;
		int minutes = totalDry % 60;

		if (minutes < 10) {
			cout << hours << ":" << "0" << minutes << endl;
		} else {
			cout << hours << ":" << minutes << endl;
		}
		cin >> numLoads;
	}

	return 0;
}
