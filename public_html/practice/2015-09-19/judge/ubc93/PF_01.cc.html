#include <iostream>
#include <sstream>
#include <algorithm>
#include <string>
#include <vector>
using namespace std;

int main() {
	int numCases;
	cin >> numCases;

	for (int i = 0; i < numCases; i++) {
		string line;
		int numArgs;
		vector<int> args;
		int largestHeight = 0;

		cin >> numArgs;
		for (int i = 0; i < numArgs; i++) {
			int height;
			cin >> height;
			args.push_back(height);
		}

		sort(args.begin(), args.end());

		for (int i = 0; i < numArgs; i++) {
				if (i != numArgs - 1) {
				int diff = args.at(i + 1) - args.at(i);
				if (diff > 0) {
					cout << diff << " " << i + 1 << endl;
				}
			}
		}
		cout << endl;
	}

	return 0;
}
