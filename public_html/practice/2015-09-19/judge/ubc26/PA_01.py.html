while True:
    a = int(raw_input())
    if not a: break
    xs, df = sorted(map(int, raw_input().split()), reverse=True), 0

    for i, dt in enumerate(xs):
        nxtw = (i + 1) * 30
        d0 = max(nxtw, df)
        df = d0 + dt

    hr, mn = df / 60, df % 60
    print("%d:%s" % (hr, str(mn).zfill(2)))
