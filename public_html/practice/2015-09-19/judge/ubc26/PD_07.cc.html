#define SL(ss, mmm) \

typedef pair<int, int> pt;

int main() {
  int I(mmm);
  L(ii, mmm) {
    int I(n);
    vector<pt> ps(n);
    L(j, n) {
      int x, y;
      cin >> x >> y;
      ps[j] = make_pair(x, y);
    }

    if (n <= 2) {
      cout << "Case #" << (ii+1) << ": " << n << endl;
      continue;
    }

#define x first
#define y second
    int best = 2;
    for (auto& p1 : ps) {
      map<pt, int> mm;
      for (auto& p2 : ps) {
        int dx = p2.x - p1.x, dy = p2.y - p1.y;
        if (dx == 0 && dy == 0) continue;

        int gg = (dx == 0 || dy == 0) ? max(abs(dx), abs(dy)) : abs(__gcd(dx, dy));
        dx /= gg, dy /= gg;
        if (dx < 0) dx *= -1, dy *= -1;
        else if (dx == 0 && dy == -1) dy = 1;
        else if (dx == -1 && dy == 0) dx = 1;

        pt dp = make_pair(dx, dy);
        int f;
        if (mm.find(dp) == mm.end()) {
          f = mm[dp] = 2;
        } else {
          f = ++mm[dp];
        }

        best = max(f, best);
      }
      for (auto pp : mm)
        printf("(%d,%d): %d\n", pp.x.x, pp.x.y, pp.y);
      printf("---------\n");
    }

    cout << "Case #" << (ii+1) << ": " << best << endl;
  }
}
