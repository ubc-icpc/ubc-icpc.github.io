#include <iostream>
#include <algorithm>
#include <utility>
#include <vector>

using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        int barLengths[n];
        vector<pair<int, int> > results;
        for (int i = 0; i < n; i++) {
            cin >> barLengths[i];
        }
        sort(barLengths, barLengths+n);
        for (int i = 1; i < n; i++) {
            if (barLengths[i] != barLengths[i-1]) {
                int length = i;
                int count = barLengths[i] - barLengths[i-1];
                results.push_back(make_pair(length, count));
            }
        }
        sort(results.begin(), results.end());
        for (auto result : results) {
            cout << result.second << " " << result.first << endl;
        }
        if (t > 0) {
            cout << endl;
        }
    }
}
