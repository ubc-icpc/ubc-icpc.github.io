#include <iostream>
#include <algorithm>

using namespace std;

int main() {
    int n;
    while (cin >> n && n) {
        int dry_times[n];
        int estimated_time = 0;
        for (int i = 0; i < n; i++) {
            cin >> dry_times[i];
        }
        sort(dry_times, dry_times+n);
        reverse(dry_times, dry_times+n);
        for (int i = 0; i < n; i++) {
            int finish = (i + 1) * 30; // minimal next possible load for dryer
            if (finish < estimated_time) {
                // if dryer won't be done on time, delay next dry
                finish = estimated_time;
            }
            estimated_time = finish + dry_times[i];
        }
        cout << estimated_time/60 << ":";
        int mins = estimated_time % 60;
        if (mins < 10) {
            cout << "0" << mins << endl;
        } else {
            cout << mins << endl;
        }
    }
    return 0;
}
