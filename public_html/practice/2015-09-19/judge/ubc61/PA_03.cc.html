#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int n;
	while(cin >> n && n!= 0) {
		int h=0, m=0;
		vector<int> loads;
		for(int i=0; i<n; i++) {
			int l; cin >> l;
			loads.push_back(l);
		}
		sort(loads.begin(), loads.end());
		reverse(loads.begin(), loads.end());
		int last = 0;
		for (int i=0; i<n; i++) {
			m = max((i+1)*30,m) + loads[i];
		}
		while(m >= 60) {
			m -= 60;
			h++;
		}
		cout << h << ":";
		if (m == 0) {
			cout << "00";
		} else if (m<10) {
			cout << "0" << m;
		} else {
			cout << m;
		}
		cout << endl;
	}
	return 0;
}
