#include <bits/stdtr1c++.h>
using namespace std;

int main() {
	int t; cin >> t;
	while(t--) {
		int n; cin >> n;
		vector<int> bars;
		for (int i=0; i<n; i++) {
			int b; cin >> b;
			bars.push_back(b);
		}
		sort(bars.begin(), bars.end());
		vector< pair<int,int> > ans;
		for (int i=n-1; i>=0; i--) {
			if (i==0 || bars[i] == bars[i-1]) continue;
			ans.push_back(make_pair(bars[i]-bars[i-1],i));
		}
		for (int i=ans.size()-1; i>=0; i--) {
			cout << ans[i].first << " " << ans[i].second << endl;
		}
		cout << endl;
	}
	return 0;
}
