#include <bits/stdtr1c++.h>
using namespace std;

typedef long double ld;

ld solve (int w, int l, ld p) {
	if (w == 4) return 1;
	if (l == 4) return 0;
	if (w == 3 && l == 3) return p*p/(1-2*(1-p)*p);
	return p*solve(w+1,l,p) + (1-p)*solve(w,l+1,p); 
}

int main() {
	int t; cin >> t;
	while(t--) {
		 ld p; cin >> p;
		 cout << fixed << setprecision(5) << solve(0,0,p) << endl;
	}
	return 0;
}
