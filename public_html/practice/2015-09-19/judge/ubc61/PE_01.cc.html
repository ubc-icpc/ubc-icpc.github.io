#include <bits/stdtr1c++.h>
using namespace std;

typedef long double ld;

int s, wf, lf;
double solve(int w, int l, int last) {
	if (l == 4) return 0;
	if (w == 4) return 1;
	int str = 20;
	if (last == 1) str += wf;
	if (last == -1) str -= lf;
	double p = str/ld(s+str);
	return p*(solve(w+1,l,1))+(1-p)*(solve(w,l+1,-1));
}
