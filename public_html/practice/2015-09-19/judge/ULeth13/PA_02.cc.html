#include <iostream>
#include <iomanip>
#include <algorithm>
using namespace std;
int n;
int hour, minn;
int dry_times[30];

int main(){
   
   while(cin>>n){
      if(n==0)
	 break;
      for(int i=0; i< n; i++){
	 cin>> dry_times[i];
      }
      sort(dry_times, dry_times + n);
      reverse(dry_times, dry_times + n);
      int finish = 0;
      for(int j=0; j<n; j++){
	 int next_ready = (j+1) * 30;
	 if (next_ready < finish)
	    next_ready = finish;
	 finish = next_ready + dry_times[j];
      }
      hour = finish / 60;
      minn = finish % 60 ;
     
      cout<< hour << ':'<<setfill('0')<<setw(2)<<minn<<endl;
   }
   
   return 0;
}
