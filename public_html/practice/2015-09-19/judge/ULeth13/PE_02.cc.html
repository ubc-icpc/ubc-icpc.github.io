double win, loss, strength;

double prob(int a, int b,double s){
   if (a==4) return 1.0;
   if (b==4) return 0.0;
   double p = s /(s+strength);
   return p*prob(a+1, b,win+20)  + (1-p)*prob(a,b+1,20-loss);
}
