#include<bits/stdc++.h>
using namespace std;

double prob(int a, int b, double p){
   if (a==4) return 1.0;
   if (b==4) return 0.0;
   if (a==3 and b==3) return -1*pow(p,2) / (2*p*(1-p) - 1);
   return p*prob(a+1,b,p) + (1-p)*prob(a,b+1,p);
}


int main(){
   int tc;
   double probability;
   cin >> tc;
   for(int i = 0; i< tc; i++){
      cin>> probability;
      double result;
      result = prob(0,0,probability);
      cout<< setprecision(5)<<fixed<<result<<endl;
   }
   
   return 0;
}
