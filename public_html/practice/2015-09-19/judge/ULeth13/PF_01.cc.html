#include <iostream>
#include <algorithm>
using namespace std;

int tc;
int n;
int inputs[100001];

int main(){
   cin>>tc;

   for( int i =0; i<tc; i++){
      cin>> n;

      for(int in=0; in < n; in++){
	 cin>> inputs[in];
      }
      sort(inputs, inputs + n);

      int N = 0;
      int L = 0;
      
      for(int j = 0; j< n-1; j++){
	 N = (inputs[j+1] - inputs[j]);
	 if(N > 0){
	    L = j+1;
	    cout<< N<< ' '<<L<<endl;
	    L=0;
	 }
	 
      }
      cout<< endl;
   }
 
   return 0;
}
