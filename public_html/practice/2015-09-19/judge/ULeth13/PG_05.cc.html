#include <bits/stdc++.h>
using namespace std;

string remove_head(string nums){
   int index=0;
   while(index < nums.size()){
      if (nums[index] == '.'){
	 return nums.substr(index + 1);
      }
      index++;
   }
   return nums;
}

string remove_tail(string nums){
   int index= nums.size()-1;
   while(index >=0){
      if(nums[index]!= '0'){
	 return nums.substr(0,index+1);
      }
      index--;
   }
   return nums;
}

string add_str(string nums){
   int carry = 0;
   int len = nums.size()-1;
   int last = nums[len] - '0';
   for(int i = len; i>=0; --i){
      last = nums[i] - '0';
      int sum = 2*(last)+carry;
      nums[i] = sum % 10 + '0';
      carry = sum / 10;
   }

   return nums;
}
   
	 
int main(){
   string n;
   while(cin>>n){
      bool lost= false;
      n = remove_head(n);
      n = remove_tail(n);
      while(n.size()>0){
	 int len = n.size()-1;
	 int last = n[len] - '0';
	 if(n[len]!= '5' and n[len]!='0'){
	    lost = true;
	    break;
	 }
	 if(n.size() == 1){
	    break;
	 }
	 else{
	    n = add_str(n);
	    n = remove_tail(n);
	 }
	 
      }
   
   if(lost)
      cout<< "Some precision lost."<<endl;
   else
      cout<< "Exact."<<endl;
      
   }
      
   return 0;
}
