#include<iostream>
#include<algorithm>
#include<queue>

using namespace std;


int main(void){
  int n;
  while(cin>>n) {
	if(n==0) break;
	int arr[n];
	for(int i=0; i<n;i++){
	  int x; cin>>x; arr[i]=x;
	}
	sort(arr,arr+n);
	int d=0;
	int total=0;
	for(int i=n-1;i>=0;i--){
	  if(d<=30) {
		total+=30;
		d=arr[i];
	  } else {
		total+=30;
		d-=30;
		d+=arr[i];
	  }
	}
	total+=d;
	cout << total/60 << ":";
	if(total%60 < 10) cout << "0";
	cout << total%60 << endl;
  }
  return 0;
}
