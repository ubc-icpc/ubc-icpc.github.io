#include<iostream>
#include<string>

using namespace std;

void f(string &s) {
  int carry=0;
  for(int i=s.size()-1; i>=0;i--) {
	if(s[i]=='.')break;
	int t=(s[i]-'0')*2+carry;
	//cout << t<< endl;
    s[i]='0'+t%10;
	carry=t/10;
  }
}

int main (void) {
  string s;
  while(cin>>s) {
	
	bool ok=true;
	while(s.back()=='0') s.pop_back();
	while(true) {
	  if(s.back()=='.') break;
	  
	  //cout << s << endl;
	  if(s.back()!='5') {
		ok=false; break;
	  }
	  f(s);
	  while(s.back()=='0') s.pop_back();
	}
	if(ok) cout << "Exact." << endl;
	else cout<< "Some precision lost." << endl;
  }
  return 0;
}
