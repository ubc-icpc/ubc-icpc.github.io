#include <cmath>
#include <iomanip>
#include <iostream>

using namespace std;

double chance(double p) {
        double q = 1.0f - p;
        double win = pow(p, 4.0f);
        double direct = win + 4 * win * q + 10 * win * q * q;
        double deuce = ((p * p) / (1 - 2 * p * q)) * 20 * pow(p, 3.0f) * pow(q, 3.0f);
        return direct + deuce;
}

int main() {
        double p;
        int T;

        cin >> T;
        cout << fixed << setprecision(5);
        
        while (T--) {
                cin >> p;
                cout << chance(p) << endl;
        }
        
        return 0;
}
