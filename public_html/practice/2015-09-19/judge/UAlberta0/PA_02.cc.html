#include <algorithm>
#include <iostream>
#include <queue>
#include <vector>

using namespace std;

int main() {
        vector<int> times;
        int loads, t, w_t, d_t;
        cin >> loads;

        while(loads != 0) {
                queue<int> dryer;
                times.clear();

                for(int i = 0; i < loads; ++i) {
                        cin >> t;
                        times.push_back(t);
                }
                sort(times.begin(), times.end(), greater<int>());

                for(int i = 0; i < loads; ++i) {
                        dryer.push(times[i]);
                }

                w_t = d_t = 30;
                while(!dryer.empty()) {
                        if (d_t < w_t) {
                                d_t = w_t;
                        }

                        d_t += dryer.front();
                        dryer.pop();
                        w_t += 30;
                }

                int mins = d_t % 60;
                cout << d_t / 60 << ":";
                if (mins < 10) {
                        cout << "0";
                }
                cout << mins << endl;
                
                cin >> loads;
        }
        
        return 0;
}
