#include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

int main() {
        vector<int> ns, length, num;
        int T, N, l, next;

        cin >> T;

        while (T--) {
                cin >> N;
                ns.clear();
                num.clear();
                length.clear();

                for (int i = 0; i < N; ++i) {
                        cin >> l;
                        ns.push_back(l);
                }
                sort(ns.begin(), ns.end(), greater<int>());

                for (int i = 0; i < N ; ++i) {
                        next = i;
                        while(ns[next] == ns[i] && next < N - 1) {
                                ++next;
                        }
                        
                        if (next == N - 1 && ns[i] == ns[N-1]) {
                                break;
                        }
                        
                        length.push_back(N - next);
                        num.push_back(ns[i] - ns[next]);

                        i = next - 1;
                }

                for (int i = num.size() - 1; i >= 0; --i) {
                        cout << num[i] << " " << length[i] << endl;
                }
                cout << endl;
        }
        return 0;
}
