#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
   long n, a[32],sum,i,b,c,d,e;
   while(scanf("%ld",&n)==1)
   {
      if(n==0)
	 break;
      sum=0;
      
      for(i=0;i<n;i++)
      {
	 scanf("%ld",&a[i]);
      }
      sort(a,a+n);
      reverse(a,a+n);
      for(i=0;i<n;i++)
      {  d=(i+1)*30;
	 if(d<sum)
	    d=sum;
	 sum=d+a[i];
      }
      b=sum/60;
      c=sum%60;
      printf("%ld:%02ld\n",b,c);
      
   }
   return 0;
}
