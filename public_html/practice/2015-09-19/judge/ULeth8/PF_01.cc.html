#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
   long n,t,a[1000002],c,d,i;
   scanf("%ld",&t);
   while(t--)
   {
      scanf("%ld",&n);
      for(i=0;i<n;i++)
      {
	 scanf("%ld",&a[i]);
      }
      sort(a,a+n);
      c=1;
      for(i=0;i<n;i++)
      {
	 if(a[i]<a[i+1])
	 printf("%ld %ld\n",a[i+1]-a[i],c);
         c++;
      }
      printf("\n");
   }
   return 0;
}
