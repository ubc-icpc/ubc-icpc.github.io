//orthogonal bar graphs

#include <iostream>
#include <algorithm>


using namespace std;

void graph();

int main()

{

   int t=0;
   cin >> t;
 
   for (int i=0; i < t; i++)
   {
      graph();
   }

      return 0;
}


   void graph()
   {

      int n=0;
      int A[1000000];
      cin >> n;
      for (int i=0; i<n;i++)
      {
	 cin>> A[i];
      }

      //sort in ascending
      sort (A, A+n);

      //for loops to process the array for unique elements
      int j = 0;
      for (int i=0,j;i<n;i=j)
      {
	 //j is the indexing variable for the unique bar heights
	 for(j=i+1;j<n && A[i] == A[j];j++)
	 {
	    ;
	 }
	    if (j<n)
	    {

	       cout << A[j] -  A[i] << " " << j << endl;
	    }
	 }

	 cout << endl;

      return;
   }
