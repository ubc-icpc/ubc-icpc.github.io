#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void print_bars( vector<int> heights ) {
  sort(heights.begin(), heights.end());
  for(int i = 0; i < heights.size() - 1; i++) {
    int diff = heights[i+1] - heights[i];
    if(diff > 0) {
      cout << diff << " " << i+1 << endl;
    }
  }
  cout << endl;
}

int main( int argc, char ** argv ) {
  int T;
  cin >> T;
  while( T-- ) {
    int N;
    cin >> N;
    int temp;
    vector<int> bars;
    while( N-- ) {
      cin >> temp;
      bars.push_back(temp);
    }
    print_bars( bars ); 
  }
  return 0;
}
