#include<iostream>
#include<vector>
#include<functional>
#include<algorithm>
#include<queue>
using namespace std;

int main() {
  int n; cin >> n;
  while (n) {
    vector<int> t;
    queue<int> q;
    for (int i=0; i<n; i++) { int x; cin >> x; t.push_back(x); }
    sort(t.begin(), t.end(), greater<int>());
    int total=0, tmp=30;
    for (int i=0; i<n; i++) {
      total+=30; tmp-=30;
      if (q.empty() && tmp<=0) tmp=0;
      q.push(t[i]);
      if (tmp<=0) {
	tmp+=q.front();
	q.pop();
      }
    }
    while (!q.empty()) {
      tmp+=q.front();
      q.pop();
    }
    //cout << total << " " << tmp << endl;
    total+=tmp;
    int hours=total/60; int mins=total%60;
    cout << hours << ":";
    if (mins<10) cout << 0;
    cout << mins << endl;
    cin >> n;
  }
}
