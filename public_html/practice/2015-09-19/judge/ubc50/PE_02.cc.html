#include<iostream>
#include<iomanip>
#include<string>
using namespace std;

float dfs(int str, int s, int w, int l, int nwon, int nlost) {
  if (nwon==4) return 1;
  if (nlost==4) return 0;
  float p=(float) str/(float)(str+s);
  return p*dfs(20+w,s,w,l,nwon+1,nlost)+(1-p)*dfs(20-l,s,w,l,nwon, nlost+1);
}

int main() {
  int t; cin >> t;
  while (t--) {
    string a;
    getline(cin, a); getline(cin, a);
    int s,w,l; cin >> s >> w >> l;
    float res=dfs(20,s,w,l,0,0);
    cout << fixed << setprecision(5) << res << endl;
  }
}
