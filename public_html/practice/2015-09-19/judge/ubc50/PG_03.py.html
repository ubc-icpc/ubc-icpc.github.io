import math
import sys

def gcd(a,b):
    while b!=0:
        t=b
        b=a%b
        a=t
    return a

for line in sys.stdin:
    s=line.strip().split('.')[1]
    x=int(s)
    n=len(s)
    d=10**n
    p=0
    while gcd(d,x)>1:
        tmp=gcd(d,x)
        d/=tmp
        x/=tmp
        p=int(math.log(d,2))
        # print('{a} {b} {c}'.format(a=x, b=d, c=p))
    if 2**p==d:
        print "Exact."
    else:
        print "Some precision lost."
