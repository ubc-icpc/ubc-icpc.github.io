#include<iostream>
#include<iomanip>
using namespace std;
typedef long double ld;

ld po(ld b, ld e) {
  if (e==0) return 1;
  if (e==1) return b;
  ld x=po(b,(int)e/2);
  x*=x;
  if (((int)e)%2) x*=b;
  return x;
}

int main() {
  int t; cin >> t;
  while (t--) {
    ld p; cin >> p;
    ld res=0;
    // 4-0
    res += po(p,4);
    // 4-1
    res += 4*po(p,4)*(1-p);
    // 4-2
    res += 10*po(p,4)*po(1-p,2);
    // 3-3 tie and then alternate till last win
    res += 20*po(p,3)*po(1-p,3)*po(p,2)*(1/(1-2*p*(1-p)));
    cout << fixed << setprecision(5) << res << endl;
  }
}
