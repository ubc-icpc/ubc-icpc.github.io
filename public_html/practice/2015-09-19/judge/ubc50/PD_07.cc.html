#include<iostream>
#include<vector>
#include<map>
#include<utility>
#include<cstring>
using namespace std;
typedef pair<float, float> pff;

int rise[405][405], run[405][405];

int main() {
  int t; cin >> t;
  for (int te=1; te<=t; te++) {
    int res=0;
    vector<pff> v;
    memset(rise,0,sizeof rise);
    memset(run,0,sizeof run);
    int n; cin >> n;
    for (int i=0; i<n; i++) {
      float x,y; cin >> x >> y; v.push_back(make_pair(x,y));
    }
    if (n<=2) res=n;
    else {
      for (int i=0; i<n; i++) {
	for (int j=0; j<n; j++) {
	  if (i==j) continue;
	  rise[i][j]=(v[j].second-v[i].second);
	  run[i][j]=(v[j].first-v[i].first);
	}
      }

      for (int i=0; i<n; i++) {
	for (int j=i+1; j<n; j++) {
	  int tmp=2;
	  for (int k=j+1; k<n; k++) {
	    if (rise[i][j]*run[i][k]==run[i][j]*rise[i][k])
	      tmp++;
	  }
	  res=max(res,tmp);
	}
      }
    }
    cout << "Case #" << te << ": " << res << endl;
  }
}
