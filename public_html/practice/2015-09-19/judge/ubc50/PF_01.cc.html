#include<iostream>
#include<vector>
#include<algorithm>
#include<functional>
#include<map>
using namespace std;

int main() {
  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    vector<int> v;
    map<int, int> cnt;
    for (int i=0; i<n; i++) {
      int x; cin >> x; v.push_back(x);
    }
    sort(v.begin(), v.end(), greater<int>());
    for (int i=0; i<n-1; i++) {
      if (v[i]==v[i+1]) continue;
      cnt[n-i-1]+=(v[i]-v[i+1]);
    }
    for (auto x : cnt) {
      cout << x.second << " " << x.first << endl;
    }
    cout << endl;
  }
}
