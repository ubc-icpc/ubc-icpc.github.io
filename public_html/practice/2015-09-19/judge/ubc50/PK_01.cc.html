#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

int n;
int wi[455];
int memo[455][455][455];

int dp(int idx, int left, int right) {
  if (idx==n) return 0;
  if (memo[idx][left][right]!=-1) return memo[idx][left][right];
  return memo[idx][left][right]=min(abs(wi[idx]-wi[left])+dp(idx+1,idx,right),
				    abs(wi[idx]-wi[right])+dp(idx+1,left,idx));
}

int main() {
  while (cin >> n) {
    memset(wi,0,sizeof wi);
    memset(memo,-1,sizeof memo);
    for (int i=0; i<n; i++) { int x; cin >> x >> wi[i]; }
    reverse(wi, wi+n);
    int res=dp(0,0,0);
    cout << res << endl;
  }
}
