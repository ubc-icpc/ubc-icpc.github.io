import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class laundry {

	List<String> output;

	public void printTime() {
		output = new ArrayList<String>();
		// get input
		Scanner scan = new Scanner(System.in);
		int load = 0;
		List<Integer> times;
		int input;
		do {
			times = new ArrayList<Integer>();
			load = scan.nextInt();
			if (load != 0) {
				for (int i = 0; i < load; i++) {
					input = scan.nextInt();
					times.add(input);
				}
				quickSort(times, 0, times.size() - 1);
				for (int i = 0; i < times.size(); i++) {
				}
				int totalTime = calculateTime(load, times);
				convertAndPrint(totalTime);
			}
		} while (load != 0);
		for (int k = 0; k < output.size(); k++) {
			System.out.println(output.get(k));
		}
	}

	public int calculateTime(int load, List<Integer> inputs) {
		int wash = 0;
		int dryer = 0;
		for (int i = 0; i < inputs.size(); i++) {
				int dry = inputs.get(i);
				wash = wash + 30;
				dryer = Math.max(dryer + dry, wash + dry);
		}	    
		return dryer;
	}

	public void convertAndPrint(int time) {
		int hour = 0;
		while ((time - 60) >= 0) {
			time -= 60;
			hour++;
		}

		int minutes = time % 60;
		String out;
		
		if (minutes == 0) {
			out = hour + ":00";
		}
		else if (minutes <= 9) {
			out = hour + ":0" + minutes;
		}
		else {
			out = hour + ":" + minutes;
		}
		output.add(out);
	}

	public static void quickSort (List<Integer> numbers, int low, int high)
    {
 
        int i=low;
        int j=high;
        int temp;
        int middle = numbers.get((low + high) / 2);
 
        while (i<j)
        {
            while (numbers.get(i) > middle)
            {
                i++;
            }
            while (numbers.get(j) < middle)
            {
                j--;
            }
            if (j >= i)
            {
                temp=numbers.get(i);
                numbers.set(i, numbers.get(j));
                numbers.set(j, temp);
                i++;
                j--;
            }
        }
        if (low < j)
        {
        	quickSort(numbers, low, j);
        }
        if (i < high)
        {
        	quickSort(numbers, i, high);
        }
    }
	
	public static void main(String[] args) {
		 laundry l = new laundry();
		 l.printTime();
	}
}
