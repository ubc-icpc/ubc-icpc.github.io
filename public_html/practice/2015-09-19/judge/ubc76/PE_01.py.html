def dfs(win, lose, buff, p, S, W, L):
	if win == 4:
		return p
	if lose == 4:
		return 0
	ret = 0.0
	ret += dfs(win + 1, lose, W, p * (20+buff)*1.0 / (20+buff+S), S, W, L)
	ret += dfs(win, lose + 1, -L, p * (1 - (20+buff)*1.0 / (20+buff+S)), S, W, L)
	return ret


T = int(raw_input())

for re in range(T):
	name = raw_input()
	S, W, L = map(int, raw_input().split())
	print '%.5f' % dfs(0, 0, 0, 1, S, W, L)
