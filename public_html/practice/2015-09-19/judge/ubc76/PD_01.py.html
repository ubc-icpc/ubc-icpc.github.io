# points on line

def maxP(points):
    lp = len(points)
    m = 0
    for i in range(lp):
        d = {'i':1}
        for j in range(i+1,lp):
            x,y = points[j][0],points[j][1]
            if (points[i][0] == x):
                slope = 'i'
            else:
                slope = (points[i][1] -y)*1.0/(points[i][0]-x)
            if (slope not in d):
                d[slope] = 2
                        
            else:
                d[slope] = d[slope]+1
        m = max(m,max(d.values()))
    return m

        points.append([x,y])
    print 'Case #'+str(t+1)+': '+str(maxP(points))
