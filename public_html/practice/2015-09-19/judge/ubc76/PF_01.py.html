test = int(raw_input())


for i in range(test):
    bars = [int(x) for x in raw_input().split()]
    num = bars[0]
    bar = bars[1:]
    # buld a dictionary
    d= {}
    for j in range(num):
        if (bar[j] in d):
            d[bar[j]] = d[bar[j]] + 1
        else:
            d[bar[j]] = 1
    b_set = list(set(bar))
    b_set.sort()
    o = []
    length = 0
    for k in range(len(b_set)-1):
        length = length + d[b_set[k]]
        o.append([b_set[k+1]-b_set[k],length])
    for lo in range(len(o)):
        print str(o[lo][0]) + ' ' + str(o[lo][1])
