T = int(raw_input())

def dfs(win, lose, p):
	if win >= 3 and win == lose:
		return p*p / (1-2*p*(1-p))
	if win >= 4 and win > lose + 1:
		return 1
	if lose >= 4 and lose > win + 1:
		return 0
	ret = 0.0
	ret += p * dfs(win + 1, lose, p)
	ret += (1 - p) * dfs(win, lose + 1, p)
	return ret

for re in range(T):
	p = float(raw_input())
	print '%.5f' % dfs(0, 0, p)
