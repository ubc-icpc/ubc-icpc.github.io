#include <iostream>
#include <algorithm>
#include <functional>
using namespace std;

int arr[100005];

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		for (int i = 0; i < n; i++)
		{
			cin >> arr[i];
		}
		sort(arr, arr + n);
		int last = 0;
		for (int i = 0; i < n; i++)
		{
			if (i == 0)
				continue;
			else
			{
				if (arr[i] == arr[i - 1])
					continue;
				cout << arr[i] - arr[last] << " " << i << endl;
				last = i;
			}
		}
		cout << endl;
	}
	return 0;
}
