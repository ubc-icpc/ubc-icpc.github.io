#include <iostream>

using namespace std;

typedef long double ld;

ld p;

ld lol(int win, int loss, ld percent )
{
	if (win >= loss + 2 && win >= 4)
	{
		return 1;
	}
	if (loss >= win + 2 && loss >= 4)
	{
		return 0;
	}
	else if (percent <= 1e-12)
		return 0.5;

	return p * lol(win + 1, loss, percent * p) + (1-p) * lol(win, loss + 1,percent * (1-p));

}

int main()
{
	int n;
	cin >> n;
	cout.precision(5);
	while (n--)
	{
		cin >> p;
		cout << fixed << lol(0,0,1) << endl;
	}
	return 0;
}
