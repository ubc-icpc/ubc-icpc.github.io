#include <iostream>
#include <cstring>
#include <utility>
#include <vector>
#include <algorithm>
#include <climits>
using namespace std;

typedef pair<int, int> pii;

int main()
{
	int t;
	cin >> t;
	int counts = 0;
	while (t--)
	{
		counts++;
		int maximum = 1;
		vector <pii> pts;
		int p;
		cin >> p;
		for (int i = 0; i < p; i++)
		{
			int x, y;
			cin >> x >> y;
			pts.push_back(pii(x, y));
		}
		for (int i = 0; i < pts.size(); i++)
		{
			for (int j = i+1; j < pts.size(); j++)
			{
				if (i == j)
					continue;
				int dx = pts[i].first - pts[j].first;
				int dy = pts[i].second - pts[j].second;
				int temp_max = 0;
				{
					for (int k = 0; k < pts.size(); k++)
					{
						if (dx *(pts[i].second - pts[k].second) == (pts[i].first - pts[k].first) * dy)
							temp_max++;
					}
				}
				if (temp_max > maximum)
					maximum = temp_max;
			}
		}
		if (p == 0)
		{
			cout << "Case #" << counts << ": " << 0 << endl;
		}
		else
			cout << "Case #" << counts << ": " << maximum << endl;
	}
	return 0;
}
