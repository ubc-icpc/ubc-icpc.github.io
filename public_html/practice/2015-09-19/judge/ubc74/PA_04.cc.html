#include<iostream>
#include<algorithm>

using namespace std;

int main () {
	int T; cin >> T;
	while (T) {
		int data[30] = {}, total = 0;
		for (int i = 0; i < T; i++) {
			int temp; cin >> temp;
			data[i] += temp;
		}
		sort(data, data+T);
		reverse(data, data+T);
		for (int i = 0; i < T; i++) {
			int done_washing = (1+i) * 30;
			total = max(done_washing, total) + data[i];
		}
		cout << total/60 << ":";
		if (total%60 >= 10)
			cout << total%60;
		else if (total%60 <10)
			cout << "0" << total%60;
		else
			cout << "00";
		cout << endl;
		cin >> T;
	}
}
