#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;

ld solve(int nwon, int nlost, ld pr, ld f) {
    if (nwon >= 4 && nwon-nlost >= 2) return 1;
    if (nlost >= 4 && nlost-nwon >= 2) return 0;
    if (f < 1e-12) return 0.5;
    return pr * solve(nwon+1, nlost, pr, f*pr) + (1-pr) * solve(nwon, nlost+1, pr, f*(1-pr));
}

int main() {
    int t; cin >> t;
    while (t--) {
        ld pr; cin >> pr;
        cout << fixed << setprecision(5) << solve(0, 0, pr, 1) << endl;
    }
	return 0;
}
