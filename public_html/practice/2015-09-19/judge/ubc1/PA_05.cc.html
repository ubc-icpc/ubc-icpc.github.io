#include <bits/stdtr1c++.h>

using namespace std;

int main() {
    int n;
    while (cin >> n) {
        if (n == 0) break;
        
        vector<int> v(n);
        for (int i = 0; i < n; i++) {
            cin >> v[i];
        }
        sort(v.rbegin(), v.rend());
        
        int tot = 0, tl = 30;
        queue<int> q;
        for (int i = 0; i < n; i++) {
            tot += 30;
            tl -= 30;
            if (q.empty() && tl < 0) tl = 0;
            q.push(v[i]);
            if (tl <= 0) {
                tl += q.front();
                q.pop();
            }
        }
        tot += tl;
        
        while (!q.empty()) {
            tot += q.front();
            q.pop();
        }
        cout << tot/60 << ":" << setfill('0') << setw(2) << tot%60 << endl;
    }
	return 0;
}
