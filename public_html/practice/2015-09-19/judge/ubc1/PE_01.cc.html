#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
ld f(int s, int w, int l, int nwon, int nlost, int last) {
    if (nlost == 4) return 0;
    if (nwon == 4) return 1;
    int str = 20;
    if (last == 1) {
        str = 20 + w;
    } else if (last == -1) {
        str = 20 - l;
    }
    ld pr = str/ld(s+str);
    return pr * f(s, w, l, nwon+1, nlost, 1) + (1-pr) * f(s, w, l, nwon, nlost+1, -1);
}

int main() {
    int t; cin >> t;
    while (t--) {
        string str; 
        getline(cin, str);
        getline(cin, str);
        
        int s, w, l; cin >> s >> w >> l;
        cout << fixed << setprecision(5) << f(s, w, l, 0, 0, 0) << endl;
    }
	return 0;
}
