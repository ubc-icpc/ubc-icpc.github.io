#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

int main() {
    ios::sync_with_stdio(0);
    int t; cin >> t;
    while (t--) {
        int n; cin >> n;
        vector<int> v(n);
        for (int i = 0; i < n; i++) {
            cin >> v[i];
        }
        
        sort(v.rbegin(), v.rend());
        vector<pair<int,int>> w;
        for (int i = 0; i < v.size()-1; i++) {
            int mult = v[i]-v[i+1], size = v.size()-i-1;
            if (mult == 0) continue;
            w.push_back({size, mult});
        }
        sort(w.begin(), w.end());
        
        for (auto p : w) {
            cout << p.second << " " << p.first << '\n';
        }
        cout << endl;
    }
	return 0;
}
