import sys
from fractions import Fraction

for line in sys.stdin:
    if len(line.strip()) == 0: continue
    a, b = line.strip().split('.')
    c, d = int(b), 10**len(b)
    fr = Fraction(c, d)
    x = fr.denominator
    while x%2 == 0: x /= 2
    if x == 1: print 'Exact.'
    else: print 'Some precision lost.'
