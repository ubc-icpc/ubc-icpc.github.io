#include <bits/stdtr1c++.h>

using namespace std;

typedef long long ll;

int memo[500][500][500], w[500], n;
int dp(int i, int l, int r) {
    if (i == n) return 0;
    if (memo[i][l][r] != -1) return memo[i][l][r];
    return memo[i][l][r] = min(abs(w[i]-w[l]) + dp(i+1, i, r), abs(w[i]-w[r]) + dp(i+1, l, i));
}

int main() {
    while (cin >> n) {
        memset(memo, -1, sizeof memo);
        for (int i = 0; i < n; i++) {
            int x; cin >> x >> w[i];
        }
        
        reverse(w, w+n);
        cout << dp(0, 0, 0) << endl;
    }
	return 0;
}
