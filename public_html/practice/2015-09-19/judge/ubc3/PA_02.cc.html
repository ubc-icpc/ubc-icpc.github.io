#include <iomanip>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using vi = vector<int>;

int main() {
  int n;
  while (cin >> n && n != 0) {
    vi loads(n);
    for (int i = 0; i < n; ++i)
      cin >> loads[i];
    sort(loads.begin(), loads.end(), [](int a, int b) {
          return a > b;
        });
    int next_wash = 0;
    int next_dryer = 0;
    for (int i = 0; i < n; ++i) {
      int load = loads[i];
      next_wash = next_wash + 30;
      next_dryer = max(next_dryer + load, next_wash + load);
    }
    cout << next_dryer / 60 << ':' << setw(2) << setfill('0') << next_dryer % 60 << endl;
  }
  return 0;
}
