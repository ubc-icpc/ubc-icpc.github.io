#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using vi = vector<int>;

int main() {
  int T;
  for (cin >> T; T > 0; --T) {
    int n; cin >> n;
    vi lengths(n);
    for (int i = 0; i < n; ++i)
      cin >> lengths[i];
    sort(lengths.begin(), lengths.end());
    int curr_height = lengths[0];
    for (int i = 1; i < n; ++i) {
      if (lengths[i] == curr_height)
        continue;
      int diff_height = lengths[i] - curr_height;
      cout << diff_height << ' ' << i << endl;
      curr_height = lengths[i];
    }
    cout << endl;
    
  }
  return 0;
}
