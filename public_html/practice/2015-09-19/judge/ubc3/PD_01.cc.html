#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
using vi = vector<int>;

int main() {
  int T; cin >> T;
  for (int cases = 1; cases <= T; ++cases) {
    int N; cin >> N;
    vi xx(N), yy(N);
    for (int i = 0; i < N; ++i)
      cin >> xx[i] >> yy[i];

    int max_pass = 0;
    for (int i = 0; i < N; ++i) {
      for (int j = 0; j < N; ++j) {
        int num_pass = 0;
        int dx = xx[j] - xx[i];
        int dy = yy[j] - yy[i];
        if (i == j)
          continue;
        for (int k = 0; k < N; ++k) {
          if (k == i || k == j)
            continue;
          int kdx = xx[k] - xx[i];
          int kdy = yy[k] - yy[i];
          // dx / kdx == dy / kdy
          if (dx * kdy == dy * kdx)
            num_pass++;
        }
        // cout << i << ' ' << j << ' ' << num_pass << endl;
        max_pass = max(max_pass, num_pass);
      }
    }
    cout << "Case #" << cases << ": ";
    if (N <= 1)
      cout << N << endl;
    else
      cout << max_pass + 2 << endl;
  }
  return 0;
}
