from fractions import Fraction
t=int(input())
for case in range(0,t):
    Dic={}
    m=int(input())
    L=[]
    for j in range(0,m):
        t=tuple(list(map(int,raw_input().split())))
        L.append(t)
    for x in range(0,m):
        for y in range(x+1,m):
            if L[y][0]-L[x][0]!=0:
                n=Fraction(L[y][1]-L[x][1],L[y][0]-L[x][0])
                b=L[x][1]-n*L[x][0]
                tup=(n,b)
                if tup in Dic:
                    Dic[tup]+=1
                else:
                    Dic[tup]=1
            else:
                n='i'
                b=L[x][0]
                tup=(n,b)
                if tup in Dic:
                    Dic[tup]+=1
                else:
                    Dic[tup]=1
    ans=1
    for key in Dic:
        ans=max(int((2*Dic[key])**0.5)+1,ans)
    print("Case #"+str(case+1)+": "+str(ans))
