#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>

using namespace std;

bool sort_max_to_min(int a, int b);

int main()
{

    int num_of_loads = 10;
    while (cin >> num_of_loads)
    {
        if (num_of_loads == 0) break;

        vector<int> input_drying_times(num_of_loads);
        for (int i = 0; i < num_of_loads; i++)
        {
            cin >> input_drying_times[i];
        }
        sort(input_drying_times.begin(), input_drying_times.end(), sort_max_to_min);

        // Magically (Well...) Obtained variables
        // Assume input_drying_times is sorted
        //input_drying_times = sort_max_to_min(input_drying_times);
        int washing_time_in_min = 30;
        int time_in_min = washing_time_in_min;
        for (int load_num = 0; load_num < num_of_loads; load_num++)
        {
            if (time_in_min >= (load_num+1)*washing_time_in_min)
            {
                time_in_min += input_drying_times[load_num];
            }
            else
            {
                time_in_min = (load_num + 1)*washing_time_in_min + input_drying_times[load_num];
            }
        }
        cout <<(time_in_min - time_in_min%60)/60 << ":" << setfill('0') << setw(2) << time_in_min%60 << endl;
    }

    return 0;
}

bool sort_max_to_min(int a, int b)
{
    // stub
    return a > b;
}
