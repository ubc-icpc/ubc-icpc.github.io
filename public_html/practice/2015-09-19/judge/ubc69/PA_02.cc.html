#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <algorithm>
#include <algorithm>

using namespace std;

int compare (const void * a, const void * b) {
	return ( *(int*)b - *(int*)a );
}

int main() {
	int n; cin>>n;
	while(n!=0){
		int dry[n];
		for(int i=0;i<n;i++){
			cin>>dry[i];
		}
		qsort (dry, n, sizeof(int), compare);
		int time=0;
		for(int i=0;i<n;i++){
			int temp = (i+1)*30;
			if(temp>time) time = temp;
			time += dry[i];
		}
		cout<<time/60<<':';
		if(time%60==0) cout<<"00"<<endl;
		else if(time%60 < 10) cout<<'0'<<time%60<<endl;
		else cout<<time%60<<endl;
		cin>>n;
	}
    return 0;
}
