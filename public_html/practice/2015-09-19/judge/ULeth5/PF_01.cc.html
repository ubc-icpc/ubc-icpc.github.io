#include<iostream>
#include<map>
#include<algorithm>
using namespace std;

int arr[100005];

int main()
{
   int t;
   cin >> t;
   while(t--)
   {
      int n;
      cin >> n;
      for(int i=0; i<n; i++)
	 cin >> arr[i];
      sort(arr, arr+n);
      bool finished=false;
      for(int i=0; i<n; i++)
      {
	 int next=i+1;
	 if(next>=n)
	    break;
	 while(arr[next]==arr[i])
	 {
	    next++;
	    //cerr << "Next is: " << next << endl;
	    if(next>=n)
	    {
	       finished=true;
	       break;
	    }
	 }
	 if(finished)
	    break;
	 //cerr << "next: " << next << " " << "curr: " << i << endl;
	 //if(arr[next]>arr[i])
	 //{
	    cout << arr[next]-arr[i] << " ";
	    cout << next << endl;
	    //}
	    //else
	    //{
	    //   cerr << "next: " << next << " " << "curr: " << i << endl;
	    //   cout << arr[i]-arr[next] << " ";
	    //   cout << n-next << endl;
	    //}
	 i=next-1;
      }
      cout << endl;
   }

   return 0;
}
