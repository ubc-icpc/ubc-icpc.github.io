#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
   int n;
   while(cin >> n and n)
   {
      int times[35];
      for(int i=0; i<n; i++)
	 cin >> times[i];
      sort(times, times+n);
      reverse(times, times+n);
      int etime=0;
      for(int i=0; i<n; i++)
      {
	 int next=(i+1)*30;
	 if(next<etime)
	    next=etime;
	 etime=next+times[i];
      }
      cout << etime/60 << ":";
      int m=etime%60;
      if(m<10)
	 cout << "0" << m << endl;
      else
	 cout << m << endl;
   }

   return 0;
}
