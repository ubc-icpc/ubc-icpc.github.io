#include<iostream>
#include<cmath>
#include<iomanip>
using namespace std;

double f(const int & me, const int & him, const double & p)
{
   if(me==4 and him<=2)
      return 1.0;
   else if(him==4 and me<=2)
      return 0.0;
   else if(me==3 and him==3)
      return (p*p)/(1-2*p*(1-p));
   else
      return p*f(me+1, him, p) + (1-p)*f(me, him+1, p);
}

int main()
{
   int T;
   cin >> T;
   while(T--)
   {
      double p;
      cin >> p;
      cout << fixed << setprecision(5) << f(0, 0, p) << endl;
   }

   return 0;
}
