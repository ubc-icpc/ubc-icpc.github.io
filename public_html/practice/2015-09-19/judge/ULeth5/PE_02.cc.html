#include<iostream>
#include<string>
#include<iomanip>
using namespace std;

double S, W, L;

double f(const int & c, const int & t, const bool & canWon)
{
   if(c==4)
      return 1.0;
   if(t==4)
      return 0.0;
   double pWin;
   if(c==0 and t==0)
      pWin=20.0/(S+20.0);
   else
   {
      if(canWon)
	 pWin=(20.0+W)/(S+20.0+W);
      else
	 pWin=(20.0-L)/(S+20.0-L);
   }
   return pWin*f(c+1, t, true) + (1-pWin)*f(c, t+1, false);
}

int main()
{
   int T;
   cin >> T;
   while(T--)
   {
      string opponent;
      getline(cin, opponent);
      getline(cin, opponent);
      cin >> S >> W >> L;
      cout << fixed << setprecision(5) << f(0, 0, 0) << endl;
   }
   
   return 0;
}
