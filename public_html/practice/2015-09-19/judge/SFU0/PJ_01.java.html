import java.util.* ;
import java.math.* ;
import java.io.* ;

public class TennisProb 
{
    static double p ;
    static double solve(int W, int L)
    {
        if (W == 4) return 1.0 ;
        if (L == 4) return 0.0 ;
        if (W == 3 && L == 3) return p*p/(1-2*(1-p)*p) ;
        return p * solve(W+1,L) + (1-p) * solve(W,L+1) ;
    }
    public static void main(String[] args) throws Exception
    {
        Scanner input = new Scanner(System.in) ;
        int T = input.nextInt() ;
        while (T --> 0)
        {
            p = input.nextDouble() ;
            System.out.printf("%.5f\n",solve(0,0));
        }
        
    }

}
