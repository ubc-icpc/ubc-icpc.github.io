import java.util.* ;
import java.math.* ;
import java.io.* ;

public class CanucksPsych 
{
    static int p,l,O ;
    static double solve(int W, int L, int curS)
    {
        if (W == 4) return 1.0 ;
        if (L == 4) return 0.0 ;
        return curS/(curS + O + 0.0) * solve(W+1,L,20+p) + (O)/(curS+O+0.0) * solve(W,L+1,20-l) ;
    }
    public static void main(String[] args) throws Exception
    {
        Scanner input = new Scanner(System.in) ;
        int T = input.nextInt() ;
        while (T --> 0)
        {
            input.nextLine() ;
            input.nextLine() ;
            O = input.nextInt() ; p = input.nextInt() ; l = input.nextInt() ;
            System.out.printf("%.5f\n",solve(0,0,20));
        }
    }

}
