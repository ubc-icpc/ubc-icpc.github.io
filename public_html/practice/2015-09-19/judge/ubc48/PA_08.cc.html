#include<bits/stdtr1c++.h>
using namespace std;

bool check(int a, int b){ return a > b; };

int main(){
	int n; cin >> n;
	while (n != 0){
		vector < int > temp;
		queue < int > dl;
		for (int i = 0; i < n; ++i){
			int t; cin >> t;
			temp.push_back(t);
		}
		sort(temp.begin(), temp.end(), check);
		int total = 0;
		for (int i = 0; i < n; ++i){
			total += 30;
			if (dl.size() != 0){
				dl.front() -= 30;
				while (dl.front() <= 0){
					if (dl.front() == 0){
						dl.pop();
						break;
					}
					else if (dl.front() < 0){
						int over = dl.front() * -1;
						dl.pop();
						if (dl.size() > 0) dl.front() -= over;
						else break;
					}
				}
			}
			dl.push(temp[i]);
		}
		while(dl.size()){
			total += dl.front();
			dl.pop();
		}
		if (total % 60 < 10){
			cout << floor((double)total / 60) << ":0" << total % 60 << endl;
		}
		else{
			cout << floor((double)total / 60) << ":" << total % 60 << endl;
		}
		
		cin >> n;
	}
	return 0;
}
