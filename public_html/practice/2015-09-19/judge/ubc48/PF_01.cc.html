#include<bits/stdtr1c++.h>
using namespace std;
int main(){
	int t; cin >> t;
	while (t--){
		int n; cin >> n;
		vector < int > g(n);
		vector < pair < int, int > > ans;
		for (int i = 0; i < n; ++i) cin >> g[i];
		sort(g.begin(), g.end());


		//from sorted to answer, every different encountered
		int l = g[0];
		bool start = true;
		for (int i = 0; i < g.size(); ++i){
			if (g[i] != l){
				ans.push_back(make_pair(g[i] - l, i));
				l = g[i];
			}
		}

		for (int i = 0; i < ans.size(); ++i){
			cout << ans[i].first << " " << ans[i].second << endl;
		}
		cout << endl;
	}
	return 0;
}
