#include <iostream>
#include <algorithm>

using namespace std;

int main(){

   int cases, index;
   cin >> cases;
   while(cases>0)
   {
      cin >> index;
      int *arr = new int[index];
      for (int x=0; x<index; x++)
      {
	 cin >> arr[x];
      }
      sort(arr, arr+index);
      int i, j = 0;
      for (i=0; i<index; i=j)
      {
	 for (j=i+1; j<index && (arr[i] == arr[j]); j++)
	 {
	    ;
	 }
	 if(j<index)
	 {
	    cout << arr[j]-arr[i] << " " << j << endl;
	 }
      }
      cout << endl;
      cases--;
   }
   return 0;
}
