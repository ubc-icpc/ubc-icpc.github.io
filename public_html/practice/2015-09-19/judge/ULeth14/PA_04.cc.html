#include <iostream>
#include <algorithm>
using namespace std;


void inputarray(int [], int);
int totaltime(int [], int);
void converttime(int, int&, int&);
void printtime(int, int);


int main(){

   int arr[30];
   int n, time, hours, minutes;
   cin >> n;
   if (n == 0)
   {
      return 0;
   }
   do
   {
      inputarray(arr,n);
      sort(arr, arr + n);
      reverse(arr, arr + n);
      time = totaltime(arr, n);
      converttime(time, hours, minutes);
      printtime(hours, minutes);
      cin >> n;
   }while(n != 0);

   return 0;
}

void inputarray(int a[], int n)
{
   for(int i=0; i<n;i++){
      cin >> a[i];
   }
}

int totaltime(int a[], int n)
{
   /*  int time = 30;
   int i=0;

   {
      time = time + a[i];
      i++;
   }
   return time;*/
   int finish = 0;
   for (int i=0; i<n; i++)
   {
      int next_ready=(i+1)*30;
      if (next_ready<finish)
      {
	 next_ready = finish;
      }
      finish = next_ready + a[i];
   }
   return finish;
}

void converttime(int a, int& hours, int& minutes)
{
   int b;
   hours = a/60;
   b = a - (hours*60);
   minutes = b;
}


void printtime(int a, int b)
{
   cout << a << ":";
   if (b == 0){
      cout << "00" << endl;
   }
   else if(b<10)
   {
      cout << "0" << b << endl;
   }
   else cout << b << endl;
}
