// problem E - canucks phychology

#include <iostream>
#include <string>
#include <iomanip>

using namespace std;

const int STR = 20;

double f(int w, int l, int r, int str, int wc, int lc)
{
   double p;
   if (w == 4 && l < 4)
      return 1.0;
   if (w < 4 && l == 4)
      return 0.0;
   
   if (r == 1)
      p = (double)(STR + wc) / (str + wc + STR);
   else if (r == 0)
      p = (double)(STR + lc) / (str + lc + STR);
   else
      p = (double)STR / (STR + str);
   
   return (p * f(w + 1, l, 1, str, wc, lc)) + ((1-p) * f(w, l + 1, 0, str, wc, lc));
}

void solve ()
{
   string s;
   int str, wc, lc;
   
   getline (cin, s);
   cin >> str >> wc >> lc;
   cin.ignore ();

   lc = -lc;

   cout << fixed << setprecision (5) << f (0, 0, -1, str, wc, lc) << endl;
}

int main ()
{
   int cases;

   cin >> cases;
   cin.ignore();

   while (cases--)
      solve ();

   return 0;
}
