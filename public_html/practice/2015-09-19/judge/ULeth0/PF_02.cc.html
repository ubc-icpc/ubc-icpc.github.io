// f.cc
// Problem F - orthogonal bar graphs

#include <iostream>
#include <algorithm>

using namespace std;

void graph ();

int main ()
{
   int cases;

   cin >> cases;

   while (cases--)
      graph ();

   return 0;
}

void graph ()
{
   int bars;
   int a[100001];
   int b[100001] = {0};
   
   cin >> bars;

   for (int i = 0; i < bars; i++)
      cin >> a[i];

   sort (a, a + bars);
   
   for (int i = 0; i < bars - 1; i++)
   {
      if (a[i + 1] == a[i])
	 continue;

      //  if (a[i + 1] < a[i])
//	 b[bars - i - 1] += (a[i] - a[i + 1]);

      if (a[i + 1] > a[i])
	 b[i + 1] += (a[i + 1] - a[i]);
   }

   for (int i = 0; i < bars; i++)
      if (b[i] > 0)
	 cout << b[i] << " " << i << endl;
   cout << endl;
}
