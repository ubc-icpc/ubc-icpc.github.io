// Problem A - doing laundry

#include <iostream>
#include <algorithm>

using namespace std;

void wash (int w);

int main ()
{
   int load;

   cin >> load;
   
   while (load != 0)
   {
      wash (load);
      cin >> load;
   }
   
   return 0;
}

void wash (int w)
{
   int wT, total, hours, mins;
   int loads[30];
   
   for (int i = 0; i < w; i++)
      cin >> loads[i];

   sort (loads, loads + w);
   reverse (loads, loads + w);

   total = 0;

   for (int i = 0; i < w; i++)
   {
      wT = (i + 1) * 30;
      if (wT < total)
	 wT = total;
      total = wT + loads[i];
   }
   
   hours = total / 60;
   mins = total % 60;

   if (mins == 0)
      cout << hours << ":" << "00" << endl;
   else if (mins < 10)
      cout << hours << ":" << "0" << mins << endl;
   else
      cout << hours << ":" << mins << endl;
 }
