// j.cc
// problem j - tennis probability

#include <iostream>
#include <cmath>
#include <iomanip>
#include <algorithm>

using namespace std;

double f(int w, int l, double p)
{
   if (w == 4 && l < 3)
      return 1.0;
   if (w < 3 && l == 4)
      return 0.0;
   if (w == 3 && l == 3)
      return (double)(p * p) / (1 - (2 * p * (1 - p)));

   return (p * f(w + 1, l, p)) + ((1 - p) * f(w, l + 1, p));
}

void solve ()
{
   double p;
   cin >> p;

   cout << fixed << setprecision (5) << f(0, 0, p) << endl;
}

int main ()
{
   int cases;
   cin >> cases;

   while (cases--)
      solve ();

   return 0;
}
