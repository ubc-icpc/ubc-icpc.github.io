import java.util.Collections;
import java.util.Scanner;
import java.util.ArrayList;


public class A {
	static Scanner in;
	public static void main(String[] args) {
		
		in = new Scanner(System.in);
		while(true){
			int numOfLoads = in.nextInt();
			if (numOfLoads == 0) 
				break;
			ArrayList<Integer> loadValues = new ArrayList<>();
			for (int i = 0; i < numOfLoads; i++) {
				loadValues.add(in.nextInt());
			}
			solve(loadValues);
		}
		return;
	}
	
	public static void solve(ArrayList<Integer> arr) {
		int sum = 30;
		int leftOver = 0;
		Collections.sort(arr);
		for (int i = 0; i < arr.size(); i++){
			int n = arr.get(arr.size() - i - 1);
			if (n >= 30) {
				leftOver += n - 30;
				sum += n;
			} else if (n <= leftOver) {
				leftOver -= 30 - n;
				sum += n;
			} else if (i == arr.size() - 1) {
				sum += n;
			} else {
				if (leftOver == 0){
					sum += 30;
				}
				else {
					sum += n - leftOver;
					leftOver = 0;	
				}
			}
		}
		print(sum);
	}
	
	public static void print(int n) {
		System.out.print(n/60);
		if (n%60 < 10)
			System.out.println(":0"+n%60);
		else
			System.out.println(":"+n%60);
	}
}
