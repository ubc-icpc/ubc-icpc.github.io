#include <bits/stdc++.h>

using namespace std;

typedef long double ld;

ld win(int strength, int S, int W, int L, int wins, int losses) {
  if (wins == 4) return 1.0L;
  if (losses == 4) return 0.0L;
  ld winprob = strength*1.0L / (strength + S);
  return winprob*win(20+W, S, W, L, wins + 1, losses)
          + (1.0L-winprob)*win(20-L, S, W, L, wins, losses+1);
}

int main() {
  int n; cin >> n;
  while (n--) {
    string temp; getline(cin, temp); getline(cin, temp);
    //string temp; cin >> temp;
    //cout << temp << endl;
    int S, W, L; cin >> S >> W >> L;
    cout << setprecision(5) << fixed << win(20, S, W, L, 0, 0) << endl;
  }
  return 0;
}
