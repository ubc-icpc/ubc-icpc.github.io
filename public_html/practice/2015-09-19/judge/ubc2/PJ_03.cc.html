#include <bits/stdc++.h>

using namespace std;

typedef long double ld;

ld win(ld p, int wins, int losses, ld mult) {
  if (wins >= 4 && wins > losses + 1) return 1.0L;
  if (losses >= 4 && losses > wins + 1) return 0.0L;
  if (mult < 1e-10) return p;
  return p*win(p, wins+1, losses, mult*p) + (1.0L-p)*win(p, wins, losses+1, mult*(1.0L-p));
}

int main() {
  int N; cin >> N;
  while(N--) {
    ld p; cin >> p;
    cout << fixed << setprecision(5) << win(p, 0, 0, 1.0L) << endl;
  }
  return 0;
}
