#include<iostream>
#include<string>
#include<iomanip>
using namespace std;
int S,W,L;
double solve(int w, int l, bool prev_win){
   if(w == 4)
      return 1.0;
   if(l == 4)
      return 0.0;
   int chance;
   if(prev_win){
      chance = 20+W;
   }
   else{
      chance = 20-L;
   }
   return chance/((double)S+chance)*solve(w+1,l,true) +
      S/((double)S+chance)*solve(w,l+1,false);
}

int main(){
   int t; cin >> t;
   string s;
   getline(cin, s);
   while(t--){
      getline(cin,s);
      cin >> S >> W >> L;
      //cout << S << " " << W << " " << L << endl;
      double p = 20.0/(S+20.0)*solve(1,0,true) + S/(S+20.0)*solve(0,1,false);
      cout << fixed << setprecision(5) << p << endl;
      getline(cin,s);
   }
   return 0;
}
