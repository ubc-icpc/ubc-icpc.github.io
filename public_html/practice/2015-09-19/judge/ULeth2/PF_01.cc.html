#include<iostream>
#include<algorithm>
using namespace std;
const int MAX = 1000010;
int A[MAX];
const int MAX2 = 100010;
int B[MAX2];
int main(){
   int T;
   cin >> T;
   while(T--){
      int n; cin >> n;
      for(int i = 0; i < n; i++){
	 cin >> B[i];
      }
      sort(B,B+n);
      reverse(B,B+n);
      fill(A,A+B[0]+1,0);
      int pos = n-1;
      int length = 0;
      for(int i = 0; i <= B[0]; i++){
	 //cout << i << " " << B[pos] << " " << length << endl;
	 while(i > B[pos]){
	    pos--;
	    length++;
	 }
	 A[length]++;
      }
      for(int i = 1; i <= B[0]; i++){
	 if(A[i] != 0){
	    cout << A[i] << " " << i << endl;
	 }
      }
      cout << endl;
   }
   return 0;
}
