#include<iostream>
#include<algorithm>
using namespace std;

const int MAX = 35;
int A[MAX];

int main(){
   int n;
   while(cin >> n && n){
      int time = 0;
      for(int i = 0; i < n; i++){
	 cin >> A[i];
      }
      sort(A,A+n);
      reverse(A,A+n);
      int timesave = 0;
      for(int i = 0; i < n; i++){
	 time += 30 + A[i];
	 
	 if(i != 0){
	    if(timesave > 30){
	       time -= 30;
	       timesave -=30;
	    }
	    else{
	       time -= timesave;
	       timesave = 0;
	    }
	 }
	 timesave += A[i];
      }
	 
      cout << time/60 << ":";
      if(time%60 < 10)
	 cout << "0";
      cout << time%60 << endl;
   }
   return 0;
}
