#include<iostream>
#include<string>
using namespace std;

void strip_zero(string &d){
   int q;
   for(q = d.size()-1; q > 0 && d[q] == '0'; q--) ;
   d = d.substr(0,q+1);
}

void mult2(string &d){
   int car = 0;
   for(int q = d.size()-1; q>= 0; q--){
      int num = d[q] - '0';
      num*= 2;
      num += car;
      d[q] = num%10 + '0';
      car = num/10;
   }
}

bool solve(string s){
   int p;
   for(p = 0; p < (int)s.size() && s[p] != '.'; p++) ;
   if(p == (int)s.size())
      return true;
   string d = s.substr(p+1);
   strip_zero(d);
   //cout << d << endl;
   while(d.size() > 1 and d.back() == '5'){
      mult2(d);
      strip_zero(d);
   }
   //cout << d << endl;
   if(d == "5" or d == "" or d == "0")
      return true;
   else
      return false;
}

int main(){
   string s;
   while(cin >> s){
      if(solve(s)){
	 cout << "Exact.";
      }
      else{
	 cout << "Some precision lost.";
      }
      cout << endl;
   }
   return 0;
}
