#include<iostream>
#include<iomanip>

using namespace std;
double p;
double solve(int w, int l){
   if(w == 4)
      return 1.0;
   if(l == 4)
      return 0.0;
   if(w == 3 and l == 3)
      return (p*p)/(1-2*p*(1-p));
   return p*solve(w+1,l) + (1-p)*solve(w,l+1);
}

int main(){
   int t; cin >> t;
   while(t--){
      cin >> p;
      cout << fixed << setprecision(5) << solve(0,0) << endl;
   }
   return 0;
}
