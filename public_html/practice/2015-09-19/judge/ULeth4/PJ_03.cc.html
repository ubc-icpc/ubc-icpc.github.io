#include<iostream>
#include<algorithm>
#include <iomanip>
using namespace std;


int solve(double p)
{
   double x = (p*p)/(1-2*p*(1-p));
   cout << fixed << setprecision(5);
   cout << p*p*p*p + 4*p*p*p*p*(1-p)+10*p*p*p*p*(1-p)*(1-p)
      +20*p*p*p*(1-p)*(1-p)*(1-p)*x << endl;
   
}

int main()
{
   int cases;
   double P[200]; 
   cin >> cases;
   for (int i=0; i<cases; i++)
      cin>>P[i];
   int j= 0;
   while(j!=cases)
   {
      solve(P[j]);
      j++;
   }
   return 0;
}
