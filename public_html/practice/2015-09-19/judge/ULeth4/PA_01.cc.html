#include<iostream>
#include<algorithm>
using namespace std;

int main ()
{

   int A[30];
   int loads;

   
   cin >> loads;
   while (loads!=0)
   {
      for (int i=0; i<loads; i++)
	 cin >> A[i];

      sort(A,A+loads);
      reverse(A,A+loads);
      
      int finish=0;
      
      for(int i=0; i<loads; i++)
      {
	 int next_ready=(i+1)*30;
	 if (next_ready <finish)
	    next_ready=finish;
	 finish = next_ready+A[i];
      }						\
      cout << finish/60 << ":";
      if (finish%60 < 10)
	 cout << "0" << finish%60;
      else
	 cout << finish%60;
      cout <<endl; 
      cin >> loads;
   } 

   return 0;
}
