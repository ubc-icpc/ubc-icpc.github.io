#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
   int arr[100000];
   int Tcase =0;
   
   cin >> Tcase;
   for(int k = 0; k < Tcase; k++)
   {
      int n = 0;
      cin >> n;
      for(int i = 0; i < n; i++)
      {
	 cin >> arr[i];
      }
      
      sort(arr, arr + n);
      
      int j = 0;
      for(int i = 0; i < n; i = j)
      {
	 for(j = (i+1); j < n && arr[i] == arr[j]; j++);
	 if(j<n)
	    
	    cout << (arr[j] - arr[i]) << " " << j << endl;
	 
	 
      }
   }
   cout << endl;
   return 0;
}
