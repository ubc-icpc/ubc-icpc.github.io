#include<iostream>
#include<map>
#include<string>
#include<fstream>
#include<vector>
#include<set>
#include<algorithm>
#include<stdio.h>
#include <iomanip>

using namespace std;

int main(){
	
	int m,n,t;
	cin>>m;
	while(m--){
		cin>>n;
		vector <int> v;
		for(int i=0;i<n;i++){
			cin>>t;
			v.push_back(t);
		}
		sort(v.begin(),v.end());
		
		int  p = v[0];
		for(int i=1;i<n;i++){
			if(v[i] != p){
				cout<<v[i]-p<<" "<<i<<endl;
				p = v[i];
			}
		}
		cout<<endl;
	}
	
	return 0;
}
