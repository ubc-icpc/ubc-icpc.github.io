#include<iostream>
#include<map>
#include<string>
#include<fstream>
#include<vector>
#include<set>
#include<algorithm>
#include<stdio.h>
#include <iomanip>

using namespace std;

int main(){
	
	int n,t;
	while(cin>>n){
		if(n==0)
			break;
		vector <int> v;
		for(int i=0;i<n;i++){
			cin>>t;
			v.push_back(t);
		}
		sort(v.begin(),v.end());
		int tt=0;
		int j=1;
		for(int i=n-1;i>=0;i--){
		
			tt = max( tt + v[i] , j*30+v[i] );
			j++;
		}
		if(tt%60 < 10)
			cout<<tt/60<<":0"<<tt%60<<endl;
		else
			cout<<tt/60<<":"<<tt%60<<endl;
	}
	
	
	return 0;
}
