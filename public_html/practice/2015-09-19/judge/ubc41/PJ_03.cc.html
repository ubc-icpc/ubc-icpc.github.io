#include <bits/stdtr1c++.h>
using namespace std;

using ld = long double;

ld recurse(int wins, int losses, ld win, ld st)
{
	if (wins > 3 && wins - losses >= 2) { return 1; }
	if (losses > 3 && losses - wins >= 2) { return 0; }
	if (st < 1E-12) { return 0.5; }
	return win * recurse(wins + 1, losses, win, win * st) + (1 - win) * recurse(wins, losses + 1, win, (1 - win) * st);
}

int main()
{
	int t; cin >> t;
	while (t--)
	{
		ld win;
		cin >> win;
		cout << fixed << setprecision(5) << recurse(0, 0, win, 1) << endl;
	}
}
