#include <bits/stdtr1c++.h>
using namespace std;

int main()
{
	int t; cin >> t;
	while (t--)
	{
		int n; cin >> n;
		vector<int> bar(n);
		for (int i = 0; i < n; i++)
		{
			int x; cin >> x;
			bar[i] = x;
		}

		sort(bar.begin(), bar.end());

		vector<int> ort(100001);
		for (int i = 0; i < n - 1; i++)
		{
			if (bar[i + 1] > bar[i])
			{
				ort[i + 1] += bar[i + 1] - bar[i];
			}
		}

		for (int i = 0; i < ort.size(); i++)
		{
			if (ort[i] > 0)
			{
				cout << ort[i] << " " << i << endl;
			}
		}
		cout << endl;
	}
}
