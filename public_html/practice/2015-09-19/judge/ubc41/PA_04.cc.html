#include <bits/stdtr1c++.h>
using namespace std;

int main()
{
    while (1)
    {
		int n; cin >> n;
		if (n == 0) return 0;

		vector<int> times(n);
		for (int i = 0; i < n; i++)
		{
			int x; cin >> x;
			times[i] = x;
		}

		sort(times.begin(), times.end());
		int time = 0;
		int loads = 0;
		int dryer = 0;
		int washer = 30;
		while (1)
		{
			if (--washer <= 0)
			{
				washer = 30;
				loads++;
			}

			if (--dryer <= 0)
			{
				if (times.size() == 0)
				{
					break;
				}

				if (loads > 0)
				{
					loads--;
					dryer = times.back();
					times.pop_back();
				}
			}

			time++;
		}
		time += 1;

		cout << time / 60 << ":" << (time % 60 < 10 ? "0" : "") << time % 60 << endl;
	}
}
