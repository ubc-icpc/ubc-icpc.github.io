#include <bits/stdc++.h>
#include <stdint.h>
using namespace std;
int main() {
	int n;
	while(cin >> n && n) {
		vector<int> ld(n);
		for(int i = 0; i < n; ++i) cin >> ld[i];

		sort(ld.begin(), ld.end());
		reverse(ld.begin(), ld.end());

		int finish = 0;
		for(int i = 0; i < n; ++i) {
			int next_ready = (i+1)*30;
			if(next_ready < finish) {
				next_ready = finish;
			}
			finish = next_ready + ld[i];
		}

		cout << finish / 60 << ":" << setw(2) << setfill('0') << finish % 60 << endl;
	}
	return 0;
}
