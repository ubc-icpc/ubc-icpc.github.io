#include <bits/stdc++.h>
#include <stdint.h>
using namespace std;
string add(string a, string b) {
	string l;
	l.resize(a.size());
	int car = 0;
	for(int i = a.size()-1; i >= 0; --i) {
		int sum = a[i]-'0' + b[i]-'0'+car;
		l[i] = sum%10+'0';
		car = sum/10;
	}
	return l;
}
int main() {
	string l;
	while(getline(cin, l)) {
		for(int i = 0; i < l.size(); ++i) {
			if(l[i] == '.') {
				l = l.substr(i+1, l.size()-i-1);
				break;
			}
		}
		
		while(l.size() > 1 && l.back() == '0') l.pop_back();
		while(l.size() > 1 && (l.back() == '0' || l.back() == '5')) {
			l = add(l, l);
			while(l.size() > 1 && l.back() == '0') l.pop_back();
		}
		if(l.back() == '0' || l.back() == '5') {
			cout << "Exact.";
		} else {
			cout << "Some precision lost.";
		}
		cout << endl;
	}
	return 0;
}
