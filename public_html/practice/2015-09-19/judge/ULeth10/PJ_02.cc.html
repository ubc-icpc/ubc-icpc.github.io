#include <bits/stdc++.h>
#include <stdint.h>
using namespace std;
double prob(int a, int b, double p) {
	if(a==4) return 1.0;
	if(b==4) return 0.0;
	if(a==3&&b==3) return (-p*p)/(2*p*(1-p)-1);
	return p*prob(a+1, b, p)+(1-p)*prob(a, b+1, p);
}
int main() {
	int T;
	cin >> T;
	while(T--) {
		double p;
		cin >> p;
		cout << fixed << setprecision(5) << prob(0,0,p) << endl;
	}
	return 0;
}
