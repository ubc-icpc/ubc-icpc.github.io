#include <bits/stdc++.h>
#include <stdint.h>
using namespace std;
int S, w, l;
double prob(int a, int b, int win) {
	if(a >= 4) return 1.0;
	if(b >= 4) return 0.0;
	int s = 20;
	if(win != -1) {
		if(win) s += w;
		else s -= l;
	}
	return    (double)(s)/(S+s)  * prob(a+1, b, true) + 
	       (1-(double)(s)/(S+s)) * prob(a, b+1, false);
}
int main() {
	int n;
	cin >> n;
	while(n--) {
		string team;
		getline(cin, team);
		getline(cin, team);
		cin >> S >> w >> l;
		cout << setprecision(5) << fixed << prob(0, 0, -1) << endl;
	}
	return 0;
}
