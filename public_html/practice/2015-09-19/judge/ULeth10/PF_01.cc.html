#include <bits/stdc++.h>
#include <stdint.h>
using namespace std;
int main() {
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		vector<int> bl(n);
		for(int i = 0; i < n; ++i) cin >> bl[i];

		sort(bl.begin(), bl.end());

		for(int i = 0; i < n; ++i) {
			int j;
			for(j = i+1; j < n && bl[j] <= bl[i]; ++j);
			if(j < n) {
				cout << bl[j]-bl[i] << " " << j << endl;
			}
			i = j-1;
		}
		cout << endl;
	}
	return 0;
}
