#include <iostream>
#include <cmath>
#include <iomanip>

using namespace std;

double win, loss, strength;

double prob(int a, int b, double s)
{
   double p = s/(s+strength);
   if(a==4)
      return 1.0;
   if(b==4)
      return 0.0;
   
   return p*prob(a+1, b, 20+win) + (1-p)*prob(a, b+1, 20-loss);

}
      



int main()
{
   int n;
   string s;
   cin >> n;

   for(int i=0; i < n; i++)
   {
      getline(cin, s);
      getline(cin, s);
      cin >> strength >> win >> loss;
      cout << setprecision(5) << fixed << prob(0, 0, 20) << endl;
   }

   return 0;
}
