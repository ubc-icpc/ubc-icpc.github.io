#include <algorithm>

using namespace std;

int main(){

   int A[100000];
   int t, n;

   for(int i=0; i < t; i++)
   {

      cin >> n;

      for(int i=0; i < n; i++)
      {
	 cin >> A[i];
      }
      sort(A, A+n);
      int number, length, index, position;

      for(int i=0; i < n-1; i++)
      {
	 if(A[i] < A[i+1])
	 {
	    number = A[i+1] - A[i];
	    length = i +1;
	    cout << number << ' ' << length << endl;
	 }
      }
      cout << endl;
   }
	 
	 
	 
	 
   

   

   
}
