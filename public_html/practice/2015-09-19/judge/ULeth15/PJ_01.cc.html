#include <iostream>
#include <iomanip>
#include <cmath>


using namespace std;

double prob(int a, int b, double p)
{
   if(a==4)
      return 1.0;
   if(b==4)
      return 0.0;
   if(a==3 && b ==3)
      return pow(p, 2)/(1-2*p*(1-p));

   return p*prob(a+1, b, p) + (1-p)*prob(a, b+1, p);
}
	      
int main()
{
   double p;
   int t;

   cin >> t;
   for(int i=0; i < t; i++)
   {
      cin >> p;
      cout << setprecision(5) << fixed << prob(0, 0, p) << endl;
   }

   return 0;
}
