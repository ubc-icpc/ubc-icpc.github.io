#include <iostream>
#include <string>

using namespace std;

void erase(string &n)
{

   if(n.length() == 0)
      return;

   if(n[n.length()-1] == '0')
   {
      
      for(int i=n.length()-1; n[i] == '0'; i--)
      {
	 n.erase(i, 1);
      }
   }
   for(int i=0; i < n.length(); i++)
   {
      if(n[i] =='.'){
	 
	 n = n.substr(i+1);
	 return;
      }
   }
}

void multiply(string &n)
{
   if(n.length() == 0)
      return;
   int carry=0, num;
   
   for(int i=n.length()-1; i >= 0; i--)
   {
      num = n[i] - '0';
      num = num*2 + carry;
      n[i] = num%10 + '0';
      carry = num/10;
   }

}
   
   

int main()
{
   string n;
   while(cin >> n)
   {
      
      erase(n);

      if(n.length() ==0 || n.back()=='5' || n.back()=='0')
      {
	 while(n.length() > 1 && (n.back()=='5' || n.back()=='0'))
	 {
	    multiply(n);
	    n.pop_back();
	 }
	 if(n.back()=='5' || n.back()=='0' || n.length() == 0)
	    cout << "Exact." << endl;
	 else
	    cout << "Some precision lost." << endl;
	 
      }
      else
	 cout << "Some precision lost." << endl;
     
   }
   return 0;
}
