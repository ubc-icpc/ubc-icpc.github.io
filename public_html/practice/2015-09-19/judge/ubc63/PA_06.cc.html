#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
int n,h,m,t,w;


bool compare(int a, int b) {
    return a > b;
}
int main() {
    cin >> n;
    while (n > 0) {
        vector<int> loads;
        for (int i = 0; i < n; i++) {
            int temp;
            cin >> temp;
            loads.push_back(temp);
        }
        t = w = 0;
        sort(loads.begin(), loads.end(), compare);
        for (int i = 0; i < n; i++) {
            w = (i+1) * 30; //minimal washing time needed for load i
            w = max (w, t); //take longest of minimal wash time or current wait time
            t = w + loads[i]; //add drying time
        }
        h = t/60;
        m = t % 60;
        if (m < 10)  cout << h << ":" << '0' <<m << endl;
        else cout << h <<":" << m << endl;
        cin >> n;
    }
    return 0;
}
