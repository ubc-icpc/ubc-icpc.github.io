#include <bits/stdc++.h>
using namespace std;

#define CAN_STR 20

int s, w, l;

long double prob(int cs, int wins, int losses) {
    if (wins == 4) return 1.0;
    if (losses == 4) return 0.0;

    long double p_win = 1.0 * cs / (cs + s);
    long double p_loss = 1.0 * s / (cs + s);    

    return (prob(CAN_STR + w, wins + 1, losses) * p_win) + (prob(CAN_STR - l, wins, losses + 1) * p_loss);
}

int main() {
    int n;
    cin >> n;
    while (n--) {
        string garbage;
        getline(cin, garbage);
        getline(cin, garbage);

        cin >> s >> w >> l;

        printf("%.5f\n", (double)prob(CAN_STR, 0, 0));
    }
}
