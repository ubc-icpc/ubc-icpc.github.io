#include <bits/stdc++.h>
using namespace std;

double deuce(double p) {
    double w = p * p;
    double l = (1.0 - p) * (1.0 - p);
    return w / (w + l);
}

double solve(double p, int w, int l) {
    if (w == 4) return 1.0;
    if (l == 4) return 0.0;
    if (w == 3 && l == 3) return deuce(p);

    return p * solve(p, w + 1, l) + (1.0 - p) * solve(p, w, l + 1);
}

int main() {
    int n;
    cin >> n;
    while (n--) {
        double p;
        cin >> p;
        printf("%.5f\n", solve(p, 0, 0));
    }
}
