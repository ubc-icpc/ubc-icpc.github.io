#include <bits/stdc++.h>

using namespace std;

#define vi vector<int>

bool comp(int a, int b) {
    return a > b;
}

int main() {
    int n;
    while (1) {
        cin >> n;
        if (!n) return 0;

        vi v;

        for (int i = 0; i < n; ++i) {
            int x;
            cin >> x;
            v.push_back(x);
        }

        sort(v.begin(), v.end(), comp);

        int time_used = 0;
        int dryer_use = 0;

        for (int i = 0; i < n; ++i) {
            time_used += 30;
            dryer_use -= 30;
            if (dryer_use < 0) dryer_use = 0;

            dryer_use += v[i];
        }

        time_used += dryer_use;

        printf("%d:%02d\n", time_used / 60, time_used % 60);

    }
}
