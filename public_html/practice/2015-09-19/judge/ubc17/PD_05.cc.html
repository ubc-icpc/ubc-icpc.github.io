#include <bits/stdc++.h>
using namespace std;

long long mn[512][512];
long long md[512][512];

long long px[512];
long long py[512];

int main() {
    int n;
    cin >> n;
    for (int qc = 1; qc <= n; ++qc) {
        cout << "Case #" << qc << ": ";
        int p;
        cin >> p;

        for (int i = 0; i < p; ++i) {
            cin >> px[i] >> py[i];
        }

        if (p < 3) {
            cout << p << "\n";
            continue;
        }

        memset(mn, 0, 512*512);
        memset(md, 0, 512*512);

        for (int i = 0; i < p; ++i) {
            for (int j = 0; j < p; ++j) {
                if (i == j) continue;

                mn[i][j] = py[i] - py[j];
                md[i][j] = px[i] - px[j];
            }
        }

        int max = 2;

        for (int i = 0; i < p; ++i) {
            for (int j = 0; j < p; ++j) {
                if (i == j) continue;
                int count = 2;
                for (int k = j+1; k < p; ++k) {
                    if (i == k) continue;
                    if (mn[i][j] * md[i][k] == mn[i][k] * md[i][j]) {
                        ++count;
                    }
                }
                if (count > max) max = count;
            }
        }

        cout << max << "\n";
    }
}
