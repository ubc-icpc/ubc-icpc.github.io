#include<bits/stdtr1c++.h>
using namespace std;

typedef long double ld;
int T, S, W, L;
string str;

ld winchance(int c, int o, int a, int b) {
    if (a <= 0) return 1.0;
    if (b <= 0) return 0.0;
    return ld(c)/(c+o) * winchance(20+W, S, a-1, b) +
           ld(o)/(c+o) * winchance(20-L, S, a, b-1);
}

int main () {
    cin >> T;
    while (T--) {
        getline(cin, str);
        getline(cin, str);
        cin >> S >> W >> L;
        cout << fixed << setprecision(5) << winchance(20, S, 4, 4) << endl;
    }
    return 0;
}
