#include<bits/stdtr1c++.h>
using namespace std;

int T, N, M[100005];
vector<pair<int, int> > P;

int main () {
    cin >> T;
    while (T--) {
        cin >> N;
        M[0] = -1;
        for (int i = 1; i <= N; ++i)
            cin >> M[i];
        sort(M, M + N + 1);
        P = vector<pair<int, int> >();
        for (int i = 1; i <= N; ++i)
            if (M[i] != M[i-1]) P.push_back(make_pair(i, M[i]));
        for (int i = 1; i < P.size(); ++i)
            cout << P[i].second - P[i-1].second << " " << P[i].first - 1 << "\n";
        cout << "\n";
    }
    return 0;
}
