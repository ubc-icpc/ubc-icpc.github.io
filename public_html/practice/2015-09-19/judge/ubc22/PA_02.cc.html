#include<bits/stdtr1c++.h>
using namespace std;

int N, M[35], O, R;

bool C(const int& a, const int& b) {
    return a > b;
}

int main () {
    while (1) {
        cin >> N;
        if (!N) break;
        O = R = 0;
        for (int i = 0; i < N; ++i) cin >> M[i];
        sort(M, M+N, C);
        for (int i = 0; i < N; ++i) {
            R += max(0, 30 - O);
            O = max(0, O - 30);
            R += M[i];
            O += M[i];
        }
        cout << R/60 << ":" << setfill('0') << setw(2) << R%60 << endl;
    }
    return 0;
}
