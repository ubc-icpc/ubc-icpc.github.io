#include <iostream>
#include <algorithm>
using namespace std;
int n;
int set[50];

int comp(int a, int b) {
    return a > b;
}

void solve() {
    sort(set, set + n, comp);
    int wash = 0;
    int ans = 0;
    for (int i = 0; i < n; i++) {
        wash += 30;
        ans = max(ans, wash);
        ans += set[i];
    }
    int hour = ans / 60;
    int minute = ans % 60;
    cout<<hour<<":";
    if(minute < 10)
        cout<<"0";
    cout<<minute<<endl;
}
int main() {
    while(true) {
        cin>>n;
        if(n == 0)
            break;
        for(int i = 0; i < n; i++) 
            cin>>set[i];
        solve();
    }
}
