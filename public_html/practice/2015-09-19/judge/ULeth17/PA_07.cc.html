#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int main() {
	int n;
	while (cin >> n  && n) {
		int dryingTime[31] = {0};
		// Start with the initial wash for 30 minutes
		int i, totalTime = 30;
		int carryover = 0;
		// Read in input
		for(i = 0; i < n; i++) {
			cin >> dryingTime[i];
		}
		// Sort drying time
		sort(dryingTime, dryingTime + n);
		for(i = n-1; i > 0; i--) {
			carryover += dryingTime[i];
			// If we the total drying time so
			if(carryover < 30) {
				totalTime += 30;
				carryover = 0;
			} else {
				totalTime += dryingTime[i];
				carryover -= 30;
			}
		}
		totalTime += dryingTime[0];
		cout << totalTime/60 << ":"; 
		if (totalTime%60 < 10) {
			cout << 0 << totalTime%60;
		} else
			cout << totalTime%60;
		cout << endl;
	}

	return 0;
}
