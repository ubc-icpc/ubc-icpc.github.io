#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

string mult(string input) {
	string out = "";
	int carry = 0, mult, temp;
	for(int i = input.length()-1; i >= 0; i--) {
		if(input[i] == '.') {
			out = "0." + out;
			break;
		}
		temp = input[i]-'0';
		mult = temp * 2 + carry;
		carry = mult/10;
		out = to_string(mult%10) + out;
	}
	for(int i = 2; i < input.length(); i++) { 
		if(input[i] != '0')
			return out;
	}
	return "1";
}

int main() {
	string in;
	while(cin >> in) {
		bool printed = false;
		for(int i = 0; i < 120; i++) {
			in = mult(in);
			if(in == "1"){
				cout << "Exact." << endl;
				printed = true;
				break;
			}
		}
		if(!printed)
			cout << "Some precision lost." << endl;
	}
	return 0;
}
