#include <iostream>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

int main () {
	int t, n, i, j, k;
	cin >> t;
	for(i = 0; i < t; i++) {
		cin >> n;
		vector<int> bars;
		for(j = 0; j < n; j++) {
			int temp;
			cin >> temp;
			bars.push_back(temp);
		}
		sort(bars.begin(), bars.end());
		int currLength = 0;
		int prev = bars[0];
		int count = 0;

		for(j = 0; j < bars.size(); j++) {
			if (bars[j] == prev) {
				currLength++;
			} else {
				
				cout << bars[j] - prev << " " << currLength << endl;
				currLength++;
				prev = bars[j];
			}
		}
		if(i < t - 1)
			cout << endl;
	}
	return 0;
}
