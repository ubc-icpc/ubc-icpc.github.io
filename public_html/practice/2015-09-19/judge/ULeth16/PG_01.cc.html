#include <iostream>
#include <string>
#include <stdlib.h>
using namespace std;

string func(string x){
	string final = x;
	int carry = 0;
	for(int i = x.length()-1;i >= 0;i--){
		int temp = (x[i]-'0')*2 + carry;
		carry = temp/10;
		final[i] = to_string(temp%10)[0];
	}
	return final;
}

int main(){
	string x;
	while(getline(cin, x)){
		x = x.substr(x.find('.')+1);
		int zero = x.length()-1;
		while(true){
			int val = -1;
			for(int i = zero;i >= 0;i--){
				if(x[i] != '0'){
					val = (x[i]-'0');
					break;
				}
				else
					zero = i;
			}
			if(val == 5)
				x = func(x);
			else if(val == -1){
				cout << "Exact." << endl;
				break;
			}
			else{
				cout << "Some precision lost." << endl;
				break;
			}
		}
	}
	return 0;
}
