#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main(){
	int n;

	while(cin >> n && n != 0){
		int arr[30] = {0};
		int total = 0;
		int buffer = 0;
		for(int i = 0;i < n;i++){
			cin >> arr[i];
		}
		sort(arr, arr+30, greater<int>());

		for(int i = 0;i < n;i++){
			if(i == 0){
				total += arr[i] + 30;
				buffer = arr[i];
			}
			else{
				buffer -= 30;
				if(buffer < 0){
					total += arr[i] + (buffer*-1);
					buffer = arr[i];
				}
				else{
					total += arr[i];
					buffer += arr[i];
				}
			}
		}
		
		cout << (total/60) << ':' << ((total%60 < 10) ? ("0"+to_string(total%60)) : (to_string(total%60))) << endl;

	}

	return 0;
}
