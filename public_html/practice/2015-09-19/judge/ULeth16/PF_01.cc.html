#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main(){
	int t;
	cin >> t;
	for(int i = 0;i < t;i++){
		if(i > 0)
			cout << endl;
		int n;
		cin >> n;
		vector<int> arr;
		for(int i = 0;i < n;i++){
			int a;
			cin >> a;
			arr.push_back(a);
		}
		sort(arr.begin(),arr.end());
		for(int i = 1;i < arr.size();i++){
			if(arr[i] > arr[i-1]){
				cout << arr[i]-arr[i-1] << ' ' << i << endl;
			}
		}
	}
	return 0;
}
