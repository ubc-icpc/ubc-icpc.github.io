import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Scanner;

 */
public class B {
    public B() throws IOException {
        Scanner sc = new Scanner(new BufferedReader(new InputStreamReader(System.in)));

        int n = sc.nextInt();
        int counter = 0;
        String []operations = new String[n];
        int []operands = new int[n];
        for (int i = 0; i < n; i++) {
            operations[i] = sc.next();
            operands[i] = sc.nextInt();
        }

        for (int j = 1; j <= 100; j++) {
            double val = j;
            for (int i = 0; i < n; i++) {
                val = doOperation(val,operands[i] , operations[i]);
                if(val<0 || ( (int)val != val  ) ){
                    counter++;
                    break;
                }

            }

        }

        System.out.println(counter);

    }

    public double doOperation(double num, int operand, String operation){
        double num1 = (double) num;
        double operand1 = (double) operand;

        if(operation.equals("MULTIPLY")){
            return num1*operand1;
        }
        else if(operation.equals("ADD")){
            return num1 + operand1;
        }
        else if(operation.equals("SUBTRACT")){
            return num1 - operand1;
        }
        else{
            return num1 / operand1;
        }

    }


    public static void main(String[] args) throws IOException {
        new B();
//        double x = 3.0;

    }



}
