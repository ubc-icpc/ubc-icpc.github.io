#include <bits/stdc++.h>
using namespace std;

int a[3];
int b[3];
set<int> st;

int main() {

        cin >> a[0] >> a[1] >> a[2];
        cin >> b[0] >> b[1] >> b[2];

        for (int i = 0; i < 3; i++) {
                st.insert(a[i]);
                st.insert(b[i]);
        }

        sort(a, a + 3);

        if (st.size() != 3 || a[0] * a[0] + a[1] * a[1] != a[2] * a[2]) {
                cout << "NO\n";
        } else {
                cout << "YES\n";
        }

        return 0;
}
