#include <bits/stdc++.h>
using namespace std;

int main() {
        int n, k, f;
        string sb;
        cin >> n >> k;
        int ls = 1;
        int hb = k;
        for (int i = 0; i < n; i++) {
                cin >> f >> sb;
                if (sb[0] == 'S') {
                        ls = max(f, ls);
                } else {
                        hb = min(f, hb);
                }
        }
        cout << ls + 1 << ' ' << hb - 1 << '\n';
        return 0;
}
