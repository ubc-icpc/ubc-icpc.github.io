#include <bits/stdc++.h>
using namespace std;

vector<int> v;
int main() {
        int n, s;
        cin >> n;
        while (n--) {
                cin >> s;
                v.push_back(s);
        }
        sort(v.begin(), v.end());

        int mmax = 0x3f3f3f;
        for (int i = 0; i < v.size() / 2; i++) {
                mmax = min(mmax, v[i] + v[v.size() - 1 - i]);
        }
        cout << mmax << '\n';
        return 0;
}
