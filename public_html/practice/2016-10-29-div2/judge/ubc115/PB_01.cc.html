#include <bits/stdc++.h>
using namespace std;

string s;
int p;
vector<pair<string, int>> v;


int mess(int ii) {
        for (auto p : v) {
                int o = p.second;
                switch (p.first[0]) {
                        case 'A':
                                ii += o;
                                break;
                        case 'S':
                                ii -= o;
                                break;
                        case 'D':
                                if (ii % o) {
                                       return 1;
                                } 
                                ii /= o;
                                break;
                        case 'M':
                                ii *= o;
                                break;
                        default:
                                break;
                }
                if (ii < 0) {
                        return 1;
                }
        }
        return 0;
}



int main() {
        int n;
        cin >> n;
               for (int i = 0; i < n; i++) {
                cin >> s >> p;
                v.push_back({s, p});
        }

        int count = 0;
        for (int ii = 1; ii <= 100; ii++) {
                count += mess(ii);
        }

        cout << count << '\n';
        
        return 0;
}
