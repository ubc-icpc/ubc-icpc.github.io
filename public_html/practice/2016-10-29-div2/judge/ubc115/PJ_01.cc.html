#include <bits/stdc++.h>
using namespace std;

int m[105][105][2];

int main() {
        int w, h, b;
        cin >> w >> h >> b;
        int base = 0;
        int tmp = 1;
        for (int j = 0; j < h; j++) {
                for (int i = 0; i < w; i++) {
                        cin >> m[j][i][base];
                }
        }

        for (int z = 0; z < b; z++) { 
                for (int j = 0; j < h; j++) {
                        for (int i = 0; i < w; i++) {
                                int sum = 0;
                                for (int k = -1; k <= 1; k++) {
                                        for (int l = -1; l <= 1; l++) {
                                                sum += m[(j+k+h) % h][(i+l+w) % w][base];
                                        }
                                }
                                m[j][i][tmp] = sum;
                        }
                }
                tmp ^= 1;
                base ^= 1;
        }

        set<int> ss;
        for (int j = 0; j < h; j++) {
                for (int i = 0; i < w; i++) {
                        ss.insert(m[j][i][base]);
                }
        }

        cout << ss.size() << '\n';
        return 0;
}
