#include <bits/stdc++.h>
using namespace std;

typedef double ld;

int main() {
        int n, w;
        cin >> n >> w;
        string s;
        ld p, v;

        ld la = 0.0;
        ld pp = 1.0;
        ld pl = 1.0;
        ld msum = 0.0;
        ld val = 0.0;

        for (int i = 0; i < n; i++) {
                cin >> s >> p >> v;
                pl = pp * (1.0 - p);
                pp *= p;
                val += pl * log(1.0 + la / w);
                msum = max(msum, pp * log(1.0 + v / w) + val);
                
                if (s[0] == 's') {
                        la = v;
                }
        }
        cout << '$' << fixed << setprecision(2) << (exp(msum) - 1.0) * w << endl;
        return 0;
}
