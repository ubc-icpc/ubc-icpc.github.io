#include <bits/stdc++.h>
using namespace std;

map<char, int> m;
int main() {
        string s;
        cin >> s;
        for (int i = 0; i < s.length(); i++) {
                m[s[i]]++;
        }
        int c = m.size();
        if (c <= 2) {
                cout << 0 << '\n';
        } else {
                vector<int> v;
                for (auto p : m) {
                        v.push_back(p.second);
                }
                sort(v.begin(), v.end());
                int sum = 0;
                for (int i = 0; i < v.size() - 2; i++) {
                        sum += v[i];
                }
                cout << sum << '\n';
        }

        return 0;
}
