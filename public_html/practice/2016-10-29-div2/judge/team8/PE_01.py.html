import sys
input = sys.stdin

def solve(s):
	s_chars = set(list(s))
	s_counts = [(i, s.count(i)) for i in s_chars]
	sorted_s_counts = sorted(s_counts, key=lambda i: i[1], reverse=True)
	if len(sorted_s_counts) <= 2:
		return 0
	return sum([i[1] for i in sorted_s_counts[2:]])

s = input.readline().strip()
print solve(s)
