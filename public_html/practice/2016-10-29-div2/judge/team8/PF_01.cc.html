#include <algorithm>
#include <iostream>
#include <vector>
#include <climits>

using namespace std;

int solve(std::vector<int> members) {
	std::sort(members.begin(), members.end());
	int min = INT_MAX;
	for (int i = 0; i < members.size()/2; i++) {
		int tmp = members[i] + members[members.size()-1-i];
		if (tmp < min) {
			min = tmp;
		}
	}
	return min;
}

int main() {
	int N;
	cin >> N;
	std::vector<int> members(N, 0);
	for (int i = 0; i < N; i++) {
		cin >> members[i];
	}

	cout << solve(members) << endl;
}
