#include <stdio.h>
#include <string.h>

int main() {
	int num, k;
	int i, temp;
	scanf("%d %d", &num, &k);
	int maxsafe, minbrkn;
	maxsafe = 1;
	minbrkn = k;
	char s[10];
	for (i = 0; i < num; i++) {
		scanf("%d %s", &temp, s);
		if (strcmp(s,"SAFE") == 0 && temp > maxsafe)
			maxsafe = temp;
		else if (strcmp(s,"BROKEN") == 0 && temp < minbrkn)
			minbrkn = temp;
	}
	printf("%d %d", maxsafe+1, minbrkn-1);
	return 0;
}
