#include <stdio.h>
#include <string.h>

int main() {
	int cnum, i, k, op;
	char command[12];
	int j[100];
	for (i = 0; i < 100; i++)
		j[i] = i+1;
	scanf("%d", &cnum);
	for (i = 0; i < cnum; i++) {
		scanf("%s %d", command, &op);
		if (strcmp(command,"SUBTRACT") == 0) {
			for (k = 0; k < 100; k++) {
				if (j[k] >= 0)
					j[k] -= op;
			}
		}
		else if (strcmp(command,"DIVIDE") == 0) {
			for (k = 0; k < 100; k++) {
				if (j[k]%op != 0)
					j[k] = -1;
				else
					j[k] /= op;
			}
		}
		else if (strcmp(command,"ADD") == 0) {
			for (k = 0; k < 100; k++)
				if (j[k] >= 0)
					j[k] += op;
		}
		else if (strcmp(command,"MULTIPLY") == 0) {
			for (k = 0; k < 100; k++)
				if (j[k] >= 0)
					j[k] *= op;
		}
	}
	cnum = 0;
	for (i = 0; i < 100; i++)
		if (j[i] < 0)
			cnum++;
	printf("%d", cnum);
	return 0;
}
