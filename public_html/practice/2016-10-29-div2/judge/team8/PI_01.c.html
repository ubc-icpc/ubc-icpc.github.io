#include <stdio.h>
#include <stdlib.h>

int cmpf(const void *a, const void *b) {
	return *(int *) a - *(int *) b;
}

int main() {
	int a[3], b[3];
	scanf("%d %d %d", a, a+1, a+2);
	scanf("%d %d %d", b, b+1, b+2);
	qsort(a,3,sizeof(int),cmpf);
	qsort(b,3,sizeof(int),cmpf);
	if (a[0] == b[0] && a[1] == b[1] && a[2] == b[2]) {
		if (a[0]*a[0] + a[1]*a[1] == a[2]*a[2]) {
			printf("YES");
			return 0;
		}
	}
	printf("NO");
	return 0;
}
