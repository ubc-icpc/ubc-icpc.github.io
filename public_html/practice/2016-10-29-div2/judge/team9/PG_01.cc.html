using namespace std;

bool fncomp(pair<string, string> s1, pair<string, string> s2) {
	if (s2.second == s1.second) { // same last name
		if (s2.first == s2.first) {// same first name
			return s1 < s2;
		}
		else {
			return s1.first < s2.first;
		}
	}
	else {
		return s1.second < s2.second;
	}
}


int main() {
	ios::sync_with_stdio(0);
	int n;
	cin >> n;
	string fn, ln;

	bool(*fn_pt)(pair<string, string>, pair<string, string>) = fncomp;
	set<pair<string, string>, bool(*)(pair<string, string>, pair<string, string>) > students(fn_pt);
	for (int i = 0; i < n; i++) {
		cin >> fn >> ln;
		students.insert(make_pair(fn, ln));
	}

	for (set<pair<string, string>>::iterator it = students.begin(); it != students.end(); ++it)
		cout << (*it).first << " " << (*it).second << endl;

	return 0;
}
