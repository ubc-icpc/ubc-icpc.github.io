#include <iostream>

using namespace std;




int main() {
	ios::sync_with_stdio(0);
	int n, k;
	cin >> n >> k;
	int safe = 1, broken = k;
	string res;
	int Cfloor;

	for (int i = 0; i < n; i++) {
		cin >> Cfloor >> res;
		if (res == "BROKEN" && Cfloor < broken) {
			// if broken then choose the smaller floor
			broken = Cfloor;
		}
		if (res == "SAFE" && Cfloor > safe) {
			safe = Cfloor;
		}
	}
	// here we get the lowest broken and highest safe


	if (abs(broken - safe) > 1) {// if they are not consecutive it's safe for them to be lower
		cout << safe+1 << " " << broken -1 << endl;
	}
	else {
		cout << broken << " " << safe << endl;
	}


	return 0;
}
