#include <iostream>
#include <string>

using namespace std;




int main() {
	ios::sync_with_stdio(0);
	string s;
	cin >> s;
	int v[26] = { 0 };
	int maxIndex = -1, secondMaxIndex = -1;
	int max = -1, secMax = -1;

	for (int i = 0; i < s.length(); i++) {
		v[int(s[i] - 'a')]++;
	}

	// compute complexity
	int complex = 0;
	for (int i = 0; i < 26; i++) {
		if (v[i] != 0) {
			complex++;
		}

		if (max < v[i]) {
			secMax = max;
			secondMaxIndex = maxIndex;

			max = v[i];
			maxIndex = i;
		}
		else {
			if (secMax < v[i]) {
				secMax = v[i];
				secondMaxIndex = i;
			}

		}
	}

	if (complex > 2) {
		int sum = 0;
		for (int i = 0; i < 26; i++) {
			if (i != secondMaxIndex && i != maxIndex) {
				sum += v[i];
			}
		}
		cout << sum << endl;
	}
	else {
		cout << 0 << endl;
	}


	return 0;
}
