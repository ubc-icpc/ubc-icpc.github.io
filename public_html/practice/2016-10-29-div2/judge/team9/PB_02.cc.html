#include <iostream>
#include <algorithm> 
#include <utility>
#include <math.h>
#include <cmath>
#include <climits>
#include <vector>
#include <set>
#include<string>
#include <tuple>
#include <iostream>
#include <string>

using namespace std;




int main() {
	ios::sync_with_stdio(0);
	string s;
	int n, operand;
	cin >> n;

	vector<pair<char, int>> operations;


	for (int i = 0; i < n; i++) {
		cin >> s;
		cin >> operand;
		char c = s[0];

		operations.push_back(make_pair(c, operand));
	}

	int count = 0;

	for (int k = 1; k <= 100; k++) {
		float number = k;
		bool stop = false;


		for (int i = 0; i < operations.size(); i++) {
			if (stop) { break; }

			switch (operations[i].first) {
			case 'A': number += operations[i].second; break;
			case 'M': number *= operations[i].second; break;
			case 'S': number -= operations[i].second; if (number < 0) { count++; stop = true; } break;
			case 'D': number /= operations[i].second; if (floor(number) != number) { count++; stop = true; } break;
			}
		}

	}

	cout << count << endl;
	return 0;
}
