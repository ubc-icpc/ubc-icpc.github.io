int main() {
	array<int, 3> A;
	array<int, 3> B;
	cin >> A[0] >> A[1] >> A[2] >> B[0] >> B[1] >> B[2];
	sort(A.begin(), A.end());
	sort(B.begin(), B.end());
	if (A[0] == B[0] && A[1] == B[1] && A[2] == B[2]) {
		if (A[0]*A[0] + A[1]*A[1] == A[2]*A[2])	cout << "YES" << endl;
		else cout << "NO" << endl;
	} else {
		cout << "NO" << endl;
	}
	return 0;
	

}
