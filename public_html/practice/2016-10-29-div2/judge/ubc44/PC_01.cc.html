#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
typedef pair<pii, pii> piipii;
typedef pair<piipii, bool> piipii_O; //pair and orientation! (T: horiz; F:vert)
typedef vector<ll> vi;
typedef vector<vi > vvi;
typedef vector<bool> vb;
typedef vector<vb > vvb;

int main () {
	int n, k;
	cin >> n >> k;
	int maxsafe = 1;
	int minbreak = k;	

	string result; int f;
	while (n--) {
		cin >> f >> result;
		if (result == "BROKEN" && f<minbreak) {
			minbreak = f;
		}
		if (result == "SAFE" && f>maxsafe) {
			maxsafe = f;
		}
	}

	cout << maxsafe+1  << " " << minbreak-1 << endl;
}
