int main() {
	array<int, 26> count;
	for (int i=0; i<26; i++)
		count[i] = 0;
	string s;
	cin >> s;
	for (int i=0; i<s.size(); i++) {
		count[s[i]-'a']++;
	}
	sort(count.begin(), count.end());
