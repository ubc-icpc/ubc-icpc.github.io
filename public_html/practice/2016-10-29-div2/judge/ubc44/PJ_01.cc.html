#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
typedef pair<pii, pii> piipii;
typedef pair<piipii, bool> piipii_O;
typedef vector<ll> vi;
typedef vector<vi > vvi;
typedef vector<bool> vb;
typedef vector<vb > vvb;


int main() {
	int c,r,b;
	cin >> c >> r >> b;
	int A[r][c];
	for (int i=0; i<r; i++) {
			for (int j=0; j<c; j++) {
				cin >> A[i][j];
			}
		}
	int temp[r][c];
	while (b--) {
		for (int i=0; i<r; i++) {
			for (int j=0; j<c; j++) {
				//cout << A[i][j] << " ";
				temp[i][j] = A[(i-1+r) % r][(j-1+c)%c] + A[(i-1+r)%r][j%c] + A[(i-1+r)%r][(j+1)%c]
							+ A[i % r][(j-1+c)%c] + A[i%r][j%c] + A[i%r][(j+1)%c]
							+ A[(i+1) % r][(j-1+c)%c] + A[(i+1)%r][j%c] + A[(i+1)%r][(j+1)%c];

			}
		}
		for (int i=0; i<r; i++) {
			for (int j=0; j<c; j++) {
				A[i][j] = temp[i][j];
				//cout << A[i][j] << " ";
			}
			//cout << endl;
		}
	}


	unordered_set<int> all;
	for (int i=0; i<r; i++) {
			for (int j=0; j<c; j++) {
				all.insert(A[i][j]);
			}
		}
	
	//all = unique(all.begin(), all.end());
	cout << all.size() << endl;

	return 0;
}
