import sys

n, k = map(int, sys.stdin.readline().split())

safe = []
broken = []

for i in range(n):
  floor, st = sys.stdin.readline().split()
  floor = int(floor)
  if st == 'SAFE':
    safe.append(floor)
  else:
    broken.append(floor)

min_b = min(broken) if broken else k 
max_s = max(safe) if safe else 1

print max_s + 1, min_b - 1
