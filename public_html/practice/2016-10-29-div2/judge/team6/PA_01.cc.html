#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;

int main() {
  long n,w; cin >> n >> w;
  long safe = 0;
  double p = 1;
  double max = 0;
  double c = 0;
  while(n--){
    string s; double cp; long m;
    cin >> s >> cp >> m;
    double lose = c + (1 - cp) * p * log(1. + (double)safe/(double)w);
    // cout << "lose: " << lose << endl;
    // if (lose > max) max = lose;
    double stop = lose + cp * p * log(1. + (double)m/(double)w);
    // cout << "stop: " << stop << endl;
    if (stop > max) max = stop;

    c = lose;
    p *= cp;
    if (s[0] == 's') safe = m;
  }
  double q = (exp(max) - 1) * w;
  printf("$%.2f\n",q);
}
