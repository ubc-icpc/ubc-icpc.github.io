#include <iostream>
#include <vector>
using namespace std;
vector<char> commands;
vector<int> vals;

bool ok(long long p) {
  for (int j = 0; j < commands.size(); j++) {
    switch(commands[j]){
      case 'A':
        p += vals[j];
        break;
      case 'S':
        p -= vals[j];
        if (p < 0) return true;
        break;
      case 'M':
        p *= vals[j];
        break;
      case 'D':
        if (p % vals[j]) return true;
        p /= vals[j];
        break;
    }
  }

  return false;
}



int main() {
  int n; cin >> n;
  while (n--) {
    string c; int d; cin >> c >> d;
    commands.push_back(c[0]);
    vals.push_back(d);
  }
  int q = 0;
  for (int i = 1; i <= 100; i++) {
    long long p = i;
    q += ok(p);
  }

  cout << q << endl;
}
