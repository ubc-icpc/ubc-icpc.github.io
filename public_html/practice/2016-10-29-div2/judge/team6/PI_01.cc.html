#include <bits/stdc++.h>
using namespace std;

int main() {
	vector<int> a, b;
	for (int i = 0; i < 3; i++) {
		int c; cin >> c;
		a.push_back(c);
	}
	for (int i = 0; i < 3; i++) {
		int c; cin >> c;
		b.push_back(c);
	}
	sort(a.begin(), a.end());
	sort(b.begin(), b.end());
	for (int i = 0; i < 3; i++) {
		if (a[i] != b[i]) {
			cout << "NO" << endl;
			return 0;
		}	
	}
	if (a[0]*a[0] + a[1]*a[1] != a[2]*a[2]) {
		cout << "NO" << endl;
		return 0;
	}
	cout << "YES" << endl;
	return 0;
}
