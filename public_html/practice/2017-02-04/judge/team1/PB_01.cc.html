#include <queue>
#include <cstdio>
#include <string>
#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	
	int ngames;
	cin >> ngames;
	
	while(ngames--){
		
		cout << "3 1 3 6" << endl;
		
		string aux;
		while(true){
			
			cin >> aux;
			if(aux[0] == 'G') break;
			
			int a1, b1, a2, b2;
			
			cin >> a1 >> b1 >> a2 >> b2;
			
			int ra1, rb1, ra2, rb2;
			
			ra1 = 6 - a1;
			ra2 = 6 - a2;
			rb1 = 7 - b1;
			rb2 = 7 - b2;
			
			if(ra1 == ra2 && rb1 > rb2) swap(rb1, rb2);
			if(rb1 == rb2 && ra1 > ra2) swap(ra1, ra2);
			
			cout << ra1 << " " << rb1 << " " << ra2 << " " << rb2 << endl;
		}
		
	}
	
	return 0;
}
