#include<bits/stdtr1c++.h>
using namespace std;

typedef long long ll;
ll T;
string A;

ll gete(int i) {
  if (i == 0) return 1;
  return gete(i-1) * 10;
}

ll calc() {
  ll total = 0;
  ll tmp = 0;
  for (int i = 0; i < (int)A.size(); ++i) {
    for (int j = 0; j < (int)A[i]-'0'; ++j) {
      total += (tmp + j) * gete(A.size()-i-1);
      if (i < (int)A.size()-1)
        total += gete(A.size()-i-1) * (A.size()-i-1) * 9 / 2;

    //cout << total << endl;
    }
    tmp += A[i] - '0';
  }
  for (int i = 0; i < (int)A.size(); ++i)
    total += A[i]-'0';
  return total;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> T;
  while (T--) {
    ll total = 0;
    cin >> A;
    //ll temp = calc();
    //cout << temp << endl;
    total -= calc();
    for (int i = 0; i < (int)A.size(); ++i) {
      total += A[i]-'0';
    }
    cin >> A;
    //temp = calc();
    //cout << temp << endl;
    total += calc();
    cout << total << endl;
  }
  return 0;
}
