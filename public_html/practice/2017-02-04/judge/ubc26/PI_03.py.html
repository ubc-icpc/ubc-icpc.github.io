import sys
from collections import deque
t = int(raw_input())
for _ in xrange(t):
    t = raw_input()
    n = int(raw_input())
    xs = deque(input())
    r = False
    try:
        for c in t:
            if c == 'R': r = not r
            elif c == 'D':
                if r: xs.pop()
                else: xs.popleft()
        sys.stdout.write("[")
        r, z = (xrange(len(xs)), len(xs)-1) if not r else (xrange(len(xs)-1, -1, -1), 0)
        for i in r:
            sys.stdout.write(str(xs[i]))
            if i != z: sys.stdout.write(",")
        sys.stdout.write("]\n")
    except IndexError: print("error")
