#include <bits/stdc++.h>
#define endl '\n'
typedef long long ll;
typedef long double ld;
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int T; cin >> T;
	while (T--) {
		string p; cin >> p;
		vector<bool> instructions;
		for (char c : p)
			instructions.push_back(c == 'R');
		int n; cin >> n;
		string s; cin >> s;
		deque<int> list;
		istringstream in(s);
		char c; in >> c;
		for (int i = 0; i < n; i++) {
			int k; in >> k;
			list.push_back(k);
			in >> c;
		}
		bool reverse = 0, err = 0;
		for (bool b : instructions) {
			if (b)
				reverse = !reverse;
			else {
				if (list.empty()) {
					err = 1;
					break;
				}
				if (reverse)
					list.pop_back();
				else
					list.pop_front();
			}
		}
		if (err)
			cout << "error" << endl;
		else {
			cout << '[';
			if (list.empty())
				cout << ']';
			while (!list.empty()) {
				int i;
				if (reverse) {
					i = list.back(); list.pop_back();
				} else {
					i = list.front(); list.pop_front();
				}
				cout << i;
				if (!list.empty())
					cout << ',';
				else
					cout << ']';
			}
			cout << endl;
		}
	}
	return 0;
}
