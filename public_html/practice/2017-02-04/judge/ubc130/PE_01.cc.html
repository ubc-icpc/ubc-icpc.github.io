#include <bits/stdc++.h>
#define endl '\n'
typedef long long ll;
typedef long double ld;
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int T; cin >> T;
	while (T--) {
		string s; cin >> s;
		int len = (int) (sqrt(s.length()) + 1e-10);
		for (int i = len-1; i >= 0; i--) {
			for (int j = 0; j < len; j++) {
				cout << s[j*len + i];
			}
		}
		cout << endl;
	}
	return 0;
}
