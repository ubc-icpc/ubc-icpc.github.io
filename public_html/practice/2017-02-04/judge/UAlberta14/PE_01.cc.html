#include<iostream>
#include<string>
#include<cmath>
#include<sstream>

using namespace std;

int N;
char myA[105][105];
string line;
int main(){
	cin >>N;cin.ignore();
	while(N--){
		getline(cin, line);
		stringstream ss(line);
		int a = sqrt(line.size());
		for(int i=0; i<a; i++){
			for(int j=0; j<a; j++){
				ss>>myA[i][j];				
			}
		}	
		for(int i=a-1; i>=0; i--){
			for(int j=0; j<a; j++){
				cout <<myA[j][i];
			}
		}	
		cout <<endl;
	}
	return 0;
}
