#include<iostream>
#include<string>
#include<sstream>

using namespace std;
typedef long long ll;

int N;
ll a, b, dp;


ll s(ll x){
	ll ans=0;
	stringstream ss;
	ss<<x;
	string myS = ss.str();
	for(char s: myS){
		ans+= s -'0';	
	}

	return ans;
}

ll f(ll x){
	if (x<=0)	return 0;
	if(x%10==0)	return	10*f(x/10) + 45*x/10;
	return f(x-1)+s(x-1);
}


int main(){
	cin >>N;
	while(N--){
		cin >>a>>b;
		
		cout<<f(b+1)-f(a)<<endl;


	}
	return 0;
}
