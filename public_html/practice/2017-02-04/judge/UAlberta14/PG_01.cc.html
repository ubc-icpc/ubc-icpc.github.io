#include<iostream>
#include<iomanip>
#include<cmath>
#include<cstring>

using namespace std;

double dp[151][151];

int N, ns, s[151];
double p[151];

double dpc(int n, int m){
	if(dp[n][m]!=dp[n][m]){	
//		cout <<"here"<<endl;
		if(m<=0)	return 1.0;
		if(n>ns)	return 0;
		else{
			dp[n][m] = max(dpc(n+1, m), dpc(n+1, m-s[n-1])*p[n-1]);
		}	
	}

//		cout <<"there"<<endl;
	return dp[n][m];
}

int main(){
	cin >>N;
	while(N--){
		cin >>ns;
		for(int i=0; i<ns; i++){
			cin >>s[i]>>p[i]; p[i]/=100.0;
//	  cout <<s[i]<<"  "<<p[i]<<endl;
		}

		memset(dp, -1, sizeof dp);
//		cout <<dp[1][1]<<endl;

		double curB = 0;

		for(int i=76; i<151; i++){
			curB = max(curB, dpc(1, i));
		}
		//cout <<curB<<endl;
    cout << fixed << setprecision(6) << curB * 100 << endl;
	}
	return 0;
}
