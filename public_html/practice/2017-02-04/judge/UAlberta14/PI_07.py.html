N = int(raw_input())
for i in range(N):
	S = raw_input()
	le = int(raw_input())
	mys = raw_input()
	mys = mys[1:-1]

	nums = mys.split(',')
	
	for i in range(le):
		nums[i] = int(nums[i])

	err = False
	rev = 0
	start = 0
	end =0
	for ele in S:
		if ele=='R':
			rev = 1-rev
			
		elif ele=='D':
			if le==0:
				err = True
				break
			else:
				if rev==0:
					start+=1
				else:
					end+=1
				le-=1;
	if err:
		print("error")
	else:
		if rev ==1:
			nums.reverse()
			ss = end
			ee = -start
		else:
			ss = start
			ee = -end

		if le==0:
			print("[]")
		elif ee==0:
			print(nums[ss:])
		else:
			print(nums[ss:ee])
