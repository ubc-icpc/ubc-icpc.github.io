#include <bits/stdc++.h>
using namespace std;

char a[128][128];
int main() {
    ios::sync_with_stdio(0);
    int t;
    string s;
    cin >> t;
    while (t--) {
        cin >> s;
        int n = sqrt(s.length());
        for (int j = 0; j < n; j++) {
            for (int i = 0; i < n; i++) {
                a[j][i] = s[j * n + i];
            }
        }
        for (int i = n - 1; i >= 0; i--) {
            for (int j = 0; j < n; j++) {
                cout << a[j][i];
            }
        }
        cout << '\n';
    }
    return 0;
}
