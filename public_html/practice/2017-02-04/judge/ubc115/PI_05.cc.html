#include <bits/stdc++.h>
using namespace std;

int a[1 << 17];

int main() {
    ios::sync_with_stdio(0);
    int t, n;
    string s, l;
    cin >> t;
    while (t--) {
        cin >> s;
        cin >> n;
        char c;

        cin >> c;
        for (int i = 0; i < n; i++) {
            cin >> a[i];
            cin >> c;
        }
        if (n == 0) {
            cin >> c;
        }
        int h = 0;
        int p = 0;
        int q = n - 1;
        bool error = false;

        for (int i = 0; i < s.length(); i++) {
            if (s[i] == 'D') {
                if (q < p) {
                    error = true;
                    break;
                }
                if (h == p) {
                    h++;
                    p++;
                } else {
                    h--;
                    q--;
                }
            } else {
                if (h == q) {
                    h = p;
                } else {
                    h = q;
                }
            }
        }

        if (!error) {
            cout << '[';
            if (h == p) {
                if (p <= q) {
                    for (int i = p; i < q; i++) {
                        cout << a[i] << ',';
                    }
                    cout << a[q];
                }
            } else {
                if (q >= p) {
                    for (int i = q; i > p; i--) {
                        cout << a[i] << ',';
                    }
                    cout << a[p];
                }
            }
            cout << ']';
            cout << '\n';
        } else {
            cout << "error\n";
        }
        
    }
    return 0;
}
