#include <iostream>
#include <string>
using namespace std;

string ntos(long long n)
{
    string s="";
    while(n!=0)s=(char)(n%10+'0')+s,n/=10;
    return s;
}

long long ston(string s)
{
    long long n=0,base=1;
    int i;
    for(i=s.size()-1;i>=0;i--)
    {
        n+=base*(s[i]-'0');
        base*=10;
    }
    return n;
}

long long su(long long a,long long b)
{
    long long s=0;
    long long i;
    for(i=a;i<=b;i++)s+=i;
    return s;
}

long long dec(long long n)
{
    if(n==0)return 0;
    string s=ntos(n);
    long long base=1,ss=0;
    int i;
    for(i=s.size()-1;i>=0;i--)
    {
        long long sl=0,sm=0,sr=0;
        if(i!=0)sl=ston(s.substr(0,i))*45*base;
        if(base!=1)sm=su(1,s[i]-'0'-1)*base;
        else sm=su(1,s[i]-'0');
        if(i!=s.size()-1)sr=(1+ston(s.substr(i+1)))*(s[i]-'0');
        ss+=sl+sm+sr;
        base*=10;
    }
    return ss;
}

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        long long a,b;
        cin>>a>>b;
        cout<<dec(b)-dec(a-1)<<endl;
    }
    return 0;
}
