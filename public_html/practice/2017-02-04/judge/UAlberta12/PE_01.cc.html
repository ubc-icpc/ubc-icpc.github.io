#include <iostream>
#include <cmath>
using namespace std;

int main() {
	int T;
	cin >> T;
	while(T--) {
		string s;
		cin >> s;
		string o = s;
		int side = int(sqrt(s.size()));
		for (int i = 0; i < side; i++) {
			for (int j = 0; j < side; j++) {
				o[i*side + j] = s[j*side + (side-i-1)];
			}
		}
		cout << o << endl;
	}
}
