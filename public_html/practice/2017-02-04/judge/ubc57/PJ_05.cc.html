#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    int t; cin>>t;
    while (t--){
        ll ans=0;
        int n; cin>>n;
        vector<int> v;
        for (int i=0;i<n;i++){
            int x; cin>>x;
            v.push_back(x);
        }
        for (int i=1;i<n;i++){
            if (v[i]<v[i-1]){
                ans++;
                int x=v[i]; 
                //imagine moving x to the front
                if (x>v[0]){
                    vector<ll> freq;
                    int now=v[0]; ll f=1;
                    for (int j=1;j<i;j++) if (v[j]<x){
                        if (v[j]==now) f++;
                        else{
                            freq.push_back(f);
                            now=v[j],f=1;
                        }
                    }
                    freq.push_back(f);
                    ll asd=freq[0];
                    for (int j=1;j<freq.size();j++) asd=asd+freq[j]*asd+freq[j];
                    ans+=asd;
                }
                v.erase(v.begin()+i);
                v.insert(v.begin(),x);
                sort(v.begin(),v.begin()+i+1);
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
