#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(0);
    int t; cin>>t;
    while (t--){
        cout<<"3 1 3 6"<<endl;
        while (true){
            string s; cin>>s; 
            if (s[0]=='G') break;
            int a,b,c,d; cin>>a>>b>>c>>d;
            cout<<6-c<<' '<<7-d<<' '<<6-a<<' '<<7-b<<endl;
        }
    }
    return 0;
}
