#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef long double ld;

ld DP[151][100];
bool seen[151][100];
int s[151],p[151]; 
ld dp(int i, int k){
    if (k<=0) return 1.0;
    if (i==1){
        if (k>s[i]) return 0.0;
        return ld(p[i])/100.0;
    }
    if (seen[i][k]) return DP[i][k];
    ld ans=max(dp(i-1,k),dp(i-1,k-s[i])*ld(p[i])/100.0); 
    seen[i][k]=true;
    return DP[i][k]=ans;
}
int main(){
    ios::sync_with_stdio(0);
    int t; cin>>t;
    while (t--){
        memset(seen,0,sizeof seen);
        int n; cin>>n;
        for (int i=1;i<=n;i++) cin>>s[i]>>p[i];
        cout<<fixed<<setprecision(10)<<100.0*dp(n,76)<<endl;
    }
    return 0;
}
