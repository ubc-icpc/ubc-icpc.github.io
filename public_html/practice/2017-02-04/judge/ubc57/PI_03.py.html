t=int(raw_input())
for ca in range(t):
    s=raw_input()
    le=int(raw_input())
    ls=raw_input()
    l=0
    direc=1
    for i in range(len(s)):
        if s[i]=='R':
            l=l+(le-1)*direc
            direc=-direc
        else:
            le-=1
            l+=direc
    if le<0:
        print("error")
    else:
        ls=ls[1:len(ls)-1]
        if len(ls)==0:
            print("[]")
        else:
            ans=""
            L=list(map(int,ls.split(',')))
            for i in range(le):
                ans+=","+str(L[l+i*direc])
            print("["+ans[1:]+"]")
