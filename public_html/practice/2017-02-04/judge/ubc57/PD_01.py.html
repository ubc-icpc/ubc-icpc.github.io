    a,b=s[0],s[1]
    while len(a)<len(b):
        a='0'+a
    ans=0
    a='0'+a
    b='0'+b
    for d in range(1,len(a)):
        for i in range(1,10):
            if i<int(a[d]):
                l=(int(a[:d])+1)*(10**(len(a)-d-1))
            elif i==int(a[d]):
                l=int(a[:d]+a[d+1:])
            else:
                l=(int(a[:d]))*(10**(len(a)-d-1))
            if i<int(b[d]):
                r=(int(b[:d])+1)*(10**(len(a)-d-1))
            elif i==int(b[d]):
                r=int(b[:d]+b[d+1:])+1
            else:
                r=(int(b[:d]))*(10**(len(a)-d-1))
            ans+=(r-l)*i
            ##print(d,i,l,r)
    print(ans)
