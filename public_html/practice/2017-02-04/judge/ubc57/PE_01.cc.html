#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    int t; cin>>t;
    while (t--){
        string s; cin>>s; 
        int n=1;
        while ((n*n)!=s.size()) n++; 
        for (int i=n-1;i>=0;i--) for (int j=0;j<n;j++) cout<<s[j*n+i]; cout<<endl;
    }
    return 0;
}
