#include<bits/stdc++.h>
#define pb   push_back
using namespace std;
typedef long long ll;
typedef long double ld;

ll solve(ll a, ll b){
    //right exclusive
    //cout << a << " "<< b <<endl;
    if (a>b) return 0;
    ll ans=0;
    if (b-a<=10) {
        for(ll i=a;i<=b;i++){
            for(ll d=i;d;d/=10)
                ans+=d%10;
        }
        return ans;
    }
    ll aup = ((a/10) +1)*10, bdown = ((b/10))*10;
    if(a%10==0) aup = a;
    if(b%10==0) bdown = b;
    //cout << a << " " << aup << " " << bdown << " " << b <<endl;
    ans += solve(bdown,b);
    ans += solve(a,aup-1);
    ll s = solve(aup/10,bdown/10-1);
    ans += 10*s + 45*(-aup/10+bdown/10);
    return ans;
}

int main(){
    ios_base::sync_with_stdio(false);cin.tie(0);
    int T;
    cin >> T;
    while(T--){
        ll a, b;
        cin >> a >> b;
        ll ans = solve(a,b);
        cout << ans <<endl;
    }
}
