#include <iostream>
#include <vector>
#include <math.h>
#include <string>
#include <set>
#include <iomanip>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0; i<t; i++)
    {
        string a;
        cin>>a;
        int uzunluq = a.length();
        uzunluq = sqrt(uzunluq);
        string arr[uzunluq] ;
        for(int i=0; i<uzunluq; i++)
        {
            arr[i]="";
        }
        for(int i=0; i<a.length(); i++)
        {
            arr[i%uzunluq]+=a[i];
        }
        for(int i=uzunluq-1; i>=0; i--)
        {
            cout<<arr[i];
        }
        cout<<endl;
    }
}
