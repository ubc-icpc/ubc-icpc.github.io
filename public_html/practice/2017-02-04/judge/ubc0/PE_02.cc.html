#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

char grid[105][105];
int main() {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	while (t--) {
		string s; cin >> s;
		int sz = int(sqrt(s.size()) + 1e-5);
		int k = 0;
		
		for (int i = 0; i < sz; i++) {
			for (int j = 0; j < sz; j++) {
				grid[i][j] = s[k++];
			}
		}
		
		for (int j = sz-1; j >= 0; j--) {
			for (int i = 0; i < sz; i++) {
				cout << grid[i][j];
			}
		}
		cout << endl;
	}
	return 0;
}
