#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int main() {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	while (t--) {
		cout << "3 1 3 6" << endl;
		while (true) {
			string s; cin >> s;
			if (s == "GAME") break;
			int r0, c0, r1, c1; cin >> r0 >> c0 >> r1 >> c1;
			cout << 6-r1 << " " << 7-c1 << " " << 6-r0 << " " << 7-c0 << endl;
		}
	}
	return 0;
}
