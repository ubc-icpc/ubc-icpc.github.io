t = int(raw_input())
for c in range(t):
	prog = raw_input().strip()
	n, L = raw_input(), eval(raw_input())
	
	rev, i, j = 0, 0, len(L)
	for p in prog:
		if p == 'R':
			rev ^= 1
		else:
			if rev:
				j -= 1
			else:
				i += 1
	
	if i > j:
		print 'error'
	else:
		res = L[i:j]
		if rev:
			res.reverse()
		print '['+','.join(str(i) for i in res)+']'
