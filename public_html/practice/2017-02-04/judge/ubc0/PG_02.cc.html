#include <bits/stdtr1c++.h>
using namespace std;

typedef long double ld;
typedef pair<ld,int> pli;

const int N = 76;

ld mp[N+1];

int main() {
  ios::sync_with_stdio(0);
  int t; cin >> t;
  while (t--) {
    memset(mp, 0, sizeof mp);
    mp[N] = 1.0L;
    int n; cin >> n;
    vector<pli> v;
    for (int i = 0; i < n; i++) {
      int a, b; cin >> a >> b;
      v.push_back(make_pair(b/100.0L,a));
    }
    sort(v.begin(),v.end(),greater<pli>());
    for (const auto& p : v) {
      for (int i = 1; i <= N; i++) {
        if (mp[i] > 0) {
          int ni = max(0,i-p.second);
          mp[ni] = max(mp[ni],mp[i]*p.first);
        }
      }
    }
    cout << 100.0L*mp[0] << "\n";
  }
  return 0;
}
