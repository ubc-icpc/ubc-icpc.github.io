def foo(c, l):
	front = True
	for command in c:
		if command == 'R':
			if front:
				front = False
			else:
				front = True
		else:
			if len(l) == 0:
				return 'error'
			if front:
				l.pop(0)
			else:
				l.pop()

	if front:
		return l
	else:
		return l[::-1]

cases = int(raw_input())
outputs = []

for case in range(cases):
	
	commands = raw_input()
	elements = int(raw_input())
	L = raw_input()
	L = L[1:-1].split(',')
	
	if elements == 0:
		L = []
	L = [ int(x) for x in L ]
	
	outputs.append(foo(commands, L))

for output in outputs:
	print output
