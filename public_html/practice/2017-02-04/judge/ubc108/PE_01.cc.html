#include <bits/stdc++.h>
using namespace std;

int main() {
	char line[10001];
	char arr[100][100];
	int side, len, testcases; 
	scanf("%d", &testcases);
	while (testcases--) {
		scanf("%s", line);
		len = strlen(line);
		side = sqrt(len);
		int k = 0;
		for (int i = 0; i < side; i++)
			for (int j = 0; j < side; j++)
				arr[i][j] = line[k++];
		k = 0;
		for (int j = side-1; j >= 0; j--) 
			for (int i = 0; i < side; i++)
				line[k++] = arr[i][j];
		printf("%s \n", line);
	}	
	return 0;
}
