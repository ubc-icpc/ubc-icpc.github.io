#include <iostream>
#include <math.h>

using namespace std;

int main() {
	int t; cin >> t;
	while(t--) {
		string s; cin >> s;
		int n; cin >> n;
		int arr[n]; char c; cin >> c;
		for(int i = 0; i < n; i++) {
			cin >> arr[i] >> c;
		}
		if(n==0) cin >> c;
		bool front = true; int start = 0, end = n;
		for(int i = 0; i < s.size(); i++) {
			if(s[i] == 'R') front = !front;
			else {
				if(front)
					start++;
				else
					end--;
			}
		}
		if(start > end) {
			cout << "error" << endl;
			continue;
		}
		if(start == end) {
			cout << "[]" << endl;
			continue;
		}
		cout << '[';
		if(front) {
			for(int i = start; i < end - 1; i++) {
				cout << arr[i] << ',';
			}
			cout << arr[end - 1];
		}
		else {
			for(int i = end - 1; i > start; i--) {
				cout << arr[i] << ',';
			}
			cout << arr[start];
		}
		cout << ']' << endl;
	}
}
