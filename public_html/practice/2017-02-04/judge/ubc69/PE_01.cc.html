#include <iostream>
#include <math.h>

using namespace std;

int main() {
	int t; cin >> t;
	while(t--) {
		string s; cin >> s;
		int sl = 0;
		while(sl*sl < s.size()) sl++;
		char sq[sl][sl];
		for(int i = 0; i < s.size(); i++) {
			sq[i/sl][i%sl] = s[i];
		}
		for(int i = sl - 1; i >= 0; i--) {
			for(int j = 0; j < sl; j++) {
				cout << sq[j][i];
			}
		}
		cout << endl;
	}
}
