#include<bits/stdtr1c++.h>
using namespace std;

string S;

int solve() {
  int l = S.size();
  for (int k = 1; k <= l; ++k) {
    if (l % k) continue;
    int fail = false;
    for (int i = 0; i < l; ++i) {
      fail |= (S[i] != S[(i - i/k) % k]);
    }
    if (!fail) return k;
  }
  return -1;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> S;
  cout << solve() << endl;

  return 0;
}
