#include<bits/stdtr1c++.h>
using namespace std;

string S;
int dp[55], res = 1;

int main() {
  ios::sync_with_stdio(0);
  cin.tie();

  cin >> S;
  for (int i = 0; i < 55; ++i) {
    dp[i] = 1;
  } 
  for (int i = (int)S.size()-2; i >= 0; --i) {
    for (int j = i+1; j < (int)S.size(); ++j) {
      if (S[i] < S[j])
        dp[i] = max(dp[i], dp[j] + 1);
    }
    res = max(res, dp[i]);
  }

  cout << 26 - res << endl;

  return 0;
}
