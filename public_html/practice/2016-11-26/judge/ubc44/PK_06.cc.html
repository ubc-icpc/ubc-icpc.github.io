#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
typedef pair<pii, pii> piipii;
typedef pair<piipii, bool> piipii_O;
typedef vector<ll> vi;
typedef vector<vi > vvi;
typedef vector<bool> vb;
typedef vector<vb > vvb;

ll seq[1000005];
ll dpup[1000005]; 
ll dpdown[1000005]; 

int main() {
    memset(seq, 0, sizeof seq);
    memset(dpup, 0, sizeof dpup);
    memset(dpdown, 0, sizeof dpdown);
    int n;
    cin >> n;

    for (int i=0; i<n; i++) {
        cin >> seq[i];
    }

	dpup[0] = 1;
	dpdown[0] = 1;
    for (int i=1; i<n; i++) {
		if (seq[i-1] < seq[i]) {
			dpup[i] = dpdown[i-1] + 1;
			dpdown[i] = dpdown[i-1];
		} else if (seq[i-1] == seq[i]) {
			dpup[i] = dpup[i-1];
			dpdown[i] = dpdown[i-1];	
		} else if (seq[i-1] > seq[i]) {
			dpup[i] = dpup[i-1];
			dpdown[i] = dpup[i-1] + 1;
		}
    }

	cout << max(dpup[n-1], dpdown[n-1]) << endl;

}
