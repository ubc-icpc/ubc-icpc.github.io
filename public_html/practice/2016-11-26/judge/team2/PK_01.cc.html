#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define debug printf

int n;
int nums[1000000];

int solve(int d) {
    int total = 1;
    int last = nums[0];
    for (int i = 1; i < n; i++) {
        if (nums[i]*d > last*d) {
            total += 1, last = nums[i], d = -d;
        } else {
            last = nums[i];
        }
    }
    return total;
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }
    printf("%d\n", max(solve(1), solve(-1)));
    return 0;
}
