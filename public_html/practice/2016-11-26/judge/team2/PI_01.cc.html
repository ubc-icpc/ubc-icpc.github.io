#include<bits/stdc++.h>

using namespace std;

int main(){
    string s;
    cin >>s;
    int n = s.size();
    int ans = n;
    for(int i=1;i<n;i++){
        int pos = 0;
        int it = 0;
        bool length = true;
        for(auto c:s){
            int tocheck = (pos-it)%i;
            //cout <<i << " " << tocheck <<" "<< c << " " << s[tocheck] << endl;
            if(c!=s[tocheck]) {
                length = false;
                break;
            }
            pos++;
            if (!(pos%i)) it++;
        }
        if (length){
            ans = i;
            break;
        }
    }
    cout << ans << endl;
}
