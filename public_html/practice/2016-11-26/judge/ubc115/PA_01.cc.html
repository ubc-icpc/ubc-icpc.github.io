#include <bits/stdc++.h>
using namespace std;

char m[64];
int main() {
        ios::sync_with_stdio(0);
        string s;
        cin >> s;
        int count = 1;
        int mc = 1;
        memset(m, 1, sizeof (m));
        for (int i = 1; i < s.length(); i++) {
                for (int j = 0; j < i; j++) {
                        if (s[i] > s[j] && m[i] < m[j] + 1) {
                                m[i] = m[j] + 1;
                                mc = max(mc, (int) m[i]);
                        }
                }
        }
        
        if (mc >= 26) {
                cout << 0;
        } else {
                cout << 26 - mc;
        }
        return 0;
}
