#include <bits/stdc++.h>
using namespace std;

char a[64][64];
int v[64][64];

int dir[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
int m, n;

int dfs(int y, int x) {
        v[y][x] = 1;
        for (int i = 0; i < 4; i++) {
                int yp = y + dir[i][0];
                int xp = x + dir[i][1];
                if (xp >= 0 && yp >= 0 && xp < n && yp < m && !v[yp][xp] && (a[yp][xp] == 'L' || a[yp][xp] == 'C')) {
                        dfs(yp, xp);
                }
        }
}

int main() {
        ios::sync_with_stdio(0);
        cin >> m >> n;

        for (int j = 0; j < m; j++) {
                for (int i = 0; i < n; i++) {
                        cin >> a[j][i];
                }
        }

        int count = 0;
        for (int j = 0; j < m; j++) {
                for (int i = 0; i < n; i++) {
                        if (a[j][i] == 'L' && !v[j][i]) {
                                dfs(j, i);
                                count++;
                        }
                }
        }
        cout << count;
        return 0;
}
