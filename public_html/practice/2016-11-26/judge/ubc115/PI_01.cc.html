#include <bits/stdc++.h>
using namespace std;

string s;

bool ok(int a, int b, int k) {
        for (int i = 0; i < k; i++) {
                if (s[a * k + i] != s[b * k + (i + 1) % k]) {
                        return false;
                }
        }
        return true;
}

int main() {
        ios::sync_with_stdio(0);
        cin >> s;
        int sl = s.length();
        int mink = sl;
        int n;
        for (int k = sl - 1; k > 0; k--) {
                if (sl % k == 0) {
                        n = sl / k;
                        bool make_sense = true;
                        for (int i = 0; i < n - 1; i++) {
                                make_sense = ok(i, i + 1, k);
                                if (!make_sense) {
                                        break;
                                }
                        }
                        if (make_sense) {
                                mink = k;
                        }
                }
        }
        cout << mink;
        return 0;
}
