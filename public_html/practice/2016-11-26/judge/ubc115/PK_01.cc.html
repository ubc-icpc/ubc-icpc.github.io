#include <bits/stdc++.h>
using namespace std;

int a[1000001];
int main() {
        ios::sync_with_stdio(0);
        int n;
        cin >> n;
        for (int i = 0; i < n; i++) {
                cin >> a[i];
        }

        int count = 1;
        int inc = a[1] > a[0];
        for (int i = 1; i < n; i++) {
                if ((inc && a[i] > a[i - 1]) || (!inc && a[i] < a[i - 1])) {
                        count++;
                        inc ^= 1;
                }
        }
        
        cout << count;
        return 0;
}
