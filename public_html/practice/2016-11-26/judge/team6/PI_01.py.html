import sys

def solve(s):
  for k in range(1,len(s)+1):
    if len(s) % k:
      continue

    parts = [s[i*k:(i+1)*k] for i in range(len(s)/k)]
    ok = True
    for i in range(1,len(s)/k):
      prev = parts[i-1]
      cur = parts[i]
      if prev[:-1] != cur[1:] or prev[-1] != cur[0]:
        ok = False
        break

    if ok:
      return k

  return -1

print solve(sys.stdin.readline().strip())
