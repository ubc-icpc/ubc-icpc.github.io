import sys

def solve(arr):
  if len(arr) == 1:
    return 1
  # count number of flips
  flips = 1
  prev = arr[0]
  dr = not (arr[1] > arr[0])
  for e in arr[1:]:
    if e == prev:
      continue

    if dr != (e > prev):
      dr = (e > prev)
      flips += 1

    prev = e
  return flips

n = int(sys.stdin.readline())
arr = [int(sys.stdin.readline()) for i in range(n)]

print solve(arr)
