#include <iostream>
#include <vector>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int arr[n];
	for(int i=0; i<n; i++)
	{
		cin>>arr[i];
	} 
	int kohne = arr[0];
	int say = 1;
	bool bir = true;
	bool iki = true;
	for(int i=1; i<n; i++)
	{
		if(arr[i]>kohne)
		{
			if(bir)
			{
			   ++say;
			   bir =false;
			   iki = true;
		    }
		    kohne = arr[i];
		}
		else if(arr[i]<kohne)
		{
			if(iki)
			{
				++say;
				iki = false;
				bir = true;
			}
			kohne = arr[i];
		}
	}
	cout<<say;
}
