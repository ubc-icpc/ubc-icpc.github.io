#include <iostream>
#include <vector>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	char arr[n][m];
	vector<pair<int,int> > islands;
	int hamisi[n][m];
	for(int i=0; i<n; i++)
	{
		for(int j=0; j<m; j++)
		{
			hamisi[i][j]=0;
			cin>>arr[i][j];
			if(arr[i][j]=='L')
			{
				islands.push_back(make_pair(i,j));
			}
		}
	}
	int say = 0;
	for(int i=0; i<islands.size(); i++)
	{
		int x = islands[i].first;
		int y = islands[i].second;
		if(hamisi[x][y]!=1)
		{
			++say;
			vector<pair<int,int> > qow;
			qow.push_back(make_pair(x,y));
			int j = 0;
			while(j<qow.size())
			{
				x = qow[j].first;
				y = qow[j].second;
				if(x>0 && arr[x-1][y]!='W' && hamisi[x-1][y]!=1)
				{
					hamisi[x-1][y]=1;
					qow.push_back(make_pair(x-1,y));
				}
				if(y>0 && arr[x][y-1]!='W' && hamisi[x][y-1]!=1)
				{
					hamisi[x][y-1]=1;
					qow.push_back(make_pair(x,y-1));
				}
				if(x<n-1 && arr[x+1][y]!='W' && hamisi[x+1][y]!=1)
				{
					hamisi[x+1][y]=1;
					qow.push_back(make_pair(x+1,y));
				}
				if(y<m-1 && arr[x][y+1]!='W' && hamisi[x][y+1]!=1)
				{
					hamisi[x][y+1]=1;
					qow.push_back(make_pair(x,y+1));
				}
				++j;
			}
		}
	}
	cout<<say;
}
