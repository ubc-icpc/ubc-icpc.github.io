#include <iostream>
#include <vector>
#include <math.h>
#include <string>
using namespace std;
int main()
{
	string a;
	cin>>a;
	int n = a.length();
    int arr[n];
    for(int i=0; i<n; i++)
    {
    	arr[i]=1;
	}
	int max = 0;
	for(int i=0;i<n; i++)
	{
		int bir = a[i] - 'a';
		for(int j=0; j<i; j++)
		{
			int iki = a[j] - 'a';
			if(iki<bir && arr[j]+1>arr[i])
			{
				arr[i]= arr[j]+1;
			}
		}
		if(arr[i]>max)
		{
			max = arr[i];
		}
	}
	cout<<26-max;
}
