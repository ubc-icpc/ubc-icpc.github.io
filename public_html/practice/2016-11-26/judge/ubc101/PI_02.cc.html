#include <iostream>
#include <string>
#include <vector>
#include <math.h>
using namespace std;
int main()
{
	string a;
	cin>>a;
	int n = a.length();
	vector<int> mumkun;
	mumkun.push_back(1);
	int kok = sqrt(n);
	for(int i=2; i<=kok; i++)
	{
		if(!(n%i))
		{
			mumkun.push_back(i);
				mumkun.push_back(n/i);
		}
	}
	int min = n;
	for(int i=0; i<mumkun.size(); i++)
	{
		int nece = mumkun[i];
		bool John = true;
	//	cout<<nece<<endl;
		for(int j=nece; j<n; j+=nece)
		{
			int yoxlanasi = j-nece;
			int bir = yoxlanasi ;
			int iki = j+1;
			for(int x=0; x<nece; x++)
			{
			    if(iki==j+nece)
			    {
			    	iki = j;
				}
				if(a[bir]!=a[iki])
				{
	//				cout<<a[bir]<<" "<<a[iki]<<" "<<bir<<" "<<iki<<endl;
					John=false;
				}
				++bir;
				++iki;
			}
		}
		if(John && nece<min)
		{
			min = nece;
		}
	}
	cout<<min;
}
