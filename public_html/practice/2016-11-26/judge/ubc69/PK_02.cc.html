#include <iostream>
#include <string>
#include <vector>
#include <climits>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> s(n, 1);
    for (int i = 0; i < n; ++i)
        cin >> s[i];
    if (n == 1) {
        cout << "1\n";
        return 0;
    }

    bool up = true;
    int next = 1;
    while (next < s.size() && s[next] == s[0])
        ++next;
    if (next == s.size()) {
        cout << "1\n";
        return 0;
    }
    if (s[next] < s[0])
        up = false;

    int res = 1;
    for (int i = 1; i < s.size(); ++i) {
        if (up && s[i] > s[i-1]) {
            ++res;
            up = false;
        } else if (!up && s[i] < s[i-1]) {
            ++res;
            up = true;
        }
    }
    cout << res << '\n';
}
