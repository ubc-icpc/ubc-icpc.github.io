#include <iostream>
#include <string>

using namespace std;

int main() {
	string s; cin >> s;
	for(int i = 1; i <= s.size(); i++) {
		if(s.size() % i != 0) continue;
		string last = s.substr(0, i);
		bool found = true;
		for(int j = 1; j < s.size()/i; j++) {
			string curr = s.substr(i*j, i);
			//cout << curr << ' ' << last << ' ' << curr.substr(1, i-1) << ' ' << last.substr(0, i-1) << endl;
			if(last[i-1] == curr[0] && last.substr(0, i-1) == curr.substr(1, i-1)) {
				last = curr;
			}
			else {
				found = false;
				break;
			}
		}
		if(found) {
			cout << i << endl;
			return 0;
		}
	}
	return 0;
}
