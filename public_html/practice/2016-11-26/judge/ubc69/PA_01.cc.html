#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {
    string s;
    cin >> s;
    vector<int> arr(s.size(), 1);
    for (int i = 0; i < s.size(); ++i) {
        for (int j = 0; j < i; ++j)
            if (s[i] > s[j])
                arr[i] = max(arr[i], 1 + arr[j]);
    }

    int res = 0;
    for (int i = 0; i < arr.size(); ++i)
        res = max(res, arr[i]);
    cout << max(0, 26 - res) << '\n';
}
