#include <bits/stdc++.h>
#define endl '\n'
using namespace std;

int lis(string s) {
	vector<int> longest(s.length(), 1);
	for (int i = 1; i < s.length(); i++) {
		for (int j = 0; j < i; j++) {
			if (s[i] > s[j] && longest[i] < longest[j] + 1)
				longest[i] = longest[j] + 1;
		}
	}
	int max = 0;
	for (int i : longest) {
		if (i > max)
			max = i;
	}
	return max;
}

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	string s; cin >> s;
	cout << 26 - lis(s) << endl;
	return 0;
}
