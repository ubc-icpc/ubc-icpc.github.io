#include <bits/stdc++.h>
#define endl '\n'
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	string s; cin >> s;
	for (int i = 1; i < s.length(); i++) {
		if (s.length() % i == 0) {
			string prev = s.substr(0, i);
			bool good = 1;
			for (int j = i; j < s.length(); j += i) {
				string cur = s.substr(j, i);
				for (int k = 1; k <= cur.length(); k++) {
					if (cur[k % cur.length()] != prev[k-1]) {
						good = 0;
						break;
					}
				}
				if (!good)
					break;
				prev = cur;
			}
			if (good) {
				cout << i << endl;
				return 0;
			}
		}
	}
	cout << s.length() << endl;
	return 0;
}
