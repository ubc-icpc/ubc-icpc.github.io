#include <bits/stdc++.h>
#define endl '\n'
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	int n; cin >> n;
	vector<int> seq;
	while (n--) {
		int k; cin >> k;
		seq.push_back(k);
	}
	bool incr = seq[1] > seq[0];
	int len = (seq.size() == 1 || seq[1] == seq[0] ? 1 : 2);
	for (int i = 1; i < seq.size()-1; i++) {
		if ((seq[i+1] > seq[i] && !incr)
			|| (seq[i+1] < seq[i] && incr)) {
			len++;
			incr = !incr;
		}
	}
	cout << len << endl;
	return 0;
}
