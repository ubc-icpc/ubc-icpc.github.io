#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

string s;
int DP[50];
int dp(int i){
    if (i==0) return 1; 
    if (DP[i]!=-1) return DP[i];
    int ans=1;
    for (int x=0;x<i;x++) if (s[x]<s[i]) ans=max(ans,dp(x)+1);
    return DP[i]=ans;
}
int main(){
    ios::sync_with_stdio(0);
    cin>>s;
    memset(DP,-1,sizeof DP);
    int ans=0;
    for (int i=0;i<s.size();i++) ans=max(ans,dp(i));
    cout<<(26-ans)<<endl;
    return 0;
}
