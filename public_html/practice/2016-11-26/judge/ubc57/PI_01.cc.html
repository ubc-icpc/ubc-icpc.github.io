#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(0);
    string s; cin>>s;
    int ans=s.size();
    for (int i=1;i<=s.size();i++) if ((s.size()%i)==0){
        vector<string> v;
        for (int j=0;j<s.size();j+=i){
            string now="";
            for (int a=0;a<i;a++) now+=s[j+a];
            v.push_back(now);
        }
        bool good=true;
        for (int j=0;j<v.size()-1;j++){
            for (int a=0;a<i;a++) if (v[j][a]!=v[j+1][(a+1)%i]) good=false;
        }
        if (good){
            ans=i; break;
        }
        //for (string s:v) cout<<s<<' '; cout<<endl;
    }
    cout<<ans<<endl;
    return 0;
}
