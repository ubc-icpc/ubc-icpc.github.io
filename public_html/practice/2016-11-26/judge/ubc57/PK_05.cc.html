#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int A[1000005];
int main(){
    ios::sync_with_stdio(0);
    int n; cin>>n;
    for (int i=1;i<=n;i++) cin>>A[i];
    if (n<=2) cout<<n<<endl;
    else{
        vector<int> v; v.push_back(A[1]); v.push_back(A[2]);
        for (int i=3;i<=n;i++){
            if (v[v.size()-1]>v[v.size()-2]){
                if (A[i]>v[v.size()-1]) v[v.size()-1]=A[i];
                if (A[i]<v[v.size()-1]) v.push_back(A[i]);
            }
            else{
                if (A[i]<v[v.size()-1]) v[v.size()-1]=A[i];
                if (A[i]>v[v.size()-1]) v.push_back(A[i]);
            }
        }
        if (A[1]==A[2]) v.pop_back();
        cout<<v.size()<<endl;
    }
    return 0;
}
