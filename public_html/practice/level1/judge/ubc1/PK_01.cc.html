#include <bits/stdtr1c++.h>

using namespace std;

typedef long double ld;
typedef long long ll;
typedef pair<ll, ll> pii;
typedef complex<ld> pt;

int main() {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	map<int, ll> ans;
	ll clvl = 0, csum = 0;
	for (int i = 1; i <= 1000; i++) {
		clvl += i;
		csum += clvl;
		ans[i] = csum;
		if (i < 10) cerr << i << " " << clvl << " " << csum << endl;
	}
	
	while (t--) {
		static int ca = 0;
		int x; cin >> x;
		cout << ++ca << ": " << x << " ";
		cout << ans[x] << endl;
	}
	return 0;
}
