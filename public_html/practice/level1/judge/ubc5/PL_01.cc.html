#include <bits/stdtr1c++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(0);
  int t; cin >> t;
  while (t--) {
    int n, b; string d;
    cin >> n >> b >> d;
    int tot = 0;
    for (int i = 0; i < d.size(); i++)
      tot += d[i]-'0';
    cout << n << " " << tot%(b-1) << "\n";
  }
  return 0;
}
