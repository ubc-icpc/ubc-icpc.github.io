#include <bits/stdtr1c++.h>
using namespace std;

typedef long double ld;

const ld EPS = 1e-6;

ld l, lp;

inline ld Lp(ld d) {
  ld h = (l*l/4-d*d)/2/d;
  return (h+d)*2*atan2(l,2*h);
}

int main() {
  ios::sync_with_stdio(0);
  cout << fixed << setprecision(3);
  ld n, c;
  while (cin >> l >> n >> c && l >= 0) {
    lp = (1+n*c)*l;
    if (n == 0) {
      cout << "0.000\n";
      continue;
    }
    ld lo = 0, hi = 1e20;
    while (hi-lo > EPS) {
      ld mid = lo+(hi-lo)/2;
      if (Lp(mid) < lp)
        lo = mid;
      else
        hi = mid;
    }
    cout << lo << endl;
  }
  return 0;
}
