#include <bits/stdc++.h>
using namespace std;

int main() {
	ios::sync_with_stdio(0);

	double a, b, c, d, m, t;

	while (cin >> a >> b >> c >> d >> m >> t) {
		double epsilon = 1e-9;
		double left, mid, right;
		left = 0;
		right = t/(m*c); // this always works because positive numbers
		while (right-left > epsilon) {
			mid = (left + right) / 2;
			if (m*(a*mid*mid*mid + b*mid*mid + c*mid + d) < t)
				left = mid;
			else
				right = mid;
		}
		cout << fixed << setprecision(2) << ((int)(100*mid))/100.0 << '\n';
	}

	return 0;
}
