import java.io.PrintWriter;
import java.util.Scanner;

public class SpeedRacer {

    public static void main(String[] args) {
        Scanner reader = new Scanner(System.in);
        PrintWriter out = new PrintWriter(System.out);
        double a, b, c, d, m, t;

        while (reader.hasNextFloat()) {
            a = reader.nextFloat();
            b = reader.nextFloat();
            c = reader.nextFloat();
            d = reader.nextFloat();
            m = reader.nextFloat();
            t = reader.nextFloat();

            double v = binarySearch(a, b, c, d, m, t);

            double truncated = Math.floor(100 * v) / 100;

            out.printf("%.2f\n", truncated);
        }
        out.flush();
    }

    public static double binarySearch(double a, double b, double c, double d, double m, double t) {
        double lower = 0;
        double upper = t / (c * m);
        double x = t / m;

        while (upper - lower >= 1e-6) {
            double mid = (upper + lower) / 2;
            if ((a * mid * mid * mid) + (b * mid * mid) + (c * mid) + d > x) {
                upper = mid;
            } else {
                lower = mid;
            }
        }

        return upper;
    }
}
