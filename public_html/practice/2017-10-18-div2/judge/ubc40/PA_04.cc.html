#include <bits/stdc++.h>


using namespace std;
typedef long long int ll;

double a, b, c, d, m;

double bs(double goal) {
	double bot = 0;
	double top = 1e10;
	for(int i=0;i<1000000;++i) {
		double mid = (bot+top)/2;
		double val = (m/mid)*(a*mid*mid*mid*mid+b*mid*mid*mid+c*mid*mid+d*mid);
		if(val >= goal)
			top = mid;
		else
			bot = mid;
	}
	return bot;
}

int main()
{
	double t;
	cin >> a;
	while(cin) {
		cin >> b >> c >> d >> m >> t;
		double v = bs(t);
		if(v > .005) v -= .005;
		cout << fixed << setprecision(2) << v << endl;
		cin >> a;
	}

	return 0;
}
