#include <bits/stdc++.h>
using namespace std;

#define double long double
inline double fph(double a, double b, double c, double d, double v) {
    return a * v * v * v * v + b * v * v * v + c * v * v + d * v;
}

int main() {
    double a, b, c, d, m, t;
    while (cin >> a >> b >> c >> d >> m >> t) {
        double lo_v = 0.0001;
        double hi_v = 0.0001;
        double mid_v;
        while (hi_v * t > m * fph(a, b, c, d, hi_v)) {
            hi_v *= 10;
        }
        while (lo_v + 1e-6 < hi_v) {
            mid_v = (lo_v + hi_v) / 2.0;
            if (mid_v * t > m * fph(a, b, c, d, mid_v)) {
                lo_v = mid_v;
            } else {
                hi_v = mid_v;
            }
        }
        cout << fixed << setprecision(2) << ((int) (mid_v * 100)) / 100.0 << '\n';
    }
    return 0;
}
